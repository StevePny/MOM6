142 subroutine extractfluxes1d(G, fluxes, optics, nsw, j, dt, &
143  depthbeforescalingfluxes, useriverheatcontent, usecalvingheatcontent, &
144  h, t, net_h, net_heat, net_salt, pen_sw_bnd, tv)
145  type(ocean_grid_type), intent(in) :: g
146  type(forcing), intent(in) :: fluxes
147  type(optics_type), pointer :: optics
148  integer, intent(in) :: nsw, j
149  real, intent(in) :: dt, depthbeforescalingfluxes
150  logical, intent(in) :: useriverheatcontent, &
151  usecalvingheatcontent
152  real, dimension(NIMEM_,NKMEM_), intent(in) :: h, t
153  real, dimension(NIMEM_), intent(out) :: net_h, net_heat, net_salt
154  real, dimension(:,:), intent(out) :: pen_sw_bnd
155  type(thermo_var_ptrs), intent(inout) :: tv
156 ! This subroutine extracts the relevant buoyancy fluxes from the surface
157 ! fluxes type.
158 
159 ! (in) G - The ocean's grid structure.
160 ! (in) fluxes - A structure containing pointers to any possible
161 ! forcing fields. Unused fields have NULL ptrs.
162 ! (in) nsw - The number of bands of penetrating shortwave radiation.
163 ! (in) j - The j-index to work on.
164 ! (in) dt - The time step in s.
165 ! (in) DepthBeforeScalingFluxes - Minimum ocean thickness to allow before scaling away fluxes
166 ! (in) useRiverHeatContent - If true, apply river heat content
167 ! (in) useCalvingHeatContent& - If true, apply calving heat content
168 ! (in) h - Layer thickness, in m or kg m-2.
169 ! (in) T - Layer temperatures, in deg C.
170 ! (out) Net_H - The net mass flux (if non-Boussinsq) or volume flux (if
171 ! Boussinesq - i.e. the fresh water flux (P+R-E)) into the
172 ! ocean over a time step, in H.
173 ! (out) Net_heat - The net heating at the surface over a time step,
174 ! exclusive of heating that appears in Pen_SW, in K H.
175 ! (out) Net_salt - The surface salt flux into the ocean over a time step, psu H.
176 ! (out) Pen_SW_bnd - The penetrating shortwave heating at the sea surface
177 ! in each penetrating band, in K H, size nsw x NIMEM_.
178 ! (inout) tv - A structure containing pointers to any available
179 ! thermodynamic fields. Here it is used to keep track of the
180 ! actual heat flux associated with net mass fluxes into the
181 ! ocean.
182 
183  real :: htot(szi_(g)) ! The total depth of the ocean in m or kg m-2.
184  real :: pen_sw_tot(szi_(g)) ! The sum across all bands of Pen_SW, K H.
185  real :: ih_limit ! The inverse of the total depth at which the
186  ! surface fluxes start to be limited, in H-1.
187  real :: scale ! Scale scales away the fluxes when the total depth is
188  ! thinner than DepthBeforeScalingFluxes.
189  real :: irho_cp ! 1.0 / rho_0 * c_p.
190  real :: irho0, i_cp
191  character(len=200) :: mesg
192  integer :: is, ie, nz, i, k, n
193  ih_limit = 1.0 / (depthbeforescalingfluxes * g%m_to_H)
194  irho0 = 1.0 / g%Rho0 ; i_cp = 1.0 / fluxes%C_p
195  irho_cp = 1.0 / (g%H_to_kg_m2 * fluxes%C_p)
196  is = g%isc ; ie = g%iec ; nz = g%ke
197 
198  ! A whole load of error checking
199  if (nsw > 0) then ; if (nsw /= optics%nbands) call mom_error(warning, &
200  "mismatch in the number of bands of shortwave radiation in MOM_forcing_type extract_fluxes.")
201  endif
202 
203  if (.not.ASSOCIATED(fluxes%sw)) call mom_error(fatal, &
204  "MOM_forcing_type extractFluxes1d: fluxes%sw is not associated.")
205  if (.not.ASSOCIATED(fluxes%lw)) call mom_error(fatal, &
206  "MOM_forcing_type extractFluxes1d: fluxes%lw is not associated.")
207  if (.not.ASSOCIATED(fluxes%latent)) call mom_error(fatal, &
208  "MOM_forcing_type extractFluxes1d: fluxes%latent is not associated.")
209  if (.not.ASSOCIATED(fluxes%sens)) call mom_error(fatal, &
210  "MOM_forcing_type extractFluxes1d: fluxes%sens is not associated.")
211 
212  if (.not.ASSOCIATED(fluxes%evap)) call mom_error(fatal, &
213  "MOM_forcing_type extractFluxes1d: No evaporation defined in mixedlayer.")
214  if (.not.ASSOCIATED(fluxes%virt_precip)) call mom_error(fatal, &
215  "MOM_forcing_type extractFluxes1d: fluxes%virt_precip not defined in mixedlayer.")
216 
217  if ((.not.ASSOCIATED(fluxes%liq_precip)) .or. &
218  (.not.ASSOCIATED(fluxes%froz_precip))) call mom_error(fatal, &
219  "MOM_forcing_type extractFluxes1d: No precipitation defined in mixedlayer.")
220 
221  if (useriverheatcontent .and. &
222  .not.ASSOCIATED(fluxes%runoff_hflx)) call mom_error(fatal, &
223  "MOM_forcing_type extractFluxes1d: fluxes%runoff_hflx must be "//&
224  "assocated if USE_RIVER_HEAT_CONTENT is true.")
225 
226  if (usecalvingheatcontent .and. &
227  .not.ASSOCIATED(fluxes%calving_hflx)) call mom_error(fatal, &
228  "MOM_forcing_type extractFluxes1d: fluxes%calving_hflx must be "//&
229  "assocated if USE_CALVING_HEAT_CONTENT is true.")
230  do i=is,ie ; htot(i) = h(i,1) ; enddo
231  do k=2,nz ; do i=is,ie ; htot(i) = htot(i) + h(i,k) ; enddo ; enddo
232 
233  do i=is,ie
234  scale = 1.0
235  if (htot(i)*ih_limit < 1.0) scale = htot(i)*ih_limit
236 
237  ! Convert the penetrating shortwave forcing to K m.
238  pen_sw_tot(i) = 0.0
239  if (nsw >= 1) then
240  do n=1,nsw
241  pen_sw_bnd(n,i) = irho_cp*scale*dt * &
242  max(0.0, optics%sw_pen_band(n,i,j))
243  pen_sw_tot(i) = pen_sw_tot(i) + pen_sw_bnd(n,i)
244  enddo
245  else
246  pen_sw_bnd(1,i) = 0.0
247  endif
248 
249  ! Volume/mass fluxes
250  net_h(i) = dt * (scale * ((((( fluxes%liq_precip(i,j) &
251  + fluxes%froz_precip(i,j) ) &
252  + fluxes%evap(i,j) ) &
253  + fluxes%liq_runoff(i,j) ) &
254  + fluxes%virt_precip(i,j) ) &
255  + fluxes%froz_runoff(i,j) ) )
256  net_h(i) = g%kg_m2_to_H * net_h(i)
257  if (.not.g%Boussinesq .and. ASSOCIATED(fluxes%salt_flux)) &
258  net_h(i) = net_h(i) + (dt * g%kg_m2_to_H) * &
259  (scale * fluxes%salt_flux(i,j))
260 
261  ! Heat fluxes
262  net_heat(i) = scale * dt * irho_cp * ( fluxes%sw(i,j) + &
263  ((fluxes%lw(i,j) + fluxes%latent(i,j)) + fluxes%sens(i,j)) )
264 
265  if (ASSOCIATED(fluxes%heat_restore)) net_heat(i) = net_heat(i) + &
266  (scale * (dt * irho_cp)) * fluxes%heat_restore(i,j)
267 
268  if (useriverheatcontent) then
269  net_heat(i) = (net_heat(i) + (scale*(dt*irho_cp)) * fluxes%runoff_hflx(i,j)) - &
270  (g%kg_m2_to_H * (scale * dt)) * fluxes%liq_runoff(i,j) * t(i,1)
271  if (ASSOCIATED(tv%TempxPmE)) then
272  tv%TempxPmE(i,j) = tv%TempxPmE(i,j) + (scale * dt) * &
273  (i_cp*fluxes%runoff_hflx(i,j) - fluxes%liq_runoff(i,j)*t(i,1))
274  endif
275  endif
276 
277  if (usecalvingheatcontent) then
278  net_heat(i) = net_heat(i) + (scale*(dt*irho_cp)) * fluxes%calving_hflx(i,j) - &
279  (g%kg_m2_to_H * (scale * dt)) * fluxes%froz_runoff(i,j) * t(i,1)
280  if (ASSOCIATED(tv%TempxPmE)) then
281  tv%TempxPmE(i,j) = tv%TempxPmE(i,j) + (scale * dt) * &
282  (i_cp*fluxes%calving_hflx(i,j) - fluxes%froz_runoff(i,j)*t(i,1))
283  endif
284  endif
285 
286  if (num_msg<max_msg) then
287  if (pen_sw_tot(i) > 1.000001*irho_cp*scale*dt*fluxes%sw(i,j)) then
288  num_msg = num_msg + 1
289  write(mesg,'("Penetrating shortwave of ",1pe17.10, &
290  &" exceeds total shortwave of ",1pe17.10,&
291  &" at ",1pg11.4,"E, "1pg11.4,"N.")') &
292  pen_sw_tot(i),irho_cp*scale*dt*fluxes%sw(i,j),&
293  g%geoLonT(i,j),g%geoLatT(i,j)
294  call mom_error(warning,mesg)
295  endif
296  endif
297 
298  net_heat(i) = net_heat(i) - pen_sw_tot(i)
299 
300  ! Salt fluxes
301  net_salt(i) = 0.0
302  ! Convert salt_flux from kg (salt) m-2 s-1 to PSU m s-1.
303  if (ASSOCIATED(fluxes%salt_flux)) &
304  net_salt(i) = (scale * dt * (1000.0 * fluxes%salt_flux(i,j))) * g%kg_m2_to_H
305 
306  enddo ! i-loop
307 
308 end subroutine extractfluxes1d

362 subroutine calculatebuoyancyflux1d(G, fluxes, optics, h, Temp, Salt, tv, j, buoyancyFlux, netHeatMinusSW, netSalt )
363 ! Calculates buoyancy flux by adding up the heat, FW and salt fluxes and linearizing
364 ! about the surface state.
365 
366 ! Arguments
367  type(ocean_grid_type), intent(in) :: g ! Ocean grid
368  type(forcing), intent(in) :: fluxes ! Surface fluxes/forcing type
369  type(optics_type), pointer :: optics ! Optics for penetrating SW
370  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h ! Layer/level thicknesses (units of H)
371  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: temp ! Pot. temperature (degrees C)
372  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: salt ! Salinity (ppt)
373  type(thermo_var_ptrs), intent(inout) :: tv ! Thermodynamics type (out needed for tv%TempxPmE ????)
374  integer, intent(in) :: j ! j-index of row to work on
375  real, dimension(NIMEM_,NK_INTERFACE_), intent(inout) :: buoyancyflux ! Buoyancy flux (m2/s3)
376  real, dimension(NIMEM_), intent(inout) :: netheatminussw ! Heat flux excluding SW (K m/s)
377  real, dimension(NIMEM_), intent(inout) :: netsalt ! Salt flux (ppt m/s)
378 
379 ! Local variables
380  integer :: nsw, start, npts, k
381  real, parameter :: dt = 1. ! This is set to unity to return a rate from extractFluxes1d
382  real, dimension( SZI_(G) ) :: neth, netheat ! FW, heat fluxes in (m/s, K m/s, ppt m/s)
383  real, dimension( optics%nbands, SZI_(G) ) :: penswbnd ! SW penetration bands
384  real, dimension( SZI_(G) ) :: pressure ! Pressurea the surface ( Pa )
385  real, dimension( SZI_(G) ) :: drhodt, drhods ! Derivatives of density
386  logical :: useriverheatcontent, usecalvingheatcontent
387  real :: depthbeforescalingfluxes, gorho
388  integer, dimension(SZI_(G),SZK_(G)) :: ksort
389  real, dimension(SZI_(G),SZK_(G)+1) :: netpen
390  real :: h_limit_fluxes
391 
392  nsw = optics%nbands
393  useriverheatcontent = .false. ! ????????????????
394  usecalvingheatcontent = .false. ! ????????????????
395  depthbeforescalingfluxes = max( g%Angstrom, 1.e-30 )
396  pressure(:) = 0. ! Ignore atmospheric pressure
397  gorho = g%g_Earth / g%Rho0
398  start = 1 + g%isc - g%isd
399  npts = 1 + g%iec - g%isc
400 
401  do k=1, g%ke
402  ksort(:,k) = k
403  enddo
404  h_limit_fluxes = depthbeforescalingfluxes ! ?????????????
405 
406  ! Fetch the fresh-water, heat and salt fluxes
407  ! netH is the fresh-water flux
408  ! netSalt is the salt flux (typically zero except under sea-ice)
409  ! netHeat is the heat flux EXCEPT the penetrating SW
410  ! penSWbnd is the surface SW for each band
411  call extractfluxes1d(g, fluxes, optics, nsw, j, dt, &
412  depthbeforescalingfluxes, useriverheatcontent, usecalvingheatcontent, &
413  h(:,j,:), temp(:,j,:), neth, netheatminussw, netsalt, penswbnd, tv)
414 
415  ! Sum over bands and attenuate as a function of depth
416  ! netPen is the netSW as a function of depth
417  call sumswoverbands(g, h(:,j,:), 0.*h(:,j,:), 0.*h(:,j,1), optics%opacity_band(:,:,j,:), nsw, j, dt, &
418  h_limit_fluxes, .true., .true., &
419  ksort, penswbnd, netpen)
420 
421  ! Density derivatives
422  call calculate_density_derivs(temp(:,j,1), salt(:,j,1), pressure, &
423  drhodt, drhods, start, npts, tv%eqn_of_state)
424 
425  ! Adjust netSalt to reflect dillution effect of FW flux
426  netsalt(:) = netsalt(:) - salt(:,j,1) * neth * g%H_to_m
427 
428  ! Add in the SW heating for purposes of calculating the net surface buoyancy flux
429  !netHeat(:) = netHeatMinusSW(:) + sum( penSWbnd(:,:), dim=1 )
430  netheat(:) = netheatminussw(:) + netpen(:,1)
431 
432  ! Convert to a buoyancy flux, excluding penetrating SW heating
433  buoyancyflux(:,1) = - gorho * ( drhods(:) * netsalt(:) + drhodt(:) * netheat(:) ) ! m2/s3
434  ! We also have a penetrative buoyancy flux associaed with penetrative SW
435  do k=2, g%ke+1
436  buoyancyflux(:,k) = - gorho * ( drhodt(:) * netpen(:,k) ) ! m2/s3
437  enddo
438 
439 end subroutine calculatebuoyancyflux1d

473 subroutine absorbremainingsw(G, h, eps, htot, opacity_band, nsw, j, dt, &
474  h_limit_fluxes, correctabsorption, absorballsw, &
475  ksort, t, ttot, pen_sw_bnd)
476  type(ocean_grid_type), intent(in) :: g
477  real, dimension(NIMEM_,NKMEM_), intent(in) :: h, eps
478  real, dimension(NIMEM_), intent(in) :: htot
479  real, dimension(:,:,:), intent(in) :: opacity_band
480  integer, intent(in) :: nsw
481  integer, intent(in) :: j
482  real, intent(in) :: dt
483  real, intent(in) :: h_limit_fluxes
484  logical, intent(in) :: correctabsorption
485  logical, intent(in) :: absorballsw
486  integer, dimension(NIMEM_,NKMEM_), intent(in) :: ksort
487  real, dimension(NIMEM_,NKMEM_), intent(inout) :: t
488  real, dimension(NIMEM_), intent(inout) :: ttot
489  real, dimension(:,:), intent(inout) :: pen_sw_bnd
490 ! This subroutine applies shortwave heating below the mixed layer. In
491 ! addition, it causes all of the remaining SW radiation to be absorbed,
492 ! provided that the total water column thickness is greater than
493 ! H_limit_fluxes. For thinner water columns, the heating is scaled down
494 ! proportionately, the assumption being that the remaining heating (which is
495 ! left in Pen_SW) should go into an (unincluded) ocean bottom sediment layer.
496 
497 ! Arguments:
498 ! (in) G - The ocean's grid structure.
499 ! (in) h - Layer thickness, in m or kg m-2. (Intent in) The units
500 ! of h are referred to as H below.
501 ! (in) eps - The (small) thickness that must remain in each layer, and
502 ! which will not be subject to heating, in H.
503 ! (in) htot - The total mixed layer thickness, in H.
504 ! (in) opacity_band - The opacity in each band of penetrating shortwave
505 ! radiation, in H-1. The indicies are band, i, k.
506 ! (in) nsw - The number of bands of penetrating shortwave radiation.
507 ! (in) j - The j-index to work on.
508 ! (in) dt - The time step in s.
509 ! (inout) ksort - The density-sorted k-indicies.
510 ! (inout) T - The layer potential temperatures, in deg C.
511 ! (inout) Ttot - The depth integrated mixed layer temperature, in K H.
512 ! (inout) Pen_SW_bnd - The penetrating shortwave heating in each band that
513 ! hits the bottom and will be redistributed through
514 ! the water column, in K H, size nsw x NIMEM_.
515 
516  real :: h_heat(szi_(g)) ! The thickness of the water column that receives
517  ! the remaining shortwave radiation, in H.
518  real :: t_chg_above(szi_(g),szk_(g)) ! The temperature change of all the
519  ! thick layers above a given layer, in K. The net
520  ! change in the heat of a layer is the sum of
521  ! T_chg_above from all the layers below, plus any
522  ! contribution from absorbing radiation that hits
523  ! the bottom.
524  real :: t_chg(szi_(g)) ! The temperature change of thick layers due to
525  ! the remaining shortwave radiation, in K.
526  real :: pen_sw_rem(szi_(g)) ! The sum across all wavelength bands of the
527  ! penetrating shortwave heating that hits the bottom
528  ! and will be redistributed through the water column
529  ! in units of K H.
530  real :: sw_trans ! The fraction of shortwave radiation that is not
531  ! absorbed in a layer, nondimensional.
532  real :: unabsorbed ! The fraction of the shortwave radiation that
533  ! is not absorbed because the layers are too thin.
534  real :: ih_limit ! The inverse of the total depth at which the
535  ! surface fluxes start to be limited, in H-1.
536  real :: h_min_heat ! The minimum thickness layer that should get
537  ! heated, in H.
538  real :: opt_depth ! The optical depth of a layer, non-dim.
539  real :: exp_od ! exp(-opt_depth), non-dim.
540  real :: heat_bnd ! The heating due to absorption in the current
541  ! layer by the current band, including any piece that
542  ! is moved upward, in K H.
543  real :: swa ! The fraction of the absorbed shortwave that is
544  ! moved to layers above with correctAbsorption, ND.
545  logical :: sw_remains ! If true, some column has shortwave radiation that
546  ! was not entirely absorbed.
547  integer :: is, ie, nz, i, k, ks, n
548  sw_remains = .false.
549 
550  h_min_heat = 2.0*g%Angstrom + g%H_subroundoff
551  is = g%isc ; ie = g%iec ; nz = g%ke
552 
553  do i=is,ie ; h_heat(i) = htot(i) ; enddo
554 
555  ! Apply penetrating SW radiation to remaining parts of layers. Excessively thin
556  ! isopycnal layers are not heated.
557  do ks=1,nz
558 
559  do i=is,ie ; if (ksort(i,ks) > 0) then
560  k = ksort(i,ks)
561 
562  t_chg_above(i,k) = 0.0
563 
564  if (h(i,k) > 1.5*eps(i,k)) then
565  do n=1,nsw ; if (pen_sw_bnd(n,i) > 0.0) then
566  ! SW_trans is the SW that is transmitted THROUGH the layer
567  opt_depth = h(i,k) * opacity_band(n,i,k)
568  exp_od = exp(-opt_depth)
569  sw_trans = exp_od
570  ! Heating at a rate of less than 10-4 W m-2 = 10-3 K m / Century,
571  ! and of the layer in question less than 1 K / Century, can be
572  ! absorbed without further penetration.
573  if ((nsw*pen_sw_bnd(n,i)*sw_trans < g%m_to_H*2.5e-11*dt) .and. &
574  (nsw*pen_sw_bnd(n,i)*sw_trans < h(i,k)*dt*2.5e-8)) &
575  sw_trans = 0.0
576 
577  if (correctabsorption .and. (h_heat(i) > 0.0)) then
578  if (opt_depth > 1e-5) then
579  swa = ((opt_depth + (opt_depth + 2.0)*exp_od) - 2.0) / &
580  ((opt_depth + opacity_band(n,i,k) * h_heat(i)) * &
581  (1.0 - exp_od))
582  else
583  ! Use a Taylor's series expansion of the expression above for a
584  ! more accurate form with very small layer optical depths.
585  swa = h(i,k) * (opt_depth * (1.0 - opt_depth)) / &
586  ((h_heat(i) + h(i,k)) * (6.0 - 3.0*opt_depth))
587  endif
588  heat_bnd = pen_sw_bnd(n,i) * (1.0 - sw_trans)
589  t_chg_above(i,k) = t_chg_above(i,k) + (swa * heat_bnd) / h_heat(i)
590  t(i,k) = t(i,k) + ((1.0 - swa) * heat_bnd) / h(i,k)
591  else
592  t(i,k) = t(i,k) + pen_sw_bnd(n,i) * (1.0 - sw_trans) / h(i,k)
593  endif
594 
595  pen_sw_bnd(n,i) = pen_sw_bnd(n,i) * sw_trans
596  endif ; enddo
597  endif
598 
599  ! Add to the accumulated thickness above that could be heated.
600  ! Only layers greater than h_min_heat thick should get heated.
601  if (h(i,k) >= 2.0*h_min_heat) then
602  h_heat(i) = h_heat(i) + h(i,k)
603  elseif (h(i,k) > h_min_heat) then
604  h_heat(i) = h_heat(i) + (2.0*h(i,k) - 2.0*h_min_heat)
605  endif
606  endif ; enddo ! i loop
607  enddo ! k loop
608 
609  ! Apply heating above the layers in which it should have occurred to get the
610  ! correct mean depth of the shortwave that should be absorbed by each layer.
611 ! if (correctAbsorption) then
612 ! endif
613 
614 ! if (.not.absorbAllSW .and. .not.correctAbsorption) return
615  if (absorballsw) then
616 
617  ! Unless modified, there is no temperature change due to fluxes from the
618  ! bottom.
619  do i=is,ie ; t_chg(i) = 0.0 ; enddo
620 
621  ! If there is still shortwave radiation at this point, it could go into
622  ! the bottom (with a bottom mud model), or it could be redistributed back
623  ! through the water column.
624  do i=is,ie
625  pen_sw_rem(i) = pen_sw_bnd(1,i)
626  do n=2,nsw ; pen_sw_rem(i) = pen_sw_rem(i) + pen_sw_bnd(n,i) ; enddo
627  enddo
628  do i=is,ie ; if (pen_sw_rem(i) > 0.0) sw_remains = .true. ; enddo
629 
630  ih_limit = 1.0 / (h_limit_fluxes * g%m_to_H)
631  do i=is,ie ; if ((pen_sw_rem(i) > 0.0) .and. (h_heat(i) > 0.0)) then
632  if (h_heat(i)*ih_limit >= 1.0) then
633  t_chg(i) = pen_sw_rem(i) / h_heat(i) ; unabsorbed = 0.0
634  else
635  t_chg(i) = pen_sw_rem(i) * ih_limit
636  unabsorbed = 1.0 - h_heat(i)*ih_limit
637  endif
638  do n=1,nsw ; pen_sw_bnd(n,i) = unabsorbed * pen_sw_bnd(n,i) ; enddo
639  endif ; enddo
640 
641  do ks=nz,1,-1 ; do i=is,ie ; if (ksort(i,ks) > 0) then
642  k = ksort(i,ks)
643  if (t_chg(i) > 0.0) then
644  ! Only layers greater than h_min_heat thick should get heated.
645  if (h(i,k) >= 2.0*h_min_heat) then ; t(i,k) = t(i,k) + t_chg(i)
646  elseif (h(i,k) > h_min_heat) then
647  t(i,k) = t(i,k) + t_chg(i) * (2.0 - 2.0*h_min_heat/h(i,k))
648  endif
649  endif
650  ! Increase the heating for layers above.
651  t_chg(i) = t_chg(i) + t_chg_above(i,k)
652  endif ; enddo ; enddo
653  do i=is,ie ; ttot(i) = ttot(i) + t_chg(i) * htot(i) ; enddo
654  endif ! absorbAllSW
655 
656 end subroutine absorbremainingsw

663  real, dimension(NIMEM_,NKMEM_), intent(in) :: h, eps
664  real, dimension(NIMEM_), intent(in) :: htot

716  real :: exp_od ! exp(-opt_depth), non-dim.
717  real :: heat_bnd ! The heating due to absorption in the current
718  ! layer by the current band, including any piece that
719  ! is moved upward, in K H.
720  real :: swa ! The fraction of the absorbed shortwave that is
721  ! moved to layers above with correctAbsorption, ND.
722  logical :: sw_remains ! If true, some column has shortwave radiation that
723  ! was not entirely absorbed.

729 
730  pen_sw_bnd(:,:) = ipen_sw_bnd(:,:)
731  do i=is,ie ; h_heat(i) = htot(i) ; enddo
732  netpen(:,1) = sum( pen_sw_bnd(:,:), dim=1 ) ! Surface interface
733

735  ! isopycnal layers are not heated.

737 
738  do i=is,ie ; if (ksort(i,ks) > 0) then
739  k = ksort(i,ks)
740  netpen(i,k+1) = 0.
741 
742  if (h(i,k) > 1.5*eps(i,k)) then
743  do n=1,nsw ; if (pen_sw_bnd(n,i) > 0.0) then
744  ! SW_trans is the SW that is transmitted THROUGH the layer
745  opt_depth = h(i,k) * opacity_band(n,i,k)
746  exp_od = exp(-opt_depth)
747  sw_trans = exp_od
748  ! Heating at a rate of less than 10-4 W m-2 = 10-3 K m / Century,
749  ! and of the layer in question less than 1 K / Century, can be
750  ! absorbed without further penetration.
751  if ((nsw*pen_sw_bnd(n,i)*sw_trans < g%m_to_H*2.5e-11*dt) .and. &
752  (nsw*pen_sw_bnd(n,i)*sw_trans < h(i,k)*dt*2.5e-8)) &

754

805  else
806  unabsorbed = 0.0
807  endif
808  do n=1,nsw ; pen_sw_bnd(n,i) = unabsorbed * pen_sw_bnd(n,i) ; enddo
809  endif ; enddo
810 
811  endif ! absorbAllSW

816 subroutine mom_forcing_chksum(mesg, fluxes, G, haloshift)
817  character(len=*), intent(in) :: mesg
818  type(forcing), intent(in) :: fluxes
819  type(ocean_grid_type), intent(in) :: g
820  integer, optional, intent(in) :: haloshift
821 ! This subroutine writes out chksums for the model's basic state variables.
822 ! Arguments: mesg - A message that appears on the chksum lines.
823 ! (in) u - Zonal velocity, in m s-1.
824 ! (in) v - Meridional velocity, in m s-1.
825 ! (in) h - Layer thickness, in m.
826 ! (in) uh - Volume flux through zonal faces = u*h*dy, m3 s-1.

828 ! (in) G - The ocean's grid structure.

831

835  ! counts, there must be no redundant points, so all variables use is..ie
836  ! and js...je as their extent.

838  call uchksum(fluxes%taux, mesg//" fluxes%taux",g,haloshift=1)
839  if (associated(fluxes%tauy)) &

844  call hchksum(fluxes%buoy, mesg//" fluxes%buoy ",g,haloshift=hshift)

848  call hchksum(fluxes%sw_vis_dir, mesg//" fluxes%sw_vis_dir",g,haloshift=hshift)
849  if (associated(fluxes%sw_vis_dif)) &
850  call hchksum(fluxes%sw_vis_dif, mesg//" fluxes%sw_vis_dif",g,haloshift=hshift)
851  if (associated(fluxes%sw_nir_dir)) &
852  call hchksum(fluxes%sw_nir_dir, mesg//" fluxes%sw_nir_dir",g,haloshift=hshift)
853  if (associated(fluxes%sw_nir_dif)) &
854  call hchksum(fluxes%sw_nir_dif, mesg//" fluxes%sw_nir_dif",g,haloshift=hshift)
855  if (associated(fluxes%lw)) &
856  call hchksum(fluxes%lw, mesg//" fluxes%lw",g,haloshift=hshift)
857  if (associated(fluxes%latent)) &
858  call hchksum(fluxes%latent, mesg//" fluxes%latent",g,haloshift=hshift)
859  if (associated(fluxes%sens)) &
860  call hchksum(fluxes%sens, mesg//" fluxes%sens",g,haloshift=hshift)
861  if (associated(fluxes%evap)) &
862  call hchksum(fluxes%evap, mesg//" fluxes%evap",g,haloshift=hshift)
863  if (associated(fluxes%liq_precip)) &
864  call hchksum(fluxes%liq_precip, mesg//" fluxes%liq_precip",g,haloshift=hshift)
865  if (associated(fluxes%froz_precip)) &
866  call hchksum(fluxes%froz_precip, mesg//" fluxes%froz_precip",g,haloshift=hshift)
867  if (associated(fluxes%virt_precip)) &
868  call hchksum(fluxes%virt_precip, mesg//" fluxes%virt_precip",g,haloshift=hshift)
869  if (associated(fluxes%p_surf)) &
870  call hchksum(fluxes%p_surf, mesg//" fluxes%p_surf",g,haloshift=hshift)
871  if (associated(fluxes%salt_flux)) &
