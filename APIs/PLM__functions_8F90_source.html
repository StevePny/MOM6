24 subroutine plm_reconstruction( N, h, u, ppoly )
25 !------------------------------------------------------------------------------
26 ! Reconstruction by linear polynomials within each cell.
27 !
28 ! grid: one-dimensional grid (properly initialized)
29 ! ppoly: piecewise linear polynomial to be reconstructed (properly initialized)
30 ! u: cell averages
31 !
32 ! It is assumed that the size of the array 'u' is equal to the number of cells
33 ! defining 'grid' and 'ppoly'. No consistency check is performed here.
34 !------------------------------------------------------------------------------
35 
36  ! Arguments
37  integer, intent(in) :: n ! Number of cells
38  real, dimension(:), intent(in) :: h ! cell widths (size N)
39  real, dimension(:), intent(in) :: u ! cell averages (size N)
40  type(ppoly_t), intent(inout) :: ppoly
41 
42  ! Local variables
43  integer :: k ! loop index
44  real :: u_l, u_c, u_r ! left, center and right cell averages
45  real :: h_l, h_c, h_r ! left, center and right cell widths
46  real :: sigma_l, sigma_c, sigma_r ! left, center and right
47  ! van Leer slopes
48  real :: slope ! retained PLM slope
49  real :: a, b ! auxiliary variables
50  real, dimension(:,:), allocatable :: e_old
51 
52  ! Loop on interior cells
53  do k = 2,n-1
54 
55  ! Get cell averages
56  u_l = u(k-1)
57  u_c = u(k)
58  u_r = u(k+1)
59 
60  ! Get cell widths
61  h_l = h(k-1)
62  h_c = h(k)
63  h_r = h(k+1)
64 
65  ! Compute limited slope
66  sigma_l = 2.0 * ( u_c - u_l ) / h_c
67  sigma_c = 2.0 * ( u_r - u_l ) / ( h_l + 2.0*h_c + h_r )
68  sigma_r = 2.0 * ( u_r - u_c ) / h_c
69 
70  if ( (sigma_l * sigma_r) .GT. 0.0 ) then
71  slope = sign( min(abs(sigma_l),abs(sigma_c),abs(sigma_r)), sigma_c )
72  else
73  slope = 0.0
74  end if
75 
76  ! Determine coefficients of straight line.
77  ! CAUTION: the slope 'slope' is computed with respect to the global
78  ! coordinate x. The slope with respect to the normalized local
79  ! coordinate is sigma * h_c (d_P/d_xi = d_P/d_x * d_x/d_xi = d_P/d_x * h_c)
80  slope = slope * h_c
81 
82  a = u_c - 0.5 * slope
83  b = slope
84 
85  ppoly%coefficients(k,1) = a
86  ppoly%coefficients(k,2) = b
87  ppoly%E(k,1) = a
88  ppoly%E(k,2) = a+b
89 
90  end do ! end loop on interior cells
91 
92  ! In both boundary cells, a piecewise constant approximation is used.
93  ! Extrapolation -- if any -- to increase the accuracy is considered
94  ! in another routine, namely 'PLM_boundary_extrapolation'
95 
96  ! Left (top) boundary cell
97  slope = u(2) - u(1)
98  slope = 0.0
99  a = u(1) - 0.5 * slope
100  b = slope
101  ppoly%coefficients(1,1) = a
102  ppoly%coefficients(1,2) = b
103  ppoly%E(1,1) = a
104  ppoly%E(1,2) = a+b
105 
106  ! Right (bottom) boundary cell
107  slope = u(n) - u(n-1)
108  slope = 0.0
109  a = u(n) - 0.5 * slope
110  b = slope
111  ppoly%coefficients(n,1) = a
112  ppoly%coefficients(n,2) = b
113  ppoly%E(n,1) = a
114  ppoly%E(n,2) = a+b
115 
116  ! Second pass: we need to check for nonmonotonic discontinuous edge values.
117  ! When this occurs, the PLM slope is redefined so as to ensure monotonic edge
118  ! values across edges.
119  allocate( e_old(n,2) )
120  e_old = ppoly%E
121 
122  do k = 2,n-1
123 
124  ! By default, the right and left slopes are both equal to the original slope
125  slope = ppoly%coefficients(k,2)
126  sigma_l = slope
127  sigma_r = slope
128 
129  ! If the edge values across the left edge are nonmonotonic (relative to the
130  ! cell averages), the PLM slope is redefined in terms of the edge value
131  ! lying in the neighboring cell (the left cell).
132  if ( (u(k)-u(k-1)) * (e_old(k,1)-e_old(k-1,2)) .LT. 0.0 ) then
133  sigma_l = 2.0 * ( u(k) - e_old(k-1,2) )
134  end if
135 
136  ! If the edge values across the right edge are nonmonotonic (relative to the
137  ! cell averages), the PLM slope is redefined in terms of the edge value
138  ! lying in the neighboring cell (the right cell).
139  if ( (u(k+1)-u(k)) * (e_old(k+1,1)-e_old(k,2)) .LT. 0.0 ) then
140  sigma_r = 2.0 * ( e_old(k+1,1) - u(k) )
141  end if
142 
143  ! Take the minimum of both new slopes. If all edge values are
144  ! monotonic, the new slope is simply equal to the old one.
145  slope = sign( min( abs(sigma_l), abs(sigma_r) ), slope )
146 
147  a = u(k) - 0.5 * slope
148  b = slope
149 
150  ppoly%coefficients(k,1) = a
151  ppoly%coefficients(k,2) = b
152  ppoly%E(k,1) = a
153  ppoly%E(k,2) = a+b
154 
155  end do ! end loop on interior cells
156 
157  deallocate( e_old )
158 
159 end subroutine plm_reconstruction

163 ! plm boundary extrapolation
164 ! -----------------------------------------------------------------------------
165 subroutine plm_boundary_extrapolation( N, h, u, ppoly )
166 !------------------------------------------------------------------------------
167 ! Reconstruction by linear polynomials within boundary cells.
168 ! The left and right edge values in the left and right boundary cells,
169 ! respectively, are estimated using a linear extrapolation within the cells.
170 !
171 ! This extrapolation is EXACT when the underlying profile is linear.
172 !
173 ! grid: one-dimensional grid (properly initialized)
174 ! ppoly: piecewise linear polynomial to be reconstructed (properly initialized)
175 ! u: cell averages
176 !
177 ! It is assumed that the size of the array 'u' is equal to the number of cells
178 ! defining 'grid' and 'ppoly'. No consistency check is performed here.
179 !------------------------------------------------------------------------------
180 
181  ! Arguments
182  integer, intent(in) :: n ! Number of cells
183  real, dimension(:), intent(in) :: h ! cell widths (size N)
184  real, dimension(:), intent(in) :: u ! cell averages (size N)
185  type(ppoly_t), intent(inout) :: ppoly
186 
187  ! Local variables
188  real :: u0, u1 ! cell averages
189  real :: h0, h1 ! corresponding cell widths
190  real :: slope ! retained PLM slope
191 
192  ! -----------------------------------------
193  ! Left edge value in the left boundary cell
194  ! -----------------------------------------
195  h0 = h(1)
196  h1 = h(2)
197 
198  u0 = u(1)
199  u1 = u(2)
200 
201  ! The h2 scheme is used to compute the right edge value
202  ppoly%E(1,2) = (u0*h1 + u1*h0) / (h0 + h1)
203 
204  ! The standard PLM slope is computed as a first estimate for the
205  ! reconstruction within the cell
206  slope = 2.0 * ( ppoly%E(1,2) - u0 )
207 
208  ppoly%E(1,1) = u0 - 0.5 * slope
209  ppoly%E(1,2) = u0 + 0.5 * slope
210 
211  ppoly%coefficients(1,1) = ppoly%E(1,1)
212  ppoly%coefficients(1,2) = ppoly%E(1,2) - ppoly%E(1,1)
213 
214  ! ------------------------------------------
215  ! Right edge value in the left boundary cell
216  ! ------------------------------------------
217  h0 = h(n-1)
218  h1 = h(n)
219 
220  u0 = u(n-1)
221  u1 = u(n)
222 
223  ! The h2 scheme is used to compute the right edge value
224  ppoly%E(n,1) = (u0*h1 + u1*h0) / (h0 + h1)
225 
226  ! The standard PLM slope is computed as a first estimate for the
227  ! reconstruction within the cell
228  slope = 2.0 * ( u1 - ppoly%E(n,1) )
229 
230  ppoly%E(n,1) = u1 - 0.5 * slope
231  ppoly%E(n,2) = u1 + 0.5 * slope
232 
233  ppoly%coefficients(n,1) = ppoly%E(n,1)
234  ppoly%coefficients(n,2) = ppoly%E(n,2) - ppoly%E(n,1)
235 
236 end subroutine plm_boundary_extrapolation
