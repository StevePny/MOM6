23 !********+*********+*********+*********+*********+*********+*********+**
24 !* *
25 !* By Robert Hallberg, 1993-2012 *
26 !* *
27 !* This file contains code that does the time-stepping of the *
28 !* adiabatic dynamic core, in this case with an unsplit third-order *
29 !* Runge-Kutta time stepping scheme for the momentum and a forward- *
30 !* backward coupling between the momentum and continuity equations. *
31 !* This was the orignal unsplit time stepping scheme used in early *
32 !* versions of HIM and its precuror. While it is very simple and *
33 !* accurate, it is much less efficient that the split time stepping *
34 !* scheme for realistic oceanographic applications. It has been *
35 !* retained for all of these years primarily to verify that the split *
36 !* scheme is giving the right answers, and to debug the failings of *
37 !* the split scheme when it is not. The split time stepping scheme *
38 !* is now sufficiently robust that it should be first choice for *
39 !* almost any conceivable application, except perhaps from cases *
40 !* with just a few layers for which the exact timing of the high- *
41 !* frequency barotropic gravity waves is of paramount importance. *
42 !* This scheme is slightly more efficient than the other unsplit *
43 !* scheme that can be found in MOM_dynamics_unsplit_RK2.F90. *
44 !* *
45 !* The subroutine step_MOM_dyn_unsplit actually does the time *
46 !* stepping, while register_restarts_dyn_unsplit sets the fields *
47 !* that are found in a full restart file with this scheme, and *
48 !* initialize_dyn_unsplit initializes the cpu clocks that are * *
49 !* used in this module. For largely historical reasons, this module *
50 !* does not have its own control structure, but shares the same *
51 !* control structure with MOM.F90 and the other MOM_dynamics_... *
52 !* modules. *
53 !* *
54 !* Macros written all in capital letters are defined in MOM_memory.h. *
55 !* *
56 !* A small fragment of the grid is shown below: *
57 !* *
58 !* j+1 x ^ x ^ x At x: q, CoriolisBu *
59 !* j+1 > o > o > At ^: v, PFv, CAv, vh, diffv, tauy, vbt, vhtr *
60 !* j x ^ x ^ x At >: u, PFu, CAu, uh, diffu, taux, ubt, uhtr *
61 !* j > o > o > At o: h, bathyT, eta, T, S, tr *
62 !* j-1 x ^ x ^ x *
63 !* i-1 i i+1 *
64 !* i i+1 *
65 !* *
66 !* The boundaries always run through q grid points (x). *
67 !* *
68 !********+*********+*********+*********+*********+*********+*********+**

201 ! Arguments: u - The input and output zonal velocity, in m s-1.
202 ! (inout) v - The input and output meridional velocity, in m s-1.
203 ! (inout) h - The input and output layer thicknesses, in m or kg m-2,
204 ! depending on whether the Boussinesq approximation is made.
205 ! (in) tv - a structure pointing to various thermodynamic variables.
206 ! (inout) visc - A structure containing vertical viscosities, bottom drag
207 ! viscosities, and related fields.
208 ! (in) Time_local - The model time at the end of the time step.
209 ! (in) dt - The time step in s.
210 ! (in) fluxes - A structure containing pointers to any possible
211 ! forcing fields. Unused fields have NULL ptrs.
212 ! (in) p_surf_begin - A pointer (perhaps NULL) to the surface pressure
213 ! at the beginning of this dynamic step, in Pa.
214 ! (in) p_surf_end - A pointer (perhaps NULL) to the surface pressure
215 ! at the end of this dynamic step, in Pa.
216 ! (inout) uh - The zonal volume or mass transport, in m3 s-1 or kg s-1.
217 ! (inout) vh - The meridional volume or mass transport, in m3 s-1 or kg s-1.
218 ! (inout) uhtr - The accumulated zonal volume or mass transport since the last
219 ! tracer advection, in m3 or kg.
220 ! (inout) vhtr - The accumulated meridional volume or mass transport since the last
221 ! tracer advection, in m3 or kg.
222 ! (out) eta_av - The time-mean free surface height or column mass, in m or
223 ! kg m-2.
224 ! (in) G - The ocean's grid structure.
225 ! (in) CS - The control structure set up by initialize_dyn_unsplit.
226 ! (in) VarMix - A pointer to a structure with fields that specify the
227 ! spatially variable viscosities.
228 ! (inout) MEKE - A pointer to a structure containing fields related to
229 ! the Mesoscale Eddy Kinetic Energy.

231  real, dimension(SZI_(G),SZJ_(G),SZK_(G)) :: h_av, hp
232  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)) :: up, upp
233  real, dimension(SZI_(G),SZJB_(G),SZK_(G)) :: vp, vpp
234  real, dimension(:,:), pointer :: p_surf
235  real :: dt_pred ! The time step for the predictor part of the baroclinic
236  ! time stepping.
237  logical :: dyn_p_surf
238  integer :: i, j, k, is, ie, js, je, isq, ieq, jsq, jeq, nz
239  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
240  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
241  dt_pred = dt / 3.0
242 
243  h_av(:,:,:) = 0; hp(:,:,:) = 0
244  up(:,:,:) = 0; upp(:,:,:) = 0
245  vp(:,:,:) = 0; vpp(:,:,:) = 0
246 
247  dyn_p_surf = associated(p_surf_begin) .and. associated(p_surf_end)
248  if (dyn_p_surf) then
249  call safe_alloc_ptr(p_surf,g%isd,g%ied,g%jsd,g%jed) ; p_surf(:,:) = 0.0
250  else
251  p_surf => fluxes%p_surf
252  endif
253 
254 ! Matsuno's third order accurate three step scheme is used to step
255 ! all of the fields except h. h is stepped separately.
256 
257  if (cs%debug) then
258  call mom_state_chksum("Start First Predictor ", u, v, h, uh, vh, g)
259  endif
260 
261 ! diffu = horizontal viscosity terms (u,h)
262  call enable_averaging(dt,time_local, cs%diag)
263  call cpu_clock_begin(id_clock_horvisc)
264  call horizontal_viscosity(u, v, h, cs%diffu, cs%diffv, meke, varmix, &
265  g, cs%hor_visc_CSp)
266  call cpu_clock_end(id_clock_horvisc)
267  call disable_averaging(cs%diag)
268 
269 ! uh = u*h
270 ! hp = h + dt/2 div . uh
271  call cpu_clock_begin(id_clock_continuity)
272  call continuity(u, v, h, hp, uh, vh, dt*0.5, g, cs%continuity_CSp, obc=cs%OBC)
273  call cpu_clock_end(id_clock_continuity)
274  call cpu_clock_begin(id_clock_pass)
275  call pass_var(hp, g%Domain)
276  call pass_vector(uh, vh, g%Domain)
277  call cpu_clock_end(id_clock_pass)
278 
279  call enable_averaging(0.5*dt,time_local-set_time(int(0.5*dt)), cs%diag)
280 ! Here the first half of the thickness fluxes are offered for averaging.
281  if (cs%id_uh > 0) call post_data(cs%id_uh, uh, cs%diag)
282  if (cs%id_vh > 0) call post_data(cs%id_vh, vh, cs%diag)
283  call disable_averaging(cs%diag)
284 
285 ! h_av = (h + hp)/2
286 ! u = u + dt diffu
287  call cpu_clock_begin(id_clock_mom_update)
288  do k=1,nz
289  do j=js-2,je+2 ; do i=is-2,ie+2
290  h_av(i,j,k) = (h(i,j,k) + hp(i,j,k)) * 0.5
291  enddo ; enddo
292  do j=js,je ; do i=isq,ieq
293  u(i,j,k) = u(i,j,k) + dt * cs%diffu(i,j,k) * g%mask2dCu(i,j)
294  enddo ; enddo
295  do j=jsq,jeq ; do i=is,ie
296  v(i,j,k) = v(i,j,k) + dt * cs%diffv(i,j,k) * g%mask2dCv(i,j)
297  enddo ; enddo
298  do j=js-2,je+2 ; do i=isq-2,ieq+2
299  uhtr(i,j,k) = uhtr(i,j,k) + 0.5*dt*uh(i,j,k)
300  enddo ; enddo
301  do j=jsq-2,jeq+2 ; do i=is-2,ie+2
302  vhtr(i,j,k) = vhtr(i,j,k) + 0.5*dt*vh(i,j,k)
303  enddo ; enddo
304  enddo
305  call cpu_clock_end(id_clock_mom_update)
306  call cpu_clock_begin(id_clock_pass)
307  call pass_vector(u, v, g%Domain)
308  call cpu_clock_end(id_clock_pass)
309 
310 ! CAu = -(f+zeta)/h_av vh + d/dx KE
311  call cpu_clock_begin(id_clock_cor)
312  call coradcalc(u, v, h_av, uh, vh, cs%CAu, cs%CAv, cs%ADp, g, &
313  cs%CoriolisAdv_CSp)
314  call cpu_clock_end(id_clock_cor)
315 
316 ! PFu = d/dx M(h_av,T,S)
317  call cpu_clock_begin(id_clock_pres)
318  if (dyn_p_surf) then ; do j=js-2,je+2 ; do i=is-2,ie+2
319  p_surf(i,j) = 0.75*p_surf_begin(i,j) + 0.25*p_surf_end(i,j)
320  enddo ; enddo ; endif
321  call pressureforce(h_av, tv, cs%PFu, cs%PFv, g, &
322  cs%PressureForce_CSp, cs%ALE_CSp, p_surf)
323  call cpu_clock_end(id_clock_pres)
324 
325 ! up = u + dt_pred * (PFu + CAu)
326  call cpu_clock_begin(id_clock_mom_update)
327  do k=1,nz ; do j=js,je ; do i=isq,ieq
328  up(i,j,k) = g%mask2dCu(i,j) * (u(i,j,k) + dt_pred * &
329  (cs%PFu(i,j,k) + cs%CAu(i,j,k)))
330  enddo ; enddo ; enddo
331  do k=1,nz ; do j=jsq,jeq ; do i=is,ie
332  vp(i,j,k) = g%mask2dCv(i,j) * (v(i,j,k) + dt_pred * &
333  (cs%PFv(i,j,k) + cs%CAv(i,j,k)))
334  enddo ; enddo ; enddo
335  call cpu_clock_end(id_clock_mom_update)
336 
337  if (cs%debug) then
338  call mom_state_chksum("Predictor 1", up, vp, h_av, uh, vh, g)
339  call mom_accel_chksum("Predictor 1 accel", cs%CAu, cs%CAv, cs%PFu, cs%PFv,&
340  cs%diffu, cs%diffv, g)
341  endif
342 
343 ! visc contains viscosity and BBL thickness (u_in,h_in)
344  if (visc%calc_bbl) then
345  call enable_averaging(visc%bbl_calc_time_interval, &
346  time_local-set_time(int(dt)), cs%diag)
347  call set_viscous_bbl(u, v, h_av, tv, visc, g, cs%set_visc_CSp)
348  call cpu_clock_begin(id_clock_pass)
349  if (associated(visc%Ray_u) .and. associated(visc%Ray_v)) &

351  to_all+scalar_pair, cgrid_ne)
352  if (associated(visc%kv_bbl_u) .and. associated(visc%kv_bbl_v)) then
353  call pass_vector(visc%bbl_thick_u, visc%bbl_thick_v, g%Domain, &
354  to_all+scalar_pair, cgrid_ne, complete=.false.)
355  call pass_vector(visc%kv_bbl_u, visc%kv_bbl_v, g%Domain, &
356  to_all+scalar_pair, cgrid_ne)
357  endif
358  call cpu_clock_end(id_clock_pass)
359  call disable_averaging(cs%diag)
360  visc%calc_bbl = .false.
361  endif
362 
363  ! up <- up + dt/2 d/dz visc d/dz up
364  call cpu_clock_begin(id_clock_vertvisc)
365  call enable_averaging(dt, time_local, cs%diag)
366  call set_viscous_ml(u, v, h_av, tv, fluxes, visc, dt*0.5, g, &
367  cs%set_visc_CSp)
368  call disable_averaging(cs%diag)
369  call vertvisc_coef(up, vp, h_av, fluxes, visc, dt*0.5, g, cs%vertvisc_CSp)
370  call vertvisc(up, vp, h_av, fluxes, visc, dt*0.5, cs%OBC, cs%ADp, cs%CDp, g, &
371  cs%vertvisc_CSp)
372  call cpu_clock_end(id_clock_vertvisc)
373  call cpu_clock_begin(id_clock_pass)
374  call pass_vector(up, vp, g%Domain)
375  call cpu_clock_end(id_clock_pass)
376 
377 ! uh = up * hp
378 ! h_av = hp + dt/2 div . uh
379  call cpu_clock_begin(id_clock_continuity)
380  call continuity(up, vp, hp, h_av, uh, vh, &
381  (0.5*dt), g, cs%continuity_CSp, obc=cs%OBC)
382  call cpu_clock_end(id_clock_continuity)
383  call cpu_clock_begin(id_clock_pass)
384  call pass_var(h_av, g%Domain)
385  call pass_vector(uh, vh, g%Domain)
386  call cpu_clock_end(id_clock_pass)

388 ! h_av <- (hp + h_av)/2
389  do k=1,nz ; do j=js-2,je+2 ; do i=is-2,ie+2
390  h_av(i,j,k) = (hp(i,j,k) + h_av(i,j,k)) * 0.5
391  enddo ; enddo ; enddo
392 
393 ! CAu = -(f+zeta(up))/h_av vh + d/dx KE(up)
394  call cpu_clock_begin(id_clock_cor)
395  call coradcalc(up, vp, h_av, uh, vh, cs%CAu, cs%CAv, cs%ADp, &
396  g, cs%CoriolisAdv_CSp)
397  call cpu_clock_end(id_clock_cor)

399 ! PFu = d/dx M(h_av,T,S)
400  call cpu_clock_begin(id_clock_pres)
401  if (dyn_p_surf) then ; do j=js-2,je+2 ; do i=is-2,ie+2
402  p_surf(i,j) = 0.25*p_surf_begin(i,j) + 0.75*p_surf_end(i,j)
403  enddo ; enddo ; endif
404  call pressureforce(h_av, tv, cs%PFu, cs%PFv, g, &
405  cs%PressureForce_CSp, cs%ALE_CSp, p_surf)
406  call cpu_clock_end(id_clock_pres)
407 
408 ! upp = u + dt/2 * ( PFu + CAu )
409  call cpu_clock_begin(id_clock_mom_update)
410  do k=1,nz ; do j=js,je ; do i=isq,ieq
411  upp(i,j,k) = g%mask2dCu(i,j) * (u(i,j,k) + dt * 0.5 * &
412  (cs%PFu(i,j,k) + cs%CAu(i,j,k)))
413  enddo ; enddo ; enddo
414  do k=1,nz ; do j=jsq,jeq ; do i=is,ie
415  vpp(i,j,k) = g%mask2dCv(i,j) * (v(i,j,k) + dt * 0.5 * &
416  (cs%PFv(i,j,k) + cs%CAv(i,j,k)))
417  enddo ; enddo ; enddo
418  call cpu_clock_end(id_clock_mom_update)
419 
420  if (cs%debug) then
421  call mom_state_chksum("Predictor 2", upp, vpp, h_av, uh, vh, g)
422  call mom_accel_chksum("Predictor 2 accel", cs%CAu, cs%CAv, cs%PFu, cs%PFv,&
423  cs%diffu, cs%diffv, g)
424  endif
425 
426 ! upp <- upp + dt/2 d/dz visc d/dz upp
427  call cpu_clock_begin(id_clock_vertvisc)
428  call vertvisc_coef(upp, vpp, hp, fluxes, visc, dt*0.5, g, cs%vertvisc_CSp)
429  call vertvisc(upp, vpp, hp, fluxes, visc, dt*0.5, cs%OBC, cs%ADp, cs%CDp, g, &
430  cs%vertvisc_CSp)
431  call cpu_clock_end(id_clock_vertvisc)
432  call cpu_clock_begin(id_clock_pass)
433  call pass_vector(upp, vpp, g%Domain)
434  call cpu_clock_end(id_clock_pass)
435 
436 ! uh = upp * hp
437 ! h = hp + dt/2 div . uh
438  call cpu_clock_begin(id_clock_continuity)
439  call continuity(upp, vpp, hp, h, uh, vh, &
440  (dt*0.5), g, cs%continuity_CSp, obc=cs%OBC)
441  call cpu_clock_end(id_clock_continuity)
442  call cpu_clock_begin(id_clock_pass)
443  call pass_var(h, g%Domain)
444  call pass_vector(uh, vh, g%Domain)
445  call cpu_clock_end(id_clock_pass)
446 
447  call enable_averaging(0.5*dt, time_local, cs%diag)
448 ! Here the second half of the thickness fluxes are offered for averaging.
449  if (cs%id_uh > 0) call post_data(cs%id_uh, uh, cs%diag)
450  if (cs%id_vh > 0) call post_data(cs%id_vh, vh, cs%diag)
451  call disable_averaging(cs%diag)
452  call enable_averaging(dt, time_local, cs%diag)
453 
454 ! h_av = (h + hp)/2
455  do k=1,nz
456  do j=js-2,je+2 ; do i=is-2,ie+2
457  h_av(i,j,k) = 0.5*(h(i,j,k) + hp(i,j,k))
458  enddo ; enddo
459  do j=js-2,je+2 ; do i=isq-2,ieq+2
460  uhtr(i,j,k) = uhtr(i,j,k) + 0.5*dt*uh(i,j,k)
461  enddo ; enddo
462  do j=jsq-2,jeq+2 ; do i=is-2,ie+2
463  vhtr(i,j,k) = vhtr(i,j,k) + 0.5*dt*vh(i,j,k)
464  enddo ; enddo
465  enddo
466 
467 ! CAu = -(f+zeta(upp))/h_av vh + d/dx KE(upp)
468  call cpu_clock_begin(id_clock_cor)
469  call coradcalc(upp, vpp, h_av, uh, vh, cs%CAu, cs%CAv, cs%ADp, &
470  g, cs%CoriolisAdv_CSp)
471  call cpu_clock_end(id_clock_cor)
472 
473 ! PFu = d/dx M(h_av,T,S)
474  call cpu_clock_begin(id_clock_pres)
475  call pressureforce(h_av, tv, cs%PFu, cs%PFv, g, &
476  cs%PressureForce_CSp, cs%ALE_CSp, p_surf)
477  call cpu_clock_end(id_clock_pres)
478 
479 ! u = u + dt * ( PFu + CAu )
480  do k=1,nz ; do j=js,je ; do i=isq,ieq
481  u(i,j,k) = g%mask2dCu(i,j) * (u(i,j,k) + dt * &
482  (cs%PFu(i,j,k) + cs%CAu(i,j,k)))
483  enddo ; enddo ; enddo
484  do k=1,nz ; do j=jsq,jeq ; do i=is,ie
485  v(i,j,k) = g%mask2dCv(i,j) * (v(i,j,k) + dt * &
486  (cs%PFv(i,j,k) + cs%CAv(i,j,k)))
487  enddo ; enddo ; enddo
488 
489 ! u <- u + dt d/dz visc d/dz u
490  call cpu_clock_begin(id_clock_vertvisc)
491  call vertvisc_coef(u, v, h_av, fluxes, visc, dt, g, cs%vertvisc_CSp)
492  call vertvisc(u, v, h_av, fluxes, visc, dt, cs%OBC, cs%ADp, cs%CDp, g, &
493  cs%vertvisc_CSp, cs%taux_bot, cs%tauy_bot)
494  call cpu_clock_end(id_clock_vertvisc)
495  call cpu_clock_begin(id_clock_pass)
496  call pass_vector(u, v, g%Domain)
497  call cpu_clock_end(id_clock_pass)
498 
499  if (cs%debug) then
500  call mom_state_chksum("Corrector", u, v, h, uh, vh, g)
501  call mom_accel_chksum("Corrector accel", cs%CAu, cs%CAv, cs%PFu, cs%PFv, &
502  cs%diffu, cs%diffv, g)
503  endif
504 
505  if (g%Boussinesq) then
506  do j=js,je ; do i=is,ie ; eta_av(i,j) = -g%bathyT(i,j) ; enddo ; enddo
507  else
508  do j=js,je ; do i=is,ie ; eta_av(i,j) = 0.0 ; enddo ; enddo
509  endif
510  do k=1,nz ; do j=js,je ; do i=is,ie
511  eta_av(i,j) = eta_av(i,j) + h_av(i,j,k)
512  enddo ; enddo ; enddo
513 
514  if (dyn_p_surf) deallocate(p_surf)
515 
516 ! Here various terms used in to update the momentum equations are
517 ! offered for averaging.
518  if (cs%id_PFu > 0) call post_data(cs%id_PFu, cs%PFu, cs%diag)
519  if (cs%id_PFv > 0) call post_data(cs%id_PFv, cs%PFv, cs%diag)
520  if (cs%id_CAu > 0) call post_data(cs%id_CAu, cs%CAu, cs%diag)
521  if (cs%id_CAv > 0) call post_data(cs%id_CAv, cs%CAv, cs%diag)
522 
523 end subroutine step_mom_dyn_unsplit
