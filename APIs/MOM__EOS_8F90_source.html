348 subroutine int_specific_vol_dp(T, S, p_t, p_b, alpha_ref, G, EOS, &
349  dza, intp_dza, intx_dza, inty_dza, halo_size)
350  real, dimension(NIMEM_,NJMEM_), intent(in) :: t, s, p_t, p_b
351  real, intent(in) :: alpha_ref
352  type(ocean_grid_type), intent(in) :: g
353  type(eos_type), pointer :: eos
354  real, dimension(NIMEM_,NJMEM_), intent(out) :: dza
355  real, dimension(NIMEM_,NJMEM_), optional, intent(out) :: intp_dza
356  real, dimension(NIMEMB_,NJMEM_), optional, intent(out) :: intx_dza
357  real, dimension(NIMEM_,NJMEMB_), optional, intent(out) :: inty_dza
358  integer, optional, intent(in) :: halo_size
359 ! This subroutine calculates analytical and nearly-analytical integrals in
360 ! pressure across layers of geopotential anomalies, which are required for
361 ! calculating the finite-volume form pressure accelerations in a non-Boussinesq
362 ! model. There are essentially no free assumptions, apart from the use of
363 ! Bode's rule to do the horizontal integrals, and from a truncation in the
364 ! series for log(1-eps/1+eps) that assumes that |eps| < .
365 !
366 ! Arguments: T - potential temperature relative to the surface in C.
367 ! (in) S - salinity in PSU.
368 ! (in) p_t - pressure at the top of the layer in Pa.
369 ! (in) p_b - pressure at the top of the layer in Pa.
370 ! (in) alpha_ref - A mean specific volume that is subtracted out to reduce
371 ! the magnitude of each of the integrals, m3 kg-1.
372 ! The calculation is mathematically identical with
373 ! different values of alpha_ref, but this reduces the
374 ! effects of roundoff.
375 ! (in) G - The ocean's grid structure.
376 ! (in) EOS - The equation of state type.
377 ! (out) dza - The change in the geopotential anomaly across the layer,
378 ! in m2 s-2.
379 ! (out,opt) intp_dza - The integral in pressure through the layer of the
380 ! geopotential anomaly relative to the anomaly at the
381 ! bottom of the layer, in Pa m2 s-2.
382 ! (out,opt) intx_dza - The integral in x of the difference between the
383 ! geopotential anomaly at the top and bottom of the layer
384 ! divided by the x grid spacing, in m2 s-2.
385 ! (out,opt) inty_dza - The integral in y of the difference between the
386 ! geopotential anomaly at the top and bottom of the layer
387 ! divided by the y grid spacing, in m2 s-2.
388 ! (in,opt) halo_size - The width of halo points on which to calculate dza.
389 
390  if (.not.associated(eos)) call mom_error(fatal, &
391  "int_specific_vol_dp called with an unassociated EOS_type EOS.")
392 
393  if (eos%EOS_quadrature) then
394  call int_spec_vol_dp_generic(t, s, p_t, p_b, alpha_ref, g, eos, &
395  dza, intp_dza, intx_dza, inty_dza, halo_size)
396  else ; select case (eos%form_of_EOS)
397  case (eos_linear)
398  call int_spec_vol_dp_linear(t, s, p_t, p_b, alpha_ref, g, eos%Rho_T0_S0, &
399  eos%dRho_dT, eos%dRho_dS, dza, intp_dza, &
400  intx_dza, inty_dza, halo_size)
401  case (eos_wright)
402  call int_spec_vol_dp_wright(t, s, p_t, p_b, alpha_ref, g, dza, &
403  intp_dza, intx_dza, inty_dza, halo_size)
404  case default
405  call int_spec_vol_dp_generic(t, s, p_t, p_b, alpha_ref, g, eos, &
406  dza, intp_dza, intx_dza, inty_dza, halo_size)
407  end select ; endif
408 
409 end subroutine int_specific_vol_dp

411 subroutine int_density_dz(T, S, z_t, z_b, rho_ref, rho_0, G_e, G, EOS, &
412  dpa, intz_dpa, intx_dpa, inty_dpa)
413  real, dimension(NIMEM_,NJMEM_), intent(in) :: t, s, z_t, z_b
414  real, intent(in) :: rho_ref, rho_0, g_e
415  type(ocean_grid_type), intent(in) :: g
416  type(eos_type), pointer :: eos
417  real, dimension(NIMEM_,NJMEM_), intent(out) :: dpa
418  real, dimension(NIMEM_,NJMEM_), optional, intent(out) :: intz_dpa
419  real, dimension(NIMEMB_,NJMEM_), optional, intent(out) :: intx_dpa
420  real, dimension(NIMEM_,NJMEMB_), optional, intent(out) :: inty_dpa
421 ! This subroutine calculates analytical and nearly-analytical integrals of
422 ! pressure anomalies across layers, which are required for calculating the
423 ! finite-volume form pressure accelerations in a Boussinesq model. The one
424 ! potentially dodgy assumtion here is that rho_0 is used both in the denominator
425 ! of the accelerations, and in the pressure used to calculated density (the
426 ! latter being -z*rho_0*G_e). These two uses could be separated if need be.
427 !
428 ! Arguments: T - potential temperature relative to the surface in C.
429 ! (in) S - salinity in PSU.
430 ! (in) z_t - height at the top of the layer in m.
431 ! (in) z_b - height at the top of the layer in m.
432 ! (in) rho_ref - A mean density, in kg m-3, that is subtracted out to reduce
433 ! the magnitude of each of the integrals.
434 ! (The pressure is calucated as p~=-z*rho_0*G_e.)
435 ! (in) rho_0 - A density, in kg m-3, that is used to calculate the pressure
436 ! (as p~=-z*rho_0*G_e) used in the equation of state.
437 ! (in) G_e - The Earth's gravitational acceleration, in m s-2.
438 ! (in) G - The ocean's grid structure.
439 ! (in) EOS - The equation of state type.
440 ! (out) dpa - The change in the pressure anomaly across the layer,
441 ! in Pa.
442 ! (out,opt) intz_dpa - The integral through the thickness of the layer of the
443 ! pressure anomaly relative to the anomaly at the top of
444 ! the layer, in Pa m.
445 ! (out,opt) intx_dpa - The integral in x of the difference between the
446 ! pressure anomaly at the top and bottom of the layer
447 ! divided by the x grid spacing, in Pa.
448 ! (out,opt) inty_dpa - The integral in y of the difference between the
449 ! pressure anomaly at the top and bottom of the layer
450 ! divided by the y grid spacing, in Pa.
451 
452  if (.not.associated(eos)) call mom_error(fatal, &
453  "int_density_dz called with an unassociated EOS_type EOS.")
454 
455  if (eos%EOS_quadrature) then
456  call int_density_dz_generic(t, s, z_t, z_b, rho_ref, rho_0, g_e, g, &
457  eos, dpa, intz_dpa, intx_dpa, inty_dpa)
458  else ; select case (eos%form_of_EOS)
459  case (eos_linear)
460  call int_density_dz_linear(t, s, z_t, z_b, rho_ref, rho_0, g_e, g, &
461  eos%Rho_T0_S0, eos%dRho_dT, eos%dRho_dS, &
462  dpa, intz_dpa, intx_dpa, inty_dpa)
463  case (eos_wright)
464  call int_density_dz_wright(t, s, z_t, z_b, rho_ref, rho_0, g_e, g, &

594 subroutine int_density_dz_generic(T, S, z_t, z_b, rho_ref, rho_0, G_e, G, &
595  eos, dpa, intz_dpa, intx_dpa, inty_dpa)
596  real, dimension(NIMEM_,NJMEM_), intent(in) :: t, s, z_t, z_b
597  real, intent(in) :: rho_ref, rho_0, g_e
598  type(ocean_grid_type), intent(in) :: g
599  type(eos_type), pointer :: eos
600  real, dimension(NIMEM_,NJMEM_), intent(out) :: dpa
601  real, dimension(NIMEM_,NJMEM_), optional, intent(out) :: intz_dpa
602  real, dimension(NIMEMB_,NJMEM_), optional, intent(out) :: intx_dpa
603  real, dimension(NIMEM_,NJMEMB_), optional, intent(out) :: inty_dpa
604 ! This subroutine calculates (by numerical quadrature) integrals of
605 ! pressure anomalies across layers, which are required for calculating the
606 ! finite-volume form pressure accelerations in a Boussinesq model. The one
607 ! potentially dodgy assumtion here is that rho_0 is used both in the denominator
608 ! of the accelerations, and in the pressure used to calculated density (the
609 ! latter being -z*rho_0*G_e). These two uses could be separated if need be.
610 !
611 ! Arguments: T - potential temperature relative to the surface in C.
612 ! (in) S - salinity in PSU.
613 ! (in) z_t - height at the top of the layer in m.
614 ! (in) z_b - height at the top of the layer in m.
615 ! (in) rho_ref - A mean density, in kg m-3, that is subtracted out to reduce
616 ! the magnitude of each of the integrals.
617 ! (The pressure is calucated as p~=-z*rho_0*G_e.)
618 ! (in) rho_0 - A density, in kg m-3, that is used to calculate the pressure
619 ! (as p~=-z*rho_0*G_e) used in the equation of state.
620 ! (in) G_e - The Earth's gravitational acceleration, in m s-2.
621 ! (in) G - The ocean's grid structure.
622 ! (in) EOS - type that selects the eqn of state.
623 ! (out) dpa - The change in the pressure anomaly across the layer,
624 ! in Pa.
625 ! (out,opt) intz_dpa - The integral through the thickness of the layer of the
626 ! pressure anomaly relative to the anomaly at the top of
627 ! the layer, in Pa m.
628 ! (out,opt) intx_dpa - The integral in x of the difference between the
629 ! pressure anomaly at the top and bottom of the layer
630 ! divided by the x grid spacing, in Pa.
631 ! (out,opt) inty_dpa - The integral in y of the difference between the
632 ! pressure anomaly at the top and bottom of the layer
633 ! divided by the y grid spacing, in Pa.
634 
635  real :: t5(5), s5(5), p5(5), r5(5)
636  real :: rho_anom
637  real :: w_left, w_right, intz(5)
638  real, parameter :: c1_90 = 1.0/90.0 ! Rational constants.
639  real :: gxrho, i_rho
640  real :: dz
641  integer :: isq, ieq, jsq, jeq, i, j, m, n
642 
643  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
644 
645  gxrho = g_e * rho_0
646  i_rho = 1.0 / rho_0
647 
648  do j=jsq,jeq+1 ; do i=isq,ieq+1
649  dz = z_t(i,j) - z_b(i,j)
650  do n=1,5
651  t5(n) = t(i,j) ; s5(n) = s(i,j)
652  p5(n) = -gxrho*(z_t(i,j) - 0.25*real(n-1)*dz)
653  enddo
654  call calculate_density(t5, s5, p5, r5, 1, 5, eos)
655 
656  ! Use Bode's rule to estimate the pressure anomaly change.
657  rho_anom = c1_90*(7.0*(r5(1)+r5(5)) + 32.0*(r5(2)+r5(4)) + 12.0*r5(3)) - &
658  rho_ref
659  dpa(i,j) = g_e*dz*rho_anom
660  ! Use a Bode's-rule-like fifth-order accurate estimate of the double integral of
661  ! the pressure anomaly.
662  if (present(intz_dpa)) intz_dpa(i,j) = 0.5*g_e*dz**2 * &
663  (rho_anom - c1_90*(16.0*(r5(4)-r5(2)) + 7.0*(r5(5)-r5(1))) )
664  enddo ; enddo
665 
666  if (present(intx_dpa)) then ; do j=g%jsc,g%jec ; do i=isq,ieq
667  intz(1) = dpa(i,j) ; intz(5) = dpa(i+1,j)
668  do m=2,4
669  w_left = 0.25*real(5-m) ; w_right = 1.0-w_left
670  dz = w_left*(z_t(i,j) - z_b(i,j)) + w_right*(z_t(i+1,j) - z_b(i+1,j))
671  t5(1) = w_left*t(i,j) + w_right*t(i+1,j)
672  s5(1) = w_left*s(i,j) + w_right*s(i+1,j)
673  p5(1) = -gxrho*(w_left*z_t(i,j) + w_right*z_t(i+1,j))
674  do n=2,5
675  t5(n) = t5(1) ; s5(n) = s5(1)
676  p5(n) = p5(n-1) + gxrho*0.25*dz
677  enddo
678  call calculate_density(t5, s5, p5, r5, 1, 5, eos)
679 
680  ! Use Bode's rule to estimate the pressure anomaly change.
681  intz(m) = g_e*dz*( c1_90*(7.0*(r5(1)+r5(5)) + 32.0*(r5(2)+r5(4)) + &
682  12.0*r5(3)) - rho_ref)
683  enddo
684  ! Use Bode's rule to integrate the bottom pressure anomaly values in x.
685  intx_dpa(i,j) = c1_90*(7.0*(intz(1)+intz(5)) + 32.0*(intz(2)+intz(4)) + &
686  12.0*intz(3))
687  enddo ; enddo ; endif
688 
689  if (present(inty_dpa)) then ; do j=jsq,jeq ; do i=g%isc,g%iec
690  intz(1) = dpa(i,j) ; intz(5) = dpa(i,j+1)
691  do m=2,4
692  w_left = 0.25*real(5-m) ; w_right = 1.0-w_left
693  dz = w_left*(z_t(i,j) - z_b(i,j)) + w_right*(z_t(i,j+1) - z_b(i,j+1))
694  t5(1) = w_left*t(i,j) + w_right*t(i,j+1)
695  s5(1) = w_left*s(i,j) + w_right*s(i,j+1)
696  p5(1) = -gxrho*(w_left*z_t(i,j) + w_right*z_t(i,j+1))
697  do n=2,5
698  t5(n) = t5(1) ; s5(n) = s5(1)
699  p5(n) = p5(n-1) + gxrho*0.25*dz
700  enddo
701  call calculate_density(t5, s5, p5, r5, 1, 5, eos)
702 
703  ! Use Bode's rule to estimate the pressure anomaly change.
704  intz(m) = g_e*dz*( c1_90*(7.0*(r5(1)+r5(5)) + 32.0*(r5(2)+r5(4)) + &
705  12.0*r5(3)) - rho_ref)
706  enddo
707  ! Use Bode's rule to integrate the values.
708  inty_dpa(i,j) = c1_90*(7.0*(intz(1)+intz(5)) + 32.0*(intz(2)+intz(4)) + &
709  12.0*intz(3))
710  enddo ; enddo ; endif
711 end subroutine int_density_dz_generic

734  real :: s_t(2), s_b(2) ! top and bottom salinities
