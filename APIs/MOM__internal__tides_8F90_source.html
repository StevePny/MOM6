107 subroutine propagate_int_tide(cg1, En_in, vel_btTide, dt, G, CS)
108  real, dimension(NIMEM_,NJMEM_), intent(in) :: cg1, en_in, vel_bttide
109  real, intent(in) :: dt
110  type(ocean_grid_type), intent(inout) :: g
111  type(int_tide_cs), pointer :: cs
112 ! This subroutine calls any of the other subroutines in this file
113 ! that are needed to specify the current surface forcing fields.
114 !
115 ! Arguments: cg1 - The first mode internal gravity wave speed, in m s-1.
116 ! (in) En_in - The energy input to the internal waves, in W m-2.
117 ! (in) dt - Length of time over which these fluxes will be applied, in s.
118 ! (in) G - The ocean's grid structure.
119 ! (in) CS - A pointer to the control structure returned by a previous
120 ! call to int_tide_init.
121  real, dimension(SZI_(G),SZJ_(G),2) :: &
122  test
123  real, dimension(SZI_(G),SZJ_(G),CS%nMode) :: &
124  c1
125  real, dimension(SZI_(G),SZJB_(G)) :: &
126  flux_heat_y, &
127  flux_prec_y
128  real, dimension(SZI_(G),SZJ_(G)) :: &
129  tot_en, drag_scale
130  real :: frac_per_sector, f2, i_rho0, i_d_here
131 
132  integer :: id_test
133  integer :: a, m, fr, i, j, is, ie, js, je, isd, ied, jsd, jed, nangle
134 
135  if (.not.associated(cs)) return
136  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
137  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed ; nangle = cs%NAngle
138  i_rho0 = 1.0 / g%Rho0
139 
140  ! Set the wave speeds for the modes, using that cg(n) ~ cg(1)/n. This is
141  ! wrong, of course, but it works reasonably in some cases.
142  do m=1,cs%nMode ; do j=jsd,jed ; do i=isd,ied
143  c1(i,j,m) = cg1(i,j) / real(m)
144  enddo ; enddo ; enddo
145 
146  ! Add the forcing.
147  if (cs%energized_angle <= 0) then
148  frac_per_sector = 1.0 / real(cs%nangle * cs%nmode * cs%nfreq)
149  do m=1,cs%nMode ; do fr=1,cs%nFreq ; do a=1,cs%nAngle ; do j=js,je ; do i=is,ie
150  f2 = 0.25*((g%CoriolisBu(i,j)**2 + g%CoriolisBu(i-1,j-1)**2) + &
151  (g%CoriolisBu(i,j-1)**2 + g%CoriolisBu(i-1,j)**2))
152  if (cs%frequency(fr)**2 > f2) &
153  cs%En(i,j,a,fr,m) = cs%En(i,j,a,fr,m) + dt*frac_per_sector*en_in(i,j)
154  enddo ; enddo ; enddo ; enddo ; enddo
155  elseif (cs%energized_angle <= cs%nAngle) then
156  frac_per_sector = 1.0 / real(cs%nmode * cs%nfreq)
157  a = cs%energized_angle
158  do m=1,cs%nMode ; do fr=1,cs%nFreq ; do j=js,je ; do i=is,ie
159  f2 = 0.25*((g%CoriolisBu(i,j)**2 + g%CoriolisBu(i-1,j-1)**2) + &
160  (g%CoriolisBu(i,j-1)**2 + g%CoriolisBu(i-1,j)**2))
161  if (cs%frequency(fr)**2 > f2) &
162  cs%En(i,j,a,fr,m) = cs%En(i,j,a,fr,m) + dt*frac_per_sector*en_in(i,j)
163  enddo ; enddo ; enddo ; enddo
164 
165  !### Delete this later.
166  do m=1,cs%nMode ; do j=jsd,jed ; do i=isd,ied
167  c1(i,j,m) = 1.0 / real(m)
168  enddo ; enddo ; enddo
169  else
170  call mom_error(warning, "Internal tide energy is being put into a angular "//&
171  "band that does not exist.")
172  endif
173 
174 
175  ! Pass a test vector to check for grid rotation in the halo updates.
176  do j=jsd,jed ; do i=isd,ied ; test(i,j,1) = 1.0 ; test(i,j,2) = 0.0 ; enddo ; enddo
177  id_test = pass_vector_start(test(:,:,1), test(:,:,2), g%domain, stagger=agrid)
178 
179  ! Apply half the refraction.
180  do m=1,cs%nMode ; do fr=1,cs%nFreq
181  call refract(cs%En(:,:,:,fr,m), c1(:,:,m), cs%frequency(fr), 0.5*dt, g, cs%nAngle)
182 
183  call pass_var(cs%En(:,:,:,fr,m), g%domain, complete=(m==cs%nMode.and.fr==cs%nFreq))
184  enddo ; enddo
185  call pass_vector_complete(id_test, test(:,:,1), test(:,:,2), g%domain, stagger=agrid)
186 
187  ! Rotate points in the halos as necessary.
188  call correct_halo_rotation(cs%En, test, g, cs%nAngle)
189 
190  ! Propagate the waves.
191  do m=1,cs%NMode ; do fr=1,cs%Nfreq
192  call propagate(cs%En(:,:,:,fr,m), c1(:,:,m), cs%frequency(fr), dt, g, cs, cs%NAngle)
193  enddo ; enddo
194 
195  ! Apply the other half of the refraction.
196  do m=1,cs%NMode ; do fr=1,cs%Nfreq
197  call refract(cs%En(:,:,:,fr,m), c1(:,:,m), cs%frequency(fr), 0.5*dt, g, cs%NAngle)
198  enddo ; enddo
199 
200  if (cs%apply_drag .or. (cs%id_tot_En > 0)) then
201  tot_en(:,:) = 0.0
202  do m=1,cs%NMode ; do fr=1,cs%Nfreq ; do a=1,cs%nAngle
203  do j=js,je ; do i=is,ie
204  tot_en(i,j) = tot_en(i,j) + cs%En(i,j,a,fr,m)
205  enddo ; enddo
206  enddo ; enddo ; enddo
207  endif
208 
209  ! Extract the energy for mixing.
210  if (cs%apply_drag) then
211  do j=jsd,jed ; do i=isd,ied
212  i_d_here = 1.0 / max(g%bathyT(i,j), 1.0)
213  drag_scale(i,j) = cs%decay_rate + cs%cdrag * &
214  sqrt(max(0.0, vel_bttide(i,j)**2 + tot_en(i,j) * i_rho0 * i_d_here)) * i_d_here
215  enddo ; enddo
216  else
217  do j=jsd,jed ; do i=isd,ied
218  drag_scale(i,j) = cs%decay_rate
219  enddo ; enddo
220  endif
221 
222  do m=1,cs%nMode ; do fr=1,cs%nFreq ; do a=1,cs%nAngle ; do j=jsd,jed ; do i=isd,ied
223  cs%En(i,j,a,fr,m) = cs%En(i,j,a,fr,m) / (1.0 + dt*drag_scale(i,j))
224  enddo ; enddo ; enddo ; enddo ; enddo
225 
226  if (query_averaging_enabled(cs%diag)) then
227  if (cs%id_tot_En > 0) call post_data(cs%id_tot_En, tot_en, cs%diag)
228 
229  if (cs%id_itide_drag > 0) call post_data(cs%id_itide_drag, drag_scale, cs%diag)
230 
231  do m=1,cs%NMode ; do fr=1,cs%Nfreq ; if (cs%id_En_mode(fr,m) > 0) then
232  tot_en(:,:) = 0.0
233  do a=1,cs%nAngle ; do j=js,je ; do i=is,ie
234  tot_en(i,j) = tot_en(i,j) + cs%En(i,j,a,fr,m)
235  enddo ; enddo ; enddo
236  call post_data(cs%id_En_mode(fr,m), tot_en, cs%diag)
237  endif ; enddo ; enddo
238  do m=1,cs%NMode ; do fr=1,cs%Nfreq ; if (cs%id_En_mode(fr,m) > 0) then
239  call post_data(cs%id_En_ang_mode(fr,m), cs%En(:,:,:,fr,m) , cs%diag)
240  endif ; enddo ; enddo
241  endif
242 
243 end subroutine propagate_int_tide

245 subroutine refract(En, cg, freq, dt, G, NAngle)
246  type(ocean_grid_type), intent(in) :: g
247  integer, intent(in) :: nangle
248  real, dimension(G%isd:G%ied,G%jsd:G%jed,NAngle), intent(inout) :: en
249  real, dimension(G%isd:G%ied,G%jsd:G%jed), intent(in) :: cg
250  real, intent(in) :: freq
251  real, intent(in) :: dt
252 ! This subroutine does refraction on the internal waves at a single frequency.
253 
254 ! Arguments: En - the internal gravity wave energy density as a function of space
255 ! and angular resolution, in J m-2 radian-1.
256  integer, parameter :: stensil = 2
257  real, dimension(SZI_(G),1-stensil:NAngle+stensil) :: &
258  en2d
259  real, dimension(1-stensil:NAngle+stensil) :: &
260  cos_angle, sin_angle
261  real, dimension(SZI_(G)) :: &
262  dk_dt_kmag, dl_dt_kmag
263  real, dimension(SZI_(G),0:nAngle) :: &
264  flux_e
265  real :: f2 ! The squared Coriolis parameter, in s-2.
266  real :: df2_dy, df2_dx ! The x- and y- gradients of the squared Coriolis parameter, in s-2 m-1.
267  real :: dlncg_dx ! The x-gradient of the wave speed divided by itself in m-1.
268  real :: dlncg_dy ! The y-gradient of the wave speed divided by itself in m-1.
269  real :: angle_size, dt_angle_size, angle
270  real :: ifreq, kmag2, i_kmag, cfl_ang
271  real, parameter :: cg_subro = 1e-100
272  integer :: is, ie, js, je, asd, aed, na
273  integer :: i, j, a
274 
275  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; na = size(en,3)
276  asd = 1-stensil ; aed = nangle+stensil
277 
278  ifreq = 1.0 / freq
279 
280  angle_size = (8.0*atan(1.0)) / (real(nangle))
281  dt_angle_size = dt / angle_size
282 
283  do a=0,na
284  angle = (real(A) - 0.5) * angle_size
285  cos_angle(a) = cos(angle) ; sin_angle(a) = sin(angle)
286  enddo
287 
288 !### There should also be refraction due to cg.grad(grid_orientation).
289 
290  do j=js,je
291  ! Copy En into angle space with halos.
292  do a=1,na ; do i=is,ie
293  en2d(i,a) = en(i,j,a)
294  enddo ; enddo
295  do a=asd,0 ; do i=is,ie
296  en2d(i,a) = en2d(i,a+nangle)
297  en2d(i,nangle+stensil+a) = en2d(i,stensil+a)
298  enddo ; enddo
299 
300  ! Do the refraction.
301  do i=is,ie
302  f2 = 0.25*((g%CoriolisBu(i,j)**2 + g%CoriolisBu(i-1,j-1)**2) + &
303  (g%CoriolisBu(i,j-1)**2 + g%CoriolisBu(i-1,j)**2))
304  df2_dx = 0.5*((g%CoriolisBu(i,j)**2 + g%CoriolisBu(i,j-1)**2) - &
305  (g%CoriolisBu(i-1,j)**2 + g%CoriolisBu(i-1,j-1)**2)) * &
306  g%IdxT(i,j)
307  dlncg_dx = 0.5*( g%IdxCu(i,j) * (cg(i+1,j) - cg(i,j)) / &
308  (cg(i+1,j) + cg(i,j) + cg_subro) + &
309  g%IdxCu(i-1,j) * (cg(i,j) - cg(i-1,j)) / &
310  (cg(i,j) + cg(i-1,j) + cg_subro) )
311  df2_dy = 0.5*((g%CoriolisBu(i,j)**2 + g%CoriolisBu(i-1,j)**2) - &
312  (g%CoriolisBu(i,j-1)**2 + g%CoriolisBu(i-1,j-1)**2)) * &
313  g%IdyT(i,j)
314  dlncg_dy = 0.5*( g%IdyCv(i,j) * (cg(i,j+1) - cg(i,j)) / &
315  (cg(i,j+1) + cg(i,j) + cg_subro) + &
316  g%IdyCv(i,j-1) * (cg(i,j) - cg(i,j-1)) / &
317  (cg(i,j) + cg(i,j-1) + cg_subro) )
318  kmag2 = (freq**2 - f2) / (cg(i,j)**2 + cg_subro**2)
319  if (kmag2 > 0.0) then
320  i_kmag = 1.0 / sqrt(kmag2)
321  dk_dt_kmag(i) = -ifreq * (df2_dx + (freq**2 - f2) * dlncg_dx) * i_kmag
322  dl_dt_kmag(i) = -ifreq * (df2_dy + (freq**2 - f2) * dlncg_dy) * i_kmag
323  else
324  dk_dt_kmag(i) = 0.0
325  dl_dt_kmag(i) = 0.0
326  endif
327  enddo
328  ! Determine the energy fluxes in angular orientation space.
329  do a=0,na ; do i=is,ie
330  cfl_ang = (sin_angle(a) * dk_dt_kmag(i) - cos_angle(a) * dl_dt_kmag(i)) * &
331  dt_angle_size
332  ! This is upwind for now. Replace it with monotonic PPM later.
333  if (abs(cfl_ang) > 1.0) then
334  call mom_error(warning, "refract: CFL exceeds 1.", .true.)
335  if (cfl_ang > 0.0) then ; cfl_ang = 1.0 ; else ; cfl_ang = -1.0 ; endif
336  endif
337  if (cfl_ang > 0.0) then
338  flux_e(i,a) = cfl_ang * en2d(i,a)
339  else
340  flux_e(i,a) = cfl_ang * en2d(i,a+1)
341  endif
342  enddo ; enddo
343 
344  ! Update and copy back to En.
345  do a=1,na ; do i=is,ie
346  en(i,j,a) = en2d(i,a) + (flux_e(i,a-1) - flux_e(i,a))
347  enddo ; enddo
348  end do ! j-loop
349 
350 end subroutine refract

352 subroutine propagate(En, cg, freq, dt, G, CS, NAngle)
353  type(ocean_grid_type), intent(in) :: g
354  integer, intent(in) :: nangle
355  real, dimension(G%isd:G%ied,G%jsd:G%jed,NAngle), intent(inout) :: en
356  real, dimension(G%isd:G%ied,G%jsd:G%jed), intent(in) :: cg
357  real, intent(in) :: freq
358  real, intent(in) :: dt
359  type(int_tide_cs), pointer :: cs
360 ! This subroutine does refraction on the internal waves at a single frequency.
361 
362 ! Arguments: En - the internal gravity wave energy density as a function of space
363 ! and angular resolution, in J m-2 radian-1.
364  integer, parameter :: stensil = 2
365  real, dimension(SZIB_(G),SZJ_(G)) :: &
366  speed_x ! The magnitude of the group velocity at the Cu points, in m s-1.
367  real, dimension(SZI_(G),SZJB_(G)) :: &
368  speed_y ! The magnitude of the group velocity at the Cv points, in m s-1.
369  real, dimension(0:NAngle) :: &
370  cos_angle, sin_angle
371  real, dimension(NAngle) :: &
372  cgx_av, cgy_av, dcgx, dcgy
373  real :: f2 ! The squared Coriolis parameter, in s-2.
374  real :: angle_size, i_angle_size, angle
375  real :: ifreq, freq2
376  real, parameter :: cg_subro = 1e-100
377  type(loop_bounds_type) :: lb
378  integer :: is, ie, js, je, asd, aed, na
379  integer :: ish, ieh, jsh, jeh
380  integer :: i, j, a
381 
382  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; na = size(en,3)
383  asd = 1-stensil ; aed = nangle+stensil
384 
385  ifreq = 1.0 / freq
386  freq2 = freq**2
387 
388  angle_size = (8.0*atan(1.0)) / real(nangle)
389  i_angle_size = 1.0 / angle_size
390 
391  ! These could be in the control structure, as they do not vary.
392  do a=0,na
393  ! These are the angles at the cell edges...
394  angle = (real(A) - 0.5) * angle_size
395  cos_angle(a) = cos(angle) ; sin_angle(a) = sin(angle)
396  enddo
397 
398  do a=1,na
399  cgx_av(a) = (sin_angle(a) - sin_angle(a-1)) * i_angle_size
400  cgy_av(a) = -(cos_angle(a) - cos_angle(a-1)) * i_angle_size
401  dcgx(a) = sqrt(0.5 + 0.5*(sin_angle(a)*cos_angle(a) - &
402  sin_angle(a-1)*cos_angle(a-1)) * i_angle_size - &
403  cgx_av(a)**2)
404  dcgy(a) = sqrt(0.5 - 0.5*(sin_angle(a)*cos_angle(a) - &
405  sin_angle(a-1)*cos_angle(a-1)) * i_angle_size - &
406  cgy_av(a)**2)
407  enddo
408 
409  jsh = js-2 ; jeh = je+2 ; ish = is ; ieh = ie
410 
411  do j=jsh,jeh ; do i=ish-1,ieh
412  f2 = 0.5*(g%CoriolisBu(i,j)**2 + g%CoriolisBu(i,j-1)**2)
413  speed_x(i,j) = 0.5*(cg(i,j) + cg(i+1,j)) * g%mask2dCu(i,j) * &
414  sqrt(max(freq2 - f2, 0.0)) * ifreq
415  enddo ; enddo
416  do j=jsh-1,jeh ; do i=ish,ieh
417  f2 = 0.5*(g%CoriolisBu(i,j)**2 + g%CoriolisBu(i-1,j)**2)
418  speed_y(i,j) = 0.5*(cg(i,j) + cg(i,j+1)) * g%mask2dCv(i,j) * &
419  sqrt(max(freq2 - f2, 0.0)) * ifreq
420  enddo ; enddo
421 
422 
423  do a=1,na
424  ! Apply the propagation in alternating directions.
425  lb%jsh = js-2 ; lb%jeh = je+2 ; lb%ish = is ; lb%ieh = ie
426  call propagate_x(en(:,:,a), speed_x, cgx_av(a), dcgx(a), dt, g, cs, lb)
427 
428  lb%jsh = js ; lb%jeh = je ; lb%ish = is ; lb%ieh = ie
429  call propagate_y(en(:,:,a), speed_y, cgy_av(a), dcgy(a), dt, g, cs, lb)
430  end do ! a-loop
431 
432 end subroutine propagate

543 subroutine zonal_flux_en(u, h, hL, hR, uh, dt, G, j, &
544  ish, ieh, vol_cfl)
545  real, dimension(NIMEMB_), intent(in) :: u
546  real, dimension(NIMEM_), intent(in) :: h, hl, hr
547  real, dimension(NIMEMB_), intent(inout) :: uh
548  real, intent(in) :: dt
549  type(ocean_grid_type), intent(in) :: g
550  integer, intent(in) :: j, ish, ieh
551  logical, intent(in) :: vol_cfl
552 ! This subroutines evaluates the zonal mass or volume fluxes in a layer.
553 !
554 ! Arguments: u - Zonal velocity, in m s-1.
555 ! (in) h - Energy density used to calculate the fluxes, in W m-2.
556 ! (in) hL, hR - Left- and right- Energy densities in the reconstruction, in W m-2.
557 ! (out) uh - The zonal energy transport, in J s-1.
558 ! (in) dt - Time increment in s.
559 ! (in) G - The ocean's grid structure.
560 ! (in) j, ish, ieh - The index range to work on.
561 ! (in) vol_CFL - If true, rescale the ratio of face areas to the cell
562 ! areas when estimating the CFL number.
563  real :: cfl ! The CFL number based on the local velocity and grid spacing, ND.
564  real :: curv_3 ! A measure of the thickness curvature over a grid length,
565  ! with the same units as h_in.
566  integer :: i
567 
568  do i=ish-1,ieh
569  ! Set new values of uh and duhdu.
570  if (u(i) > 0.0) then
571  if (vol_cfl) then ; cfl = (u(i) * dt) * (g%dy_Cu(i,j) * g%IareaT(i,j))
572  else ; cfl = u(i) * dt * g%IdxT(i,j) ; endif
573  curv_3 = (hl(i) + hr(i)) - 2.0*h(i)
574  uh(i) = g%dy_Cu(i,j) * u(i) * &
575  (hr(i) + cfl * (0.5*(hl(i) - hr(i)) + curv_3*(cfl - 1.5)))
576  elseif (u(i) < 0.0) then
577  if (vol_cfl) then ; cfl = (-u(i) * dt) * (g%dy_Cu(i,j) * g%IareaT(i+1,j))
578  else ; cfl = -u(i) * dt * g%IdxT(i+1,j) ; endif
579  curv_3 = (hl(i+1) + hr(i+1)) - 2.0*h(i+1)
580  uh(i) = g%dy_Cu(i,j) * u(i) * &
581  (hl(i+1) + cfl * (0.5*(hr(i+1)-hl(i+1)) + curv_3*(cfl - 1.5)))
582  else
583  uh(i) = 0.0
584  endif
585  enddo

689 subroutine ppm_reconstruction_x(h_in, h_l, h_r, G, LB, simple_2nd)
690  real, dimension(NIMEM_,NJMEM_), intent(in) :: h_in
691  real, dimension(NIMEM_,NJMEM_), intent(out) :: h_l, h_r
692  type(ocean_grid_type), intent(in) :: g
693  type(loop_bounds_type), intent(in) :: lb
694  logical, optional, intent(in) :: simple_2nd
695 ! This subroutine calculates left/right edge values for PPM reconstruction.
696 ! Arguments: h_in - Energy density in a sector (2D)
697 ! (out) h_l,h_r - left/right edge value of reconstruction (2D)
698 ! (in) G - The ocean's grid structure.
699 ! (in) LB - A structure with the active loop bounds.
700 ! (in, opt) simple_2nd - If true, use the arithmetic mean energy densities as
701 ! default edge values for a simple 2nd order scheme.
702 
703 ! Local variables with useful mnemonic names.
704  real, dimension(SZI_(G),SZJ_(G)) :: slp ! The slopes.
705  real, parameter :: onesixth = 1./6.
706  real :: h_ip1, h_im1
707  real :: dmx, dmn
708  logical :: use_cw84, use_2nd
709  character(len=256) :: mesg
710  integer :: i, j, isl, iel, jsl, jel, stensil
711 
712  use_2nd = .false. ; if (present(simple_2nd)) use_2nd = simple_2nd
713  isl = lb%ish-1 ; iel = lb%ieh+1 ; jsl = lb%jsh ; jel = lb%jeh
714 
715  ! This is the stensil of the reconstruction, not the scheme overall.
716  stensil = 2 ; if (use_2nd) stensil = 1
717 
718  if ((isl-stensil < g%isd) .or. (iel+stensil > g%ied)) then
719  write(mesg,'("In MOM_internal_tides, PPM_reconstruction_x called with a ", &
720  & "x-halo that needs to be increased by ",i2,".")') &
721  stensil + max(g%isd-isl,iel-g%ied)
722  call mom_error(fatal,mesg)
723  endif
724  if ((jsl < g%jsd) .or. (jel > g%jed)) then
725  write(mesg,'("In MOM_internal_tides, PPM_reconstruction_x called with a ", &
726  & "y-halo that needs to be increased by ",i2,".")') &
727  max(g%jsd-jsl,jel-g%jed)
728  call mom_error(fatal,mesg)
729  endif
730 
731  if (use_2nd) then
732  do j=jsl,jel ; do i=isl,iel
733  h_im1 = g%mask2dT(i-1,j) * h_in(i-1,j) + (1.0-g%mask2dT(i-1,j)) * h_in(i,j)
734  h_ip1 = g%mask2dT(i+1,j) * h_in(i+1,j) + (1.0-g%mask2dT(i+1,j)) * h_in(i,j)
735  h_l(i,j) = 0.5*( h_im1 + h_in(i,j) )
736  h_r(i,j) = 0.5*( h_ip1 + h_in(i,j) )
737  enddo ; enddo
738  else
739  do j=jsl,jel ; do i=isl-1,iel+1
740  if ((g%mask2dT(i-1,j) * g%mask2dT(i,j) * g%mask2dT(i+1,j)) == 0.0) then
741  slp(i,j) = 0.0
742  else
743  ! This uses a simple 2nd order slope.
744  slp(i,j) = 0.5 * (h_in(i+1,j) - h_in(i-1,j))
745  ! Monotonic constraint, see Eq. B2 in Lin 1994, MWR (132)
746  dmx = max(h_in(i+1,j), h_in(i-1,j), h_in(i,j)) - h_in(i,j)
747  dmn = h_in(i,j) - min(h_in(i+1,j), h_in(i-1,j), h_in(i,j))
748  slp(i,j) = sign(1.,slp(i,j)) * min(abs(slp(i,j)), 2. * min(dmx, dmn))
749  ! * (G%mask2dT(i-1,j) * G%mask2dT(i,j) * G%mask2dT(i+1,j))
750  endif
751  enddo; enddo
752 
753  do j=jsl,jel ; do i=isl,iel
754  ! Neighboring values should take into account any boundaries. The 3
755  ! following sets of expressions are equivalent.
756  ! h_im1 = h_in(i-1,j,k) ; if (G%mask2dT(i-1,j) < 0.5) h_im1 = h_in(i,j)
757  ! h_ip1 = h_in(i+1,j,k) ; if (G%mask2dT(i+1,j) < 0.5) h_ip1 = h_in(i,j)
758  h_im1 = g%mask2dT(i-1,j) * h_in(i-1,j) + (1.0-g%mask2dT(i-1,j)) * h_in(i,j)
759  h_ip1 = g%mask2dT(i+1,j) * h_in(i+1,j) + (1.0-g%mask2dT(i+1,j)) * h_in(i,j)
760  ! Left/right values following Eq. B2 in Lin 1994, MWR (132)
761  h_l(i,j) = 0.5*( h_im1 + h_in(i,j) ) + onesixth*( slp(i-1,j) - slp(i,j) )
762  h_r(i,j) = 0.5*( h_ip1 + h_in(i,j) ) + onesixth*( slp(i,j) - slp(i+1,j) )
763  enddo; enddo
764  endif
765 
766  call ppm_limit_pos(h_in, h_l, h_r, 0.0, isl, iel, jsl, jel)
767 
768 end subroutine ppm_reconstruction_x

850 subroutine ppm_limit_pos(h_in, h_L, h_R, h_min, iis, iie, jis, jie)
851  real, dimension(NIMEM_,NJMEM_), intent(in) :: h_in
852  real, dimension(NIMEM_,NJMEM_), intent(inout) :: h_l, h_r
853  real, intent(in) :: h_min
854  integer, intent(in) :: iis, iie, jis, jie
855 ! This subroutine limits the left/right edge values of the PPM reconstruction
856 ! to give a reconstruction that is positive-definite. Here this is
857 ! reinterpreted as giving a constant thickness if the mean thickness is less
858 ! than h_min, with a minimum of h_min otherwise.
859 ! Arguments: h_in - thickness of layer (2D)
860 ! (inout) h_L - left edge value (2D)
861 ! (inout) h_R - right edge value (2D)
862 ! (in) h_min - The minimum thickness that can be obtained by a
863 ! concave parabolic fit.
864 ! (in) iis, iie, jis, jie - Index range for computation.
865 
866 ! Local variables
867  real :: curv, dh, scale
868  character(len=256) :: mesg
869  integer :: i,j
870 
871  do j=jis,jie ; do i=iis,iie
872  ! This limiter prevents undershooting minima within the domain with
873  ! values less than h_min.
874  curv = 3.0*(h_l(i,j) + h_r(i,j) - 2.0*h_in(i,j))
875  if (curv > 0.0) then ! Only minima are limited.
876  dh = h_r(i,j) - h_l(i,j)
877  if (abs(dh) < curv) then ! The parabola's minimum is within the cell.
878  if (h_in(i,j) <= h_min) then
879  h_l(i,j) = h_in(i,j) ; h_r(i,j) = h_in(i,j)
880  elseif (12.0*curv*(h_in(i,j) - h_min) < (curv**2 + 3.0*dh**2)) then
881  ! The minimum value is h_in - (curv^2 + 3*dh^2)/(12*curv), and must
882  ! be limited in this case. 0 < scale < 1.
883  scale = 12.0*curv*(h_in(i,j) - h_min) / (curv**2 + 3.0*dh**2)
884  h_l(i,j) = h_in(i,j) + scale*(h_l(i,j) - h_in(i,j))
885  h_r(i,j) = h_in(i,j) + scale*(h_r(i,j) - h_in(i,j))
886  endif
887  endif
888  endif
889  enddo ; enddo
890 
891 end subroutine ppm_limit_pos
