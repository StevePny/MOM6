23 !* *
24 !* By Robert Hallberg, April 1994 - October 2006 *
25 !* *
26 !* This file contains the subroutine that implements vertical *
27 !* viscosity (vertvisc). *
28 !* *
29 !* The vertical diffusion of momentum is fully implicit. This is *
30 !* necessary to allow for vanishingly small layers. The coupling *
31 !* is based on the distance between the centers of adjacent layers, *
32 !* except where a layer is close to the bottom compared with a *
33 !* bottom boundary layer thickness when a bottom drag law is used. *
34 !* A stress top b.c. and a no slip bottom b.c. are used. There *
35 !* is no limit on the time step for vertvisc. *
36 !* *
37 !* Near the bottom, the horizontal thickness interpolation scheme *
38 !* changes to an upwind biased estimate to control the effect of *
39 !* spurious Montgomery potential gradients at the bottom where *
40 !* nearly massless layers layers ride over the topography. Within a *
41 !* few boundary layer depths of the bottom, the harmonic mean *
42 !* thickness (i.e. (2 h+ h-) / (h+ + h-) ) is used if the velocity *
43 !* is from the thinner side and the arithmetic mean thickness *
44 !* (i.e. (h+ + h-)/2) is used if the velocity is from the thicker *
45 !* side. Both of these thickness estimates are second order *
46 !* accurate. Above this the arithmetic mean thickness is used. *
47 !* *
48 !* In addition, vertvisc truncates any velocity component that *
49 !* exceeds maxvel to truncvel. This basically keeps instabilities *
50 !* spatially localized. The number of times the velocity is *
51 !* truncated is reported each time the energies are saved, and if *
52 !* exceeds CS%Maxtrunc the model will stop itself and change the time *
53 !* to a large value. This has proven very useful in (1) diagnosing *
54 !* model failures and (2) letting the model settle down to a *
55 !* meaningful integration from a poorly specified initial condition. *
56 !* *
57 !* The same code is used for the two velocity components, by *
58 !* indirectly referencing the velocities and defining a handful of *
59 !* direction-specific defined variables. *
60 !* *
61 !* Macros written all in capital letters are defined in MOM_memory.h. *
62 !* *
63 !* A small fragment of the grid is shown below: *
64 !* *
65 !* j+1 x ^ x ^ x At x: q *
66 !* j+1 > o > o > At ^: v, frhatv, tauy *
67 !* j x ^ x ^ x At >: u, frhatu, taux *
68 !* j > o > o > At o: h *
69 !* j-1 x ^ x ^ x *
70 !* i-1 i i+1 At x & ^: *
71 !* i i+1 At > & o: *
72 !* *
73 !* The boundaries always run through q grid points (x). *
74 !* *

364  enddo ; enddo ; endif

424  if (present(taux_bot) .and. (cs%id_taux_bot > 0)) &
425  call post_data(cs%id_taux_bot, taux_bot, cs%diag)
426  if (present(tauy_bot) .and. (cs%id_tauy_bot > 0)) &
427  call post_data(cs%id_tauy_bot, tauy_bot, cs%diag)

431 subroutine vertvisc_remnant(visc, visc_rem_u, visc_rem_v, dt, G, CS)
432 ! This subroutine does a fully implicit vertical diffusion
433 ! of momentum. Stress top and bottom b.c.s are used.
434  type(vertvisc_type), intent(in) :: visc
435  real, intent(inout), dimension(NIMEMB_,NJMEM_,NKMEM_) :: visc_rem_u
436  real, intent(inout), dimension(NIMEM_,NJMEMB_,NKMEM_) :: visc_rem_v
437  real, intent(in) :: dt
438  type(ocean_grid_type), intent(in) :: g
439  type(vertvisc_cs), pointer :: cs
440 ! Arguments: visc - The vertical viscosity type, containing information about
441 ! viscosities and bottom drag-related quantities, intent in.
442 ! (out) visc_rem_u - Both the fraction of the momentum originally in a
443 ! (out) visc_rem_v - layer that remains after a time-step of viscosity,
444 ! and the fraction of a time-step's worth of a
445 ! barotropic acceleration that a layer experiences
446 ! after viscosity is applied, in the zonal (_u) and
447 ! meridional (_v) directions. Nondimensional between
448 ! 0 (at the bottom) and 1 (far above the bottom).
449 ! (in) dt - Time increment in s.
450 ! (in) G - The ocean's grid structure.
451 ! (in) CS - The control structure returned by a previous call to
452 ! vertvisc_init.
453 !
454 
455  real :: b1(szib_(g)) ! b1 and c1 are variables used by the
456  real :: c1(szib_(g),szk_(g)) ! tridiagonal solver. c1 is nondimensional,
457  ! while b1 has units of inverse thickness.
458  real :: d1(szib_(g)) ! d1=1-c1 is used by the tridiagonal solver, ND.
459  real :: ray(szib_(g),szk_(g)) ! Ray is the Rayleigh-drag velocity times the
460  ! time step, in m.
461  real :: b_denom_1 ! The first term in the denominator of b1, in m or kg m-2.
462  real :: dt_m_to_h ! The time step times the conversion from m to the
463  ! units of thickness - either s or s m3 kg-1.
464  logical :: do_i(szib_(g))
465 
466  integer :: i, j, k, is, ie, isq, ieq, jsq, jeq, nz
467  is = g%isc ; ie = g%iec
468  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB ; nz = g%ke
469 
470  if (.not.associated(cs)) call mom_error(fatal,"MOM_vert_friction(visc): "// &
471  "Module must be initialized before it is used.")
472 
473  dt_m_to_h = dt*g%m_to_H
474 
475  do k=1,nz ; do i=isq,ieq ; ray(i,k) = 0.0 ; enddo ; enddo
476 
477  ! Find the zonal viscous using a modification of a standard tridagonal solver.
478  do j=g%jsc,g%jec
479  do i=isq,ieq ; do_i(i) = (g%mask2dCu(i,j) > 0) ; enddo
480 
481  if (cs%Channel_drag) then ; do k=1,nz ; do i=isq,ieq
482  ray(i,k) = visc%Ray_u(i,j,k)
483  enddo ; enddo ; endif
484 
485  do i=isq,ieq ; if (do_i(i)) then
486  b_denom_1 = cs%h_u(i,j,1) + dt_m_to_h * (ray(i,1) + cs%a_u(i,j,1))
487  b1(i) = 1.0 / (b_denom_1 + dt_m_to_h*cs%a_u(i,j,2))
488  d1(i) = b_denom_1 * b1(i)
489  visc_rem_u(i,j,1) = b1(i) * cs%h_u(i,j,1)
490  endif ; enddo
491  do k=2,nz ; do i=isq,ieq ; if (do_i(i)) then
492  c1(i,k) = dt_m_to_h * cs%a_u(i,j,k)*b1(i)
493  b_denom_1 = cs%h_u(i,j,k) + dt_m_to_h * (ray(i,k) + cs%a_u(i,j,k)*d1(i))
494  b1(i) = 1.0 / (b_denom_1 + dt_m_to_h * cs%a_u(i,j,k+1))
495  d1(i) = b_denom_1 * b1(i)
496  visc_rem_u(i,j,k) = (cs%h_u(i,j,k) + dt_m_to_h * cs%a_u(i,j,k) * visc_rem_u(i,j,k-1)) * b1(i)
497  endif ; enddo ; enddo
498  do k=nz-1,1,-1 ; do i=isq,ieq ; if (do_i(i)) then
499  visc_rem_u(i,j,k) = visc_rem_u(i,j,k) + c1(i,k+1)*visc_rem_u(i,j,k+1)
500  endif ; enddo ; enddo ! i and k loops
501 
502  enddo ! end u-component j loop
503 
504  ! Now find the meridional viscous using a modification.
505  do j=jsq,jeq
506  do i=is,ie ; do_i(i) = (g%mask2dCv(i,j) > 0) ; enddo
507 
508  if (cs%Channel_drag) then ; do k=1,nz ; do i=is,ie
509  ray(i,k) = visc%Ray_v(i,j,k)
510  enddo ; enddo ; endif
511 
512  do i=is,ie ; if (do_i(i)) then
513  b_denom_1 = cs%h_v(i,j,1) + dt_m_to_h * (ray(i,1) + cs%a_v(i,j,1))
514  b1(i) = 1.0 / (b_denom_1 + dt_m_to_h*cs%a_v(i,j,2))
515  d1(i) = b_denom_1 * b1(i)
516  visc_rem_v(i,j,1) = b1(i) * cs%h_v(i,j,1)
517  endif ; enddo
518  do k=2,nz ; do i=is,ie ; if (do_i(i)) then
519  c1(i,k) = dt_m_to_h * cs%a_v(i,j,k)*b1(i)
520  b_denom_1 = cs%h_v(i,j,k) + dt_m_to_h * (ray(i,k) + cs%a_v(i,j,k)*d1(i))
521  b1(i) = 1.0 / (b_denom_1 + dt_m_to_h * cs%a_v(i,j,k+1))
522  d1(i) = b_denom_1 * b1(i)
523  visc_rem_v(i,j,k) = (cs%h_v(i,j,k) + dt_m_to_h * cs%a_v(i,j,k) * visc_rem_v(i,j,k-1)) * b1(i)
524  endif ; enddo ; enddo
525  do k=nz-1,1,-1 ; do i=is,ie ; if (do_i(i)) then
526  visc_rem_v(i,j,k) = visc_rem_v(i,j,k) + c1(i,k+1)*visc_rem_v(i,j,k+1)
527  endif ; enddo ; enddo ! i and k loops
528  enddo ! end of v-component J loop
529 
530 
531  if (cs%debug) then
532  call uchksum(visc_rem_u,"visc_rem_u",g,haloshift=0)
533  call vchksum(visc_rem_v,"visc_rem_v",g,haloshift=0)
534  endif
535 
536 end subroutine vertvisc_remnant

539 subroutine vertvisc_coef(u, v, h, fluxes, visc, dt, G, CS)
540 ! This subroutine calculates the coupling coefficients (CS%a_u and CS%a_v)
541 ! and effective layer thicknesses (CS%h_u and CS%h_v) for later use in the
542 ! applying the implicit vertical viscosity via vertvisc.
543  real, intent(in), dimension(NIMEMB_,NJMEM_,NKMEM_) :: u
544  real, intent(in), dimension(NIMEM_,NJMEMB_,NKMEM_) :: v
545  real, intent(in), dimension(NIMEM_,NJMEM_,NKMEM_) :: h
546  type(forcing), intent(in) :: fluxes
547  type(vertvisc_type), intent(in) :: visc
548  real, intent(in) :: dt
549  type(ocean_grid_type), intent(in) :: g
550  type(vertvisc_cs), pointer :: cs
551 
552 ! Arguments: u - Zonal velocity, in m s-1. Intent in.
553 ! (in) v - Meridional velocity, in m s-1.
554 ! (in) h - Layer thickness, in m.
555 ! (in) fluxes - A structure containing pointers to any possible
556 ! forcing fields. Unused fields have NULL ptrs.
557 ! (in) visc - The vertical viscosity type, containing information about
558 ! viscosities and bottom drag-related quantities.
559 ! (in) dt - Time increment in s.
560 ! (in) G - The ocean's grid structure.
561 ! (in/out) CS - The control structure returned by a previous call to
562 ! vertvisc_init.
563 !
564 ! Field from fluxes used in this subroutine:
565 ! ustar: the friction velocity in m s-1, used here as the mixing
566 ! velocity in the mixed layer if G%nkml > 1 in a bulk mixed layer.
567 !
568  real, dimension(SZIB_(G),SZK_(G)) :: &
569  h_harm, & ! Harmonic mean of the thicknesses around a velocity grid point,
570  ! given by 2*(h+ * h-)/(h+ + h-), in m or kg m-2 (H for short).
571  hvel, & ! hvel is the thickness used at a velocity grid point, in H.
572  hvel_shelf ! The equivalent of hvel under shelves, in H.
573  real, dimension(SZIB_(G),SZK_(G)+1) :: &
574  a, & ! The drag coefficients across interfaces, in m s-1. a times
575  ! the velocity difference gives the stress across an interface.
576  a_shelf, & ! The drag coefficients across interfaces in water columns under
577  ! ice shelves, in m s-1.
578  z_i ! An estimate of each interface's height above the bottom,
579  ! normalized by the bottom boundary layer thickness, nondim.
580  real, dimension(SZIB_(G)) :: &
581  kv_bbl, & ! The bottom boundary layer viscosity in m2 s-1.
582  bbl_thick, & ! The bottom boundary layer thickness in m or kg m-2.
583  i_hbbl, & ! The inverse of the bottom boundary layer thickness, in units
584  ! of H-1 (i.e., m-1 or m2 kg-1).
585  i_htbl, & ! The inverse of the top boundary layer thickness, in units
586  ! of H-1 (i.e., m-1 or m2 kg-1).
587  zcol1, & ! The height of the interfaces to the north and south of a
588  zcol2, & ! v-point, in m or kg m-2.
589  ztop_min, & ! The deeper of the two adjacent surface heights, in H.
590  dmin, & ! The shallower of the two adjacent bottom depths converted to
591  ! thickness units, in m or kg m-2.
592  zh ! An estimate of the interface's distance from the bottom
593  ! based on harmonic mean thicknesses, in m or kg m-2.
594  real, allocatable, dimension(:,:) :: hml_u, hml_v
595  real :: zcol(szi_(g)) ! The height of an interface at h-points, in m or kg m-2.
596  real :: botfn ! A function which goes from 1 at the bottom to 0 much more
597  ! than Hbbl into the interior.
598  real :: topfn ! A function which goes from 1 at the top to 0 much more
599  ! than Htbl into the interior.
600  real :: z2 ! The distance from the bottom, normalized by Hbbl, nondim.
601  real :: h_neglect ! A thickness that is so small it is usually lost
602  ! in roundoff and can be neglected, in H.
603 
604  real :: h_arith ! The arithmetic mean thickness, in m or kg m-2.
605  logical, dimension(SZIB_(G)) :: do_i, do_i_shelf
606  logical :: do_any_shelf
607 
608  integer :: i, j, k, is, ie, js, je, isq, ieq, jsq, jeq, nz
609  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
610  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB ; nz = g%ke
611 
612  if (.not.associated(cs)) call mom_error(fatal,"MOM_vert_friction(coef): "// &
613  "Module must be initialized before it is used.")
614 
615  h_neglect = g%H_subroundoff
616  i_hbbl(:) = 1.0 / (cs%Hbbl * g%m_to_H + h_neglect)
617 
618  if (cs%debug .or. (cs%id_hML_u > 0)) then
619  allocate(hml_u(g%IsdB:g%IedB,g%jsd:g%jed)) ; hml_u(:,:) = 0.0
620  endif
621  if (cs%debug .or. (cs%id_hML_v > 0)) then
622  allocate(hml_v(g%isd:g%ied,g%JsdB:g%JedB)) ; hml_v(:,:) = 0.0
623  endif
624 
625  if ((associated(visc%taux_shelf) .or. associated(fluxes%frac_shelf_u)) .and. &
626  .not.associated(cs%a1_shelf_u)) then
627  allocate(cs%a1_shelf_u(g%IsdB:g%IedB,g%jsd:g%jed)) ; cs%a1_shelf_u(:,:)=0.0
628  endif
629  if ((associated(visc%tauy_shelf) .or. associated(fluxes%frac_shelf_u)) .and. &
630  .not.associated(cs%a1_shelf_v)) then
631  allocate(cs%a1_shelf_v(g%isd:g%ied,g%JsdB:g%JedB)) ; cs%a1_shelf_v(:,:)=0.0
632  endif
633 
634  do j=g%Jsc,g%Jec
635  do i=isq,ieq ; do_i(i) = (g%mask2dCu(i,j) > 0) ; enddo
636 
637  if (cs%bottomdraglaw) then ; do i=isq,ieq
638  kv_bbl(i) = visc%kv_bbl_u(i,j)
639  bbl_thick(i) = visc%bbl_thick_u(i,j) * g%m_to_H
640  if (do_i(i)) i_hbbl(i) = 1.0 / (bbl_thick(i) + h_neglect)
641  enddo ; endif
642 
643 ! The following block calculates the thicknesses at velocity
644 ! grid points for the vertical viscosity (hvel[k]). Near the
645 ! bottom an upwind biased thickness is used to control the effect
646 ! of spurious Montgomery potential gradients at the bottom where
647 ! nearly massless layers layers ride over the topography.
648  if (cs%harmonic_visc) then
649  do i=isq,ieq ; z_i(i,nz+1) = 0.0 ; enddo
650  do k=nz,1,-1 ; do i=isq,ieq ; if (do_i(i)) then
651  h_harm(i,k) = 2.0*h(i,j,k)*h(i+1,j,k) / (h(i,j,k)+h(i+1,j,k)+h_neglect)
652  h_arith = 0.5*(h(i+1,j,k)+h(i,j,k))
653 
654  hvel(i,k) = h_harm(i,k)
655  if (u(i,j,k) * (h(i+1,j,k)-h(i,j,k)) < 0) then
656  z2 = z_i(i,k+1) ; botfn = 1.0 / (1.0 + 0.09*z2*z2*z2*z2*z2*z2)
657  hvel(i,k) = (1.0-botfn)*h_harm(i,k) + botfn*h_arith
658  endif
659  z_i(i,k) = z_i(i,k+1) + h_harm(i,k)*i_hbbl(i)
660  endif ; enddo ; enddo ! i & k loops
661  else ! Not harmonic_visc
662  do i=isq,ieq
663  zh(i) = 0.0 ; z_i(i,nz+1) = 0.0
664  dmin(i) = min(g%bathyT(i,j), g%bathyT(i+1,j)) * g%m_to_H
665  enddo
666  do i=isq,ieq+1 ; zcol(i) = -g%bathyT(i,j) * g%m_to_H ; enddo
667  do k=nz,1,-1
668  do i=isq,ieq+1 ; zcol(i) = zcol(i) + h(i,j,k) ; enddo
669  do i=isq,ieq ; if (do_i(i)) then
670  h_harm(i,k) = 2.0*h(i,j,k)*h(i+1,j,k) / (h(i,j,k)+h(i+1,j,k)+h_neglect)
671  h_arith = 0.5*(h(i+1,j,k)+h(i,j,k))
672  zh(i) = zh(i) + h_harm(i,k)
673  z_i(i,k) = max(zh(i), max(zcol(i),zcol(i+1)) + dmin(i)) * i_hbbl(i)
674 
675  hvel(i,k) = h_arith
676  if (u(i,j,k) * (h(i+1,j,k)-h(i,j,k)) > 0) then
677  z2 = max(zh(i), max(zcol(i),zcol(i+1)) + dmin(i)) * i_hbbl(i)
678  botfn = 1.0 / (1.0 + 0.09*z2*z2*z2*z2*z2*z2)
679  hvel(i,k) = (1.0-botfn)*h_arith + botfn*h_harm(i,k)
680  endif
681 
682  endif ; enddo ! i loop
683  enddo ! k loop
684  endif
685 
686  call find_coupling_coef(a, hvel, do_i, work_on_u=.true.)
687  do_any_shelf = .false.
688  if (associated(fluxes%frac_shelf_u)) then
689  do i=isq,ieq
690  cs%a1_shelf_u(i,j) = 0.0
691  do_i_shelf(i) = (do_i(i) .and. fluxes%frac_shelf_u(i,j) > 0.0)
692  if (do_i_shelf(i)) do_any_shelf = .true.
693  enddo
694  if (do_any_shelf) then
695  if (cs%harmonic_visc) then
696  call find_coupling_coef(a_shelf, hvel, do_i_shelf, work_on_u=.true., shelf=.true.)
697  else ! Find upwind-biased thickness near the surface.
698  ! Perhaps this needs to be done more carefully, via find_eta.
699  do i=isq,ieq ; if (do_i_shelf(i)) then
700  zh(i) = 0.0 ; ztop_min(i) = min(zcol(i), zcol(i+1))
701  i_htbl(i) = 1.0 / (visc%tbl_thick_shelf_u(i,j)*g%m_to_H + h_neglect)
702  endif ; enddo
703  do k=1,nz
704  do i=isq,ieq+1 ; zcol(i) = zcol(i) - h(i,j,k) ; enddo
705  do i=isq,ieq ; if (do_i_shelf(i)) then
706  h_arith = 0.5*(h(i+1,j,k)+h(i,j,k))
707  zh(i) = zh(i) + h_harm(i,k)
708 
709  hvel_shelf(i,k) = hvel(i,k)
710  if (u(i,j,k) * (h(i+1,j,k)-h(i,j,k)) > 0) then
711  z2 = max(zh(i), ztop_min(i) - min(zcol(i),zcol(i+1))) * i_htbl(i)
712  topfn = 1.0 / (1.0 + 0.09*z2**6)
713  hvel_shelf(i,k) = min(hvel(i,k), (1.0-topfn)*h_arith + topfn*h_harm(i,k))
714  endif
715  endif ; enddo
716  enddo
717  call find_coupling_coef(a_shelf, hvel_shelf, do_i_shelf, work_on_u=.true., shelf=.true.)
718  endif
719  do i=isq,ieq ; if (do_i_shelf(i)) cs%a1_shelf_u(i,j) = a_shelf(i,1) ; enddo
720  endif
721  endif
722 
723  if (do_any_shelf) then
724  do k=1,nz+1 ; do i=isq,ieq ; if (do_i_shelf(i)) then
725  cs%a_u(i,j,k) = fluxes%frac_shelf_u(i,j) * a_shelf(i,k) + &
726  (1.0-fluxes%frac_shelf_u(i,j)) * a(i,k)
727  elseif (do_i(i)) then
728  cs%a_u(i,j,k) = a(i,k)
729  endif ; enddo ; enddo
730  do k=1,nz ; do i=isq,ieq ; if (do_i_shelf(i)) then
731  ! Should we instead take the inverse of the average of the inverses?
732  cs%h_u(i,j,k) = fluxes%frac_shelf_u(i,j) * hvel_shelf(i,k) + &
733  (1.0-fluxes%frac_shelf_u(i,j)) * hvel(i,k)
734  elseif (do_i(i)) then
735  cs%h_u(i,j,k) = hvel(i,k)
736  endif ; enddo ; enddo
737  else
738  do k=1,nz+1 ; do i=isq,ieq ; if (do_i(i)) cs%a_u(i,j,k) = a(i,k) ; enddo ; enddo
739  do k=1,nz ; do i=isq,ieq ; if (do_i(i)) cs%h_u(i,j,k) = hvel(i,k) ; enddo ; enddo
740  endif
741 
742  enddo
743 
744  ! Now work on v-points.
745  do j=jsq,jeq
746  do i=is,ie ; do_i(i) = (g%mask2dCv(i,j) > 0) ; enddo
747 
748  if (cs%bottomdraglaw) then ; do i=is,ie
749  kv_bbl(i) = visc%kv_bbl_v(i,j)
750  bbl_thick(i) = visc%bbl_thick_v(i,j) * g%m_to_H
751  if (do_i(i)) i_hbbl(i) = 1.0 / bbl_thick(i)
752  enddo ; endif
753 
754 ! The following block calculates the thicknesses at velocity
755 ! grid points for the vertical viscosity (hvel[k]). Near the
756 ! bottom an upwind biased thickness is used to control the effect
757 ! of spurious Montgomery potential gradients at the bottom where
758 ! nearly massless layers layers ride over the topography.
759  if (cs%harmonic_visc) then
760  do i=is,ie ; z_i(i,nz+1) = 0.0 ; enddo
761 
762  do k=nz,1,-1 ; do i=is,ie ; if (do_i(i)) then
763  h_harm(i,k) = 2.0*h(i,j,k)*h(i,j+1,k) / (h(i,j,k)+h(i,j+1,k)+h_neglect)
764  h_arith = 0.5*(h(i,j+1,k)+h(i,j,k))
765 
766  hvel(i,k) = h_harm(i,k)
767  if (v(i,j,k) * (h(i,j+1,k)-h(i,j,k)) < 0) then
768  z2 = z_i(i,k+1) ; botfn = 1.0 / (1.0 + 0.09*z2*z2*z2*z2*z2*z2)
769  hvel(i,k) = (1.0-botfn)*h_harm(i,k) + botfn*h_arith
770  endif
771  z_i(i,k) = z_i(i,k+1) + h_harm(i,k)*i_hbbl(i)
772  endif ; enddo ; enddo ! i & k loops
773  else ! Not harmonic_visc
774  do i=is,ie
775  zh(i) = 0.0 ; z_i(i,nz+1) = 0.0
776  dmin(i) = min(g%bathyT(i,j), g%bathyT(i,j+1)) * g%m_to_H
777  zcol1(i) = -g%bathyT(i,j) * g%m_to_H
778  zcol2(i) = -g%bathyT(i,j+1) * g%m_to_H
779  enddo
780  do k=nz,1,-1 ; do i=is,ie ; if (do_i(i)) then
781  h_harm(i,k) = 2.0*h(i,j,k)*h(i,j+1,k) / (h(i,j,k)+h(i,j+1,k)+h_neglect)
782  h_arith = 0.5*(h(i,j+1,k)+h(i,j,k))
783  zh(i) = zh(i) + h_harm(i,k)
784  zcol1(i) = zcol1(i) + h(i,j,k) ; zcol2(i) = zcol2(i) + h(i,j+1,k)
785  z_i(i,k) = max(zh(i), max(zcol1(i),zcol2(i)) + dmin(i)) * i_hbbl(i)
786 
787  hvel(i,k) = h_arith
788  if (v(i,j,k) * (h(i,j+1,k)-h(i,j,k)) > 0) then
789  z2 = max(zh(i), max(zcol1(i),zcol2(i)) + dmin(i)) * i_hbbl(i)
790  botfn = 1.0 / (1.0 + 0.09*z2*z2*z2*z2*z2*z2)
791  hvel(i,k) = (1.0-botfn)*h_arith + botfn*h_harm(i,k)
792  endif
793 
794  endif ; enddo ; enddo ! i & k loops
795  endif
796 
797  call find_coupling_coef(a, hvel, do_i, work_on_u=.false.)
798  do_any_shelf = .false.
799  if (associated(fluxes%frac_shelf_v)) then
800  do i=is,ie
801  cs%a1_shelf_v(i,j) = 0.0

808  else ! Find upwind-biased thickness near the surface.
809  ! Perhaps this needs to be done more carefully, via find_eta.
810  do i=is,ie ; if (do_i_shelf(i)) then
811  zh(i) = 0.0 ; ztop_min(i) = min(zcol1(i), zcol2(i))
812  i_htbl(i) = 1.0 / (visc%tbl_thick_shelf_v(i,j)*g%m_to_H + h_neglect)
813  endif ; enddo
814  do k=1,nz
815  do i=is,ie ; if (do_i_shelf(i)) then
816  zcol1(i) = zcol1(i) - h(i,j,k) ; zcol2(i) = zcol2(i) - h(i,j+1,k)
817  h_arith = 0.5*(h(i,j+1,k)+h(i,j,k))
818  zh(i) = zh(i) + h_harm(i,k)
819 
820  hvel_shelf(i,k) = hvel(i,k)
821  if (v(i,j,k) * (h(i,j+1,k)-h(i,j,k)) > 0) then
822  z2 = max(zh(i), ztop_min(i) - min(zcol1(i),zcol2(i))) * i_htbl(i)
823  topfn = 1.0 / (1.0 + 0.09*z2**6)
824  hvel_shelf(i,k) = min(hvel(i,k), (1.0-topfn)*h_arith + topfn*h_harm(i,k))
825  endif
826  endif ; enddo
827  enddo
828  call find_coupling_coef(a_shelf, hvel_shelf, do_i_shelf, work_on_u=.false., shelf=.true.)
829  endif
830  do i=is,ie ; if (do_i_shelf(i)) cs%a1_shelf_v(i,j) = a_shelf(i,1) ; enddo
831  endif
832  endif
833 
834  if (do_any_shelf) then
835  do k=1,nz+1 ; do i=is,ie ; if (do_i_shelf(i)) then
836  cs%a_v(i,j,k) = fluxes%frac_shelf_v(i,j) * a_shelf(i,k) + &
837  (1.0-fluxes%frac_shelf_v(i,j)) * a(i,k)
838  elseif (do_i(i)) then
839  cs%a_v(i,j,k) = a(i,k)
840  endif ; enddo ; enddo
841  do k=1,nz ; do i=is,ie ; if (do_i_shelf(i)) then
842  ! Should we instead take the inverse of the average of the inverses?
843  cs%h_v(i,j,k) = fluxes%frac_shelf_v(i,j) * hvel_shelf(i,k) + &
844  (1.0-fluxes%frac_shelf_v(i,j)) * hvel(i,k)
845  elseif (do_i(i)) then
846  cs%h_v(i,j,k) = hvel(i,k)
847  endif ; enddo ; enddo
848  else
849  do k=1,nz+1 ; do i=is,ie ; if (do_i(i)) cs%a_v(i,j,k) = a(i,k) ; enddo ; enddo
850  do k=1,nz ; do i=is,ie ; if (do_i(i)) cs%h_v(i,j,k) = hvel(i,k) ; enddo ; enddo
851  endif
852 
853  enddo ! end of v-point j loop

856  call uchksum(cs%h_u,"vertvisc_coef h_u",g,haloshift=0)
857  call vchksum(cs%h_v,"vertvisc_coef h_v",g,haloshift=0)

867  if (cs%id_h_u > 0) call post_data(cs%id_h_u, cs%h_u, cs%diag)

877  subroutine find_coupling_coef(a, hvel, do_i, work_on_u, shelf)
878 ! This subroutine calculates the 'coupling coefficient' (a[k]) at the
879 ! interfaces. If BOTTOMDRAGLAW is defined, the minimum of Hbbl and half the
880 ! adjacent layer thicknesses are used to calculate a[k] near the bottom.
881  real, dimension(NIMEMB_,NK_INTERFACE_), intent(out) :: a
882  real, dimension(NIMEMB_,NKMEM_), intent(in) :: hvel
883  logical, dimension(NIMEMB_), intent(in) :: do_i
884  logical, intent(in) :: work_on_u
885  logical, optional, intent(in) :: shelf
886 ! Arguments: a - The coupling coefficent across interfaces, in m/s. Intent out.
887 ! (in) hvel - The thickness at velocity points, in H.
888 ! (in) do_i - If true, determine the a for a column.
889 ! (in) work_on_u - If true, u-points are being worked on, otherwise this
890 ! call is for v-points.
891 ! (in) shelf - If present and true, use a surface boundary condition
892 ! appropriate for an ice shelf.
893  real, dimension(SZIB_(G)) :: &
894  u_star, & ! ustar at a velocity point, in m s-1.
895  absf, & ! The average of the neighboring absolute values of f, in s-1.
896  h_ml, & ! The mixed layer depth, in m or kg m-2.
897  nk_visc, & ! The (real) interface index of the base of mixed layer.
898  z_t, & ! The distance from the top, sometimes normalized
899  ! by Hmix, in m or nondimensional.
900  kv_tbl, &
901  tbl_thick
902  real :: h_shear ! The distance over which shears occur, m or kg m-2.
903  real :: r ! A thickness to compare with Hbbl, in m or kg m-2.
904  real :: visc_ml ! The mixed layer viscosity, in m2 s-1.
905  real :: i_hmix ! The inverse of the mixed layer thickness, in m-1 or m2 kg-1.
906  real :: a_ml ! The layer coupling coefficient across an interface in
907  ! the mixed layer, in m s-1.
908  real :: temp1 ! A temporary variable in m2 s-1.
909  real :: h_neglect ! A thickness that is so small it is usually lost
910  ! in roundoff and can be neglected, in H.
911  real :: dz_neglect ! A thickness in m that is so small it is usually lost
912  ! in roundoff and can be neglected, in m.
913  real :: z2 ! A copy of z_i, nondim.
914  real :: topfn
915  real :: a_top
916  logical :: do_shelf
917  integer :: i, k, is, ie, max_nk
918 
919  if (work_on_u) then ; is = g%IscB ; ie = g%IecB
920  else ; is = g%isc ; ie = g%iec ; endif
921  h_neglect = g%H_subroundoff
922  dz_neglect = g%H_subroundoff*g%H_to_m
923 
924  do_shelf = .false. ; if (present(shelf)) do_shelf = shelf
925 
926 ! The following loop calculates the vertical average velocity and
927 ! surface mixed layer contributions to the vertical viscosity.
928  do i=is,ie ; a(i,1) = 0.0 ; enddo
929  if ((g%nkml>0) .or. do_shelf) then ; do k=2,nz ; do i=is,ie
930  if (do_i(i)) a(i,k) = 2.0*cs%Kv
931  enddo ; enddo ; else
932  i_hmix = 1.0 / (cs%Hmix * g%m_to_H + h_neglect)
933  do i=is,ie ; z_t(i) = h_neglect*i_hmix ; enddo
934  do k=2,nz ; do i=is,ie ; if (do_i(i)) then
935  z_t(i) = z_t(i) + h_harm(i,k-1)*i_hmix
936  a(i,k) = 2.0*cs%Kv + 2.0*cs%Kvml / ((z_t(i)*z_t(i)) * &
937  (1.0 + 0.09*z_t(i)*z_t(i)*z_t(i)*z_t(i)*z_t(i)*z_t(i)))
938  endif ; enddo ; enddo
939  endif
940 
941  do i=is,ie ; if (do_i(i)) then
942  if (cs%bottomdraglaw) then
943  r = hvel(i,nz)*0.5
944  if (r < bbl_thick(i)) then
945  a(i,nz+1) = 1.0*kv_bbl(i) / (1e-10*dt*kv_bbl(i) + r*g%H_to_m)
946  else
947  a(i,nz+1) = 1.0*kv_bbl(i) / (1e-10*dt*kv_bbl(i) + bbl_thick(i)*g%H_to_m)
948  endif
949  else
950  a(i,nz+1) = 2.0*cs%Kvbbl / (hvel(i,nz)*g%H_to_m + 2.0e-10*dt*cs%Kvbbl)
951  endif
952  endif ; enddo
953 
954  if (associated(visc%Kv_turb)) then
955  if (work_on_u) then
956  do k=nz,2,-1 ; do i=is,ie ; if (do_i(i)) then
957  a(i,k) = a(i,k) + 0.5*(visc%Kv_turb(i,j,k) + visc%Kv_turb(i+1,j,k))
958  endif ; enddo ; enddo
959  else
960  do k=nz,2,-1 ; do i=is,ie ; if (do_i(i)) then
961  a(i,k) = a(i,k) + 0.5*(visc%Kv_turb(i,j,k) + visc%Kv_turb(i,j+1,k))
962  endif ; enddo ; enddo
963  endif
964  endif
965 
966  do k=nz,2,-1 ; do i=is,ie ; if (do_i(i)) then
967  ! botfn determines when a point is within the influence of the bottom
968  ! boundary layer, going from 1 at the bottom to 0 in the interior.
969  z2 = z_i(i,k)
970  botfn = 1.0 / (1.0 + 0.09*z2*z2*z2*z2*z2*z2)
971 
972  if (cs%bottomdraglaw) then
973  a(i,k) = a(i,k) + 2.0*(kv_bbl(i)-cs%Kv)*botfn
974  r = (hvel(i,k)+hvel(i,k-1))
975  if (r > 2.0*bbl_thick(i)) then
976  h_shear = ((1.0 - botfn) * r + botfn*2.0*bbl_thick(i))
977  else
978  h_shear = r
979  endif
980  else
981  a(i,k) = a(i,k) + 2.0*(cs%Kvbbl-cs%Kv)*botfn
982  h_shear = hvel(i,k) + hvel(i,k-1) + h_neglect
983  endif
984 
985  ! Up to this point a has units of m2 s-1, but now is converted to m s-1.
986  ! The term including 1e-10 in the denominators is here to avoid
987  ! truncation error problems in the tridiagonal solver. Effectively, this
988  ! sets the maximum coupling coefficient at 1e10 m.
989  a(i,k) = a(i,k) / (h_shear*g%H_to_m + 1.0e-10*dt*a(i,k))
990  endif ; enddo ; enddo ! i & k loops
991 
992  if (do_shelf) then
993  ! Set the coefficients to include the no-slip surface stress.
994  do i=is,ie ; if (do_i(i)) then
995  if (work_on_u) then
996  kv_tbl(i) = visc%kv_tbl_shelf_u(i,j)
997  tbl_thick(i) = visc%tbl_thick_shelf_u(i,j) * g%m_to_H
998  else
999  kv_tbl(i) = visc%kv_tbl_shelf_v(i,j)
1000  tbl_thick(i) = visc%tbl_thick_shelf_v(i,j) * g%m_to_H
1001  endif
1002  z_t(i) = 0.0
1003 
1004  ! If a(i,1) were not already 0, it would be added here.
1005  if (0.5*hvel(i,1) > tbl_thick(i)) then
1006  a(i,1) = kv_tbl(i) / (tbl_thick(i) *g%H_to_m + (1.0e-10*dt)*kv_tbl(i))
1007  else
1008  a(i,1) = kv_tbl(i) / (0.5*hvel(i,1)*g%H_to_m + (1.0e-10*dt)*kv_tbl(i))
1009  endif
1010  endif ; enddo
1011 
1012  do k=2,nz ; do i=is,ie ; if (do_i(i)) then
1013  z_t(i) = z_t(i) + hvel(i,k-1) / tbl_thick(i)
1014  topfn = 1.0 / (1.0 + 0.09 * z_t(i)**6)
1015 
1016  r = (hvel(i,k)+hvel(i,k-1))
1017  if (r > 2.0*tbl_thick(i)) then
1018  h_shear = ((1.0 - topfn) * r + topfn*2.0*tbl_thick(i))
1019  else
1020  h_shear = r
1021  endif
1022  ! The term including 1e-10 in the denominators is here to avoid
1023  ! truncation error problems in the tridiagonal solver. Effectively, this
1024  ! sets the maximum coupling coefficient increment to 1e10 m.
1025  a_top = 2.0 * topfn * kv_tbl(i)
1026  a(i,k) = a(i,k) + a_top / (h_shear*g%H_to_m + 1.0e-10*dt*a_top)
1027  endif ; enddo ; enddo
1028  elseif (cs%dynamic_viscous_ML .or. (g%nkml>0)) then
1029  max_nk = 0
1030  do i=is,ie ; if (do_i(i)) then
1031  if (g%nkml>0) nk_visc(i) = real(g%nkml+1)
1032  if (work_on_u) then
1033  u_star(i) = 0.5*(fluxes%ustar(i,j) + fluxes%ustar(i+1,j))
1034  absf(i) = 0.5*(abs(g%CoriolisBu(i,j-1)) + abs(g%CoriolisBu(i,j)))
1035  if (cs%dynamic_viscous_ML) nk_visc(i) = visc%nkml_visc_u(i,j) + 1
1036  else
1037  u_star(i) = 0.5*(fluxes%ustar(i,j) + fluxes%ustar(i,j+1))
1038  absf(i) = 0.5*(abs(g%CoriolisBu(i-1,j)) + abs(g%CoriolisBu(i,j)))
1039  if (cs%dynamic_viscous_ML) nk_visc(i) = visc%nkml_visc_v(i,j) + 1
1040  endif
1041  h_ml(i) = h_neglect ; z_t(i) = 0.0
1042  max_nk = max(max_nk,ceiling(nk_visc(i) - 1.0))
1043  endif ; enddo
1044 
1045  do k=1,max_nk ; do i=is,ie ; if (do_i(i)) then
1046  if (k+1 <= nk_visc(i)) then ! This layer is all in the ML.
1047  h_ml(i) = h_ml(i) + hvel(i,k)
1048  elseif (k < nk_visc(i)) then ! Part of this layer is in the ML.
1049  h_ml(i) = h_ml(i) + (nk_visc(i) - k) * hvel(i,k)
1050  endif
1051  endif ; enddo ; enddo
1052  if (work_on_u .and. allocated(hml_u)) then
1053  do i=is,ie ; if (do_i(i)) then ; hml_u(i,j) = h_ml(i) ; endif ; enddo
1054  elseif (.not.work_on_u .and. allocated(hml_v)) then
1055  do i=is,ie ; if (do_i(i)) then ; hml_v(i,j) = h_ml(i) ; endif ; enddo
1056  endif
1057 
1058  do k=2,max_nk ; do i=is,ie ; if (do_i(i)) then ; if (k < nk_visc(i)) then
1059  ! Set the viscosity at the interfaces.
1060  z_t(i) = z_t(i) + hvel(i,k-1)
1061  temp1 = (z_t(i)*h_ml(i) - z_t(i)*z_t(i)) * g%H_to_m
1062  ! This viscosity is set to go to 0 at the mixed layer top and bottom
1063  ! (in a log-layer) and be further limited by rotation to give the
1064  ! natural Ekman length.
1065  visc_ml = u_star(i) * 0.41 * (temp1*u_star(i)) / &
1066  (absf(i)*temp1 + h_ml(i)*u_star(i))
1067  a_ml = 4.0*visc_ml / ((hvel(i,k)+hvel(i,k-1) + h_neglect) * g%H_to_m + &
1068  2.0e-10*dt*visc_ml)
1069  ! Choose the largest estimate of a.
1070  if (a_ml > a(i,k)) a(i,k) = a_ml
1071  endif ; endif ; enddo ; enddo
1072  endif
1073 
1074  end subroutine find_coupling_coef
1075
