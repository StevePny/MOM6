144 subroutine initialize_ale( param_file, G, CS )
145 !------------------------------------------------------------------------------
146 ! This routine is typically called (from initialize_MOM in file MOM.F90)
147 ! before the main time integration loop to initialize the regridding stuff.
148 ! We read the MOM_input file to register the values of different
149 ! regridding/remapping parameters.
150 !------------------------------------------------------------------------------
151 
152  ! Arguments
153  type(param_file_type), intent(in) :: param_file
154  type(ocean_grid_type), intent(in) :: g
155  type(ale_cs), pointer :: cs
156 
157  ! Local variables
158  real, dimension(:), allocatable :: dz
159  character(len=40) :: mod = "MOM_ALE" ! This module's name.
160  character(len=80) :: string ! Temporary strings
161 
162  if (associated(cs)) then
163  call mom_error(warning, "initialize_ALE called with an associated "// &
164  "control structure.")
165  return
166  endif
167  allocate(cs)
168 
169  ! Memory allocation for regridding
170  call ale_memory_allocation( g, cs )
171 
172  ! --- BOUNDARY EXTRAPOLATION --
173  ! This sets whether high-order (rather than PCM) reconstruction schemes
174  ! should be used within boundary cells
175  call get_param(param_file, mod, "BOUNDARY_EXTRAPOLATION_PRESSURE", &
176  cs%boundary_extrapolation_for_pressure, &
177  "When defined, the reconstruction is extrapolated\n"//&
178  "within boundary cells rather than assume PCM for the.\n"//&
179  "calculation of pressure. e.g. if PPM is used, a\n"//&
180  "PPM reconstruction will also be used within\n"//&
181  "boundary cells.", default=.true.)
182 
183  ! --- PRESSURE GRADIENT CALCULATION ---
184  call get_param(param_file, mod, "RECONSTRUCT_FOR_PRESSURE", &
185  cs%reconstructForPressure , &
186  "If True, use vertical reconstruction of T/S within\n"//&
187  "the integrals of teh FV pressure gradient calculation.\n"//&
188  "If False, use the constant-by-layer algorithm.\n"//&
189  "By default, this is True when using ALE and False otherwise.", &
190  default=.true. )
191 
192  call get_param(param_file, mod, "PRESSURE_RECONSTRUCTION_SCHEME", &

195  "within the FV pressure gradient calculation."//&

207  "for vertical remapping for all variables.\n"//&

259 !------------------------------------------------------------------------------
260 ! Dispatching regridding routine: regridding & remapping
261 !------------------------------------------------------------------------------
262 subroutine ale_main( G, h, u, v, tv, CS )
263 !------------------------------------------------------------------------------
264 ! This routine takes care of (1) building a new grid and (2) remapping between
265 ! the old grid and the new grid. The creation of the new grid can be based
266 ! on z coordinates, target interface densities, sigma coordinates or any
267 ! arbitrary coordinate system.
268 !------------------------------------------------------------------------------
269 
270  ! Arguments
271  type(ocean_grid_type), intent(in) :: &
272  g ! Ocean grid informations
273  real, dimension(NIMEM_,NJMEM_, NKMEM_), intent(inout) :: &
274  h ! Current 3D grid obtained after the last time step
275  real, dimension(NIMEMB_,NJMEM_, NKMEM_), intent(inout) :: &
276  u ! Zonal velocity field
277  real, dimension(NIMEM_,NJMEMB_, NKMEM_), intent(inout) :: &
278  v ! Meridional velocity field
279  type(thermo_var_ptrs), intent(inout) :: &
280  tv ! Thermodynamical variables (T, S, ...)
281  type(ale_cs), intent(inout) :: cs ! Regridding parameters and options
282 
283  ! Local variables
284  integer :: nk
285 
286  ! Build new grid. The new grid is stored in h_new. The old grid is h.
287  ! Both are needed for the subsequent remapping of variables.
288  call regridding_main( cs%remapCS, cs%regridCS, g, h, tv, cs%dzRegrid )
289 
290  ! Remap all variables from old grid h onto new grid h_new
291  call remapping_main( cs%remapCS, g, h, -cs%dzRegrid, tv, u, v )
292 
293  ! Override old grid with new one. The new grid 'h_new' is built in
294  ! one of the 'build_...' routines above.
295  nk = g%ke
296  h(:,:,:) = h(:,:,:) + ( cs%dzRegrid(:,:,1:nk) - cs%dzRegrid(:,:,2:nk+1) )
297 
298 end subroutine ale_main

302 ! Use plm reconstruction for pressure gradient (determine edge values)
303 !------------------------------------------------------------------------------
304 subroutine pressure_gradient_plm( CS, S_t, S_b, T_t, T_b, G, tv, h )
305 !------------------------------------------------------------------------------
306 ! By using a PLM (limited piecewise linear method) reconstruction, this
307 ! routine determines the edge values for the salinity and temperature
308 ! within each layer. These edge values are returned and are used to compute
309 ! the pressure gradient (by computing the densities).
310 !------------------------------------------------------------------------------
311 
312  ! Arguments
313  type(ale_cs), intent(inout) :: cs ! Regridding parameters and options
314  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(inout) :: &
315  s_t, s_b ! Salinity at the top and bottom edges of each layer
316  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(inout) :: &
317  t_t, t_b ! Temperature at the top and bottom edges of each layer
318  type(ocean_grid_type), intent(in) :: g
319  type(thermo_var_ptrs), intent(in) :: tv
320  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: &
321  h ! Three-dimensional ocean grid
322 
323  ! Local variables
324  integer :: i, j, k
325  real :: htmp(g%ke)
326 
327  ! NOTE: the variables 'CS%grid_generic' and 'CS%ppoly_linear' are declared at
328  ! the module level. Memory is allocated once at the beginning of the run
329  ! in 'ALE_memory_allocation'.
330 
331  ! Determine reconstruction within each column
332  do i = g%isc,g%iec+1
333  do j = g%jsc,g%jec+1
334 
335  ! Build current grid
336  htmp(:) = h(i,j,:)
337 
338  ! Reconstruct salinity profile
339  cs%ppoly_linear%E = 0.0
340  cs%ppoly_linear%coefficients = 0.0
341  call plm_reconstruction( g%ke, htmp, tv%S(i,j,:), cs%ppoly_linear )
342  if (cs%boundary_extrapolation_for_pressure) call &
343  plm_boundary_extrapolation( g%ke, htmp, tv%S(i,j,:), cs%ppoly_linear )
344 
345  do k = 1,g%ke
346  s_t(i,j,k) = cs%ppoly_linear%E(k,1)
347  s_b(i,j,k) = cs%ppoly_linear%E(k,2)
348  end do
349 
350  ! Reconstruct temperature profile
351  cs%ppoly_linear%E = 0.0
352  cs%ppoly_linear%coefficients = 0.0
353  call plm_reconstruction( g%ke, htmp, tv%T(i,j,:), cs%ppoly_linear )
354  if (cs%boundary_extrapolation_for_pressure) call &
355  plm_boundary_extrapolation( g%ke, htmp, tv%T(i,j,:), cs%ppoly_linear )
356 
357  do k = 1,g%ke
358  t_t(i,j,k) = cs%ppoly_linear%E(k,1)
359  t_b(i,j,k) = cs%ppoly_linear%E(k,2)
360  end do
361 
362  end do
363  end do
364 
365 end subroutine pressure_gradient_plm

440 subroutine ale_memory_allocation( G, CS )
441 !------------------------------------------------------------------------------
442 ! In this routine, we allocate the memory needed to carry out regridding
443 ! steps in the course of the simulation.
444 
445 ! For example, to compute implicit edge-value estimates, a tridiagonal system
446 ! must be solved. We allocate the needed memory at the beginning of the
447 ! simulation because the number of layers never changes.
448 !------------------------------------------------------------------------------
449 
450  ! Arguments
451  type(ocean_grid_type), intent(in) :: g
452  type(ale_cs), intent(inout) :: cs
453 
454  ! Local variables
455  integer :: nz
456 
457  nz = g%ke
458 
459  ! Allocate memory for the tridiagonal system
460  call tridiagedgeworkallocate( nz, cs%edgeValueWrk )
461  call tridiagslopeworkallocate( nz, cs%edgeSlopeWrk )
462 
463  ! Generic linear piecewise polynomial
464  call ppoly_init( cs%ppoly_linear, nz, 1 )
465 
466  ! Generic parabolic piecewise polynomial
467  call ppoly_init( cs%ppoly_parab, nz, 2 )
468 
469  ! Work space
470  alloc_(cs%dzRegrid(g%isd:g%ied,g%jsd:g%jed,nz+1)); cs%dzRegrid(:,:,:) = 0.
471 
472 end subroutine ale_memory_allocation
