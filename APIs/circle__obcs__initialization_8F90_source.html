45 subroutine circle_obcs_initialize_thickness(h, G, param_file)
46  real, intent(out), dimension(NIMEM_,NJMEM_, NKMEM_) :: h
47  type(ocean_grid_type), intent(in) :: g
48  type(param_file_type), intent(in) :: param_file
49 ! Arguments: h - The thickness that is being initialized.
50 ! (in) G - The ocean's grid structure.
51 ! (in) param_file - A structure indicating the open file to parse for
52 ! model parameter values.
53 
54 ! This subroutine initializes layer thicknesses for the circle_obcs experiment
55  real :: e0(szk_(g)+1) ! The resting interface heights, in m, usually !
56  ! negative because it is positive upward. !
57  real :: eta1d(szk_(g)+1)! Interface height relative to the sea surface !
58  ! positive upward, in m. !
59 ! This include declares and sets the variable "version".
60 #include "version_variable.h"
61  character(len=40) :: mod = "circle_obcs_initialize_thickness" ! This module's name.
62  integer :: i, j, k, is, ie, js, je, nz
63  real :: diskrad, rad, xcenter, xradius, lonc, latc
64 
65  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
66 
67  call mom_mesg(" circle_obcs_initialization.F90, circle_obcs_initialize_thickness: setting thickness", 5)
68 
69  ! Parameters read by cartesian grid initialization
70  call log_version(param_file, mod, version, "")
71  call get_param(param_file, mod, "DISK_RADIUS", diskrad, &
72  "The radius of the initially elevated disk in the \n"//&
73  "circle_obcs test case.", units=g%axis_units, &
74  fail_if_missing=.true.)
75 
76  do k=1,nz
77  e0(k) = -g%max_depth * real(k-1) / real(nz)
78  enddo
79 
80  ! Uniform thicknesses for base state
81  do j=js,je ; do i=is,ie !
82  eta1d(nz+1) = -1.0*g%bathyT(i,j)
83  do k=nz,1,-1
84  eta1d(k) = e0(k)
85  if (eta1d(k) < (eta1d(k+1) + g%Angstrom_z)) then
86  eta1d(k) = eta1d(k+1) + g%Angstrom_z
87  h(i,j,k) = g%Angstrom_z
88  else
89  h(i,j,k) = eta1d(k) - eta1d(k+1)
90  endif
91  enddo
92  enddo ; enddo
93 
94  ! Perturb base state by circular anomaly in center
95  k=nz
96  latc = g%south_lat + 0.5*g%len_lat
97  lonc = g%west_lon + 0.5*g%len_lon
98  do j=js,je ; do i=is,ie
99  rad = sqrt((g%geoLonT(i,j)-lonc)**2+(g%geoLatT(i,j)-latc)**2)/(diskrad)
100  ! if (rad <= 6.*diskrad) h(i,j,k) = h(i,j,k)+10.0*exp( -0.5*( rad**2 ) )
101  rad = min( rad, 1. ) ! Flatten outside radius of diskrad
102  rad = rad*(2.*asin(1.)) ! Map 0-1 to 0-pi
103  h(i,j,k) = h(i,j,k) + 10.0*0.5*(1.+cos(rad)) ! cosine bell
104  h(i,j,k-1) = h(i,j,k-1) - 10.0*0.5*(1.+cos(rad)) ! cosine bell
105  enddo ; enddo
