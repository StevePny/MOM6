77 subroutine register_tracer(tr1, name, param_file, Reg, ad_x, ad_y, &
78  df_x, df_y, obc_inflow, obc_in_u, obc_in_v, &
79  ad_2d_x, ad_2d_y, df_2d_x, df_2d_y)
80  real, dimension(NIMEM_,NJMEM_,NKMEM_), target :: tr1
81  character(len=*), intent(in) :: name
82  type(param_file_type), intent(in) :: param_file
83  type(tracer_registry_type), pointer :: reg
84  real, pointer, dimension(:,:,:), optional :: ad_x, ad_y, df_x, df_y
85  real, intent(in), optional :: obc_inflow

88 ! This subroutine registers a tracer to be advected and horizontally
89 ! diffused.
90 
91 ! Arguments: tr1 - The pointer to the tracer, in arbitrary concentration units (CONC).
92 ! (in) name - The name to be used in messages about the tracer.
93 ! (in) param_file - A structure indicating the open file to parse for
94 ! model parameter values.
95 ! (in/out) Reg - A pointer to the tracer registry.
96 ! (in) ad_x - An array in which zonal advective fluxes are stored in
97 ! units of CONC m3 s-1.
98 ! (in) ad_y - An array in which meridional advective fluxes are stored
99 ! in units of CONC m3 s-1.
100 ! (in) df_x - An array in which zonal diffusive fluxes are stored in
101 ! units of CONC m3 s-1.
102 ! (in) df_y - An array in which meridional diffusive fluxes are stored
103 ! in units of CONC m3 s-1.
104 ! (in) OBC_inflow - The value of the tracer for all inflows via the open
105 ! boundary conditions for which OBC_in_u or OBC_in_v are
106 ! not specified, in the same units as tr (CONC).
107 ! (in) OBC_in_u - The value of the tracer at inflows through u-faces of
108 ! tracer cells, in the same units as tr (CONC).
109 ! (in) OBC_in_v - The value of the tracer at inflows through v-faces of
110 ! tracer cells, in the same units as tr (CONC).
111 ! (in,opt) ad_2d_x - An array in which the vertically summed zonal advective
112 ! fluxes are stored in units of CONC m3 s-1.
113 ! (in,opt) ad_2d_y - An array in which the vertically summed meridional advective
114 ! fluxes are stored in units of CONC m3 s-1.
115 ! (in,opt) df_2d_x - An array in which the vertically summed zonal diffusive
116 ! fluxes are stored in units of CONC m3 s-1.
117 ! (in,opt) df_2d_y - An array in which the vertically summed meridional diffusive
118 ! fluxes are stored in units of CONC m3 s-1.
119 
120  integer :: m, ntr
121  type(tracer_type) :: temp
122  character(len=256) :: mesg ! Message for error messages.
123 
124  if (.not. associated(reg)) call tracer_registry_init(param_file, reg)
125 
126  if (reg%ntr>=max_fields_) then
127  write(mesg,'("Increase MAX_FIELDS_ in MOM_memory.h to at least ",I3," to allow for &
128  &all the tracers being registered via register_tracer.")') reg%ntr+1
129  call mom_error(fatal,"MOM register_tracer: "//mesg)
130  endif
131  reg%ntr = reg%ntr + 1
132  ntr = reg%ntr
133 
134  reg%Tr(ntr)%name = trim(name)
135  reg%Tr(ntr)%t => tr1
136 
137  if (present(ad_x)) then ; if (associated(ad_x)) reg%Tr(ntr)%ad_x => ad_x ; endif
138  if (present(ad_y)) then ; if (associated(ad_y)) reg%Tr(ntr)%ad_y => ad_y ; endif
139  if (present(df_x)) then ; if (associated(df_x)) reg%Tr(ntr)%df_x => df_x ; endif
140  if (present(df_y)) then ; if (associated(df_y)) reg%Tr(ntr)%df_y => df_y ; endif
141  if (present(obc_inflow)) reg%Tr(ntr)%OBC_inflow_conc = obc_inflow
142  if (present(obc_in_u)) then ; if (associated(obc_in_u)) &
143  reg%Tr(ntr)%OBC_in_u => obc_in_u ; endif
144  if (present(obc_in_v)) then ; if (associated(obc_in_v)) &
145  reg%Tr(ntr)%OBC_in_v => obc_in_v ; endif
146  if (present(ad_2d_x)) then ; if (associated(ad_2d_x)) reg%Tr(m)%ad2d_x => ad_2d_x ; endif
147  if (present(ad_2d_y)) then ; if (associated(ad_2d_y)) reg%Tr(m)%ad2d_y => ad_2d_y ; endif
148  if (present(df_2d_x)) then ; if (associated(df_2d_x)) reg%Tr(m)%df2d_x => df_2d_x ; endif
149  if (present(df_2d_y)) then ; if (associated(df_2d_y)) reg%Tr(m)%df2d_y => df_2d_y ; endif
150 
151 end subroutine register_tracer

239 subroutine tracer_vertdiff(h_old, ea, eb, dt, tr, G, &
240  sfc_flux, btm_flux, btm_reservoir, sink_rate)
241  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h_old, ea, eb
242  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(inout) :: tr
243  real, intent(in) :: dt
244  type(ocean_grid_type), intent(in) :: g
245  real, dimension(NIMEM_,NJMEM_), optional, intent(in) :: sfc_flux
246  real, dimension(NIMEM_,NJMEM_), optional, intent(in) :: btm_flux
247  real, dimension(NIMEM_,NJMEM_), optional, intent(inout) :: btm_reservoir
248  real, optional, intent(in) :: sink_rate
249 ! Arguments: h_old - Layer thickness before entrainment, in m or kg m-2.
250 ! (in) ea - The amount of fluid entrained from the layer above, in the
251 ! same units as h_old, i.e. m or kg m-2.
252 ! (in) eb - The amount of fluid entrained from the layer below, in the
253 ! same units as h_old, i.e. m or kg m-2
254 ! (inout) tr - The tracer concentration, in concentration units (CU).
255 ! (in) dt - The amount of time covered by this call, in s.
256 ! (in) G - The ocean's grid structure.
257 ! (in,opt) sfc_flux - The surface flux of the tracer, in CU kg m-2 s-1.
258 ! (in,opt) btm_flux - The (negative upward) bottom flux of the tracer,
259 ! in units of CU kg m-2 s-1.
260 ! (inout,opt) btm_reservoir - The amount of tracer in a bottom reservoir, in
261 ! units of CU kg m-2. (was CU m)
262 ! (in,opt) sink_rate - The rate at which the tracer sinks, in m s-1.
263 
264 ! This subroutine solves a tridiagonal equation for the final tracer
265 ! concentrations after the dual-entrainments, and possibly sinking or surface
266 ! and bottom sources, are applied. The sinking is implemented with an
267 ! fully implicit upwind advection scheme.
268 
269  real :: sink_dist ! The distance the tracer sinks in a time step, in m or kg m-2.
270  real, dimension(SZI_(G)) :: &
271  sfc_src, & ! The time-integrated surface source of the tracer, in
272  ! units of m or kg m-2 times a concentration.
273  btm_src, & ! The time-integrated bottom source of the tracer, in
274  ! units of m or kg m-2 times a concentration.
275  b1, & ! b1 is used by the tridiagonal solver, in m-1 or m2 kg-1.
276  d1 ! d1=1-c1 is used by the tridiagonal solver, nondimensional.
277  real :: c1(szi_(g),szk_(g)) ! c1 is used by the tridiagonal solver, ND.
278  real :: h_minus_dsink(szi_(g),szk_(g)) ! The layer thickness minus the
279  ! difference in sinking rates across the layer, in m or kg m-2.
280  ! By construction, 0 <= h_minus_dsink < h_old.
281  real :: sink(szi_(g),szk_(g)+1) ! The tracer's sinking distances at the
282  ! interfaces, limited to prevent characteristics from
283  ! crossing within a single timestep, in m or kg m-2.
284  real :: b_denom_1 ! The first term in the denominator of b1, in m or kg m-2.
285  real :: h_tr ! h_tr is h at tracer points with a h_neglect added to
286  ! ensure positive definiteness, in m or kg m-2.
287  real :: h_neglect ! A thickness that is so small it is usually lost
288  ! in roundoff and can be neglected, in m.
289  integer :: i, j, k, is, ie, js, je, nz
290  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
291 
292  h_neglect = g%H_subroundoff
293 
294  sink_dist = 0.0
295  if (present(sink_rate)) sink_dist = (dt*sink_rate) * g%m_to_H
296  do i=is,ie ; sfc_src(i) = 0.0 ; btm_src(i) = 0.0 ; enddo
297 
298  if (present(sink_rate)) then
299  do j=js,je
300  ! Find the sinking rates at all interfaces, limiting them if necesary
301  ! so that the characteristics do not cross within a timestep.
302  ! If a non-constant sinking rate were used, that would be incorprated
303  ! here.
304  if (present(btm_reservoir)) then
305  do i=is,ie ; sink(i,nz+1) = sink_dist ; enddo
306  do k=2,nz ; do i=is,ie
307  sink(i,k) = sink_dist ; h_minus_dsink(i,k) = h_old(i,j,k)
308  enddo ; enddo
309  else
310  do i=is,ie ; sink(i,nz+1) = 0.0 ; enddo
311  ! Find the limited sinking distance at the interfaces.
312  do k=nz,2,-1 ; do i=is,ie
313  if (sink(i,k+1) >= sink_dist) then
314  sink(i,k) = sink_dist
315  h_minus_dsink(i,k) = h_old(i,j,k) + (sink(i,k+1) - sink(i,k))
316  elseif (sink(i,k+1) + h_old(i,j,k) < sink_dist) then
317  sink(i,k) = sink(i,k+1) + h_old(i,j,k)
318  h_minus_dsink(i,k) = 0.0
319  else
320  sink(i,k) = sink_dist
321  h_minus_dsink(i,k) = (h_old(i,j,k) + sink(i,k+1)) - sink(i,k)
322  endif
323  enddo ; enddo
324  endif
325  do i=is,ie
326  sink(i,1) = 0.0 ; h_minus_dsink(i,1) = (h_old(i,j,1) + sink(i,2))
327  enddo
328 
329  ! Now solve the tridiagonal equation for the tracer concentrations.
330  do i=is,ie ; if (g%mask2dT(i,j) > 0.5) then
331  b_denom_1 = h_minus_dsink(i,1) + ea(i,j,1) + h_neglect
332  b1(i) = 1.0 / (b_denom_1 + eb(i,j,1))
333  d1(i) = b_denom_1 * b1(i)
334  if (present(sfc_flux)) sfc_src(i) = (sfc_flux(i,j)*dt) * g%kg_m2_to_H
335  h_tr = h_old(i,j,1) + h_neglect
336  tr(i,j,1) = b1(i)*(h_tr*tr(i,j,1) + sfc_src(i))
337  endif ; enddo
338  do k=2,nz-1 ; do i=is,ie ; if (g%mask2dT(i,j) > 0.5) then
339  c1(i,k) = eb(i,j,k-1) * b1(i)
340  b_denom_1 = h_minus_dsink(i,k) + d1(i) * (ea(i,j,k) + sink(i,k)) + &
341  h_neglect
342  b1(i) = 1.0 / (b_denom_1 + eb(i,j,k))
343  d1(i) = b_denom_1 * b1(i)
344  h_tr = h_old(i,j,k) + h_neglect
345  tr(i,j,k) = b1(i) * (h_tr * tr(i,j,k) + &
346  (ea(i,j,k) + sink(i,k)) * tr(i,j,k-1))
347  endif ; enddo ; enddo
348  do i=is,ie ; if (g%mask2dT(i,j) > 0.5) then
349  c1(i,nz) = eb(i,j,nz-1) * b1(i)
350  b_denom_1 = h_minus_dsink(i,nz) + d1(i) * (ea(i,j,nz) + sink(i,nz)) + &
351  h_neglect
352  b1(i) = 1.0 / (b_denom_1 + eb(i,j,nz))
353  h_tr = h_old(i,j,nz) + h_neglect
354  if (present(btm_flux)) btm_src(i) = (-btm_flux(i,j)*dt) * g%kg_m2_to_H
355  tr(i,j,nz) = b1(i) * ((h_tr * tr(i,j,nz) + btm_src(i)) + &
356  (ea(i,j,nz) + sink(i,nz)) * tr(i,j,nz-1))
357  endif ; enddo
358  if (present(btm_reservoir)) then ; do i=is,ie ; if (g%mask2dT(i,j)>0.5) then
359  btm_reservoir(i,j) = btm_reservoir(i,j) + &
360  (sink(i,nz+1)*tr(i,j,nz)) * g%H_to_kg_m2
361  endif ; enddo ; endif
362 
363  do k=nz-1,1,-1 ; do i=is,ie ; if (g%mask2dT(i,j) > 0.5) then
364  tr(i,j,k) = tr(i,j,k) + c1(i,k+1)*tr(i,j,k+1)
365  endif ; enddo ; enddo
366  enddo
367  else
368  do j=js,je
369  do i=is,ie ; if (g%mask2dT(i,j) > 0.5) then
370  h_tr = h_old(i,j,1) + h_neglect
371  b_denom_1 = h_tr + ea(i,j,1)
372  b1(i) = 1.0 / (b_denom_1 + eb(i,j,1))
373  d1(i) = b_denom_1 * b1(i)
374  if (present(sfc_flux)) sfc_src(i) = (sfc_flux(i,j)*dt) * g%kg_m2_to_H
375  tr(i,j,1) = b1(i)*(h_tr*tr(i,j,1) + sfc_src(i))
376  endif ; enddo
377  do k=2,nz-1 ; do i=is,ie ; if (g%mask2dT(i,j) > 0.5) then
378  c1(i,k) = eb(i,j,k-1) * b1(i)
379  h_tr = h_old(i,j,k) + h_neglect
380  b_denom_1 = h_tr + d1(i) * ea(i,j,k)
381  b1(i) = 1.0 / (b_denom_1 + eb(i,j,k))
382  d1(i) = b_denom_1 * b1(i)
383  tr(i,j,k) = b1(i) * (h_tr * tr(i,j,k) + ea(i,j,k) * tr(i,j,k-1))
384  endif ; enddo ; enddo
385  do i=is,ie ; if (g%mask2dT(i,j) > 0.5) then
