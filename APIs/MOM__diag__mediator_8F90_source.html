278 subroutine post_data_2d(diag_field_id, field, diag, is_static, mask)
279  integer, intent(in) :: diag_field_id
280  real, intent(in) :: field(:,:)
281  type(diag_ctrl), intent(in) :: diag
282  logical, optional, intent(in) :: is_static
283  real, optional, intent(in) :: mask(:,:)
284 ! Arguments: diag_field_id - the id for an output variable returned by a
285 ! previous call to register_diag_field.
286 ! (in) field - The 2-d array being offered for output or averaging.
287 ! (inout) diag - A structure that is used to regulate diagnostic output.
288 ! (in,opt) is_static - If true, this is a static field that is always offered.
289 ! (in,opt) mask - If present, use this real array as the data mask.
290  logical :: used, is_stat
291  integer :: isv, iev, jsv, jev
292 
293  is_stat = .false. ; if (present(is_static)) is_stat = is_static
294 
295  ! Determine the propery array indices, noting that because of the (:,:)
296  ! declaration of field, symmetric arrays are using a SW-grid indexing,
297  ! but non-symmetric arrays are using a NE-grid indexing. Send_data
298  ! actually only uses the difference between ie and is to determine
299  ! the output data size and assumes that halos are symmetric.
300  isv = diag%is ; iev = diag%ie ; jsv = diag%js ; jev = diag%je
301 
302  if ( size(field,1) == diag%ied-diag%isd +1 ) then
303  isv = diag%is ; iev = diag%ie ! Data domain
304  elseif ( size(field,1) == diag%ied-diag%isd +2 ) then
305  isv = diag%is ; iev = diag%ie+1 ! Symmetric data domain
306  elseif ( size(field,1) == diag%ie-diag%is +1 ) then
307  isv = 1 ; iev = diag%ie + 1-diag%is ! Computational domain
308  elseif ( size(field,1) == diag%ie-diag%is +2 ) then
309  isv = 1 ; iev = diag%ie + 2-diag%is ! Symmetric computational domain
310  else
311  call mom_error(fatal,"post_data_2d: peculiar size in i-direction")
312  endif
313  if ( size(field,2) == diag%jed-diag%jsd +1 ) then
314  jsv = diag%js ; jev = diag%je ! Data domain
315  elseif ( size(field,2) == diag%jed-diag%jsd +2 ) then
316  jsv = diag%js ; jev = diag%je+1 ! Symmetric data domain
317  elseif ( size(field,2) == diag%je-diag%js +1 ) then
318  jsv = 1 ; jev = diag%je + 1-diag%js ! Computational domain
319  elseif ( size(field,1) == diag%je-diag%js +2 ) then
320  jsv = 1 ; jev = diag%je + 2-diag%js ! Symmetric computational domain
321  else
322  call mom_error(fatal,"post_data_2d: peculiar size in j-direction")
323  endif
324 
325  if (present(mask)) then
326  if ((size(field,1) /= size(mask,1)) .or. &
327  (size(field,2) /= size(mask,2))) then
328  call mom_error(fatal, "post_data_2d: post_data called with a mask "//&
329  "that does not match the size of field.")
330  endif
331  endif
332 
333  if (is_stat) then
334  if (present(mask)) then
335  used = send_data(diag_field_id, field, &
336  is_in=isv, js_in=jsv, ie_in=iev, je_in=jev, rmask=mask)
337  !elseif(associated(diag%maskList(diag_field_id)%mask2d)) then
338  ! used = send_data(diag_field_id, field, &
339  ! is_in=isv, js_in=jsv, ie_in=iev, je_in=jev, rmask=diag%maskList(diag_field_id)%mask2d)
340  else
341  used = send_data(diag_field_id, field, &
342  is_in=isv, js_in=jsv, ie_in=iev, je_in=jev)
343  endif
344  elseif (diag%ave_enabled) then
345  if (present(mask)) then
346  used = send_data(diag_field_id, field, diag%time_end, &
347  is_in=isv, js_in=jsv, ie_in=iev, je_in=jev, &
348  weight=diag%time_int, rmask=mask)
349  elseif(associated(diag%maskList(diag_field_id)%mask2d)) then
350  used = send_data(diag_field_id, field, diag%time_end, &
351  is_in=isv, js_in=jsv, ie_in=iev, je_in=jev, &
352  weight=diag%time_int, rmask=diag%maskList(diag_field_id)%mask2d)
353  else
354  used = send_data(diag_field_id, field, diag%time_end, &
355  is_in=isv, js_in=jsv, ie_in=iev, je_in=jev, &
356  weight=diag%time_int)
357  endif
358  endif
359 
360 end subroutine post_data_2d
