24 !* By Robert Hallberg, April 1994 - July 2000 *
25 !* *
26 !* This program contains the subroutine that, along with the *
27 !* subroutines that it calls, implements diapycnal mass and momentum *
28 !* fluxes and a bulk mixed layer. The diapycnal diffusion can be *
29 !* used without the bulk mixed layer. *
30 !* *
31 !* diabatic first determines the (diffusive) diapycnal mass fluxes *
32 !* based on the convergence of the buoyancy fluxes within each layer. *
33 !* The dual-stream entrainment scheme of MacDougall and Dewar (JPO, *
34 !* 1997) is used for combined diapycnal advection and diffusion, *
35 !* calculated implicitly and potentially with the Richardson number *
36 !* dependent mixing, as described by Hallberg (MWR, 2000). Diapycnal *
37 !* advection is fundamentally the residual of diapycnal diffusion, *
38 !* so the fully implicit upwind differencing scheme that is used is *
39 !* entirely appropriate. The downward buoyancy flux in each layer *
40 !* is determined from an implicit calculation based on the previously *
41 !* calculated flux of the layer above and an estimated flux in the *
42 !* layer below. This flux is subject to the following conditions: *
43 !* (1) the flux in the top and bottom layers are set by the boundary *
44 !* conditions, and (2) no layer may be driven below an Angstrom thick-*
45 !* ness. If there is a bulk mixed layer, the buffer layer is treat- *
46 !* ed as a fixed density layer with vanishingly small diffusivity. *
47 !* *
48 !* diabatic takes 5 arguments: the two velocities (u and v), the *
49 !* thicknesses (h), a structure containing the forcing fields, and *
50 !* the length of time over which to act (dt). The velocities and *
51 !* thickness are taken as inputs and modified within the subroutine. *
52 !* There is no limit on the time step. *
53 !* *
54 !* A small fragment of the grid is shown below: *
55 !* *
56 !* j+1 x ^ x ^ x At x: q *
57 !* j+1 > o > o > At ^: v *
58 !* j x ^ x ^ x At >: u *
59 !* j > o > o > At o: h, T, S, buoy, ustar, ea, eb, etc. *
60 !* j-1 x ^ x ^ x *
61 !* i-1 i i+1 At x & ^: *
62 !* i i+1 At > & o: *
63 !* *
64 !* The boundaries always run through q grid points (x). *
65 !* *

986  b1(i) = 1.0 / ((eb(i,j,k) + eb(i,j+1,k)) + (hval + d1(i)*eaval))

1073 subroutine make_frazil(h, tv, G, CS)
1074  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h
1075  type(thermo_var_ptrs), intent(inout) :: tv
1076  type(ocean_grid_type), intent(in) :: g
1077  type(diabatic_cs), intent(in) :: cs
1078 
1079 ! Frazil formation keeps the temperature above the freezing point.
1080 ! This subroutine warms any water that is colder than the (currently
1081 ! surface) freezing point up to the freezing point and accumulates
1082 ! the required heat (in J m-2) in tv%frazil.
1083 ! The expression, below, for the freezing point of sea water comes
1084 ! from Millero (1978) via Appendix A of Gill, 1982.
1085 
1086 ! Arguments: h - Layer thickness, in m or kg m-2.
1087 ! (in/out) tv - A structure containing pointers to any available
1088 ! thermodynamic fields. Absent fields have NULL ptrs.
1089 ! (in) G - The ocean's grid structure.
1090 ! (in) CS - The control structure returned by a previous call to
1091 ! diabatic_driver_init.
1092  real, dimension(SZI_(G)) :: &
1093  fraz_col, & ! The accumulated heat requirement due to frazil, in J.
1094  t_freeze ! The freezing potential temperature at the current salinity, C.
1095  real, dimension(SZI_(G),SZK_(G)) :: &
1096  pressure ! The pressure at the middle of each layer in Pa.
1097  real :: hc ! A layer's heat capacity in J m-2 K-1.
1098  logical :: t_fr_set ! True if the freezing point has been calculated for a
1099  ! row of points.
1100  integer :: i, j, k, is, ie, js, je, nz
1101  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
1102 
1103  call cpu_clock_begin(id_clock_frazil)
1104 
1105  if (.not.cs%pressure_dependent_frazil) then
1106  do k=1,nz ; do i=is,ie ; pressure(i,k) = 0.0 ; enddo ; enddo
1107  endif
1108 
1109  do j=js,je
1110  do i=is,ie ; fraz_col(:) = 0.0 ; enddo
1111 
1112  if (cs%pressure_dependent_frazil) then
1113  do i=is,ie
1114  pressure(i,1) = (0.5*g%H_to_Pa)*h(i,j,1)
1115  enddo
1116  do k=2,nz ; do i=is,ie
1117  pressure(i,k) = pressure(i,k-1) + &
1118  (0.5*g%H_to_Pa) * (h(i,j,k) + h(i,j,k-1))
1119  enddo ; enddo
1120  endif
1121 
1122  if (cs%reclaim_frazil) then
1123  t_fr_set = .false.
1124  do i=is,ie ; if (tv%frazil(i,j) > 0.0) then
1125  if (.not.t_fr_set) then
1126  call calculate_tfreeze(tv%S(i:,j,1), pressure(i:,1), t_freeze(i:), &
1127  1, ie-i+1, tv%eqn_of_state)
1128  t_fr_set = .true.
1129  endif
1130 
1131  if (tv%T(i,j,1) > t_freeze(i)) then
1132  ! If frazil had previously been formed, but the surface temperature is now
1133  ! above freezing, cool the surface layer with the frazil heat deficit.
1134  hc = (tv%C_p*g%H_to_kg_m2) * h(i,j,1)
1135  if (tv%frazil(i,j) - hc * (tv%T(i,j,1) - t_freeze(i)) <= 0.0) then
1136  tv%T(i,j,1) = tv%T(i,j,1) - tv%frazil(i,j)/hc
1137  tv%frazil(i,j) = 0.0
1138  else
1139  tv%frazil(i,j) = tv%frazil(i,j) - hc * (tv%T(i,j,1) - t_freeze(i))
1140  tv%T(i,j,1) = t_freeze(i)
1141  endif
1142  endif
1143  endif ; enddo
1144  endif
1145 
1146  do k=nz,1,-1
1147  t_fr_set = .false.
1148  do i=is,ie
1149  if ((g%mask2dT(i,j) > 0.0) .and. &
1150  ((tv%T(i,j,k) < 0.0) .or. (fraz_col(i) > 0.0))) then
1151  if (.not.t_fr_set) then
1152  call calculate_tfreeze(tv%S(i:,j,k), pressure(i:,k), t_freeze(i:), &
1153  1, ie-i+1, tv%eqn_of_state)
1154  t_fr_set = .true.
1155  endif
1156 
1157  hc = (tv%C_p*g%H_to_kg_m2) * h(i,j,k)
1158  if (h(i,j,k) <= 10.0*g%Angstrom) then
1159  ! Very thin layers should not be cooled by the frazil flux.
1160  if (tv%T(i,j,k) < t_freeze(i)) then

1162  tv%T(i,j,k) = t_freeze(i)

1195 ! (in) tv - A structure containing pointers to any available
1196 ! thermodynamic fields. Absent fields have NULL ptrs.

1200 ! (in) G - The ocean's grid structure.

1236  do i=is,ie
1237  i_h_int = 1.0 / (0.5 * (h(i,j,1) + h(i,j,2)) + h_neglect)

1239  mix_s(i,2) = ((dt * kd_s(i,j,2)) * g%m_to_H**2) * i_h_int
1240

1245  d1_s(i) = h_tr * b1_s(i)

1253  mix_s(i,k+1) = ((dt * kd_s(i,j,k+1)) * g%m_to_H**2) * i_h_int

1256  c1_s(i,k) = mix_s(i,k) * b1_s(i)
1257

1260  b_denom_s = h_tr + d1_s(i)*mix_s(i,k)
1261  b1_t(i) = 1.0 / (b_denom_t + mix_t(i,k+1))
1262  b1_s(i) = 1.0 / (b_denom_s + mix_s(i,k+1))
1263  d1_t(i) = b_denom_t * b1_t(i)
1264  d1_s(i) = b_denom_s * b1_s(i)
1265 
1266  t(i,j,k) = b1_t(i) * (h_tr*t(i,j,k) + mix_t(i,k)*t(i,j,k-1))
1267  s(i,j,k) = b1_s(i) * (h_tr*s(i,j,k) + mix_s(i,k)*s(i,j,k-1))
1268  enddo ; enddo
1269  do i=is,ie
1270  c1_t(i,nz) = mix_t(i,nz) * b1_t(i)
1271  c1_s(i,nz) = mix_s(i,nz) * b1_s(i)
1272 
1273  h_tr = h(i,j,nz) + h_neglect
1274  b1_t(i) = 1.0 / (h_tr + d1_t(i)*mix_t(i,nz))
1275  b1_s(i) = 1.0 / (h_tr + d1_s(i)*mix_s(i,nz))
1276 
1277  t(i,j,nz) = b1_t(i) * (h_tr*t(i,j,nz) + mix_t(i,nz)*t(i,j,nz-1))
1278  s(i,j,nz) = b1_s(i) * (h_tr*s(i,j,nz) + mix_s(i,nz)*s(i,j,nz-1))
1279  enddo
1280  do k=nz-1,1,-1 ; do i=is,ie
1281  t(i,j,k) = t(i,j,k) + c1_t(i,k+1)*t(i,j,k+1)
1282  s(i,j,k) = s(i,j,k) + c1_s(i,k+1)*s(i,j,k+1)
1283  enddo ; enddo
1284  enddo

1695 subroutine find_uv_at_h(u, v, h, u_h, v_h, G, ea, eb)
1696  real, dimension(NIMEMB_,NJMEM_,NKMEM_), intent(in) :: u
1697  real, dimension(NIMEM_,NJMEMB_,NKMEM_), intent(in) :: v
1698  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h
1699  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(out) :: u_h, v_h
1700  type(ocean_grid_type), intent(in) :: g
1701  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in), optional :: ea, eb
1702 ! This subroutine calculates u_h and v_h (velocities at thickness
1703 ! points), optionally using the entrainments (in m) passed in as arguments.
1704 
1705 ! Arguments: u - Zonal velocity, in m s-1.
1706 ! (in) v - Meridional velocity, in m s-1.
1707 ! (in) h - Layer thickness, in m or kg m-2.
1708 ! (out) u_h - The zonal velocity at thickness points after
1709 ! entrainment, in m s-1.
1710 ! (out) v_h - The meridional velocity at thickness points after
1711 ! entrainment, in m s-1.
1712 ! (in) G - The ocean's grid structure.
1713 ! (in, opt) ea - The amount of fluid entrained from the layer above within
1714 ! this time step, in units of m or kg m-2. Omitting ea is the
1715 ! same as setting it to 0.
1716 ! (in, opt) eb - The amount of fluid entrained from the layer below within
1717 ! this time step, in units of m or kg m-2. Omitting eb is the
1718 ! same as setting it to 0. ea and eb must either be both
1719 ! present or both absent.
1720 
1721  real :: b_denom_1 ! The first term in the denominator of b1 in m or kg m-2.
1722  real :: h_neglect ! A thickness that is so small it is usually lost
1723  ! in roundoff and can be neglected, in m or kg m-2.
1724  real :: b1(szi_(g)), d1(szi_(g)), c1(szi_(g),szk_(g))
1725  real :: a_n(szi_(g)), a_s(szi_(g)) ! Fractional weights of the neighboring
1726  real :: a_e(szi_(g)), a_w(szi_(g)) ! velocity points, ~1/2 in the open
1727  ! ocean, nondimensional.
1728  real :: s, idenom
1729  logical :: mix_vertically
1730  integer :: i, j, k, is, ie, js, je, nz
1731  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
1732  call cpu_clock_begin(id_clock_uv_at_h)
1733  h_neglect = g%H_subroundoff
1734 
1735  mix_vertically = present(ea)
1736  if (present(ea) .neqv. present(eb)) call mom_error(fatal, &
1737  "find_uv_at_h: Either both ea and eb or neither one must be present "// &
1738  "in call to find_uv_at_h.")
1739 
1740  do j=js,je
1741  do i=is,ie
1742  s = g%areaCu(i-1,j)+g%areaCu(i,j)
1743  if (s>0.0) then
1744  idenom = sqrt(0.5*g%IareaT(i,j)/s)
1745  a_w(i) = g%areaCu(i-1,j)*idenom
1746  a_e(i) = g%areaCu(i,j)*idenom
1747  else
1748  a_w(i) = 0.0 ; a_e(i) = 0.0
1749  endif
1750 
1751  s = g%areaCv(i,j-1)+g%areaCv(i,j)
1752  if (s>0.0) then
1753  idenom = sqrt(0.5*g%IareaT(i,j)/s)
1754  a_s(i) = g%areaCv(i,j-1)*idenom
1755  a_n(i) = g%areaCv(i,j)*idenom
1756  else
1757  a_s(i) = 0.0 ; a_n(i) = 0.0
1758  endif
1759  enddo
1760 
1761  if (mix_vertically) then
1762  do i=is,ie
1763  b_denom_1 = h(i,j,1) + h_neglect
1764  b1(i) = 1.0 / (b_denom_1 + eb(i,j,1))
1765  d1(i) = b_denom_1 * b1(i)
1766  u_h(i,j,1) = (h(i,j,1)*b1(i)) * (a_e(i)*u(i,j,1) + a_w(i)*u(i-1,j,1))
1767  v_h(i,j,1) = (h(i,j,1)*b1(i)) * (a_n(i)*v(i,j,1) + a_s(i)*v(i,j-1,1))
1768  enddo
1769  do k=2,nz ; do i=is,ie
1770  c1(i,k) = eb(i,j,k-1) * b1(i)
1771  b_denom_1 = h(i,j,k) + d1(i)*ea(i,j,k) + h_neglect
1772  b1(i) = 1.0 / (b_denom_1 + eb(i,j,k))
1773  d1(i) = b_denom_1 * b1(i)
1774  u_h(i,j,k) = (h(i,j,k) * (a_e(i)*u(i,j,k) + a_w(i)*u(i-1,j,k)) + &
1775  ea(i,j,k)*u_h(i,j,k-1))*b1(i)
1776  v_h(i,j,k) = (h(i,j,k) * (a_n(i)*v(i,j,k) + a_s(i)*v(i,j-1,k)) + &
1777  ea(i,j,k)*v_h(i,j,k-1))*b1(i)
1778  enddo ; enddo
1779  do k=nz-1,1,-1 ; do i=is,ie
1780  u_h(i,j,k) = u_h(i,j,k) + c1(i,k+1)*u_h(i,j,k+1)
1781  v_h(i,j,k) = v_h(i,j,k) + c1(i,k+1)*v_h(i,j,k+1)
1782  enddo ; enddo
1783  else
1784  do k=1,nz ; do i=is,ie
1785  u_h(i,j,k) = a_e(i)*u(i,j,k) + a_w(i)*u(i-1,j,k)
1786  v_h(i,j,k) = a_n(i)*v(i,j,k) + a_s(i)*v(i,j-1,k)
1787  enddo ; enddo
1788  endif
1789  enddo
1790 
1791  call cpu_clock_end(id_clock_uv_at_h)
1792 end subroutine find_uv_at_h

1794 subroutine applyboundaryfluxes(CS, G, dt, fluxes, optics, ea, h, tv)
