23 !* *
24 !* By Robert Hallberg, 2002 *
25 !* *
26 !* This file contains an example of the code that is needed to set *
27 !* up and use a set (in this case eleven) of dynamically passive *
28 !* tracers. These tracers dye the inflowing water or water initially *
29 !* within a range of latitudes or water initially in a range of *
30 !* depths. *
31 !* *
32 !* A single subroutine is called from within each file to register *
33 !* each of the tracers for reinitialization and advection and to *
34 !* register the subroutine that initializes the tracers and set up *
35 !* their output and the subroutine that does any tracer physics or *
36 !* chemistry along with diapycnal mixing (included here because some *
37 !* tracers may float or swim vertically or dye diapycnal processes). *
38 !* *
39 !* *
40 !* Macros written all in capital letters are defined in MOM_memory.h. *
41 !* *
42 !* A small fragment of the grid is shown below: *
43 !* *
44 !* j+1 x ^ x ^ x At x: q *
45 !* j+1 > o > o > At ^: v *
46 !* j x ^ x ^ x At >: u *
47 !* j > o > o > At o: h, tr *
48 !* j-1 x ^ x ^ x *
49 !* i-1 i i+1 At x & ^: *
50 !* i i+1 At > & o: *
51 !* *
52 !* The boundaries always run through q grid points (x). *
53 !* *
54 !********+*********+*********+*********+*********+*********+*********+**

123 function register_dome_tracer(G, param_file, CS, diag, tr_Reg, &
124  restart_cs)

126  type(param_file_type), intent(in) :: param_file

128  type(diag_ctrl), target, intent(in) :: diag

132 ! to be used with MOM.
133 ! Arguments: G - The ocean's grid structure.
134 ! (in) param_file - A structure indicating the open file to parse for

136 ! (in/out) CS - A pointer that is set to point to the control structure

139 ! (in/out) tr_Reg - A pointer to the tracer registry.

142 ! This include declares and sets the variable "version".
143 #include "version_variable.h"
144  character(len=40) :: mod = "DOME_tracer" ! This module's name.
145  character(len=200) :: inputdir
146  real, pointer :: tr_ptr(:,:,:) => null()
147  logical :: register_dome_tracer

149  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed ; nz = g%ke
150

152  call mom_error(warning, "DOME_register_tracer called with an "// &
153  "associated control structure.")
154  return
155  endif

157 
158  cs%diag => diag

160  call log_version(param_file, mod, version, "")

162  "The name of a file from which to read the initial \n"//&
163  "conditions for the DOME tracers, or blank to initialize \n"//&

191  tr_ptr => cs%tr(:,:,:,m)
192  ! Register the tracer for the restart file.
193  call register_restart_field(tr_ptr, cs%tr_desc(m),.true.,restart_cs)
194  ! Register the tracer for horizontal advection & diffusion.
195  call register_tracer(tr_ptr, cs%tr_desc(m)%name, param_file, tr_reg)

197  ! Set coupled_tracers to be true (hard-coded above) to provide the surface

200  if (cs%coupled_tracers) &
201  cs%ind_tr(m) = aof_set_coupler_flux(trim(cs%tr_desc(m)%name)//'_flux', &
202  flux_type=' ', implementation=' ', caller="register_DOME_tracer")

205  cs%tr_Reg => tr_reg
206  register_dome_tracer = .true.

209 subroutine initialize_dome_tracer(restart, day, G, h, OBC, CS, sponge_CSp, &
210  diag_to_z_csp)
211  logical, intent(in) :: restart
212  type(time_type), target, intent(in) :: day
213  type(ocean_grid_type), intent(in) :: g
214  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h
215  type(ocean_obc_type), pointer :: obc
216  type(dome_tracer_cs), pointer :: cs
217  type(sponge_cs), pointer :: sponge_csp
218  type(diag_to_z_cs), pointer :: diag_to_z_csp
219 ! This subroutine initializes the NTR tracer fields in tr(:,:,:,:)
220 ! and it sets up the tracer output.
221 
222 ! Arguments: restart - .true. if the fields have already been read from
223 ! a restart file.
224 ! (in) day - Time of the start of the run.
225 ! (in) G - The ocean's grid structure.
226 ! (in) h - Layer thickness, in m or kg m-2.
227 ! (in) OBC - This open boundary condition type specifies whether, where,
228 ! and what open boundary conditions are used.
229 ! (in/out) CS - The control structure returned by a previous call to
230 ! DOME_register_tracer.
231 ! (in/out) sponge_CSp - A pointer to the control structure for the sponges, if
232 ! they are in use. Otherwise this may be unassociated.
233 ! (in/out) diag_to_Z_Csp - A pointer to the control structure for diagnostics
234 ! in depth space.
235  real, allocatable :: temp(:,:,:)
236  real, pointer, dimension(:,:,:) :: &
237  obc_tr1_u => null(), & ! These arrays should be allocated and set to
238  obc_tr1_v => null() ! specify the values of tracer 1 that should come
239  ! in through u- and v- points through the open
240  ! boundary conditions, in the same units as tr.
241  character(len=16) :: name ! A variable's name in a NetCDF file.
242  character(len=72) :: longname ! The long name of that variable.
243  character(len=48) :: units ! The dimensions of the variable.
244  character(len=48) :: flux_units ! The units for tracer fluxes, usually
245  ! kg(tracer) kg(water)-1 m3 s-1 or kg(tracer) s-1.
246  real, pointer :: tr_ptr(:,:,:) => null()
247  real :: pi ! 3.1415926... calculated as 4*atan(1)
248  real :: tr_y ! Initial zonally uniform tracer concentrations.
249  real :: dist2 ! The distance squared from a line, in m2.
250  real :: h_neglect ! A thickness that is so small it is usually lost
251  ! in roundoff and can be neglected, in m.
252  real :: e(szk_(g)+1), e_top, e_bot, d_tr
253  integer :: i, j, k, is, ie, js, je, isd, ied, jsd, jed, nz, m
254  integer :: isdb, iedb, jsdb, jedb
255 
256  if (.not.associated(cs)) return
257  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
258  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
259  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
260  h_neglect = g%H_subroundoff
261 
262  cs%Time => day
263 
264  if (.not.restart) then
265  if (len_trim(cs%tracer_IC_file) >= 1) then
266  ! Read the tracer concentrations from a netcdf file.
267  if (.not.file_exists(cs%tracer_IC_file, g%Domain)) &
268  call mom_error(fatal, "DOME_initialize_tracer: Unable to open "// &
269  cs%tracer_IC_file)
270  do m=1,ntr
271  call read_data(cs%tracer_IC_file, trim(cs%tr_desc(m)%name), &
272  cs%tr(:,:,:,m), domain=g%Domain%mpp_domain)
273  enddo
274  else
275  do m=1,ntr
276  do k=1,nz ; do j=js,je ; do i=is,ie
277  cs%tr(i,j,k,m) = 1.0e-20 ! This could just as well be 0.
278  enddo ; enddo ; enddo
279  enddo
280 
281 ! This sets a stripe of tracer across the basin.
282  do m=2,ntr ; do j=js,je ; do i=is,ie
283  tr_y = 0.0
284  if ((m <= 6) .and. (g%geoLatT(i,j) > (300.0+50.0*real(m-1))) .and. &
285  (g%geolatt(i,j) < (350.0+50.0*real(m-1)))) tr_y = 1.0
286  do k=1,nz
287 ! This adds the stripes of tracer to every layer.
288  cs%tr(i,j,k,m) = cs%tr(i,j,k,m) + tr_y
289  enddo
290  enddo; enddo; enddo
291 
292  if (ntr > 7) then
293  do j=js,je ; do i=is,ie
294  e(nz+1) = -g%bathyT(i,j)
295  do k=nz,1,-1
296  e(k) = e(k+1) + h(i,j,k)*g%H_to_m
297  do m=7,ntr
298  e_top = -600.0*real(m-1) + 3000.0
299  e_bot = -600.0*real(m-1) + 2700.0
300  if (e_top < e(k)) then
301  if (e_top < e(k+1)) then ; d_tr = 0.0
302  elseif (e_bot < e(k+1)) then
303  d_tr = (e_top-e(k+1)) / ((h(i,j,k)+h_neglect)*g%H_to_m)
304  else ; d_tr = (e_top-e_bot) / ((h(i,j,k)+h_neglect)*g%H_to_m)
305  endif
306  elseif (e_bot < e(k)) then
307  if (e_bot < e(k+1)) then ; d_tr = 1.0
308  else ; d_tr = (e(k)-e_bot) / ((h(i,j,k)+h_neglect)*g%H_to_m)
309  endif
310  else
311  d_tr = 0.0
312  endif
313  if (h(i,j,k) < 2.0*g%Angstrom) d_tr=0.0
314  cs%tr(i,j,k,m) = cs%tr(i,j,k,m) + d_tr
315  enddo
316  enddo
317  enddo ; enddo
318  endif
319 
320  endif
321  endif ! restart
322 
323  if ( cs%use_sponge ) then
324 ! If sponges are used, this example damps tracers in sponges in the
325 ! northern half of the domain to 1 and tracers in the southern half
326 ! to 0. For any tracers that are not damped in the sponge, the call
327 ! to set_up_sponge_field can simply be omitted.
328  if (.not.associated(sponge_csp)) &
329  call mom_error(fatal, "DOME_initialize_tracer: "// &
330  "The pointer to sponge_CSp must be associated if SPONGE is defined.")
331 
332  allocate(temp(g%isd:g%ied,g%jsd:g%jed,nz))
333  do k=1,nz ; do j=js,je ; do i=is,ie
334  if (g%geoLatT(i,j) > 700.0 .and. (k > nz/2)) then
335  temp(i,j,k) = 1.0
336  else
337  temp(i,j,k) = 0.0
338  endif
339  enddo ; enddo ; enddo
340 
341 ! do m=1,NTR
342  do m=1,1
343  ! This is needed to force the compiler not to do a copy in the sponge
344  ! calls. Curses on the designers and implementers of Fortran90.
345  tr_ptr => cs%tr(:,:,:,m)
346  call set_up_sponge_field(temp,tr_ptr,nz,sponge_csp)
347  enddo
348  deallocate(temp)
349  endif
350 
351  if (associated(obc)) then
352  if (obc%apply_OBC_v) then
353  allocate(obc_tr1_v(g%isd:g%ied,g%jsd:g%jed,nz))
354  do k=1,nz ; do j=g%jsd,g%jed ; do i=g%isd,g%ied
355  if (k < nz/2) then ; obc_tr1_v(i,j,k) = 0.0
356  else ; obc_tr1_v(i,j,k) = 1.0 ; endif
357  enddo ; enddo ; enddo
358  call add_tracer_obc_values(trim(cs%tr_desc(1)%name), cs%tr_Reg, &
359  0.0, obc_in_v=obc_tr1_v)
360  else
361  ! This is not expected in the DOME example.
362  call add_tracer_obc_values(trim(cs%tr_desc(1)%name), cs%tr_Reg, 0.0)
363  endif
364  ! All tracers but the first have 0 concentration in their inflows. As this
365  ! is the default value, the following calls are unnecessary.
366  do m=2,ntr
367  call add_tracer_obc_values(trim(cs%tr_desc(m)%name), cs%tr_Reg, 0.0)
368  enddo
369  endif
370 
371  ! This needs to be changed if the units of tracer are changed above.
372  if (g%Boussinesq) then ; flux_units = "kg kg-1 m3 s-1"
373  else ; flux_units = "kg s-1" ; endif
374 
375  do m=1,ntr
376  ! Register the tracer for the restart file.
377  name = cs%tr_desc(m)%name ; longname = cs%tr_desc(m)%longname
378  units = cs%tr_desc(m)%units
379  cs%id_tracer(m) = register_diag_field("ocean_model", trim(name), cs%diag%axesTL, &
380  day, trim(longname) , trim(units))
381  cs%id_tr_adx(m) = register_diag_field("ocean_model", trim(name)//"_adx", &
382  cs%diag%axesCuL, day, trim(longname)//" advective zonal flux" , &
383  trim(flux_units))
384  cs%id_tr_ady(m) = register_diag_field("ocean_model", trim(name)//"_ady", &
385  cs%diag%axesCvL, day, trim(longname)//" advective meridional flux" , &
386  trim(flux_units))
387  cs%id_tr_dfx(m) = register_diag_field("ocean_model", trim(name)//"_dfx", &
388  cs%diag%axesCuL, day, trim(longname)//" diffusive zonal flux" , &
389  trim(flux_units))
390  cs%id_tr_dfy(m) = register_diag_field("ocean_model", trim(name)//"_dfy", &
391  cs%diag%axesCvL, day, trim(longname)//" diffusive zonal flux" , &
392  trim(flux_units))
393  if (cs%id_tr_adx(m) > 0) call safe_alloc_ptr(cs%tr_adx(m)%p,isdb,iedb,jsd,jed,nz)
394  if (cs%id_tr_ady(m) > 0) call safe_alloc_ptr(cs%tr_ady(m)%p,isd,ied,jsdb,jedb,nz)
395  if (cs%id_tr_dfx(m) > 0) call safe_alloc_ptr(cs%tr_dfx(m)%p,isdb,iedb,jsd,jed,nz)
396  if (cs%id_tr_dfy(m) > 0) call safe_alloc_ptr(cs%tr_dfy(m)%p,isd,ied,jsdb,jedb,nz)
397 
398 ! Register the tracer for horizontal advection & diffusion.
399  if ((cs%id_tr_adx(m) > 0) .or. (cs%id_tr_ady(m) > 0) .or. &
400  (cs%id_tr_dfx(m) > 0) .or. (cs%id_tr_dfy(m) > 0)) &
401  call add_tracer_diagnostics(name, cs%tr_Reg, cs%tr_adx(m)%p, &
402  cs%tr_ady(m)%p, cs%tr_dfx(m)%p, cs%tr_dfy(m)%p)
403 
404  call register_z_tracer(cs%tr(:,:,:,m), trim(name)//"_z", longname, units, &
405  day, g, diag_to_z_csp)
406  enddo
407 
408 end subroutine initialize_dome_tracer

410 subroutine dome_tracer_column_physics(h_old, h_new, ea, eb, fluxes, dt, G, CS)
411  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h_old, h_new, ea, eb
412  type(forcing), intent(in) :: fluxes
413  real, intent(in) :: dt
414  type(ocean_grid_type), intent(in) :: g
415  type(dome_tracer_cs), pointer :: cs
416 ! This subroutine applies diapycnal diffusion and any other column
417 ! tracer physics or chemistry to the tracers from this file.
418 ! This is a simple example of a set of advected passive tracers.
419 
420 ! Arguments: h_old - Layer thickness before entrainment, in m or kg m-2.
421 ! (in) h_new - Layer thickness after entrainment, in m or kg m-2.
422 ! (in) ea - an array to which the amount of fluid entrained
423 ! from the layer above during this call will be
424 ! added, in m or kg m-2.
425 ! (in) eb - an array to which the amount of fluid entrained
426 ! from the layer below during this call will be
427 ! added, in m or kg m-2.
428 ! (in) fluxes - A structure containing pointers to any possible
429 ! forcing fields. Unused fields have NULL ptrs.
430 ! (in) dt - The amount of time covered by this call, in s.
431 ! (in) G - The ocean's grid structure.
432 ! (in) CS - The control structure returned by a previous call to
433 ! DOME_register_tracer.
434 !
435 ! The arguments to this subroutine are redundant in that
436 ! h_new[k] = h_old[k] + ea[k] - eb[k-1] + eb[k] - ea[k+1]
437 
438  real :: b1(szi_(g)) ! b1 and c1 are variables used by the
439  real :: c1(szi_(g),szk_(g)) ! tridiagonal solver.
440  integer :: i, j, k, is, ie, js, je, nz, m
441  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
442 
443  if (.not.associated(cs)) return
444 
445  do m=1,ntr
446  call tracer_vertdiff(h_old, ea, eb, dt, cs%tr(:,:,:,m), g)
447  enddo
