177 subroutine mom_grid_init(G, param_file)
178  type(ocean_grid_type), intent(inout) :: g
179  type(param_file_type), intent(in) :: param_file
180 ! Arguments: G - The ocean's grid structure.
181 ! (in) param_file - A structure indicating the open file to parse for
182 ! model parameter values.
183 ! This include declares and sets the variable "version".
184 #include "version_variable.h"
185  integer :: isd, ied, jsd, jed, nk, idg_off, jdg_off
186  integer :: isdb, iedb, jsdb, jedb
187 
188  ! get_domain_extent ensures that domains start at 1 for compatibility between
189  ! static and dynamically allocated arrays.
190  call get_domain_extent(g%Domain, g%isc, g%iec, g%jsc, g%jec, &
191  g%isd, g%ied, g%jsd, g%jed, &
192  g%isg, g%ieg, g%jsg, g%jeg, &
193  idg_off, jdg_off, g%symmetric)
194  g%isd_global = g%isd+idg_off ; g%jsd_global = g%jsd+jdg_off
195 
196  ! Read all relevant parameters and write them to the model log.
197  call log_version(param_file, "MOM_grid", version, &
198  "Parameters providing information about the vertical grid.")
199  call get_param(param_file, "MOM", "G_EARTH", g%g_Earth, &
200  "The gravitational acceleration of the Earth.", &
201  units="m s-2", default = 9.80)
202  call get_param(param_file, "MOM", "RHO_0", g%Rho0, &
203  "The mean ocean density used with BOUSSINESQ true to \n"//&
204  "calculate accelerations and the mass for conservation \n"//&
205  "properties, or with BOUSSINSEQ false to convert some \n"//&
206  "parameters from vertical units of m to kg m-2.", &
207  units="kg m-3", default=1035.0)
208  call get_param(param_file, "MOM_grid", "FIRST_DIRECTION", g%first_direction, &
209  "An integer that indicates which direction goes first \n"//&
210  "in parts of the code that use directionally split \n"//&
211  "updates, with even numbers (or 0) used for x- first \n"//&
212  "and odd numbers used for y-first.", default=0)
213  call get_param(param_file, "MOM_grid", "BOUSSINESQ", g%Boussinesq, &
214  "If true, make the Boussinesq approximation.", default=.true.)
215  call get_param(param_file, "MOM_grid", "ANGSTROM", g%Angstrom_z, &
216  "The minumum layer thickness, usually one-Angstrom.", &
217  units="m", default=1.0e-10)
218  if (.not.g%Boussinesq) &
219  call get_param(param_file, "MOM_grid", "H_TO_KG_M2", g%H_to_kg_m2,&
220  "A constant that translates thicknesses from the model's \n"//&
221  "internal units of thickness to kg m-2.", units="kg m-2 H-1", &
222  default=1.0)
223  call get_param(param_file, "MOM_grid", "BATHYMETRY_AT_VEL", g%bathymetry_at_vel, &
224  "If true, there are separate values for the basin depths \n"//&
225  "at velocity points. Otherwise the effects of of \n"//&
226  "topography are entirely determined from thickness points.", &
227  default=.false.)
228 #ifdef STATIC_MEMORY_
229  ! Here NK_ is a macro, while nk is a variable.
230  call get_param(param_file, "MOM_grid", "NK", nk, &
231  "The number of model layers.", units="nondim", &
232  static_value=nk_)
233  if (nk /= nk_) call mom_error(fatal, "MOM_grid_init: " // &
234  "Mismatched number of layers NK_ between MOM_memory.h and param_file")
235 
236 #else
237  call get_param(param_file, "MOM_grid", "NK", nk, &
238  "The number of model layers.", units="nondim", fail_if_missing=.true.)
239 #endif
240 
241  g%ks = 1 ; g%ke = nk
242 
243  g%nonblocking_updates = g%Domain%nonblocking_updates
244 
245  g%IscB = g%isc ; g%JscB = g%jsc
246  g%IsdB = g%isd ; g%JsdB = g%jsd
247  g%IsgB = g%isg ; g%JsgB = g%jsg
248  if (g%symmetric) then
249  g%IscB = g%isc-1 ; g%JscB = g%jsc-1
250  g%IsdB = g%isd-1 ; g%JsdB = g%jsd-1
251  g%IsgB = g%isg-1 ; g%JsgB = g%jsg-1
252  endif
253  g%IecB = g%iec ; g%JecB = g%jec
254  g%IedB = g%ied ; g%JedB = g%jed
255  g%IegB = g%ieg ; g%JegB = g%jeg
256 
257  call mom_mesg(" MOM_grid.F90, MOM_grid_init: allocating metrics", 5)
258 
259  call allocate_metrics(g)
260 
261  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
262  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
263  alloc_(g%bathyT(isd:ied, jsd:jed)) ; g%bathyT(:,:) = g%Angstrom_z
264  alloc_(g%g_prime(nk+1)) ; g%g_prime(:) = 0.0
265  alloc_(g%Rlay(nk+1)) ; g%Rlay(:) = 0.0
266 
267  if (g%bathymetry_at_vel) then
268  alloc_(g%Dblock_u(isdb:iedb, jsd:jed)) ; g%Dblock_u(:,:) = 0.0
269  alloc_(g%Dopen_u(isdb:iedb, jsd:jed)) ; g%Dopen_u(:,:) = 0.0
270  alloc_(g%Dblock_v(isd:ied, jsdb:jedb)) ; g%Dblock_v(:,:) = 0.0
271  alloc_(g%Dopen_v(isd:ied, jsdb:jedb)) ; g%Dopen_v(:,:) = 0.0
272  endif
273 
274  if (g%Boussinesq) then
275  g%H_to_kg_m2 = g%Rho0
276  g%kg_m2_to_H = 1.0/g%Rho0
277  g%m_to_H = 1.0
278  g%H_to_m = 1.0
279  g%Angstrom = g%Angstrom_z
280  else
281  g%kg_m2_to_H = 1.0 / g%H_to_kg_m2
282  g%m_to_H = g%Rho0 * g%kg_m2_to_H
283  g%H_to_m = g%H_to_kg_m2 / g%Rho0
284  g%Angstrom = g%Angstrom_z*1000.0*g%kg_m2_to_H
285  endif
286  g%H_subroundoff = 1e-20 * max(g%Angstrom,g%m_to_H*1e-17)
287  g%H_to_Pa = g%g_Earth * g%H_to_kg_m2
288 
289  allocate(g%gridLatT(g%jsg:g%jeg))
290  allocate(g%gridLatB(g%JsgB:g%JegB))
291  g%gridLatT(:) = 0.0 ; g%gridLatB(:) = 0.0
292  allocate(g%gridLonT(g%isg:g%ieg))
293  allocate(g%gridLonB(g%IsgB:g%IegB))
294  g%gridLonT(:) = 0.0 ; g%gridLonB(:) = 0.0
295 
296 ! Log derivative values.
297  call log_param(param_file, "MOM_grid", "M to THICKNESS", g%m_to_H)
298 
299  call verticalgridinit( param_file, g%GV )
300 
301 end subroutine mom_grid_init

303 logical function ispointincell(G, i, j, x, y)
304 ! Returns true if the coordinates (x,y) are within the h-cell (i,j)
305  type(ocean_grid_type), intent(in) :: g
306  integer, intent(in) :: i, j
307  real, intent(in) :: x, y
308 ! This is a crude calculation that assume a geographic coordinate system
309 ispointincell = &
310  ( g%geoLonBu(i-1,j-1) <= x ) &
311  .and. ( g%geoLonBu(i-1,j) <= x ) &
312  .and. ( g%geoLonBu(i,j-1) >= x ) &
313  .and. ( g%geoLonBu(i,j) >= x ) &
314  .and. ( g%geoLatBu(i-1,j-1) <= y ) &
315  .and. ( g%geoLatBu(i,j-1) <= y ) &
316  .and. ( g%geoLatBu(i-1,j) >= y ) &
317  .and. ( g%geoLatBu(i,j) >= y )
318 end function ispointincell

361 function get_tr_flux_units(G, tr_units, tr_vol_conc_units,tr_mass_conc_units)
362  character(len=48) :: get_tr_flux_units
363  type(ocean_grid_type), intent(in) :: g
364  character(len=*), optional, intent(in) :: tr_units
365  character(len=*), optional, intent(in) :: tr_vol_conc_units
366  character(len=*), optional, intent(in) :: tr_mass_conc_units
367 ! This subroutine returns the appropriate units for thicknesses and fluxes,
368 ! depending on whether the model is Boussinesq or not and the scaling for
369 ! the vertical thickness.
370 
371 ! Arguments: G - The ocean's grid structure.
372 ! One of the following three arguments must be present.
373 ! (in,opt) tr_units - Units for a tracer, for example Celsius or PSU.
374 ! (in,opt) tr_vol_conc_units - The concentration units per unit volume, for
375 ! example if the units are umol m-3,
376 ! tr_vol_conc_units would be umol.
377 ! (in,opt) tr_mass_conc_units - The concentration units per unit mass of sea
378 ! water, for example if the units are mol kg-1,
379 ! tr_vol_conc_units would be mol.
380 ! (ret) get_tr_flux_units - The model's flux units for a tracer.
381  integer :: cnt
382 
383  cnt = 0
384  if (present(tr_units)) cnt = cnt+1
385  if (present(tr_vol_conc_units)) cnt = cnt+1
386  if (present(tr_mass_conc_units)) cnt = cnt+1
387 
388  if (cnt == 0) call mom_error(fatal, "get_tr_flux_units: One of the three "//&
389  "arguments tr_units, tr_vol_conc_units, or tr_mass_conc_units "//&
390  "must be present.")
391  if (cnt > 1) call mom_error(fatal, "get_tr_flux_units: Only one of "//&
392  "tr_units, tr_vol_conc_units, and tr_mass_conc_units may be present.")
393  if (present(tr_units)) then
394  if (g%Boussinesq) then
395  get_tr_flux_units = trim(tr_units)//" meter3 second-1"
396  else
397  get_tr_flux_units = trim(tr_units)//" kilogram second-1"
398  endif
399  endif
400  if (present(tr_vol_conc_units)) then
401  if (g%Boussinesq) then
402  get_tr_flux_units = trim(tr_vol_conc_units)//" second-1"
403  else
404  get_tr_flux_units = trim(tr_vol_conc_units)//" m-3 kg s-1"
405  endif
406  endif
407  if (present(tr_mass_conc_units)) then
408  if (g%Boussinesq) then
409  get_tr_flux_units = trim(tr_mass_conc_units)//" kg-1 m3 s-1"
410  else
411  get_tr_flux_units = trim(tr_mass_conc_units)//" second-1"
412  endif
413  endif
414 
415 end function get_tr_flux_units
