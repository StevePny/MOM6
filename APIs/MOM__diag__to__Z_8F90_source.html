27 !* This subroutine maps tracers and velocities into depth space *
28 !* for output as diagnostic quantities. Currently, a piecewise *
29 !* linear subgrid structure is used for tracers, while velocities can *
30 !* use either piecewise constant or piecewise linear structures. *
31 !* *
32 !* A small fragment of the grid is shown below: *
33 !* *
34 !* j+1 x ^ x ^ x At x: q, CoriolisBu *
35 !* j+1 > o > o > At ^: v *
36 !* j x ^ x ^ x At >: u *
37 !* j > o > o > At o: h, bathyT *
38 !* j-1 x ^ x ^ x *
39 !* i-1 i i+1 At x & ^: *
40 !* i i+1 At > & o: *
41 !* *
42 !* The boundaries always run through q grid points (x). *

425  integer :: i, j, k, is, ie, js, je, isq, ieq, jsq, jeq, nk, nk_z
426  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nk = g%ke
427  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
428 
429  if (.not.associated(cs)) call mom_error(fatal, &
430  "calculate_Z_transport: Module must be initialized before it is used.")
431  if ((cs%id_uh_Z <= 0) .and. (cs%id_vh_Z <= 0)) return
432 
433  idt = 1.0 ; if (dt > 0.0) idt = 1.0 / dt
434  nk_z = cs%nk_zspace
435 
436  if (nk_z <= 0) return
437 
438  ! Determine how much the layers will be dilated in recasting them into z*
439  ! coordiantes. (-G%D < z* < 0).
440  do j=jsq,jeq+1 ; do i=isq,ieq+1
441  htot(i,j) = g%H_subroundoff
442  enddo ; enddo
443  do k=1,nk ; do j=jsq,jeq+1 ; do i=isq,ieq+1
444  htot(i,j) = htot(i,j) + h(i,j,k)
445  enddo ; enddo ; enddo
446  do j=jsq,jeq+1 ; do i=isq,ieq+1
447  dilate(i,j) = g%bathyT(i,j) / htot(i,j)
448  enddo ; enddo
449 
450  if (cs%id_uh_Z > 0) then ; do j=js,je
451  do i=isq,ieq
452  kz(i) = nk_z ; z_int_above(i) = -0.5*(g%bathyT(i,j)+g%bathyT(i+1,j))
453  enddo
454  do k=nk_z,1,-1 ; do i=isq,ieq
455  uh_z(i,k) = 0.0
456  if (cs%Z_int(k) < z_int_above(i)) kz(i) = k-1
457  enddo ; enddo
458  do k=nk,1,-1 ; do i=isq,ieq
459  h_rem = 0.5*(dilate(i,j)*h(i,j,k) + dilate(i+1,j)*h(i+1,j,k))
460  uh_rem = uh_int(i,j,k)
461  z_int_above(i) = z_int_above(i) + h_rem
462 
463  do ! Distribute this layer's transport into the depth-levels.
464  h_above = z_int_above(i) - cs%Z_int(kz(i))
465  if ((kz(i) == 1) .or. (h_above <= 0.0) .or. (h_rem <= 0.0)) then
466  ! The entire remaining transport is on this level.
467  uh_z(i,kz(i)) = uh_z(i,kz(i)) + uh_rem ; exit
468  else
469  h_here = h_rem - h_above
470  uh_here = uh_rem * (h_here / h_rem)
471 
472  h_rem = h_rem - h_here ! = h_above
473  uh_z(i,kz(i)) = uh_z(i,kz(i)) + uh_here
474  uh_rem = uh_rem - uh_here
475  kz(i) = kz(i) - 1
476  endif
477  enddo ! End of loop through the target depth-space levels.
478  enddo ; enddo
479  do k=1,nk_z ; do i=isq,ieq
480  cs%uh_z(i,j,k) = uh_z(i,k)*idt
481  enddo ; enddo
482  enddo ; endif
483  if (cs%id_vh_Z > 0) then ; do j=jsq,jeq
484  do i=is,ie
485  kz(i) = nk_z ; z_int_above(i) = -0.5*(g%bathyT(i,j)+g%bathyT(i,j+1))
486  enddo
487  do k=nk_z,1,-1 ; do i=is,ie
488  vh_z(i,k) = 0.0
489  if (cs%Z_int(k) < z_int_above(i)) kz(i) = k-1
490  enddo ; enddo
491  do k=nk,1,-1 ; do i=is,ie
492  h_rem = 0.5*(dilate(i,j)*h(i,j,k) + dilate(i,j+1)*h(i,j+1,k))
493  vh_rem = vh_int(i,j,k)
494  z_int_above(i) = z_int_above(i) + h_rem
495 
496  do ! Distribute this layer's transport into the depth-levels.
497  h_above = z_int_above(i) - cs%Z_int(kz(i))
498  if ((kz(i) == 1) .or. (h_above <= 0.0) .or. (h_rem <= 0.0)) then
499  ! The entire remaining transport is on this level.
500  vh_z(i,kz(i)) = vh_z(i,kz(i)) + vh_rem ; exit
501  else
502  h_here = h_rem - h_above
503  vh_here = vh_rem * (h_here / h_rem)
504 
505  h_rem = h_rem - h_here ! = h_above
506  vh_z(i,kz(i)) = vh_z(i,kz(i)) + vh_here
507  vh_rem = vh_rem - vh_here
508  kz(i) = kz(i) - 1
509  endif
510  enddo ! End of loop through the target depth-space levels.
511  enddo ; enddo
512  do k=1,nk_z ; do i=is,ie
513  cs%vh_z(i,j,k) = vh_z(i,k)*idt
514  enddo ; enddo
515  enddo ; endif
516 
517  if (cs%id_uh_Z > 0) call post_data(cs%id_uh_Z, cs%uh_z, cs%diag)
518  if (cs%id_vh_Z > 0) call post_data(cs%id_vh_Z, cs%vh_z, cs%diag)
519 
520 end subroutine calculate_z_transport

522 subroutine find_overlap(e, Z_top, Z_bot, k_max, k_start, k_top, k_bot, wt, z1, z2)
523  real, dimension(:), intent(in) :: e
524  real, intent(in) :: z_top, z_bot
525  integer, intent(in) :: k_max, k_start
526  integer, intent(inout) :: k_top, k_bot
527  real, dimension(:), intent(out) :: wt, z1, z2
528 
529 ! This subroutine determines the layers bounded by interfaces e that overlap
530 ! with the depth range between Z_top and Z_bot, and also the fractional weights
531 ! of each layer. It also calculates the normalized relative depths of the range
532 ! of each layer that overlaps that depth range.
533 ! Note that by convention, e decreases with increasing k and Z_top > Z_bot.
534 !
535 ! Arguments: e - A column's interface heights, in m.
536 ! (in) Z_top - The top of the range being mapped to, in m.
537 ! (in) Z_bot - The bottom of the range being mapped to, in m.
538 ! (in) k_max - The number of valid layers.
539 ! (in) k_start - The layer at which to start searching.
540 ! (out) k_top, k_bot - The indices of the top and bottom layers that
541 ! overlap with the depth range.
542 ! (out) wt - The relative weights of each layer from k_top to k_bot.
543 ! (out) z1, z2 - z1 and z2 are the depths of the top and bottom limits of
544 ! the part of a layer that contributes to a depth level,
545 ! relative to the cell center and normalized by the cell
546 ! thickness, nondim. Note that -1/2 <= z1 < z2 <= 1/2.
547  real :: ih, e_c, tot_wt, i_totwt
548  integer :: k
549 
550  do k=k_start,k_max ; if (e(k+1)<z_top) exit ; enddo
551  k_top = k
552  if (k>k_max) return
553 
554  ! Determine the fractional weights of each layer.
555  ! Note that by convention, e and Z_int decrease with increasing k.
556  if (e(k+1)<=z_bot) then
557  wt(k) = 1.0 ; k_bot = k
558  ih = 1.0 / (e(k)-e(k+1))
559  e_c = 0.5*(e(k)+e(k+1))
560  z1(k) = (e_c - min(e(k),z_top)) * ih
561  z2(k) = (e_c - z_bot) * ih
562  else
563  wt(k) = min(e(k),z_top) - e(k+1) ; tot_wt = wt(k) ! These are always > 0.
564  z1(k) = (0.5*(e(k)+e(k+1)) - min(e(k),z_top)) / (e(k)-e(k+1))
565  z2(k) = 0.5
566  k_bot = k_max
567  do k=k_top+1,k_max
568  if (e(k+1)<=z_bot) then
569  k_bot = k
570  wt(k) = e(k) - z_bot ; z1(k) = -0.5
571  z2(k) = (0.5*(e(k)+e(k+1)) - z_bot) / (e(k)-e(k+1))
572  else
573  wt(k) = e(k) - e(k+1) ; z1(k) = -0.5 ; z2(k) = 0.5
574  endif
575  tot_wt = tot_wt + wt(k) ! wt(k) is always > 0.
576  if (k>=k_bot) exit
577  enddo
578 
579  i_totwt = 1.0 / tot_wt
580  do k=k_top,k_bot ; wt(k) = i_totwt*wt(k) ; enddo
581  endif
582 
583 end subroutine find_overlap

586 subroutine find_limited_slope(val, e, slope, k)
587  real, dimension(:), intent(in) :: val
588  real, dimension(:), intent(in) :: e
589  real, intent(out) :: slope
590  integer, intent(in) :: k
591 ! This subroutine determines a limited slope for val to be advected with
592 ! a piecewise limited scheme.
593 
594 ! Arguments: val - An column the values that are being interpolated.
595 ! (in) e - A column's interface heights, in m.
596 ! (in) slope - The normalized slope in the intracell distribution of val.
597 ! (in) k - The layer whose slope is being determined.
598  real :: d1, d2
599 
600  if ((val(k)-val(k-1)) * (val(k)-val(k+1)) >= 0.0) then
601  slope = 0.0 ! ; curvature = 0.0
602  else
603  d1 = 0.5*(e(k-1)-e(k+1)) ; d2 = 0.5*(e(k)-e(k+2))
604  slope = (d1**2*(val(k+1) - val(k)) + d2**2*(val(k) - val(k-1))) * &
605  ((e(k) - e(k+1)) / (d1*d2*(d1+d2)))
606  ! slope = 0.5*(val(k+1) - val(k-1))
607  ! This is S.J. Lin's form of the PLM limiter.
608  slope = sign(1.0,slope) * min(abs(slope), &
609  2.0*(max(val(k-1),val(k),val(k+1)) - val(k)), &
610  2.0*(val(k) - min(val(k-1),val(k),val(k+1))))
611  ! curvature = 0.0
612  endif
613 
614 end subroutine find_limited_slope

616 subroutine calc_zint_diags(h, in_ptrs, ids, num_diags, &
617  g, cs)
618  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h
619  type(p3d), dimension(:), intent(in) :: in_ptrs
620  integer, dimension(:), intent(in) :: ids
621  integer, intent(in) :: num_diags
622  type(ocean_grid_type), intent(in) :: g
623  type(diag_to_z_cs), pointer :: cs
624 
625  real, dimension(SZI_(G),SZJ_(G),max(CS%nk_zspace+1,1),max(num_diags,1)) :: &
626  diag_on_z ! The diagnostics interpolated to depth space.
627  real, dimension(SZI_(G),SZK_(G)+1) :: e
628  real, dimension(max(num_diags,1),SZI_(G),SZK_(G)+1) :: diag2d
629  real, dimension(SZI_(G)) :: &
630  htot, & ! The summed layer thicknesses in m or kg m-2.
631  dilate ! The proportion by which to dilate every layer.
632  real :: wt ! The weighting of the interface above in the
633  ! interpolation to the target depths.
634  integer :: kl(szi_(g)) ! The layer-space index of the shallowest interface
635  ! below the target depth.
636  integer :: i, j, k, k2, kz, is, ie, js, je, nk, m
637  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nk = g%ke
638 
639  if (num_diags < 1) return
640  if (.not.associated(cs)) call mom_error(fatal, &
641  "calc_Zint_diags: Module must be initialized before it is used.")
642 
643  do j=js,je
644  ! Calculate the stretched z* interface depths.
645  do i=is,ie ; htot(i) = 0.0 ; kl(i) = 1 ; enddo
646  do k=1,nk ; do i=is,ie ; htot(i) = htot(i) + h(i,j,k) ; enddo ; enddo
647  do i=is,ie
648  dilate(i) = 0.0
649  if (htot(i)*g%H_to_m > 0.5) dilate(i) = (g%bathyT(i,j) - 0.0) / htot(i)
650  e(i,nk+1) = -g%bathyT(i,j)
651  enddo
652  do k=nk,1,-1 ; do i=is,ie
653  e(i,k) = e(i,k+1) + h(i,j,k) * dilate(i)
654  enddo ; enddo
655  ! e(i,1) should be 0 as a consistency check.
656 
657  do k=1,nk+1 ; do i=is,ie ; do m=1,num_diags
658  diag2d(m,i,k) = in_ptrs(m)%p(i,j,k)
659  enddo ; enddo ; enddo
660 
661  do kz=1,cs%nk_zspace+1 ; do i=is,ie
662  ! Find the interface below the target Z-file depth, kL.
663  if (cs%Z_int(kz) < e(i,nk+1)) then
664  kl(i) = nk+2
665  else
666  do k=kl(i),nk+1 ; if (cs%Z_int(kz) > e(i,k)) exit ; enddo
667  kl(i) = k
668  endif
669  if (kl(i)>1) then
670  if (cs%Z_int(kz) > e(i,kl(i)-1)) call mom_error(fatal, &
671  "calc_Zint_diags: Interface depth mapping is incorrect.")
672  endif
673  if ((kl(i)>1) .and. (kl(i)<=nk+1)) then
674  if (e(i,kl(i)-1) == e(i,kl(i))) call mom_error(warning, &
675  "calc_Zint_diags: Interface depths equal.", all_print=.true.)
676  if (e(i,kl(i)-1) - e(i,kl(i)) < 0.0) call mom_error(fatal, &
677  "calc_Zint_diags: Interface depths inverted.")
678  endif
679 
680  if (kl(i) <= 1) then
681  do m=1,num_diags
682  diag_on_z(i,j,kz,m) = diag2d(m,i,1)
683  enddo
684  elseif (kl(i) > nk+1) then
685  do m=1,num_diags
686  diag_on_z(i,j,kz,m) = cs%missing_value
687  enddo
688  else
689  wt = 0.0 ! This probably should not happen?
690  if (e(i,kl(i)-1) - e(i,kl(i)) > 0.0) &
691  wt = (cs%Z_int(kz) - e(i,kl(i))) / (e(i,kl(i)-1) - e(i,kl(i)))
692  if ((wt < 0.0) .or. (wt > 1.0)) call mom_error(fatal, &
693  "calc_Zint_diags: Bad value of wt found.")
694  do m=1,num_diags
695  diag_on_z(i,j,kz,m) = wt * diag2d(m,i,kl(i)-1) + &
696  (1.0-wt) * diag2d(m,i,kl(i))
697  enddo
698  endif
699  enddo ; enddo
700 
701  enddo
702 
703  do m=1,num_diags
704  if (ids(m) > 0) call post_data(ids(m), diag_on_z(:,:,:,m), cs%diag)
705  enddo
706 
707 end subroutine calc_zint_diags
