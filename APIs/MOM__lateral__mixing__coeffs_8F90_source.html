98 subroutine calc_resoln_function(h, tv, G, CS)
99  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h
100  type(thermo_var_ptrs), intent(in) :: tv
101  type(ocean_grid_type), intent(inout) :: g
102  type(varmix_cs), pointer :: cs
103 ! This subroutine determines a function of the ratio of the grid
104 ! spacing to the deformation radius that is used to scale horizontal
105 ! viscosities or diffusivities.
106 ! In the current implementation, the function is F(R) = 1/(1+R) where
107 ! R = cg^2/(f^2*(dx^2+dy^2))
108 ! Arguments: h - Layer thickness, in m or kg m-2.
109 ! (in) tv - A structure containing the thermodynamic variables.
110 ! (in) G - The ocean's grid structure.
111 ! (in) CS - The control structure returned by a previous call to
112 ! VarMix_init.
113 
114  real :: cg1_q ! The gravity wave speed interpolated to q points, in m s-1.
115  real :: dx_term
116  integer :: mod_power_2, power_2
117  integer :: is, ie, js, je, isq, ieq, jsq, jeq, nz
118  integer :: i, j, k
119  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
120  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
121 
122  if (.not. ASSOCIATED(cs)) call mom_error(fatal, "calc_resoln_function:"// &
123  "Module must be initialized before it is used.")
124  if (.not. (cs%Resoln_scaled_Kh .or. cs%Resoln_scaled_KhTh .or. &
125  cs%Resoln_scaled_KhTr)) return
126  if (.not. ASSOCIATED(cs%cg1)) call mom_error(fatal, &
127  "calc_resoln_function: %cg1 is not associated with Resoln_scaled_Kh.")
128  if (.not. ASSOCIATED(cs%Res_fn_h)) call mom_error(fatal, &
129  "calc_resoln_function: %Res_fn_h is not associated with Resoln_scaled_Kh.")
130  if (.not. ASSOCIATED(cs%Res_fn_q)) call mom_error(fatal, &
131  "calc_resoln_function: %Res_fn_q is not associated with Resoln_scaled_Kh.")
132  if (.not. ASSOCIATED(cs%f2_dx2_h)) call mom_error(fatal, &
133  "calc_resoln_function: %f2_dx2_h is not associated with Resoln_scaled_Kh.")
134  if (.not. ASSOCIATED(cs%beta_dx2_h)) call mom_error(fatal, &
135  "calc_resoln_function: %beta_dx2_h is not associated with Resoln_scaled_Kh.")
136  if (.not. ASSOCIATED(cs%f2_dx2_q)) call mom_error(fatal, &
137  "calc_resoln_function: %f2_dx2_q is not associated with Resoln_scaled_Kh.")
138  if (.not. ASSOCIATED(cs%beta_dx2_q)) call mom_error(fatal, &
139  "calc_resoln_function: %beta_dx2_q is not associated with Resoln_scaled_Kh.")
140 
141  call wave_speed(h, tv, g, cs%cg1, cs%wave_speed_CSp)
142 
143  call pass_var(cs%cg1, g%Domain)
144 
145  ! Do this calculation on the extent used in MOM_hor_visc.F90, and
146  ! MOM_tracer.F90 so that no halo update is needed.
147  mod_power_2 = mod(cs%Res_fn_power, 2)
148  if (cs%Res_fn_power >= 100) then
149  do j=js-1,je+1 ; do i=is-1,ie+1
150  dx_term = cs%f2_dx2_h(i,j) + cs%cg1(i,j)*cs%beta_dx2_h(i,j)
151  if ((cs%Res_coef * cs%cg1(i,j))**2 > dx_term) then
152  cs%Res_fn_h(i,j) = 0.0
153  else
154  cs%Res_fn_h(i,j) = 1.0
155  endif
156  enddo ; enddo
157 
158  do j=js-1,jeq ; do i=is-1,ieq
159  cg1_q = 0.25 * ((cs%cg1(i,j) + cs%cg1(i+1,j+1)) + &
160  (cs%cg1(i+1,j) + cs%cg1(i,j+1)))
161  dx_term = cs%f2_dx2_q(i,j) + cg1_q * cs%beta_dx2_q(i,j)
162  if ((cs%Res_coef * cg1_q)**2 > dx_term) then
163  cs%Res_fn_q(i,j) = 0.0
164  else
165  cs%Res_fn_q(i,j) = 1.0
166  endif
167  enddo ; enddo
168  elseif (cs%Res_fn_power == 2) then
169  do j=js-1,je+1 ; do i=is-1,ie+1
170  dx_term = cs%f2_dx2_h(i,j) + cs%cg1(i,j)*cs%beta_dx2_h(i,j)
171  cs%Res_fn_h(i,j) = dx_term / (dx_term + (cs%Res_coef * cs%cg1(i,j))**2)
172  enddo ; enddo
173 
174  do j=js-1,jeq ; do i=is-1,ieq
175  cg1_q = 0.25 * ((cs%cg1(i,j) + cs%cg1(i+1,j+1)) + &
176  (cs%cg1(i+1,j) + cs%cg1(i,j+1)))
177  dx_term = cs%f2_dx2_q(i,j) + cg1_q * cs%beta_dx2_q(i,j)
178  cs%Res_fn_q(i,j) = dx_term / (dx_term + (cs%Res_coef * cg1_q)**2)
179  enddo ; enddo
180  elseif (mod_power_2 == 0) then
181  power_2 = cs%Res_fn_power / 2
182  do j=js-1,je+1 ; do i=is-1,ie+1
183  dx_term = (cs%f2_dx2_h(i,j) + cs%cg1(i,j)*cs%beta_dx2_h(i,j))**power_2
184  cs%Res_fn_h(i,j) = dx_term / &
185  (dx_term + (cs%Res_coef * cs%cg1(i,j))**cs%Res_fn_power)
186  enddo ; enddo
187 
188  do j=js-1,jeq ; do i=is-1,ieq
189  cg1_q = 0.25 * ((cs%cg1(i,j) + cs%cg1(i+1,j+1)) + &
190  (cs%cg1(i+1,j) + cs%cg1(i,j+1)))
191  dx_term = (cs%f2_dx2_q(i,j) + cg1_q * cs%beta_dx2_q(i,j))**power_2
192  cs%Res_fn_q(i,j) = dx_term / &
193  (dx_term + (cs%Res_coef * cg1_q)**cs%Res_fn_power)
194  enddo ; enddo
195  else
196  do j=js-1,je+1 ; do i=is-1,ie+1
197  dx_term = (sqrt(cs%f2_dx2_h(i,j) + &
198  cs%cg1(i,j)*cs%beta_dx2_h(i,j)))**cs%Res_fn_power
199  cs%Res_fn_h(i,j) = dx_term / &
200  (dx_term + (cs%Res_coef * cs%cg1(i,j))**cs%Res_fn_power)
201  enddo ; enddo
202 
203 
204  do j=js-1,jeq ; do i=is-1,ieq
205  cg1_q = 0.25 * ((cs%cg1(i,j) + cs%cg1(i+1,j+1)) + &
206  (cs%cg1(i+1,j) + cs%cg1(i,j+1)))
207  dx_term = (sqrt(cs%f2_dx2_q(i,j) + &
208  cg1_q * cs%beta_dx2_q(i,j)))**cs%Res_fn_power
209  cs%Res_fn_q(i,j) = dx_term / &
210  (dx_term + (cs%Res_coef * cg1_q)**cs%Res_fn_power)
211  enddo ; enddo
212  endif
213 
214  ! Calculate and store the ratio between deformation radius and grid-spacing
215  ! at h-points (non-dimensional).
216  do j=js-1,je+1 ; do i=is-1,ie+1
217  cs%Rd_dx_h(i,j) = cs%cg1(i,j) / &
218  (sqrt(cs%f2_dx2_h(i,j) + cs%cg1(i,j)*cs%beta_dx2_h(i,j)))
219  enddo ; enddo
220 
221  if (query_averaging_enabled(cs%diag)) then
222  if (cs%id_Res_fn > 0) call post_data(cs%id_Res_fn, cs%Res_fn_h, cs%diag)
223  if (cs%id_Rd_dx > 0) call post_data(cs%id_Rd_dx, cs%Rd_dx_h, cs%diag)
224  endif
225 
226 end subroutine calc_resoln_function

228 subroutine calc_slope_function_need_e(h, tv, G, CS)

230  type(thermo_var_ptrs), intent(in) :: tv

245 subroutine calc_slope_function_(h, tv, G, CS, e)
246  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(inout) :: h
247  type(thermo_var_ptrs), intent(in) :: tv
248  type(ocean_grid_type), intent(inout) :: g
249  type(varmix_cs), pointer :: cs
250  real, dimension(NIMEM_,NJMEM_,NK_INTERFACE_), intent(in) :: e
251 ! This subroutine calculates the slope function S*N for the Visbeck et
252 ! al. style scaling for the various horizontal diffusivities.
253 
254 ! Arguments: h - Layer thickness, in m or kg m-2.
255 ! (in) tv - A structure containing pointers to any available
256 ! thermodynamic fields. Absent fields have NULL ptrs.
257 ! (in) G - The ocean's grid structure.
258 ! (inout) CS - The control structure returned by a previous call to
259 ! thickness_diffuse_init.
260 ! (in) e - The interface heights, in m.
261  real :: e_x(szib_(g), szj_(g)) ! X-slope of interface at u points (for diagnostics)
262  real :: e_y(szi_(g), szjb_(g)) ! Y-slope of interface at u points (for diagnostics)
263  real :: khth_loc ! Locally calculated thickness mixing coefficient (m2/s)
264  real :: h_cutoff ! Local estimate of a minimum thickness for masking (m)
265  real :: h_neglect ! A thickness that is so small it is usually lost
266  ! in roundoff and can be neglected, in H.
267  real :: s2 ! Interface slope squared (non-dim)
268  real :: n2 ! Brunt-Vaisala frequency (1/s)
269  real :: hup, hdn ! Thickness from above, below (m or kg m-2)
270  real :: h_geom ! The geometric mean of Hup*Hdn, in m or kg m-2.
271  real :: one_meter ! One meter in thickness units of m or kg m-2.
272  integer :: is, ie, js, je, nz
273  integer :: i, j, k, kb_max
274 
275  if (.not. ASSOCIATED(cs)) call mom_error(fatal, "calc_slope_function:"// &
276  "Module must be initialized before it is used.")
277  if (.not. cs%use_variable_mixing) return
278  if (.not. ASSOCIATED(cs%SN_u)) call mom_error(fatal, "calc_slope_function:"// &
279  "%SN_u is not associated with use_variable_mixing.")
280  if (.not. ASSOCIATED(cs%SN_v)) call mom_error(fatal, "calc_slope_function:"// &
281  "%SN_v is not associated with use_variable_mixing.")
282 
283  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
284 
285  one_meter = 1.0 * g%m_to_H
286  h_neglect = g%H_subroundoff
287  h_cutoff = real(2*nz) * (g%angstrom + h_neglect)
288 
289  do j=js-1,je+1 ; do i=is-1,ie+1
290  cs%SN_u(i,j) = 0.0
291  cs%SN_v(i,j) = 0.0
292  enddo ; enddo
293 
294  ! To set the length scale based on the deformation radius, use wave_speed to
295  ! calculate the first-mode gravity wave speed and then blend the equatorial
296  ! and midlatitude deformation radii, using calc_resoln_function as a template.
297 
298  ! Set the length scale at u-points.
299  do j=js,je ; do i=is-1,ie
300  cs%L2u(i,j) = cs%Visbeck_L_scale**2
301  enddo ; enddo
302  ! Set length scale at v-points
303  do j=js-1,je ; do i=is,ie
304  cs%L2v(i,j) = cs%Visbeck_L_scale**2
305  enddo ; enddo
306 
307  do k=nz,cs%VarMix_Ktop,-1
308 
309  ! Calculate the interface slopes E_x and E_y and u- and v- points respectively
310  do j=js-1,je+1 ; do i=is-1,ie
311  e_x(i,j) = (e(i+1,j,k)-e(i,j,k))*g%IdxCu(i,j)
312  ! Mask slopes where interface intersects topography
313  if (min(h(i,j,k),h(i+1,j,k)) < h_cutoff) e_x(i,j) = 0.
314  enddo ; enddo
315  do j=js-1,je ; do i=is-1,ie+1
316  e_y(i,j) = (e(i,j+1,k)-e(i,j,k))*g%IdyCv(i,j)
317  ! Mask slopes where interface intersects topography
318  if (min(h(i,j,k),h(i,j+1,k)) < h_cutoff) e_y(i,j) = 0.
319  enddo ; enddo
320 
321  ! Calculate N*S*h from this layer and add to the sum
322  do j=js,je ; do i=is-1,ie
323  s2 = ( e_x(i,j)**2 + 0.25*( &
324  (e_y(i,j)**2+e_y(i+1,j-1)**2)+(e_y(i+1,j)**2+e_y(i,j-1)**2) ) )
325  hdn = 2.*h(i,j,k)*h(i,j,k-1) / (h(i,j,k) + h(i,j,k-1) + h_neglect)
326  hup = 2.*h(i+1,j,k)*h(i+1,j,k-1) / (h(i+1,j,k) + h(i+1,j,k-1) + h_neglect)
327  h_geom = sqrt(hdn*hup)
328  n2 = g%g_prime(k) / (g%H_to_m * max(hdn,hup,one_meter))
329  if (min(h(i,j,k-1), h(i+1,j,k-1), h(i,j,k), h(i+1,j,k)) < h_cutoff) &
330  s2 = 0.0
331  cs%SN_u(i,j) = cs%SN_u(i,j) + (h_geom * g%H_to_m) * s2 * n2
332  enddo ; enddo
333  do j=js-1,je ; do i=is,ie
334  s2 = ( e_y(i,j)**2 + 0.25*( &
335  (e_x(i,j)**2+e_x(i-1,j+1)**2)+(e_x(i,j+1)**2+e_x(i-1,j)**2) ) )
336  hdn = 2.*h(i,j,k)*h(i,j,k-1) / (h(i,j,k) + h(i,j,k-1) + h_neglect)
337  hup = 2.*h(i,j+1,k)*h(i,j+1,k-1) / (h(i,j+1,k) + h(i,j+1,k-1) + h_neglect)
338  h_geom = sqrt(hdn*hup)
339  n2 = g%g_prime(k) / (g%H_to_m * max(hdn,hup,one_meter))
340  if (min(h(i,j,k-1), h(i,j+1,k-1), h(i,j,k), h(i,j+1,k)) < h_cutoff) &
341  s2 = 0.0
342  cs%SN_v(i,j) = cs%SN_v(i,j) + (h_geom * g%H_to_m) * s2 * n2
343  enddo ; enddo
344 
345  enddo ! k
346 
347  ! SN above contains S^2*N^2*H, convert to vertical average of S*N
348  do j=js,je ; do i=is-1,ie
349  !SN_u(I,j) = sqrt( SN_u(I,j) / ( max(G%bathyT(I,j), G%bathyT(I+1,j)) + G%Angstrom ) )
350  !The code below behaves better than the line above. Not sure why? AJA
351  if ( min(g%bathyT(i,j), g%bathyT(i+1,j)) > h_cutoff ) then
352  cs%SN_u(i,j) = sqrt( cs%SN_u(i,j) / max(g%bathyT(i,j), g%bathyT(i+1,j)) )
353  else
354  cs%SN_u(i,j) = 0.0
355  endif
356  enddo ; enddo
357  do j=js-1,je ; do i=is,ie
358  !SN_v(i,J) = sqrt( SN_v(i,J) / ( max(G%bathyT(i,J), G%bathyT(i,J+1)) + G%Angstrom ) )
359  !The code below behaves better than the line above. Not sure why? AJA
360  if ( min(g%bathyT(i,j), g%bathyT(i+1,j)) > h_cutoff ) then
361  cs%SN_v(i,j) = sqrt( cs%SN_v(i,j) / max(g%bathyT(i,j), g%bathyT(i,j+1)) )
362  else
363  cs%SN_v(i,j) = 0.0
364  endif
365  enddo ; enddo
366 
367 ! Offer diagnostic fields for averaging.
368  if (query_averaging_enabled(cs%diag)) then
369  if (cs%id_SN_u > 0) call post_data(cs%id_SN_u, cs%SN_u, cs%diag)
370  if (cs%id_SN_v > 0) call post_data(cs%id_SN_v, cs%SN_v, cs%diag)
371  if (cs%id_L2u > 0) call post_data(cs%id_L2u, cs%L2u, cs%diag)
372  if (cs%id_L2v > 0) call post_data(cs%id_L2v, cs%L2v, cs%diag)
373  endif
374 
375 end subroutine calc_slope_function_

383 ! Arguments: Time - The current model time.
384 ! (in) G - The ocean's grid structure.
385 ! (in) param_file - A structure indicating the open file to parse for
386 ! model parameter values.
387 ! (in) diag - A structure that is used to regulate diagnostic output.
388 ! (in/out) CS - A pointer that is set to point to the control structure
389 ! for this module
390  real :: khtr_slope_cff, khth_slope_cff
391  real, parameter :: absurdly_small_freq2 = 1e-34 ! A miniscule frequency
