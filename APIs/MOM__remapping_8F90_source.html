97 ! General remapping routine
98 !------------------------------------------------------------------------------
99 subroutine remapping_main( CS, G, h, dxInterface, tv, u, v )
100 !------------------------------------------------------------------------------
101 ! This routine takes care of remapping all variable between the old and the
102 ! new grids. When velocity components need to be remapped, thicknesses at
103 ! velocity points are taken to be arithmetic averages of tracer thicknesses.
104 !------------------------------------------------------------------------------
105 
106  ! Arguments
107  type(remapping_cs), intent(inout) :: cs
108  type(ocean_grid_type), intent(in) :: g
109  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h
110  real, dimension(NIMEM_,NJMEM_,NK_INTERFACE_), intent(in) :: dxinterface
111  type(thermo_var_ptrs), intent(inout) :: tv
112  real, dimension(NIMEMB_,NJMEM_,NKMEM_), optional, intent(inout) :: u
113  real, dimension(NIMEM_,NJMEMB_,NKMEM_), optional, intent(inout) :: v
114 
115  ! Local variables
116  integer :: i, j, k
117  integer :: nz
118  real, dimension(G%ke+1) :: dx
119  real, dimension(G%ke) :: h1, u_column
120 
121  nz = g%ke
122 
123  ! Remap tracer
124  do j = g%jsc,g%jec
125  do i = g%isc,g%iec
126  if (g%mask2dT(i,j)>0.) then
127  ! Build the start and final grids
128  h1(:) = h(i,j,:)
129  dx(:) = dxinterface(i,j,:)
130  call remapping_core(cs, nz, h1, tv%S(i,j,:), nz, dx, u_column)
131  tv%S(i,j,:) = u_column(:)
132  call remapping_core(cs, nz, h1, tv%T(i,j,:), nz, dx, u_column)
133  tv%T(i,j,:) = u_column(:)
134  endif
135  enddo
136  enddo
137 
138  ! Remap u velocity component
139  if ( present(u) ) then
140  do j = g%jsc,g%jec
141  do i = g%iscB,g%iecB
142  if (g%mask2dCu(i,j)>0.) then
143  ! Build the start and final grids
144  h1(:) = 0.5 * ( h(i,j,:) + h(i+1,j,:) )
145  dx(:) = 0.5 * ( dxinterface(i,j,:) + dxinterface(i+1,j,:) )
146  call remapping_core(cs, nz, h1, u(i,j,:), nz, dx, u_column)
147  u(i,j,:) = u_column(:)
148  endif
149  enddo
150  enddo
151  endif
152 
153  ! Remap v velocity component
154  if ( present(v) ) then
155  do j = g%jscB,g%jecB
156  do i = g%isc,g%iec
157  if (g%mask2dCv(i,j)>0.) then
158  ! Build the start and final grids
159  h1(:) = 0.5 * ( h(i,j,:) + h(i,j+1,:) )
160  dx(:) = 0.5 * ( dxinterface(i,j,:) + dxinterface(i,j+1,:) )
161  call remapping_core(cs, nz, h1, v(i,j,:), nz, dx, u_column)
162  v(i,j,:) = u_column(:)
163  endif
164  enddo
165  enddo
166  endif
167 
168 end subroutine remapping_main

198 subroutine checkconsistantcoords(ns, xs, nf, xf, strict, msg)
199 !------------------------------------------------------------------------------
200 ! Checks that xs and xf are consistent to within roundoff.
201 ! If strict=False, the end points of xs and xf are allowed to differ by
202 ! numerical roundoff due to the nature of summation to obtain xs, xf.
203 ! If strict=True, the edn points must be identical.
204 !------------------------------------------------------------------------------
205 
206  ! Arguments
207  integer, intent(in) :: ns, nf
208  real, intent(in) :: xs(ns+1), xf(nf+1)
209  logical, intent(in) :: strict
210  character(len=*), intent(in) :: msg
211 
212  ! Local variables
213  integer :: k
214  real :: sumhs, sumhf
215 
216  sumhs = xs(ns+1)-xs(1)
217  sumhf = xf(nf+1)-xf(1)
218 
219  if (strict) then
220  if (sumhf /= sumhs) call &
221  mom_error(fatal,'MOM_remapping, checkConsistantCoords: '//&
222  'Total thickness of two grids are not exactly equal.'//&
223  ' Called from '//trim(msg) )
224  else ! not strict
225  if (ispossumerrsignificant(ns, sumhs, nf, sumhf)) then
226  write(0,*) 'Start/final/start-final grid'
227  do k = 1,max(ns,nf)+1
228  if (k<=min(ns+1,nf+1)) then
229  write(0,'(i4,3es12.3)') k,xs(k),xf(k),xs(k)-xf(k)
230  elseif (k>ns+1) then
231  write(0,'(i4,12x,1es12.3)') k,xf(k)
232  else
233  write(0,'(i4,1es12.3)') k,xs(k)
234  endif
235  enddo
236  call mom_error(fatal,'MOM_remapping, checkConsistantCoords: '//&
237  'Total thickness of two grids do not match to within round-off.'//&
238  ' Called from '//trim(msg) )
239  endif
240  endif
241 
242 end subroutine checkconsistantcoords

245 !------------------------------------------------------------------------------
246 ! Compare two summation estimates of positive data and judge if due to more
247 ! than round-off
248 !------------------------------------------------------------------------------
249 function ispossumerrsignificant(n1, sum1, n2, sum2)
250 !------------------------------------------------------------------------------
251 ! When two sums are calculated from different vectors that should add up to
252 ! the same value, the results can differ by round off. The round off error
253 ! can be bounded to be proportional to the number of operations.
254 ! This function returns true if the difference between sum1 and sum2 is
255 ! larger than than the estimated round off bound.
256 ! NOTE: This estimate/function is only valid for summation of postive data.
257 !------------------------------------------------------------------------------
258  ! Arguments
259  integer, intent(in) :: n1, n2
260  real, intent(in) :: sum1, sum2
261  logical :: ispossumerrsignificant
262  ! Local variables
263  real :: sumerr, allowederr, eps
264 
265  if (sum1<0.) call mom_error(fatal,'isPosSumErrSignificant: sum1<0 is not allowed!')
266  if (sum2<0.) call mom_error(fatal,'isPosSumErrSignificant: sum2<0 is not allowed!')
267  sumerr = abs(sum1-sum2)
268  eps = epsilon(sum1)
269  allowederr = eps*0.5*(real(n1-1)*sum1+real(n2-1)*sum2)
270  if (sumerr>allowederr) then
271  write(0,*) 'isPosSumErrSignificant: sum1,sum2=',sum1,sum2
272  write(0,*) 'isPosSumErrSignificant: eps=',eps
273  write(0,*) 'isPosSumErrSignificant: err,n*eps=',sumerr,allowederr
274  write(0,*) 'isPosSumErrSignificant: err/eps,n1,n2,n1+n2=',sumerr/eps,n1,n2,n1+n2
275  ispossumerrsignificant = .true.
276  else
277  ispossumerrsignificant = .false.
278  endif
279 end function ispossumerrsignificant

285 subroutine sumhtimesq(nz, h, q, sumHQ, sumErr)
286 !------------------------------------------------------------------------------
287 ! This routine calculates the sum of h(:)*q(:), and optionally returns a
288 ! bound on the roundoff error in the sum.
289 !------------------------------------------------------------------------------
290 
291  ! Arguments
292  integer, intent(in) :: nz
293  real, dimension(nz), intent(in) :: h, q
294  real, intent(out) :: sumhq
295  real, optional, intent(out) :: sumerr
296 
297  integer :: k
298  real :: hq, eps
299 
300  if (present(sumerr)) then
301  ! Calculate the sum and estimate errors
302  eps = epsilon(q(1))
303  sumerr=0.
304  sumhq = 0.
305  do k = 1,nz
306  hq = h(k)*q(k)
307  sumhq = sumhq + hq
308  if (k>1) sumerr = sumerr + eps*max(abs(sumhq),abs(hq))
309  end do
310  else
311  ! Calculate the sum
312  sumhq = 0.
313  do k = 1,nz
314  sumhq = sumhq + h(k)*q(k)
315  end do
316  endif
317 
318 end subroutine sumhtimesq

358 !------------------------------------------------------------------------------
359 subroutine remapping_core( CS, n0, h0, u0, n1, dx, u1 )
360 !------------------------------------------------------------------------------
361 ! This routine is basic in that it simply takes two grids and remaps the
362 ! field known on the first grid onto the second grid, following the rules
363 ! stored in the structure CS.
364 !------------------------------------------------------------------------------
365 
366  ! Arguments
367  type(remapping_cs), intent(inout) :: cs
368  integer, intent(in) :: n0 ! Number of cells on source grid
369  real, dimension(:), intent(in) :: h0 ! cell widths on source grid
370  real, dimension(:), intent(in) :: u0 ! cell averages on source grid
371  integer, intent(in) :: n1 ! Number of cells on target grid
372  real, dimension(:), intent(in) :: dx ! Change in interface positions
373  real, dimension(:), intent(out) :: u1 ! cell averages on target grid
374 
375  ! Local variables
376  integer :: imethod
377 
378 #ifdef __DO_SAFTEY_CHECKS__
379  integer :: k, remapping_scheme
380  real :: htmp, totalh0, totalhf, eps
381  real :: err0, totalhu0, err2, totalhu2
382  real :: z0, z1
383 
384  if (dx(1) /= 0.) call mom_error( fatal, 'MOM_remapping, remapping_core: '//&
385  'Non-zero surface flux!' ) ! This is techically allowed but in avoided practice
386  totalh0 = 0.
387  do k=1, n0
388  totalh0 = totalh0 + h0(k)
389  enddo
390  totalhf = 0.
391  do k=1, n1
392  if (k <= n0) then
393  htmp = h0(k) + ( dx(k+1) - dx(k) )
394  if (htmp < 0.) then
395  write(0,*) 'k,h0(k),hNew,dx(+1),dx(0)=',k,h0(k),dx(k+1),dx(k)
396  call mom_error( fatal, 'MOM_remapping, remapping_core: '//&
397  'negative h implied by fluxes' )
398  endif
399  else
400  htmp = ( dx(k+1) - dx(k) )
401  if (htmp < 0.) then
402  write(0,*) 'k,hNew,dx(+1),dx(0)=',k,dx(k+1),dx(k)
403  call mom_error( fatal, 'MOM_remapping, remapping_core: '//&
404  'negative h implied by fluxes' )
405  endif
406  endif
407  totalhf = totalhf + htmp
408  end do
409  eps = epsilon(htmp)*totalh0
410  if (abs(totalhf-totalh0) > 0.5*real(n0+n1-1)*eps) then
411  write(0,*) 'H0,Hf=',totalh0,totalhf,totalhf-totalh0,eps
412  call mom_error( fatal, 'MOM_remapping, remapping_core: '//&
413  'Total thicknesses of h0 and h2 differ by more than roundoff' )
414  endif
415 #endif
416 
417  imethod = -999
418 
419  ! Reset polynomial
420  cs%ppoly_r%E(:,:) = 0.0
421  cs%ppoly_r%S(:,:) = 0.0
422  cs%ppoly_r%coefficients(:,:) = 0.0
423 
424  remapping_scheme = cs%remapping_scheme
425  if (n0<=1) then
426  remapping_scheme = remapping_pcm
427  elseif (n0<=3) then
428  remapping_scheme = min( remapping_scheme, remapping_plm )
429  elseif (n0<=4) then
430  remapping_scheme = min( remapping_scheme, remapping_ppm_h4 )
431  endif
432  select case ( remapping_scheme )
433  case ( remapping_pcm )
434  call pcm_reconstruction( n0, u0, cs%ppoly_r )
435  imethod = integration_pcm
436  case ( remapping_plm )
437  call plm_reconstruction( n0, h0, u0, cs%ppoly_r )
438  if ( cs%boundary_extrapolation) then
439  call plm_boundary_extrapolation( n0, h0, u0, cs%ppoly_r )
440  end if
441  imethod = integration_plm
442  case ( remapping_ppm_h4 )
443  call edge_values_explicit_h4( n0, h0, u0, cs%ppoly_r%E )
444  call ppm_reconstruction( n0, h0, u0, cs%ppoly_r )
445  if ( cs%boundary_extrapolation) then
446  call ppm_boundary_extrapolation( n0, h0, u0, cs%ppoly_r )
447  end if
448  imethod = integration_ppm
449  case ( remapping_ppm_ih4 )
450  call edge_values_implicit_h4( n0, h0, u0, cs%edgeValueWrk, cs%ppoly_r%E )
451  call ppm_reconstruction( n0, h0, u0, cs%ppoly_r )
452  if ( cs%boundary_extrapolation) then
453  call ppm_boundary_extrapolation( n0, h0, u0, cs%ppoly_r )
454  end if
455  imethod = integration_ppm
456  case ( remapping_pqm_ih4ih3 )
457  call edge_values_implicit_h4( n0, h0, u0, cs%edgeValueWrk, cs%ppoly_r%E )
458  call edge_slopes_implicit_h3( n0, h0, u0, cs%edgeSlopeWrk, cs%ppoly_r%S )
459  call pqm_reconstruction( n0, h0, u0, cs%ppoly_r )
460  if ( cs%boundary_extrapolation) then
461  call pqm_boundary_extrapolation_v1( n0, h0, u0, cs%ppoly_r )
462  end if
463  imethod = integration_pqm
464  case ( remapping_pqm_ih6ih5 )
465  call edge_values_implicit_h6( n0, h0, u0, cs%edgeValueWrk, cs%ppoly_r%E )
466  call edge_slopes_implicit_h5( n0, h0, u0, cs%edgeSlopeWrk, cs%ppoly_r%S )
467  call pqm_reconstruction( n0, h0, u0, cs%ppoly_r )
468  if ( cs%boundary_extrapolation) then
469  call pqm_boundary_extrapolation_v1( n0, h0, u0, cs%ppoly_r )
470  end if
471  imethod = integration_pqm
472  case default
473  call mom_error( fatal, 'MOM_remapping, remapping_core: '//&
474  'The selected remapping method is invalid' )
475  end select
476 
477  call remapbydeltaz( n0, h0, u0, cs%ppoly_r, n1, dx, imethod, u1 )
478 ! call remapByProjection( n0, h0, u0, CS%ppoly_r, n1, h1, iMethod, u1 )
479 
480 #ifdef __DO_SAFTEY_CHECKS__
481  totalhu0 = 0.
482  err0 = 0.
483  do k = 1, n0
484  htmp = h0(k) * u0(k)
485  totalhu0 = totalhu0 + htmp
486  err0 = err0 + epsilon(err0)*max(err0,abs(htmp))
487  enddo
488  totalhu2 = 0. ; err2 = 0. ; z0 =0. ; z1 = 0.
489  do k = 1, n1
490  if (k <= n0) then
491  htmp = h0(k) + ( dx(k+1) - dx(k) )
492  else
493  htmp = ( dx(k+1) - dx(k) )
494  endif
495  htmp = htmp * u1(k)
496  totalhu2 = totalhu2 + htmp
497  err2 = err2 + epsilon(err2)*max(err2,abs(htmp))
498  if (u1(k) /= u1(k)) then
499  write(0,*) 'NaN detected at k=',k
500  write(0,*) 'h0(k)=',h0(k),' u0(k)=',u0(k)
501  write(0,*) 'z0(k)=',z0,' z0(k+1)=',z0+h0(k)
502  write(0,*) 'w(k)=',dx(k),' w(k+1)=',dx(k+1)
503  write(0,*) 'z1(k)=',z1,' z1(k+1)=',z1+(h0(k)+(dx(k+1)-dx(k)))
504  write(0,*) 'z0(k)+w(k)=',z0-dx(k),' z1(k+1)+w(k+1)=',(z0+h0(k))-dx(k+1)
505  write(0,*) 'h1(k)=',h0(k) + ( dx(k+1) - dx(k) )
506  call mom_error( fatal, 'MOM_remapping, remapping_core: '//&
507  'NaN detected!' )
508  endif
509  z0 = z0 + h0(k) ; z1 = z1 + ( h0(k) + ( dx(k+1) - dx(k) ) )
510  enddo
511  if (abs(totalhu2-totalhu0) > (err0+err2)*real(n1)) then
512  write(0,*) 'h0=',h0
513  write(0,*) 'hf=',h0+dx(2:n1+1)-dx(1:n1)
514  write(0,*) 'u0=',u0
515  write(0,*) 'u1=',u1
516  write(0,*) 'total HU0,HUf,f-0=',totalhu0,totalhu2,totalhu2-totalhu0
517  write(0,*) 'err0,errF=',err0,err2
518  call mom_error( fatal, 'MOM_remapping, remapping_core: '//&
519  'Total stuff on h0 and hF differ by more than roundoff' )
520  endif
521 #endif
522 
523 end subroutine remapping_core

538  real, intent(out) :: u1(:) ! target cell averages (size n1)

541  integer :: itarget
542  real :: xl, xr ! coordinates of target cell edges
543 
544  ! Loop on cells in target grid (grid1). For each target cell, we need to find
545  ! in which source cells the target cell edges lie. The associated indexes are
546  ! noted j0 and j1.
547  xr = 0. ! Left boundary is at x=0
548  do itarget = 1,n1
549  ! Determine the coordinates of the target cell edges
550  xl = xr
551  xr = xl + h1(itarget)
552 
553  call integraterecononinterval( n0, h0, u0, ppoly0, method, &
554  xl, xr, h1(itarget), u1(itarget) )

556  end do ! end iTarget loop on target grid cells

561 ! -----------------------------------------------------------------------------
562 ! Remap using change in interface positions
563 ! -----------------------------------------------------------------------------
564 subroutine remapbydeltaz( n0, h0, u0, ppoly0, n1, dx1, method, u1, h1 )
565 ! The new grid is defined relative to the original grid by change
566 ! dx1(:) = xNew(:) - xOld(:)
567 ! and the remapping calculated so that
568 ! hNew(k) qNew(k) = hOld(k) qOld(k) + F(k+1) - F(k)
569 ! where
570 ! F(k) = dx1(k) qAverage
571 ! and where qAverage is the average qOld in the region zOld(k) to zNew(k).
572  ! Arguments
573  integer, intent(in) :: n0 ! number of cells in source grid
574  real, intent(in) :: h0(:) ! source grid widths (size n0)
575  real, intent(in) :: u0(:) ! source cell averages (size n0)
576  type(ppoly_t), intent(in) :: ppoly0 ! source piecewise polynomial
577  integer, intent(in) :: n1 ! number of cells in target grid
578  real, intent(in) :: dx1(:) ! target grid edge positions (size n1+1)
579  integer :: method ! remapping scheme to use
580  real, intent(out) :: u1(:) ! target cell averages (size n1)
581  real, optional, intent(out) :: h1(:) ! target grid widths (size n1)
582 
583  ! Local variables
584  integer :: itarget
585  real :: xl, xr ! coordinates of target cell edges
586  real :: xold, hold, uold
587  real :: xnew, hnew
588  real :: uhnew, hflux, uave, fluxl, fluxr
589 #ifdef __DO_SAFTEY_CHECKS__
590  real :: h0total
591 
592  h0total = 0.
593  do itarget = 1, n0
594  h0total = h0total + h0(itarget)
595  enddo
596 #endif
597 
598  ! Loop on cells in target grid. For each cell, iTarget, the left flux is
599  ! the right flux of the cell to the left, iTarget-1.
600  ! The left flux is initialized by started at iTarget=0 to calclate the
601  ! right flux which can take into account the target left boundary being
602  ! in the interior of the source domain.
603  fluxr = 0.
604  do itarget = 0,n1
605  fluxl = fluxr ! This does nothing for iTarget=0
606 
607  if (itarget == 0) then
608  xold = 0. ! Left boundary is at x=0
609  hold = -1.e30 ! Should not be used for iTarget = 0
610  uold = -1.e30 ! Should not be used for iTarget = 0
611  elseif (itarget <= n0) then
612  xold = xold + h0(itarget) ! Position of right edge of cell
613  hold = h0(itarget)
614  uold = u0(itarget)
615  else
616  hold = 0. ! as if for layers>n0, they were vanished
617  uold = 1.e30 ! and the initial value should not matter
618  endif
619  xnew = xold + dx1(itarget+1)
620  xl = min( xold, xnew )
621  xr = max( xold, xnew )
622 
623 #ifdef __DO_SAFTEY_CHECKS__
624  if (xl < 0.) then
625  write(0,*) 'h0=',h0
626  write(0,*) 'dx1=',dx1
627  write(0,*) 'xOld,xNew,xL,xR,i=',xold,xnew,xl,xr,itarget
628  call mom_error(fatal,'MOM_remapping, remapByDeltaZ: xL too negative')
629  endif
630  if (xr > h0total) then
631  write(0,*) 'h0=',h0
632  write(0,*) 'dx1=',dx1
633  write(0,*) 'xOld,xNew,xL,xR,i=',xold,xnew,xl,xr,itarget
634  call mom_error(fatal,'MOM_remapping, remapByDeltaZ: xR too positive')
635  endif
636 #endif
637 
638  ! hFlux is the positive width of the remapped volume
639  hflux = abs(dx1(itarget+1))
640  call integraterecononinterval( n0, h0, u0, ppoly0, method, &
641  xl, xr, hflux, uave )
642  ! uAve is the average value of u, indendent of sign of dx1
643  fluxr = dx1(itarget+1)*uave ! Includes sign of dx1
644 
645 #ifdef XXX__DO_SAFTEY_CHECKS__
646  ! Valid for CFL<1
647  if (dx1(itarget+1)<h0(itarget+1) .and. dx1(itarget+1)>-hold) then
648  if (uave<min(u0(itarget),u0(itarget+1))) then
649  write(0,*) 'u,u(k),u(k+1)=',uave,u0(itarget),u0(itarget+1)
650  call mom_error(fatal,'MOM_remapping, remapByDeltaZ: undershoot in U')
651  endif
652  if (uave>max(u0(itarget),u0(itarget+1))) then
653  write(0,*) 'u,u(k),u(k+1)=',uave,u0(itarget),u0(itarget+1)
654  call mom_error(fatal,'MOM_remapping, remapByDeltaZ: overshoot in U')
655  endif
656  endif
657 #endif
658 
659  if (itarget>0) then
660  hnew = hold + ( dx1(itarget+1) - dx1(itarget) )
661  hnew = max( 0., hnew )
662  uhnew = ( uold * hold ) + ( fluxr - fluxl )
663  if (hnew>0.) then
664  u1(itarget) = uhnew / hnew
665  else
666  u1(itarget) = uave
667  endif
668  if (present(h1)) h1(itarget) = hnew
669  endif
670 
671  end do ! end iTarget loop on target grid cells
672 
673 end subroutine remapbydeltaz

676 ! -----------------------------------------------------------------------------
677 ! integrate the reconstructed profile over a single cell
678 ! -----------------------------------------------------------------------------
679 subroutine integraterecononinterval( n0, h0, u0, ppoly0, method, &
680  xl, xr, hc, uave )
681  ! Arguments
682  integer, intent(in) :: n0 ! number of cells in source grid
683  real, intent(in) :: h0(:) ! source grid sizes (size n0)
684  real, dimension(:), intent(in) :: u0 ! source cell averages
685  type(ppoly_t), intent(in) :: ppoly0 ! source piecewise polynomial
686  integer, intent(in) :: method ! remapping scheme to use
687  real, intent(in) :: xl, xr ! left/right edges of target cell
688  real, intent(in) :: hc ! cell width hC = xR - xL
689  real, intent(out) :: uave ! average value on target cell
690 
691  ! Local variables
692  integer :: j, k
693  integer :: jl, jr ! indexes of source cells containing target
694  ! cell edges
695  real :: q ! complete integration
696  real :: xi0, xi1 ! interval of integration (local -- normalized
697  ! -- coordinates)
698  real :: x0jll, x0jlr ! Left/right position of cell jL
699  real :: x0jrl, x0jrr ! Left/right position of cell jR
700 
701 #ifdef __DO_SAFTEY_CHECKS__
702  real :: h0total
703 
704  h0total = 0.
705  do k = 1, n0
706  h0total = h0total + h0(k)
707  enddo
708 ! if (xL < 0.) call MOM_error(FATAL, &
709 ! 'MOM_remapping, integrateReconOnInterval: '//&
710 ! 'The target cell starts beyond the left edge of the source grid')
711 ! if (xR < 0.) call MOM_error(FATAL, &
712 ! 'MOM_remapping, integrateReconOnInterval: '//&
713 ! 'The target cell ends beyond the left edge of the source grid')
714 ! if (xL > h0Total) call MOM_error(FATAL, &
715 ! 'MOM_remapping, integrateReconOnInterval: '//&
716 ! 'The target cell starts beyond the right edge of the source grid')
717 ! if (xR > h0Total) call MOM_error(FATAL, &
718 ! 'MOM_remapping, integrateReconOnInterval: '//&
719 ! 'The target cell ends beyond the right edge of the source grid')
720 #endif
721 
722  q = -1.e30
723  x0jll = -1.e30
724  x0jrl = -1.e30
725 
726  ! Find the left most cell in source grid spanned by the target cell
727  jl = -1
728  x0jlr = 0.
729  do j = 1,n0
730  x0jll = x0jlr
731  x0jlr = x0jll + h0(j)
732  ! Left edge is found in cell j
733  if ( ( xl >= x0jll ) .AND. ( xl <= x0jlr ) ) then
734  jl = j
735  exit ! once target grid cell is found, exit loop
736  endif
737  enddo
738 
739  ! If, at this point, jL is equal to -1, it means the vanished
740  ! cell lies outside the source grid. In other words, it means that
741  ! the source and target grids do not cover the same physical domain
742  ! and there is something very wrong !
743  if ( jl == -1 ) call mom_error(fatal, &
744  'MOM_remapping, integrateReconOnInterval: '//&
745  'The location of the left-most cell could not be found')
746 
747 
748  ! ============================================================
749  ! Check whether target cell is vanished. If it is, the cell
750  ! average is simply the interpolated value at the location
751  ! of the vanished cell. If it isn't, we need to integrate the
752  ! quantity within the cell and divide by the cell width to
753  ! determine the cell average.
754  ! ============================================================
755  ! 1. Cell is vanished
756  if ( abs(xr - xl) == 0.0 ) then
757 
758  ! We check whether the source cell (i.e. the cell in which the
759  ! vanished target cell lies) is vanished. If it is, the interpolated
760  ! value is set to be mean of the edge values (which should be the same).
761  ! If it isn't, we simply interpolate.
762  if ( h0(jl) == 0.0 ) then
763  uave = 0.5 * ( ppoly0%E(jl,1) + ppoly0%E(jl,2) )
764  else
765  ! WHY IS THIS NOT WRITTEN AS xi0 = ( xL - x0jLl ) / h0(jL) ---AJA
766  xi0 = xl / h0(jl) - x0jll / h0(jl)
767 
768  select case ( method )
769  case ( integration_pcm )
770  uave = ppoly0%coefficients(jl,1)
771  case ( integration_plm )
772  uave = evaluation_polynomial( ppoly0%coefficients(jl,:), 2, xi0 )
773  case ( integration_ppm )
774  uave = evaluation_polynomial( ppoly0%coefficients(jl,:), 3, xi0 )
775  case ( integration_pqm )
776  uave = evaluation_polynomial( ppoly0%coefficients(jl,:), 5, xi0 )
777  case default
778  call mom_error( fatal,'The selected integration method is invalid' )
779  end select
780 
781  end if ! end checking whether source cell is vanished
782 
783  ! 2. Cell is not vanished
784  else
785 
786  ! Find the right most cell in source grid spanned by the target cell
787  jr = -1
788  x0jrr = 0.
789  do j = 1,n0
790  x0jrl = x0jrr
791  x0jrr = x0jrl + h0(j)
792  ! Right edge is found in cell j
793  if ( ( xr >= x0jrl ) .AND. ( xr <= x0jrr ) ) then
794  jr = j
795  exit ! once target grid cell is found, exit loop
796  endif
797  enddo ! end loop on source grid cells
798 
799  ! HACK to avoid roundoff problems THIS NEEDS TO BE REMOVED ---AJA
800  if (xr>x0jrr) jr = n0
801 
802 #ifdef __DO_SAFTEY_CHECKS__
803  if ( jr == -1 ) call mom_error(fatal, &
804  'MOM_remapping, integrateReconOnInterval: '//&
805  'The location of the right-most cell could not be found')
806 #endif
807 
808  ! To integrate, two cases must be considered: (1) the target cell is
809  ! entirely comtained within a cell of the source grid and (2) the target
810  ! cell spans at least two cells of the source grid.
811 
812  if ( jl == jr ) then
813  ! The target cell is entirely contained within a cell of the source
814  ! grid. This situation is represented by the following schematic, where
815  ! the cell in which xL and xR are located has index jL=jR :
816  !
817  ! ----|-----o--------o----------|-------------
818  ! xL xR
819  !
820  ! Determine normalized coordinates
821  ! WHY IS THIS NOT WRITTEN AS xi0 = ( xL - x0jLl ) / h0(jL) ---AJA
822  ! WHY IS THIS NOT WRITTEN AS xi0 = ( xR - x0jLl ) / h0(jL) ---AJA
823  xi0 = xl / h0(jl) - x0jll / h0(jl)
824  xi1 = xr / h0(jl) - x0jll / h0(jl)
825 
826  ! Depending on which polynomial is used, integrate quantity
827  ! between xi0 and xi1. Integration is carried out in normalized
828  ! coordinates, hence: \int_xL^xR p(x) dx = h \int_xi0^xi1 p(xi) dxi
829  select case ( method )
830  case ( integration_pcm )
831  q = ppoly0%coefficients(jl,1) * ( xr - xl )
832  case ( integration_plm )
833  q = h0(jl) * &
834  integration_polynomial( xi0, xi1, ppoly0%coefficients(jl,:), 1 )
835  case ( integration_ppm )
836  q = h0(jl) * &
837  integration_polynomial( xi0, xi1, ppoly0%coefficients(jl,:), 2 )
838  case ( integration_pqm )
839  q = h0(jl) * &
840  integration_polynomial( xi0, xi1, ppoly0%coefficients(jl,:), 4 )
841  case default
842  call mom_error( fatal,'The selected integration method is invalid' )
843  end select
844 
845  else
846  ! The target cell spans at least two cells of the source grid.
847  ! This situation is represented by the following schematic, where
848  ! the cells in which xL and xR are located have indexes jL and jR,
849  ! respectively :
850  !
851  ! ----|-----o---|--- ... --|---o----------|-------------
852  ! xL xR
853  !
854  ! We first integrate from xL up to the right boundary of cell jL, then
855  ! add the integrated amounts of cells located between jL and jR and then
856  ! integrate from the left boundary of cell jR up to xR
857 
858  q = 0.0
859 
860  ! Integrate from xL up to right boundary of cell jL
861  !xi0 = xL / h0(jL) - x0jLl / h0(jL)
862  xi0 = (xl - x0jll) / h0(jl)
863  xi1 = 1.0
864  select case ( method )
865  case ( integration_pcm )
866  q = q + ppoly0%coefficients(jl,1) * ( x0jlr - xl )
867  case ( integration_plm )
868  q = q + h0(jl) * &
869  integration_polynomial( xi0, xi1, ppoly0%coefficients(jl,:), 1 )
870  case ( integration_ppm )
871  q = q + h0(jl) * &
872  integration_polynomial( xi0, xi1, ppoly0%coefficients(jl,:), 2 )
873  case ( integration_pqm )
874  q = q + h0(jl) * &
875  integration_polynomial( xi0, xi1, ppoly0%coefficients(jl,:), 4 )
876  case default
877  call mom_error( fatal, 'The selected integration method is invalid' )
878  end select
879 
880  ! Integrate contents within cells strictly comprised between jL and jR
881  if ( jr > (jl+1) ) then
882  do k = jl+1,jr-1
883  q = q + h0(k) * u0(k)
884  end do
885  end if
886 
887  ! Integrate from left boundary of cell jR up to xR
888  xi0 = 0.0
889  !xi1 = xR / h0(jR) - x0jRl / h0(jR)
890  xi1 = (xr - x0jrl) / h0(jr)
891 
892  select case ( method )
893  case ( integration_pcm )
894  q = q + ppoly0%coefficients(jr,1) * ( xr - x0jrl )
895  case ( integration_plm )
896  q = q + h0(jr) * &
897  integration_polynomial( xi0, xi1, ppoly0%coefficients(jr,:), 1 )
898  case ( integration_ppm )
899  q = q + h0(jr) * &
900  integration_polynomial( xi0, xi1, ppoly0%coefficients(jr,:), 2 )
901  case ( integration_pqm )
902  q = q + h0(jr) * &
903  integration_polynomial( xi0, xi1, ppoly0%coefficients(jr,:), 4 )
904  case default
905  call mom_error( fatal,'The selected integration method is invalid' )
906  end select
907 
908  end if ! end integration for non-vanished cells
909 
910  ! The cell average is the integrated value divided by the cell width
911  uave = q / hc
912 
913  end if ! end if clause to check if cell is vanished
914 
915 end subroutine integraterecononinterval

1078 logical function remappingunittests()

1081  integer, parameter :: n0 = 4, n1 = 3, n2 = 6
1082  real :: h0(n0), x0(n0+1), u0(n0)
