24 !* *
25 !* By Robert Hallberg, February 2001 *
26 !* *
27 !* The subroutines here calculate the heights of interfaces in a *
28 !* way that is consistent with the calculations in the pressure *
29 !* gradient acceleration calculation. In a Boussinseq model this is *
30 !* pretty simple vertical sum, but in a non-Boussinesq model it uses *
31 !* integrals of the equation of state. *
32 !* *
33 !* Macros written all in capital letters are defined in MOM_memory.h. *
34 !* *
35 !* A small fragment of the grid is shown below: *
36 !* *
37 !* j+1 x ^ x ^ x At x: q, CoriolisBu *
38 !* j+1 > o > o > At ^: v *
39 !* j x ^ x ^ x At >: u *
40 !* j > o > o > At o: h, bathyT *
41 !* j-1 x ^ x ^ x *
42 !* i-1 i i+1 At x & ^: *
43 !* i i+1 At > & o: *
44 !* *
45 !* The boundaries always run through q grid points (x). *
46 !* *
47 !********+*********+*********+*********+*********+*********+*********+**

67 subroutine find_eta_3d(h, tv, G_Earth, G, eta, eta_bt, halo_size)
68  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h
69  type(thermo_var_ptrs), intent(in) :: tv
70  real, intent(in) :: g_earth
71  type(ocean_grid_type), intent(in) :: g
72  real, dimension(NIMEM_,NJMEM_,NK_INTERFACE_), intent(out) :: eta
73  real, dimension(NIMEM_,NJMEM_), optional, intent(in) :: eta_bt
74  integer, optional, intent(in) :: halo_size
75 ! This subroutine determines the heights of all interfaces between layers,
76 ! using the appropriate form for consistency with the calculation of the
77 ! pressure gradient forces. Additionally, these height may be dilated for
78 ! consistency with the corresponding time-average quantity from the barotropic
79 ! calculation.
80 
81 ! Arguments: h - Layer thickness, in m. Intent in.
82 ! (in) tv - a structure pointing to various thermodynamic variables.
83 ! (in) G_Earth - The Earth's gravitational acceleration, in m s-2.
84 ! (in) G - The ocean's grid structure.
85 ! (out) eta - The free surface height realtive to mean sea level, in m.
86 ! (in,opt) eta_bt - An optional barotropic variable that gives the "correct"
87 ! free surface height (Boussinesq) or total water column
88 ! mass per unit aread (non-Boussinesq). This is used to
89 ! dilate the layer thicknesses when calculating interface
90 ! heights, in m or kg m-2.
91 ! (in,opt) halo_size - The width of halo points on which to calculate eta.
92 
93  real :: p(szi_(g),szj_(g),szk_(g)+1)
94  real :: dz_geo(szi_(g),szj_(g)) ! The change in geopotential height
95  ! across a layer, in m2 s-2.
96  real :: dilate(szi_(g)), htot(szi_(g))
97  real :: i_gearth
98  integer i, j, k, isv, iev, jsv, jev, nz, halo
99 
100  halo = 0 ; if (present(halo_size)) halo = max(0,halo_size)
101 
102  isv = g%isc-halo ; iev = g%iec+halo ; jsv = g%jsc-halo ; jev = g%jec+halo
103  nz = g%ke
104 
105  if ((isv<g%isd) .or. (iev>g%ied) .or. (jsv<g%jsd) .or. (jev>g%jed)) &
106  call mom_error(fatal,"find_eta called with an overly large halo_size.")
107 
108  i_gearth = 1.0 / g_earth
109 
110  do j=jsv,jev ; do i=isv,iev ; eta(i,j,nz+1) = -g%bathyT(i,j) ; enddo ; enddo
111 
112  if (g%Boussinesq) then
113  do k=nz,1,-1 ; do j=jsv,jev ; do i=isv,iev
114  eta(i,j,k) = eta(i,j,k+1) + h(i,j,k)
115  enddo ; enddo ; enddo
116  if (present(eta_bt)) then
117  ! Dilate the water column to agree with the free surface height
118  ! that is used for the dynamics.
119  do j=jsv,jev
120  do i=isv,iev
121  dilate(i) = (eta_bt(i,j)+g%bathyT(i,j)) / (eta(i,j,1)+g%bathyT(i,j))
122  enddo
123  do k=1,nz ; do i=isv,iev
124  eta(i,j,k) = dilate(i) * (eta(i,j,k) + g%bathyT(i,j)) - g%bathyT(i,j)
125  enddo ; enddo
126  enddo
127  endif
128  else
129  if (associated(tv%eqn_of_state)) then
130  ! ### THIS SHOULD BE P_SURF, IF AVAILABLE.
131  do j=jsv,jev ; do i=isv,iev ; p(i,j,1) = 0.0 ; enddo ; enddo
132  do k=1,nz ; do j=jsv,jev ; do i=isv,iev
133  p(i,j,k+1) = p(i,j,k) + g_earth*g%H_to_kg_m2*h(i,j,k)
134  enddo ; enddo ; enddo
135 
136  do k=nz,1,-1
137  call int_specific_vol_dp(tv%T(:,:,k), tv%S(:,:,k), p(:,:,k), p(:,:,k+1), &
138  0.0, g, tv%eqn_of_state, dz_geo, halo_size=halo)
139  do j=jsv,jev ; do i=isv,iev
140  eta(i,j,k) = eta(i,j,k+1) + i_gearth * dz_geo(i,j)
141  enddo ; enddo
142  enddo
143  else
144  do k=nz,1,-1 ; do j=jsv,jev ; do i=isv,iev
145  eta(i,j,k) = eta(i,j,k+1) + g%H_to_kg_m2*h(i,j,k)/g%Rlay(k)
146  enddo ; enddo ; enddo
147  endif
148  if (present(eta_bt)) then
149  ! Dilate the water column to agree with the free surface height
150  ! from the time-averaged barotropic solution.
151  do j=jsv,jev
152  do i=isv,iev ; htot(i) = g%H_subroundoff ; enddo
153  do k=1,nz ; do i=isv,iev ; htot(i) = htot(i) + h(i,j,k) ; enddo ; enddo
154  do i=isv,iev ; dilate(i) = eta_bt(i,j) / htot(i) ; enddo
155  do k=1,nz ; do i=isv,iev
156  eta(i,j,k) = dilate(i) * (eta(i,j,k) + g%bathyT(i,j)) - g%bathyT(i,j)
157  enddo ; enddo
158  enddo
159  endif
160  endif
161 
162 end subroutine find_eta_3d

164 subroutine find_eta_2d(h, tv, G_Earth, G, eta, eta_bt, halo_size)
165  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h
166  type(thermo_var_ptrs), intent(in) :: tv
167  real, intent(in) :: g_earth
168  type(ocean_grid_type), intent(in) :: g
169  real, dimension(NIMEM_,NJMEM_), intent(out) :: eta
170  real, dimension(NIMEM_,NJMEM_), optional, intent(in) :: eta_bt
171  integer, optional, intent(in) :: halo_size
172 ! This subroutine determines the free surface height, using the appropriate
173 ! form for consistency with the calculation of the pressure gradient forces.
174 ! Additionally, the sea surface height may be adjusted for consistency with the
175 ! corresponding time-average quantity from the barotropic calculation.
176 
177 ! Arguments: h - Layer thickness, in m. Intent in.
178 ! (in) tv - a structure pointing to various thermodynamic variables.
179 ! (in) G_Earth - The Earth's gravitational acceleration, in m s-2.
180 ! (in) G - The ocean's grid structure.
181 ! (out) eta - The free surface height relative to mean sea level, in m.
182 ! (in,opt) eta_bt - An optional barotropic variable that gives the "correct"
183 ! free surface height (Boussinesq) or total water column
184 ! mass per unit aread (non-Boussinesq), in m or kg m-2.
185 ! (in,opt) halo_size - The width of halo points on which to calculate eta.
186 
187  real, dimension(SZI_(G),SZJ_(G)) :: &
188  p_top, & ! The pressure at the interface above a layer, in Pa.
189  p_bot, & ! The pressure at the interface below a layer, in Pa.
190  dz_geo ! The change in geopotential height across a layer, in m2 s-2.
191  real :: htot(szi_(g)) ! The sum of all layers' thicknesses, in kg m-2 or m.
192  real :: i_gearth
193  integer i, j, k, is, ie, js, je, nz, halo
194 
195  halo = 0 ; if (present(halo_size)) halo = max(0,halo_size)
196  is = g%isc-halo ; ie = g%iec+halo ; js = g%jsc-halo ; je = g%jec+halo
197  nz = g%ke
198 
199  i_gearth = 1.0 / g_earth
200 
201  do j=js,je ; do i=is,ie ; eta(i,j) = -g%bathyT(i,j) ; enddo ; enddo
202 
203  if (g%Boussinesq) then
204  if (present(eta_bt)) then
205  do j=js,je ; do i=is,ie
206  eta(i,j) = eta_bt(i,j)
207  enddo ; enddo
208  else
209  do k=1,nz ; do j=js,je ; do i=is,ie
210  eta(i,j) = eta(i,j) + h(i,j,k)
211  enddo ; enddo ; enddo
212  endif
213  else
214  if (associated(tv%eqn_of_state)) then
215  do j=js,je ; do i=is,ie
216  p_top(i,j) = 0.0 ; p_bot(i,j) = 0.0
217  enddo ; enddo
218  do k=1,nz
219  do j=js,je ; do i=is,ie
220  p_top(i,j) = p_bot(i,j)
221  p_bot(i,j) = p_top(i,j) + g_earth*g%H_to_kg_m2*h(i,j,k)
222  enddo ; enddo
223  call int_specific_vol_dp(tv%T(:,:,k), tv%S(:,:,k), p_top, p_bot, 0.0, &
224  g, tv%eqn_of_state, dz_geo, halo_size=halo)
225  do j=js,je ; do i=is,ie
226  eta(i,j) = eta(i,j) + i_gearth * dz_geo(i,j)
227  enddo ; enddo
228  enddo
229  else
230  do k=1,nz ; do j=js,je ; do i=is,ie
231  eta(i,j) = eta(i,j) + g%H_to_kg_m2*h(i,j,k)/g%Rlay(k)
232  enddo ; enddo ; enddo
233  endif
234  if (present(eta_bt)) then
235  ! Dilate the water column to agree with the the time-averaged column
236  ! mass from the barotropic solution.
237  do j=js,je
238  do i=is,ie ; htot(i) = g%H_subroundoff ; enddo
239  do k=1,nz ; do i=is,ie ; htot(i) = htot(i) + h(i,j,k) ; enddo ; enddo
240  do i=is,ie
241  eta(i,j) = (eta_bt(i,j) / htot(i)) * (eta(i,j) + g%bathyT(i,j)) - &
242  g%bathyT(i,j)
243  enddo
244  enddo
245  endif
246  endif

248 end subroutine find_eta_2d
