25 subroutine ppm_reconstruction( N, h, u, ppoly )
26 !------------------------------------------------------------------------------
27 ! Reconstruction by quadratic polynomials within each cell.
28 !
29 ! The edge values MUST have been estimated prior to calling this routine
30 !
31 ! grid: one-dimensional grid (see grid.F90)
32 ! ppoly: piecewise quadratic polynomial to be reconstructed (see ppoly.F90)
33 ! u: cell averages
34 !
35 ! It is assumed that the dimension of 'u' is equal to the number of cells
36 ! defining 'grid' and 'ppoly'. No consistency check is performed.
37 !------------------------------------------------------------------------------
38 
39  ! Arguments
40  integer, intent(in) :: n ! Number of cells
41  real, dimension(:), intent(in) :: h ! cell widths (size N)
42  real, dimension(:), intent(in) :: u ! cell averages (size N)
43  type(ppoly_t), intent(inout) :: ppoly
44 
45  ! Local variables
46  integer :: k ! loop index
47  real :: u0_l, u0_r ! edge values (left and right)
48  real :: a, b, c ! parabola coefficients
49 
50  ! PPM limiter
51  call ppm_limiter_standard( n, h, u, ppoly )
52 
53  ! Loop on cells to construct the parabola within each cell
54  do k = 1,n
55 
56  u0_l = ppoly%E(k,1)
57  u0_r = ppoly%E(k,2)
58 
59  a = u0_l
60  b = 6.0 * u(k) - 4.0 * u0_l - 2.0 * u0_r
61  c = 3.0 * ( u0_r + u0_l - 2.0 * u(k) )
62 
63  ! Store coefficients
64  ppoly%coefficients(k,1) = a
65  ppoly%coefficients(k,2) = b
66  ppoly%coefficients(k,3) = c
67 
68  end do ! end loop on interior cells
69 
70 end subroutine ppm_reconstruction

74 ! Limit ppm
75 ! -----------------------------------------------------------------------------
76 subroutine ppm_limiter_standard( N, h, u, ppoly )
77 !------------------------------------------------------------------------------
78 ! Standard PPM limiter (Colella & Woodward, JCP 1984).
79 !
80 ! grid: one-dimensional grid (see grid.F90)
81 ! ppoly: piecewise quadratic polynomial to be reconstructed (see ppoly.F90)
82 ! u: cell averages
83 !
84 ! It is assumed that the dimension of 'u' is equal to the number of cells
85 ! defining 'grid' and 'ppoly'. No consistency check is performed.
86 !------------------------------------------------------------------------------
87 
88  ! Arguments
89  integer, intent(in) :: n ! Number of cells
90  real, dimension(:), intent(in) :: h ! cell widths (size N)
91  real, dimension(:), intent(in) :: u ! cell averages (size N)
92  type(ppoly_t), intent(inout) :: ppoly
93 
94  ! Local variables
95  integer :: k ! loop index
96  real :: u_l, u_c, u_r ! cell averages (left, center and right)
97  real :: u0_l, u0_r ! edge values (left and right)
98  real :: expr1, expr2
99 
100  ! Bound edge values
101  call bound_edge_values( n, h, u, ppoly%E )
102 
103  ! Make discontinuous edge values monotonic
104  call check_discontinuous_edge_values( n, u, ppoly%E )
105 
106  ! Loop on interior cells to apply the standard
107  ! PPM limiter (Colella & Woodward, JCP 84)
108  do k = 2,n-1
109 
110  ! Get cell averages
111  u_l = u(k-1)
112  u_c = u(k)
113  u_r = u(k+1)
114 
115  u0_l = ppoly%E(k,1)
116  u0_r = ppoly%E(k,2)
117 
118  ! Auxiliary variables
119  expr1 = (u0_r - u0_l) * (u_c - 0.5*(u0_l+u0_r))
120  expr2 = (u0_r - u0_l) * (u0_r - u0_l) / 6.0
121 
122  ! Flatten extremum
123  if ( (u_r - u_c)*(u_c - u_l) .LE. 0.0) then
124  u0_l = u_c
125  u0_r = u_c
126  end if
127 
128  if ( expr1 .GT. expr2 ) then
129  u0_l = 3.0 * u_c - 2.0 * u0_r
130  end if
131 
132  if ( expr1 .LT. -expr2 ) then
133  u0_r = 3.0 * u_c - 2.0 * u0_l
134  end if
135 
136  ppoly%E(k,1) = u0_l
137  ppoly%E(k,2) = u0_r
138 
139  end do ! end loop on interior cells
140 
141  ! PCM within boundary cells
142  ppoly%E(1,:) = u(1)
143  ppoly%E(n,:) = u(n)
144 
145 end subroutine ppm_limiter_standard

148 !------------------------------------------------------------------------------
149 ! ppm boundary extrapolation
150 ! -----------------------------------------------------------------------------
151 subroutine ppm_boundary_extrapolation( N, h, u, ppoly )
152 !------------------------------------------------------------------------------
153 ! Reconstruction by parabolas within boundary cells.
154 !
155 ! The following explanations apply to the left boundary cell. The same
156 ! reasoning holds for the right boundary cell.
157 !
158 ! A parabola needs to be built in the cell and requires three degrees of
159 ! freedom, which are the right edge value and slope and the cell average.
160 ! The right edge values and slopes are taken to be that of the neighboring
161 ! cell (i.e., the left edge value and slope of the neighboring cell).
162 ! The resulting parabola is not necessarily monotonic and the traditional
163 ! PPM limiter is used to modify one of the edge values in order to yield
164 ! a monotonic parabola.
165 !
166 ! grid: one-dimensional grid (properly initialized)
167 ! ppoly: piecewise linear polynomial to be reconstructed (properly initialized)
168 ! u: cell averages
169 !
170 ! It is assumed that the size of the array 'u' is equal to the number of cells
171 ! defining 'grid' and 'ppoly'. No consistency check is performed here.
172 !------------------------------------------------------------------------------
173 
174  ! Arguments
175  integer, intent(in) :: n ! Number of cells
176  real, dimension(:), intent(in) :: h ! cell widths (size N)
177  real, dimension(:), intent(in) :: u ! cell averages (size N)
178  type(ppoly_t), intent(inout) :: ppoly
179 
180  ! Local variables
181  integer :: i0, i1
182  real :: u0, u1
183  real :: h0, h1
184  real :: a, b, c
185  real :: u0_l, u0_r
186  real :: u1_l, u1_r
187  real :: slope
188  real :: exp1, exp2
189 
190  ! ----- Left boundary -----
191  i0 = 1
192  i1 = 2
193  h0 = h(i0)
194  h1 = h(i1)
195  u0 = u(i0)
196  u1 = u(i1)
197 
198  ! Compute the left edge slope in neighboring cell and express it in
199  ! the global coordinate system
200  b = ppoly%coefficients(i1,2)
201  u1_r = b *(h0/h1) ! derivative evaluated at xi = 0.0,
202  ! expressed w.r.t. xi (local coord. system)
203 
204  ! Limit the right slope by the PLM limited slope
205  slope = 2.0 * ( u1 - u0 )
206  if ( abs(u1_r) .GT. abs(slope) ) then
207  u1_r = slope
208  end if
209 
210  ! The right edge value in the boundary cell is taken to be the left
211  ! edge value in the neighboring cell
212  u0_r = ppoly%E(i1,1)
213 
214  ! Given the right edge value and slope, we determine the left
215  ! edge value and slope by computing the parabola as determined by
216  ! the right edge value and slope and the boundary cell average
217  u0_l = 3.0 * u0 + 0.5 * u1_r - 2.0 * u0_r
218 
219  ! Apply the traditional PPM limiter
220  exp1 = (u0_r - u0_l) * (u0 - 0.5*(u0_l+u0_r))
221  exp2 = (u0_r - u0_l) * (u0_r - u0_l) / 6.0
222 
223  if ( exp1 .GT. exp2 ) then
224  u0_l = 3.0 * u0 - 2.0 * u0_r
225  end if
226 
227  if ( exp1 .LT. -exp2 ) then
228  u0_r = 3.0 * u0 - 2.0 * u0_l
229  end if
230 
231  ppoly%E(i0,1) = u0_l
232  ppoly%E(i0,2) = u0_r
233 
234  a = u0_l
235  b = 6.0 * u0 - 4.0 * u0_l - 2.0 * u0_r
236  c = 3.0 * ( u0_r + u0_l - 2.0 * u0 )
237 
238  ppoly%coefficients(i0,1) = a
239  ppoly%coefficients(i0,2) = b
240  ppoly%coefficients(i0,3) = c
241 
242  ! ----- Right boundary -----
243  i0 = n-1
244  i1 = n
245  h0 = h(i0)
246  h1 = h(i1)
247  u0 = u(i0)
248  u1 = u(i1)
249 
250  ! Compute the right edge slope in neighboring cell and express it in
251  ! the global coordinate system
252  b = ppoly%coefficients(i0,2)
253  c = ppoly%coefficients(i0,3)
254  u1_l = (b + 2*c) ! derivative evaluated at xi = 1.0
255  u1_l = u1_l * (h1/h0)
256 
257  ! Limit the left slope by the PLM limited slope
258  slope = 2.0 * ( u1 - u0 )
259  if ( abs(u1_l) .GT. abs(slope) ) then
260  u1_l = slope
261  end if
262 
263  ! The left edge value in the boundary cell is taken to be the right
264  ! edge value in the neighboring cell
265  u0_l = ppoly%E(i0,2)
266 
267  ! Given the left edge value and slope, we determine the right
268  ! edge value and slope by computing the parabola as determined by
269  ! the left edge value and slope and the boundary cell average
270  u0_r = 3.0 * u1 - 0.5 * u1_l - 2.0 * u0_l
271 
272  ! Apply the traditional PPM limiter
273  exp1 = (u0_r - u0_l) * (u1 - 0.5*(u0_l+u0_r))
274  exp2 = (u0_r - u0_l) * (u0_r - u0_l) / 6.0
275 
276  if ( exp1 .GT. exp2 ) then
277  u0_l = 3.0 * u1 - 2.0 * u0_r
278  end if
279 
280  if ( exp1 .LT. -exp2 ) then
281  u0_r = 3.0 * u1 - 2.0 * u0_l
282  end if
283 
284  ppoly%E(i1,1) = u0_l
285  ppoly%E(i1,2) = u0_r
286 
287  a = u0_l
288  b = 6.0 * u1 - 4.0 * u0_l - 2.0 * u0_r
289  c = 3.0 * ( u0_r + u0_l - 2.0 * u1 )
290 
291  ppoly%coefficients(i1,1) = a
292  ppoly%coefficients(i1,2) = b
293  ppoly%coefficients(i1,3) = c
294 
295 end subroutine ppm_boundary_extrapolation
