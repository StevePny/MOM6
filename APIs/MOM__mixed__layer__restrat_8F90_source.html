26 !* The subroutine in this file implements a parameterization of *
27 !* unresolved viscous mixed layer restratification of the mixed layer *
28 !* as described in Fox-Kemper, Ferrari and Hallberg (JPO, 2008), and *
29 !* whose impacts are described in Fox-Kemper et al. (Ocean Modelling, *
30 !* 2011). This is derived in part from the older parameterizaton *
31 !* that is described in Hallberg (Aha Hulikoa, 2003), which this new *
32 !* parameterization surpasses, which in turn is based on the *
33 !* subinertial mixed layer theory of Young (JPO, 1994). There is no *
34 !* net horizontal volume transport due to this parameterization, and *
35 !* no direct effect below the mixed layer. *
36 !* *
37 !* This parameterization, developed in detail by Fox-Kemper, et al.,*
38 !* sets the restratification timescale to agree with his high- *
39 !* resolution studies of mixed layer restratification. The run-time *
40 !* parameter FOX_KEMPER_ML_RESTRAT_COEF is a nondimensional number *
41 !* of order a few tens, proportional to the ratio of the deformation *
42 !* radius or the gridscale (whichever is smaller to the dominant *
43 !* horizontal lengthscale of the submesoscale mixed layer *
44 !* instabilities. *
45 !* *
46 !* Macros written all in capital letters are defined in MOM_memory.h. *
47 !* *
48 !* A small fragment of the grid is shown below: *
49 !* *
50 !* j+1 x ^ x ^ x At x: q *
51 !* j+1 > o > o > At ^: v, vh, vav *
52 !* j x ^ x ^ x At >: u, uh, uav *
53 !* j > o > o > At o: h *
54 !* j-1 x ^ x ^ x *
55 !* i-1 i i+1 At x & ^: *
56 !* i i+1 At > & o: *
57 !* *
58 !* The boundaries always run through q grid points (x). *
59 !* *
60 !********+*********+*********+*********+*********+*********+*********+**

91 subroutine mixedlayer_restrat(h, uhtr, vhtr, tv, fluxes, dt, G, CS)
92  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(inout) :: h
93  real, dimension(NIMEMB_,NJMEM_,NKMEM_), intent(inout) :: uhtr
94  real, dimension(NIMEM_,NJMEMB_,NKMEM_), intent(inout) :: vhtr
95  type(thermo_var_ptrs), intent(in) :: tv
96  type(forcing), intent(in) :: fluxes
97  real, intent(in) :: dt
98  type(ocean_grid_type), intent(in) :: g
99  type(mixedlayer_restrat_cs), pointer :: cs
100 ! This subroutine does interface depth diffusion. The fluxes are
101 ! limited to give positive definiteness, and the diffusivities are
102 ! limited to guarantee stability.
103 
104 ! Arguments: h - Layer thickness, in m or kg m-2. (Intent in/out.)
105 ! The units of h are referred to as H below.
106 ! (in/out) uhtr - Accumulated zonal mass fluxes in m3 or kg.
107 ! (in/out) vhtr - Accumulated meridional mass fluxes in m3 or kg.
108 ! (in) tv - A structure containing the thermobaric variables.
109 ! (in) fluxes - A structure containing pointers to any possible
110 ! forcing fields. Unused fields have NULL ptrs.
111 ! (in) dt - Time increment in s.
112 ! (in) G - The ocean's grid structure.
113 ! (in) CS - The control structure returned by a previous call to
114 ! mixedlayer_restrat_init.
115 
116  real :: uhml(szib_(g),szj_(g),szk_(g)) ! The zonal and meridional mixed layer
117  real :: vhml(szi_(g),szjb_(g),szk_(g)) ! fluxes, in m3 s-1 or kg s-1.
118  real, dimension(SZI_(G),SZJ_(G),SZK_(G)) :: &
119  h_avail ! The volume available for diffusion out of each face of each
120  ! sublayer of the mixed layer, divided by dt, in units
121  ! of H m2 s-1 (i.e., m3 s-1 or kg s-1).
122  real, dimension(SZI_(G),SZJ_(G)) :: &
123  htot, & ! The sum of the thicknesses of layers in the mixed layer, H.
124  rml_av ! g_Rho0 times the average mixed layer density, in m s-2.
125  real :: g_rho0 ! G_Earth/Rho0 in m4 s-2 kg-1.
126  real :: rho0(szi_(g)) ! Potential density relative to the surface, in kg m-3.
127  real :: p0(szi_(g)) ! A pressure of 0, in Pa.
128 
129  real :: h_vel ! htot interpolated onto velocity points, in m (not H).
130  real :: absf ! The absolute value of f, interpolated to velocity
131  ! points, in s-1.
132  real :: u_star ! The surface friction velocity, interpolated to velocity
133  ! points, in m s-1.
134  real :: mom_mixrate ! The rate at which momentum is homogenized within the
135  ! mixed layer in s-1.
136  real :: timescale ! The mixing growth timescale in s.
137  real :: h_neglect ! A thickness that is so small it is usually lost
138  ! in roundoff and can be neglected, in H.
139  real :: dz_neglect ! A thickness in m that is so small it is usually lost
140  ! in roundoff and can be neglected, in m.
141  real :: i4dt ! 1 / 4 dt
142  real :: i2htot ! Twice the total mixed layer thickness,
143  real :: z_topx2 ! depth of the top of a layer, and
144  real :: hx2 ! layer thickness, all at velocity points and in H.
145  real :: a(szk_(g)) ! A nondimensional value relating the overall flux
146  ! magnitudes (uDml & vDml) to the realized flux in a
147  ! layer. The vertical sum of a() through the pieces of
148  ! the mixed layer must be 0.
149  real :: udml(szib_(g)) ! The zonal and meridional volume fluxes in the upper
150  real :: vdml(szi_(g)) ! half of the mixed layer in H m2 s-1 (m3 s-1 or kg s-1).
151  real :: utimescale_diag(szib_(g),szj_(g)) ! The restratification timescales
152  real :: vtimescale_diag(szi_(g),szjb_(g)) ! in the zonal and meridional
153  ! directions, in s, stored in 2-D
154  ! arrays for diagnostic purposes.
155  logical :: use_eos ! If true, density is calculated from T & S using an
156  ! equation of state.
157  integer :: i, j, k, is, ie, js, je, isq, ieq, jsq, jeq, nz
158  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
159  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
160 
161  if (.not. associated(cs)) call mom_error(fatal, "MOM_mixedlayer_restrat: "// &
162  "Module must be initialized before it is used.")
163  if ((g%nkml<2) .or. (cs%ml_restrat_coef<=0.0)) return
164 
165  udml(:) = 0.0 ; vdml(:) = 0.0
166  i4dt = 0.25 / dt
167  g_rho0 = g%g_Earth/g%Rho0
168  use_eos = associated(tv%eqn_of_state)
169  h_neglect = g%H_subroundoff
170  dz_neglect = g%H_subroundoff*g%H_to_m
171 
172  if (.not.use_eos) call mom_error(fatal, "MOM_mixedlayer_restrat: "// &
173  "An equation of state must be used with this module.")
174 
175  ! Fix this later for G%nkml >= 3.
176 
177  p0(:) = 0.0
178 
179  do j=js-1,je+1
180  do i=is-1,ie+1
181  htot(i,j) = 0.0 ; rml_av(i,j) = 0.0
182  enddo
183  do k=1,g%nkml
184  call calculate_density(tv%T(:,j,k),tv%S(:,j,k),p0,rho0(:),is-1,ie-is+3,tv%eqn_of_state)
185  do i=is-1,ie+1
186  rml_av(i,j) = rml_av(i,j) + h(i,j,k)*rho0(i)
187  htot(i,j) = htot(i,j) + h(i,j,k)
188  h_avail(i,j,k) = max(i4dt*g%areaT(i,j)*(h(i,j,k)-g%Angstrom),0.0)
189  enddo
190  enddo
191 
192  do i=is-1,ie+1
193  rml_av(i,j) = (g_rho0*rml_av(i,j)) / (htot(i,j) + h_neglect)
194  enddo
195  enddo
196 
197 ! TO DO:
198 ! 1. Mixing extends below the mixing layer to the mixed layer. Find it!
199 ! 2. Add exponential tail to streamfunction?
200 
201  do j=js,je ; do i=is,ie ; utimescale_diag(i,j) = 0.0 ; enddo ; enddo
202  do j=js,je ; do i=is,ie ; vtimescale_diag(i,j) = 0.0 ; enddo ; enddo
203 
204 ! U - Component
205  do j=js,je ; do i=is-1,ie
206  h_vel = 0.5*(htot(i,j) + htot(i+1,j)) * g%H_to_m
207 
208  u_star = 0.5*(fluxes%ustar(i,j) + fluxes%ustar(i+1,j))
209  absf = 0.5*(abs(g%CoriolisBu(i,j-1)) + abs(g%CoriolisBu(i,j)))
210  ! peak ML visc: u_star * 0.41 * (h_ml*u_star)/(absf*h_ml + 4.0*u_star)
211  ! momentum mixing rate: pi^2*visc/h_ml^2
212  ! 0.41 is the von Karmen constant, 9.8696 = pi^2.
213  mom_mixrate = (0.41*9.8696)*u_star**2 / &
214  (absf*h_vel**2 + 4.0*(h_vel+dz_neglect)*u_star)
215  timescale = 0.0625 * (absf + 2.0*mom_mixrate) / (absf**2 + mom_mixrate**2)
216 
217  timescale = timescale * cs%ml_restrat_coef
218 ! timescale = timescale*(2?)*(L_def/L_MLI)*min(EKE/MKE,1.0 + G%dyCv(i,j)**2/L_def**2))
219 
220  utimescale_diag(i,j) = timescale
221 
222  udml(i) = timescale * g%mask2dCu(i,j)*g%dyCu(i,j)* &
223  g%IdxCu(i,j)*(rml_av(i+1,j)-rml_av(i,j)) * (h_vel**2 * g%m_to_H)
224 
225  if (udml(i) == 0) then
226  do k=1,g%nkml ; uhml(i,j,k) = 0.0 ; enddo
227  else
228  i2htot = 1.0 / (htot(i,j) + htot(i+1,j) + h_neglect)
229  z_topx2 = 0.0
230  ! a(k) relates the sublayer transport to uDml with a linear profile.
231  ! The sum of a through the mixed layers must be 0.
232  do k=1,g%nkml
233  hx2 = (h(i,j,k) + h(i+1,j,k) + h_neglect)
234  a(k) = (hx2 * i2htot) * (2.0 - 4.0*(z_topx2+0.5*hx2)*i2htot)
235  z_topx2 = z_topx2 + hx2
236  if (a(k)*udml(i) > 0.0) then
237  if (a(k)*udml(i) > h_avail(i,j,k)) udml(i) = h_avail(i,j,k) / a(k)
238  else
239  if (-a(k)*udml(i) > h_avail(i+1,j,k)) udml(i) = -h_avail(i+1,j,k)/a(k)
240  endif
241  enddo
242  do k=1,g%nkml
243  uhml(i,j,k) = a(k)*udml(i)
244  uhtr(i,j,k) = uhtr(i,j,k) + uhml(i,j,k)*dt
245  enddo
246  endif
247  enddo ; enddo
248 
249 ! V- component
250  do j=js-1,je ; do i=is,ie
251  h_vel = 0.5*(htot(i,j) + htot(i,j+1)) * g%H_to_m
252 
253  u_star = 0.5*(fluxes%ustar(i,j) + fluxes%ustar(i,j+1))
254  absf = 0.5*(abs(g%CoriolisBu(i-1,j)) + abs(g%CoriolisBu(i,j)))
255  ! peak ML visc: u_star * 0.41 * (h_ml*u_star)/(absf*h_ml + 4.0*u_star)
256  ! momentum mixing rate: pi^2*visc/h_ml^2
257  ! 0.41 is the von Karmen constant, 9.8696 = pi^2.
258  mom_mixrate = (0.41*9.8696)*u_star**2 / &
259  (absf*h_vel**2 + 4.0*(h_vel+dz_neglect)*u_star)
260  timescale = 0.0625 * (absf + 2.0*mom_mixrate) / (absf**2 + mom_mixrate**2)
261 
262  timescale = timescale * cs%ml_restrat_coef
263 ! timescale = timescale*(2?)*(L_def/L_MLI)*min(EKE/MKE,1.0 + G%dyCv(i,j)**2/L_def**2))
264 
265  vtimescale_diag(i,j) = timescale
266 
267  vdml(i) = timescale * g%mask2dCv(i,j)*g%dxCv(i,j)* &
268  g%IdyCv(i,j)*(rml_av(i,j+1)-rml_av(i,j)) * (h_vel**2 * g%m_to_H)
269  if (vdml(i) == 0) then
270  do k=1,g%nkml ; vhml(i,j,k) = 0.0 ; enddo
271  else
272  i2htot = 1.0 / (htot(i,j) + htot(i,j+1) + h_neglect)
273  z_topx2 = 0.0
274  ! a relates the sublayer transport to uDml with a linear profile.
275  ! The sum of a through the mixed layers must be 0.
276  do k=1,g%nkml
277  hx2 = (h(i,j,k) + h(i,j+1,k) + h_neglect)
278  a(k) = (hx2 * i2htot) * (2.0 - 4.0*(z_topx2+0.5*hx2)*i2htot)
279  z_topx2 = z_topx2 + hx2
280  if (a(k)*vdml(i) > 0.0) then
281  if (a(k)*vdml(i) > h_avail(i,j,k)) vdml(i) = h_avail(i,j,k) / a(k)
282  else
283  if (-a(k)*vdml(i) > h_avail(i,j+1,k)) vdml(i) = -h_avail(i,j+1,k)/a(k)
284  endif
285  enddo
286  do k=1,g%nkml
287  vhml(i,j,k) = a(k)*vdml(i)
288  vhtr(i,j,k) = vhtr(i,j,k) + vhml(i,j,k)*dt
289  enddo
290  endif
291  enddo ; enddo
292 
293  do k=1,g%nkml ; do j=js,je ; do i=is,ie
294  h(i,j,k) = h(i,j,k) - dt*g%IareaT(i,j) * &
295  ((uhml(i,j,k) - uhml(i-1,j,k)) + (vhml(i,j,k) - vhml(i,j-1,k)))
296  enddo ; enddo ; enddo
297 
298 ! Offer fields for averaging.
299  if (query_averaging_enabled(cs%diag) .and. &
300  ((cs%id_urestrat_time > 0) .or. (cs%id_vrestrat_time > 0))) then
301  call post_data(cs%id_urestrat_time, utimescale_diag, cs%diag)
302  call post_data(cs%id_vrestrat_time, vtimescale_diag, cs%diag)
303  endif
304  if (query_averaging_enabled(cs%diag) .and. &
305  ((cs%id_uhml>0) .or. (cs%id_vhml>0))) then
306  do k=g%nkml+1,nz
307  do j=js,je ; do i=isq,ieq ; uhml(i,j,k) = 0.0 ; enddo ; enddo
308  do j=jsq,jeq ; do i=is,ie ; vhml(i,j,k) = 0.0 ; enddo ; enddo
309  enddo
310  if (cs%id_uhml > 0) call post_data(cs%id_uhml, uhml, cs%diag)
311  if (cs%id_vhml > 0) call post_data(cs%id_vhml, vhml, cs%diag)
312  endif
313 
314 end subroutine mixedlayer_restrat
