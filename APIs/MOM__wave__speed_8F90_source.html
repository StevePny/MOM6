66 subroutine wave_speed(h, tv, G, cg1, CS, full_halos)
67  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h
68  type(thermo_var_ptrs), intent(in) :: tv
69  real, dimension(NIMEM_,NJMEM_), intent(out) :: cg1
70  type(ocean_grid_type), intent(in) :: g
71  type(wave_speed_cs), optional, pointer :: cs
72  logical, optional, intent(in) :: full_halos
73 ! This subroutine determines the first mode internal wave speed.
74 ! Arguments: h - Layer thickness, in m or kg m-2.
75 ! (in) tv - A structure containing the thermobaric variables.
76 ! (out) cg1 - The first mode internal gravity wave speed, in m s-1.
77 ! (in) G - The ocean's grid structure.
78 ! (in) CS - The control structure returned by a previous call to
79 ! wave_speed_init.
80 ! (in,opt) full_halos - If true, do the calculation over the entire
81 ! computational domain.
82 
83 ! This subroutine solves for the first baroclinic mode wave speed. (It could
84 ! solve for all the wave speeds, but the iterative approach taken here means
85 ! that this is not particularly efficient.)
86 
87 ! If e(k) is the perturbation interface height, this means solving for the
88 ! smallest eigenvalue (lam = 1/c^2) of the system
89 !
90 ! -Igu(k)*e(k-1) + (Igu(k)+Igl(k)-lam)*e(k) - Igl(k)*e(k+1) = 0.0
91 !
92 ! with rigid lid boundary conditions e(1) = e(nz+1) = 0.0 giving
93 !
94 ! (Igu(2)+Igl(2)-lam)*e(2) - Igl(2)*e(3) = 0.0
95 ! -Igu(nz)*e(nz-1) + (Igu(nz)+Igl(nz)-lam)*e(nz) = 0.0
96 !
97 ! Here
98 ! Igl(k) = 1.0/(gprime(k)*H(k)) ; Igu(k) = 1.0/(gprime(k)*H(k-1))
99 !
100 ! Alternately, these same eigenvalues can be found from the second smallest
101 ! eigenvalue of the Montgomery potential (M(k)) calculation:
102 !
103 ! -Igl(k)*M(k-1) + (Igl(k)+Igu(k+1)-lam)*M(k) - Igu(k+1)*M(k+1) = 0.0
104 !
105 ! with rigid lid and flat bottom boundary conditions
106 !
107 ! (Igu(2)-lam)*M(1) - Igu(2)*M(2) = 0.0
108 ! -Igl(nz)*M(nz-1) + (Igl(nz)-lam)*M(nz) = 0.0
109 !
110 ! Note that the barotropic mode has been eliminated from the rigid lid
111 ! interface height equations, hence the matrix is one row smaller. Without
112 ! the rigid lid, the top boundary condition is simpler to implement with
113 ! the M equations.
114 
115  real, dimension(SZK_(G)+1) :: &
116  drho_dt, drho_ds, &
117  pres, t_int, s_int, &
118  gprime ! The reduced gravity across each interface, in m s-2.
119  real, dimension(SZK_(G)) :: &
120  igl, igu ! The inverse of the reduced gravity across an interface times
121  ! the thickness of the layer below (Igl) or above (Igu) it,
122  ! in units of s2 m-2.
123  real, dimension(SZK_(G),SZI_(G)) :: &
124  hf, tf, sf, rf
125  real, dimension(SZK_(G)) :: &
126  hc, tc, sc, rc, &
127  det, ddet
128  real :: lam, dlam, lam0
129  real :: min_h_frac
130  real :: h_to_pres
131  real, dimension(SZI_(G)) :: &
132  htot, hmin, & ! Thicknesses in m.
133  h_here, hxt_here, hxs_here, hxr_here
134  real :: speed2_tot
135  real :: i_hnew, drxh_sum
136  real, parameter :: tol1 = 0.0001, tol2 = 0.001
137  real, pointer, dimension(:,:,:) :: t, s
138  real :: g_rho0 ! G_Earth/Rho0 in m4 s-2 kg-1.
139  real :: rescale, i_rescale
140  integer :: kf(szi_(g))
141  integer, parameter :: max_itt = 10
142  real :: lam_it(max_itt), det_it(max_itt), ddet_it(max_itt)
143  logical :: use_eos ! If true, density is calculated from T & S using an
144  ! equation of state.
145  integer :: kc
146  integer :: i, j, k, k2, itt, is, ie, js, je, nz
147 
148  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
149 
150  if (present(cs)) then
151  if (.not. associated(cs)) call mom_error(fatal, "MOM_wave_speed: "// &
152  "Module must be initialized before it is used.")
153  endif
154  if (present(full_halos)) then ; if (full_halos) then
155  is = g%isd ; ie = g%ied ; js = g%jsd ; je = g%jed
156  endif ; endif
157 
158  s => tv%S ; t => tv%T
159  g_rho0 = g%g_Earth/g%Rho0
160  use_eos = associated(tv%eqn_of_state)
161 
162  h_to_pres = g%g_Earth * g%Rho0
163  rescale = 1024.0**4 ; i_rescale = 1.0/rescale
164 
165  min_h_frac = tol1 / real(nz)
166 
167  do j=js,je
168  ! First merge very thin layers with the one above (or below if they are
169  ! at the top). This also transposes the row order so that columns can
170  ! be worked upon one at a time.
171  do i=is,ie ; htot(i) = 0.0 ; enddo
172  do k=1,nz ; do i=is,ie ; htot(i) = htot(i) + h(i,j,k)*g%H_to_m ; enddo ; enddo
173 
174  do i=is,ie
175  hmin(i) = htot(i)*min_h_frac ; kf(i) = 1 ; h_here(i) = 0.0
176  hxt_here(i) = 0.0 ; hxs_here(i) = 0.0 ; hxr_here(i) = 0.0
177  enddo
178  if (use_eos) then
179  do k=1,nz ; do i=is,ie
180  if ((h_here(i) > hmin(i)) .and. (h(i,j,k)*g%H_to_m > hmin(i))) then
181  hf(kf(i),i) = h_here(i)
182  tf(kf(i),i) = hxt_here(i) / h_here(i)
183  sf(kf(i),i) = hxs_here(i) / h_here(i)
184  kf(i) = kf(i) + 1
185 
186  ! Start a new layer
187  h_here(i) = h(i,j,k)*g%H_to_m
188  hxt_here(i) = (h(i,j,k)*g%H_to_m)*t(i,j,k)
189  hxs_here(i) = (h(i,j,k)*g%H_to_m)*s(i,j,k)
190  else
191  h_here(i) = h_here(i) + h(i,j,k)*g%H_to_m
192  hxt_here(i) = hxt_here(i) + (h(i,j,k)*g%H_to_m)*t(i,j,k)
193  hxs_here(i) = hxs_here(i) + (h(i,j,k)*g%H_to_m)*s(i,j,k)
194  endif
195  enddo ; enddo
196  do i=is,ie ; if (h_here(i) > 0.0) then
197  hf(kf(i),i) = h_here(i)
198  tf(kf(i),i) = hxt_here(i) / h_here(i)
199  sf(kf(i),i) = hxs_here(i) / h_here(i)
200  endif ; enddo
201  else
202  do k=1,nz ; do i=is,ie
203  if ((h_here(i) > hmin(i)) .and. (h(i,j,k)*g%H_to_m > hmin(i))) then
204  hf(kf(i),i) = h_here(i) ; rf(kf(i),i) = hxr_here(i) / h_here(i)
205  kf(i) = kf(i) + 1
206 
207  ! Start a new layer
208  h_here(i) = h(i,j,k)*g%H_to_m
209  hxr_here(i) = (h(i,j,k)*g%H_to_m)*g%Rlay(k)
210  else
211  h_here(i) = h_here(i) + h(i,j,k)*g%H_to_m
212  hxr_here(i) = hxr_here(i) + (h(i,j,k)*g%H_to_m)*g%Rlay(k)
213  endif
214  enddo ; enddo
215  do i=is,ie ; if (h_here(i) > 0.0) then
216  hf(kf(i),i) = h_here(i) ; rf(kf(i),i) = hxr_here(i) / h_here(i)
217  endif ; enddo
218  endif
219 
220  ! From this point, we can work on individual columns without causing memory
221  ! to have page faults.
222  do i=is,ie ; if (g%mask2dT(i,j) > 0.5) then
223  if (use_eos) then
224  pres(1) = 0.0
225  do k=2,kf(i)
226  pres(k) = pres(k-1) + h_to_pres*hf(k-1,i)
227  t_int(k) = 0.5*(tf(k,i)+tf(k-1,i))
228  s_int(k) = 0.5*(sf(k,i)+sf(k-1,i))
229  enddo
230  call calculate_density_derivs(t_int, s_int, pres, drho_dt, drho_ds, 2, &
231  kf(i)-1, tv%eqn_of_state)
232 
233  ! Sum the reduced gravities to find out how small a density difference
234  ! is negligibly small.
235  drxh_sum = 0.0
236  do k=2,kf(i)
237  drxh_sum = drxh_sum + 0.5*(hf(k-1,i)+hf(k,i)) * &
238  max(0.0,drho_dt(k)*(tf(k,i)-tf(k-1,i)) + &
239  drho_ds(k)*(sf(k,i)-sf(k-1,i)))
240  enddo
241  else
242  drxh_sum = 0.0
243  do k=2,kf(i)
244  drxh_sum = drxh_sum + 0.5*(hf(k-1,i)+hf(k,i)) * &
245  max(0.0,rf(k,i)-rf(k-1,i))
246  enddo
247  endif
248  ! Find gprime across each internal interface, taking care of convective
249  ! instabilities by merging layers.
250  if (drxh_sum <= 0.0) then
251  cg1(i,j) = 0.0
252  else
253  ! Merge layers to eliminate convective instabilities or exceedingly
254  ! small reduced gravities.
255  if (use_eos) then
256  kc = 1
257  hc(1) = hf(1,i) ; tc(1) = tf(1,i) ; sc(1) = sf(1,i)
258  do k=2,kf(i)
259  if ((drho_dt(k)*(tf(k,i)-tc(kc)) + drho_ds(k)*(sf(k,i)-sc(kc))) * &
260  (hc(kc) + hf(k,i)) < 2.0 * tol2*drxh_sum) then
261  ! Merge this layer with the one above and backtrack.
262  i_hnew = 1.0 / (hc(kc) + hf(k,i))
263  tc(kc) = (hc(kc)*tc(kc) + hf(k,i)*tf(k,i)) * i_hnew
264  sc(kc) = (hc(kc)*sc(kc) + hf(k,i)*sf(k,i)) * i_hnew
265  hc(kc) = (hc(kc) + hf(k,i))
266  ! Backtrack to remove any convective instabilities above... Note
267  ! that the tolerance is a factor of two larger, to avoid limit how
268  ! far back we go.
269  do k2=kc,2,-1
270  if ((drho_dt(k2)*(tc(k2)-tc(k2-1)) + drho_ds(k2)*(sc(k2)-sc(k2-1))) * &
271  (hc(k2) + hc(k2-1)) < tol2*drxh_sum) then
272  ! Merge the two bottommost layers. At this point kc = k2.
273  i_hnew = 1.0 / (hc(kc) + hc(kc-1))
274  tc(kc-1) = (hc(kc)*tc(kc) + hc(kc-1)*tc(kc-1)) * i_hnew
275  sc(kc-1) = (hc(kc)*sc(kc) + hc(kc-1)*sc(kc-1)) * i_hnew
276  hc(kc-1) = (hc(kc) + hc(kc-1))
277  kc = kc - 1
278  else ; exit ; endif
279  enddo
280  else
281  ! Add a new layer to the column.
282  kc = kc + 1
283  drho_ds(kc) = drho_ds(k) ; drho_dt(kc) = drho_dt(k)
284  tc(kc) = tf(k,i) ; sc(kc) = sf(k,i) ; hc(kc) = hf(k,i)
285  endif
286  enddo
287  ! At this point there are kc layers and the gprimes should be positive.
288  do k=2,kc ! Revisit this if non-Boussinesq.
289  gprime(k) = g_rho0 * (drho_dt(k)*(tc(k)-tc(k-1)) + &
290  drho_ds(k)*(sc(k)-sc(k-1)))
291  enddo
292  else ! .not.use_EOS
293  ! Do the same with density directly...
294  kc = 1
295  hc(1) = hf(1,i) ; rc(1) = rf(1,i)
296  do k=2,kf(i)
297  if ((rf(k,i) - rc(kc)) * (hc(kc) + hf(k,i)) < 2.0*tol2*drxh_sum) then
298  ! Merge this layer with the one above and backtrack.
299  rc(kc) = (hc(kc)*rc(kc) + hf(k,i)*rf(k,i)) / (hc(kc) + hf(k,i))
300  hc(kc) = (hc(kc) + hf(k,i))
301  ! Backtrack to remove any convective instabilities above... Note
302  ! that the tolerance is a factor of two larger, to avoid limit how
303  ! far back we go.
304  do k2=kc,2,-1
305  if ((rc(k2)-rc(k2-1)) * (hc(k2)+hc(k2-1)) < tol2*drxh_sum) then
306  ! Merge the two bottommost layers. At this point kc = k2.
307  rc(kc-1) = (hc(kc)*rc(kc) + hc(kc-1)*rc(kc-1)) / (hc(kc) + hc(kc-1))
308  hc(kc-1) = (hc(kc) + hc(kc-1))
309  kc = kc - 1
310  else ; exit ; endif
311  enddo
312  else
313  ! Add a new layer to the column.
314  kc = kc + 1
315  rc(kc) = rf(k,i) ; hc(kc) = hf(k,i)
316  endif
317  enddo
318  ! At this point there are kc layers and the gprimes should be positive.
319  do k=2,kc ! Revisit this if non-Boussinesq.
320  gprime(k) = g_rho0 * (rc(k)-rc(k-1))
321  enddo
322  endif ! use_EOS
323 
324  ! Sum the contributions from all of the interfaces to give an over-estimate
325  ! of the first-mode wave speed.
326  if (kc >= 2) then
327  speed2_tot = 0.0
328  do k=2,kc
329  igl(k) = 1.0/(gprime(k)*hc(k)) ; igu(k) = 1.0/(gprime(k)*hc(k-1))
330  speed2_tot = speed2_tot + gprime(k)*(hc(k-1)+hc(k))
331  enddo
332 
333  ! Overestimate the speed to start with.
334  lam0 = 1.0 / speed2_tot ; lam = lam0
335  ! Find the determinant and its derivative with lam.
336  do itt=1,max_itt
337  lam_it(itt) = lam
338  det(1) = 1.0 ; ddet(1) = 0.0
339  det(2) = (igu(2)+igl(2)-lam) ; ddet(2) = -1.0
340  do k=3,kc
341  det(k) = (igu(k)+igl(k)-lam)*det(k-1) - (igu(k)*igl(k-1))*det(k-2)
342  ddet(k) = (igu(k)+igl(k)-lam)*ddet(k-1) - (igu(k)*igl(k-1))*ddet(k-2) - &
343  det(k-1)
344 
345  ! Rescale det & ddet if det is getting too large.
346  if (abs(det(k)) > rescale) then
347  det(k) = i_rescale*det(k) ; det(k-1) = i_rescale*det(k-1)
348  ddet(k) = i_rescale*ddet(k) ; ddet(k-1) = i_rescale*ddet(k-1)
349  endif
350  enddo
351  ! Use Newton's method iteration to find a new estimate of lam.
352  det_it(itt) = det(kc) ; ddet_it(itt) = ddet(kc)
353 
354  if ((ddet(kc) >= 0.0) .or. (-det(kc) > -0.5*lam*ddet(kc))) then
355  ! lam was not an under-estimate, as intended, so Newton's method
356  ! may not be reliable; lam must be reduced, but not by more
357  ! than half.
358  lam = 0.5 * lam
359  else ! Newton's method is OK.
360  dlam = - det(kc) / ddet(kc)
361  lam = lam + dlam
362  if (abs(dlam) < tol2*lam) exit
363  endif
364  enddo
365 
366  cg1(i,j) = 0.0
367  if (lam > 0.0) cg1(i,j) = 1.0 / sqrt(lam)
368  else
369  cg1(i,j) = 0.0
370  endif
371 
372  endif ! cg1 /= 0.0
373  else
374  cg1(i,j) = 0.0 ! This is a land point.
375  endif ; enddo ! i-loop
376  enddo ! j-loop
377 
378 end subroutine wave_speed
