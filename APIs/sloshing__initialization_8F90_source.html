86 ! Initialization of thicknesses
87 !------------------------------------------------------------------------------
88 subroutine sloshing_initialize_thickness ( h, G, param_file )
89 
90  real, intent(out), dimension(NIMEM_,NJMEM_, NKMEM_) :: h
91  type(ocean_grid_type), intent(in) :: g
92  type(param_file_type), intent(in) :: param_file
93 
94 ! This routine is called when THICKNESS_CONFIG is set to 'sloshing'
95 !
96 ! This routine initializes layer positions to set off a sloshing motion in
97 ! the zonal direction in a rectangular basin. All layers have initially the
98 ! same thickness but all interfaces (except bottom and sea surface) are
99 ! displaced according to a half-period cosine, with maximum value on the
100 ! left and minimum value on the right. This sets off a regular sloshing motion.
101  real :: displ(szk_(g)+1)
102  real :: z_unif(szk_(g)+1)
103  real :: z_inter(szk_(g)+1)
104  real :: x
105  real :: a0
106  real :: deltah
107  real :: total_height
108  real :: weight_z
109  real :: x1, y1, x2, y2
110  real :: t
111  integer :: n
112 
113  integer :: i, j, k, is, ie, js, je, nx, nz
114 
115  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
116 
117  deltah = g%max_depth / nz
118 
119  ! Define thicknesses
120  do j=g%jsc,g%jec ; do i=g%isc,g%iec
121 
122  ! Define uniform interfaces
123  do k = 0,nz
124  z_unif(k+1) = -real(k)/real(nz)
125  end do
126 
127  ! 1. Define stratification
128  n = 3
129  do k = 1,nz+1
130 
131  ! Thin pycnocline in the middle
132  !z_inter(k) = (2.0**(n-1)) * (z_unif(k) + 0.5)**n - 0.5
133 
134  ! Thin pycnocline in the middle (piecewise linear profile)
135  x1 = 0.30; y1 = 0.48; x2 = 0.70; y2 = 0.52
136 
137  x = -z_unif(k)
138 
139  if ( x .le. x1 ) then
140  t = y1*x/x1;
141  else if ( (x .gt. x1 ) .and. ( x .lt. x2 )) then
142  t = y1 + (y2-y1) * (x-x1) / (x2-x1)
143  else
144  t = y2 + (1.0-y2) * (x-x2) / (1.0-x2)
145  end if
146 
147  t = - z_unif(k)
148 
149  z_inter(k) = -t * g%max_depth
150 
151  end do
152 
153  ! 2. Define displacement
154  a0 = 75.0; ! Displacement amplitude (meters)
155  do k = 1,nz+1
156 
157  weight_z = - 4.0 * ( z_unif(k) + 0.5 )**2 + 1
158 
159  x = g%geoLonT(i,j) / g%len_lon
160  displ(k) = a0 * cos(acos(-1.0)*x) + weight_z;
161 
162  if ( k .EQ. 1 ) then
163  displ(k) = 0.0
164  end if
165 
166  if ( k .EQ. nz+1 ) then
167  displ(k) = 0.0
168  end if
169 
170  z_inter(k) = z_inter(k) + displ(k)
171 
172  end do
173 
174  ! 3. The last interface must coincide with the seabed
175  z_inter(nz+1) = -g%bathyT(i,j)
176 
177  ! Modify interface heights to make sure all thicknesses
178  ! are strictly positive
179  do k = nz,1,-1
180 
181  if ( z_inter(k) .LT. (z_inter(k+1) + g%Angstrom) ) then
182  z_inter(k) = z_inter(k+1) + g%Angstrom
183  end if
184 
185  end do
186 
187  ! 4. Define layers
188  total_height = 0.0
189  do k = 1,nz
190  h(i,j,k) = z_inter(k) - z_inter(k+1)
191 
192  total_height = total_height + h(i,j,k)
193  end do
194 
195  enddo ; enddo
196 
197 end subroutine sloshing_initialize_thickness
