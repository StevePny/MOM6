241 !------------------------------------------------------------------------------
242 ! Dispatching regridding routine: regridding & remapping
243 !------------------------------------------------------------------------------
244 subroutine regridding_main( remapCS, CS, G, h, tv, dzInterface )
245 !------------------------------------------------------------------------------
246 ! This routine takes care of (1) building a new grid and (2) remapping between
247 ! the old grid and the new grid. The creation of the new grid can be based
248 ! on z coordinates, target interface densities, sigma coordinates or any
249 ! arbitrary coordinate system.
250 ! The MOM6 interface positions are always calculated from the bottom up by
251 ! accumulating the layer thicknessea starting at z=-G%bathyT. z increases
252 ! upwards (decreasing k-index).
253 ! The new grid is defined by the change in position of those interfaces in z
254 ! dzInterface = zNew - zOld.
255 ! Thus, if the regridding inflates the top layer, hNew(1) > hOld(1), then the
256 ! second interface moves downward, zNew(2) < zOld(2), and dzInterface(2) < 0.
257 ! hNew(k) = hOld(k) - dzInterface(k+1) + dzInterface(k)
258 ! IMPORTANT NOTE:
259 ! This is the converse of the sign convention used in the remapping code!
260 !------------------------------------------------------------------------------
261 
262  ! Arguments
263  type(remapping_cs), intent(inout) :: remapcs ! Remapping parameters and options
264  type(regridding_cs), intent(inout) :: cs ! Regridding parameters and options
265  type(ocean_grid_type), intent(in) :: g ! Ocean grid informations
266  real, dimension(NIMEM_,NJMEM_, NKMEM_), intent(inout) :: h ! Current 3D grid obtained after the last time step
267  type(thermo_var_ptrs), intent(inout) :: tv ! Thermodynamical variables (T, S, ...)
268  real, dimension(NIMEM_,NJMEM_, NK_INTERFACE_), intent(inout) :: dzinterface ! The change in position of each interface
269 
270  real :: trickgnucompiler
271 
272  ! Local variables
273 
274  select case ( cs%regridding_scheme )
275 
276  case ( regridding_zstar )
277  call buildgridzstar( cs, g, h, dzinterface )
278 
279  case ( regridding_sigma )
280  call buildgridsigma( cs, g, h, dzinterface )
281 
282  case ( regridding_rho )
283  call convective_adjustment(g, h, tv)
284  call buildgridrho( g, h, tv, dzinterface, remapcs, cs )
285 
286  case ( regridding_arbitrary )
287  call build_grid_arbitrary( g, h, dzinterface, trickgnucompiler, cs )
288 
289  case default
290  call mom_error(fatal,'MOM_regridding, regridding_main: '//&
291  'Unknown regridding scheme selected!')
292 
293  end select ! type of grid
294 
295 #ifdef __DO_SAFTEY_CHECKS__
296  call checkgridsmatch(g, h, dzinterface)
297 #endif
298 
299 end subroutine regridding_main

307 ! This routine calculates the total thickness of
308 !------------------------------------------------------------------------------
309 
310  ! Arguments

312  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h

314

322

326  ! Total thickness of grid h

352  write(0,*) 'i,j,nz=',i,j,nz
353  do k = 1,nz
354  write(0,*) 'k,h,hnew=',k,h(i,j,k),h(i,j,k)+(dzinterface(i,j,k)-dzinterface(i,j,k+1))
355  enddo
356  write(0,*) 'Hold,Hnew,Hnew-Hold=',totalhold,totalhnewf,totalhnewf-totalhold

375 ! Build uniform z*-ccordinate grid with partial steps
376 !------------------------------------------------------------------------------
377 subroutine buildgridzstar( CS, G, h, dzInterface )
378 !------------------------------------------------------------------------------
379 ! This routine builds a grid where the distribution of levels is based on a
380 ! z* coordinate system with partial steps (Adcroft and Campin, 2004).
381 ! The module parameter coordinateResolution(:) determines the nominal
382 ! resolution, dz*(:), in the absence of topography. z* is defined by
383 ! z* = (z-eta)/(H+eta)*H s.t. z*=0 when z=eta and z*=-H when z=-H .
384 !------------------------------------------------------------------------------
385 
386  ! Arguments
387  type(regridding_cs), intent(in) :: cs
388  type(ocean_grid_type), intent(in) :: g
389  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h
390  real, dimension(NIMEM_,NJMEM_, NK_INTERFACE_), intent(inout) :: dzinterface
391 
392  ! Local variables
393  integer :: i, j, k
394  integer :: nz
395  real :: nominaldepth, totalthickness, eta, stretching, dh
396  real, dimension(SZK_(G)+1) :: zold, znew
397  real :: minthickness
398 
399  nz = g%ke
400 
401 
402  do j = g%jsc-1,g%jec+1
403  do i = g%isc-1,g%iec+1
404 
405  if (g%mask2dT(i,j)==0.) then
406  dzinterface(i,j,:) = 0.
407  cycle
408  endif
409 
410  ! Local depth (G%bathyT is positive)
411  nominaldepth = g%bathyT(i,j)
412 
413  ! Determine water column thickness
414  totalthickness = 0.0
415  do k = 1,nz
416  totalthickness = totalthickness + h(i,j,k)
417  end do
418  minthickness = min( cs%min_thickness, totalthickness/float(nz) )
419 
420  ! Position of free-surface
421  eta = totalthickness - nominaldepth
422 
423  ! z* = (z-eta) / stretching where stretching = (H+eta)/H
424  ! z = eta + stretching * z*
425  stretching = totalthickness / nominaldepth
426 
427  ! Integrate down from the top for a notional new grid, ignoring topography
428  znew(1) = eta
429  do k = 1,nz
430  dh = stretching * cs%coordinateResolution(k) ! Notional grid spacing
431  znew(k+1) = znew(k) - dh
432  enddo
433 
434  ! The rest of the model defines grids integrating up from the bottom
435  zold(nz+1) = - nominaldepth
436  znew(nz+1) = - nominaldepth
437  do k = nz,1,-1
438  ! Adjust interface position to accomodate inflating layers
439  ! without disturbing the interface above
440  if ( znew(k) < (znew(k+1) + minthickness) ) then
441  znew(k) = znew(k+1) + minthickness
442  endif
443  zold(k) = zold(k+1) + h(i,j,k)
444  enddo
445 
446  ! Define regridding in terms of a movement of interfaces
447  dzinterface(i,j,1) = 0.
448  do k = 2,nz
449  dzinterface(i,j,k) = znew(k) - zold(k)
450  enddo
451  dzinterface(i,j,nz+1) = 0.
452 
453 #ifdef __DO_SAFTEY_CHECKS__
454  dh=max(nominaldepth,totalthickness)
455  if (abs(znew(1)-zold(1))>(nz-1)*0.5*epsilon(eta)*dh) then
456  write(0,*) 'min_thickness=',minthickness
457  write(0,*) 'eta=',eta,'nominalDepth=',nominaldepth,'totalThickness=',totalthickness
458  write(0,*) 'dzInterface(1) = ',dzinterface(i,j,1),epsilon(eta),nz
459  do k=1,nz+1
460  write(0,*) k,zold(k),znew(k)
461  enddo
462  write(0,*) 'stretching=',stretching
463  do k=1,nz
464  write(0,*) k,h(i,j,k),znew(k)-znew(k+1),stretching * cs%coordinateResolution(k),cs%coordinateResolution(k)
465  enddo
466  call mom_error( fatal, &
467  'MOM_regridding, buildGridZstar: top surface has moved!!!' )
468  endif
469  dzinterface(i,j,1) = 0.
470  dzinterface(i,j,nz+1) = 0.
471 #endif
472 
473  end do
474  end do
475 
476 end subroutine buildgridzstar

488 !------------------------------------------------------------------------------

492  type(ocean_grid_type), intent(in) :: g
493  real, dimension(NIMEM_,NJMEM_, NKMEM_), intent(in) :: h
494  real, dimension(NIMEM_,NJMEM_, NK_INTERFACE_), intent(inout) :: dzinterface
495

501

559 !------------------------------------------------------------------------------
560 ! Build grid based on target interface densities
561 !------------------------------------------------------------------------------
562 subroutine buildgridrho( G, h, tv, dzInterface, remapCS, CS )
563 !------------------------------------------------------------------------------
564 ! This routine builds a new grid based on a given set of target interface
565 ! densities (these target densities are computed by taking the mean value
566 ! of given layer densities). The algorithn operates as follows within each
567 ! column:
568 ! 1. Given T & S within each layer, the layer densities are computed.
569 ! 2. Based on these layer densities, a global density profile is reconstructed
570 ! (this profile is monotonically increasing and may be discontinuous)
571 ! 3. The new grid interfaces are determined based on the target interface
572 ! densities.
573 ! 4. T & S are remapped onto the new grid.
574 ! 5. Return to step 1 until convergence or until the maximum number of
575 ! iterations is reached, whichever comes first.
576 !------------------------------------------------------------------------------
577 
578  ! Arguments
579  type(ocean_grid_type), intent(in) :: g
580  real, dimension(NIMEM_,NJMEM_, NKMEM_), intent(in) :: h
581  type(thermo_var_ptrs), intent(in) :: tv
582  real, dimension(NIMEM_,NJMEM_, NK_INTERFACE_), intent(inout) :: dzinterface
583  type(remapping_cs), intent(inout) :: remapcs
584  type(regridding_cs), intent(inout) :: cs
585 
586  ! Local variables
587  integer :: i, j, k, m
588  integer :: map_index
589  integer :: nz
590  integer :: k_found
591  integer :: count_nonzero_layers
592  real :: deviation ! When iterating to determine the final
593  ! grid, this is the deviation between two
594  ! successive grids.
595  real :: threshold
596  real :: max_thickness
597  real :: correction
598 
599  real, dimension(SZK_(G)) :: p_column, densities, t_column, s_column, tmp_column
600  integer, dimension(SZK_(G)) :: mapping
601  real :: nominaldepth, totalthickness, dh
602  real, dimension(SZK_(G)+1) :: zold, znew
603  real, dimension(SZK_(G)) :: h0, h1, htmp
604  real, dimension(SZK_(G)+1) :: x0, x1, xtmp, dx
605 
606  nz = g%ke
607  threshold = cs%min_thickness
608  p_column(:) = 0.
609 
610  ! Prescribe target values
611  cs%coordinateInterfaces(1) = g%Rlay(1)+0.5*(g%Rlay(1)-g%Rlay(2))
612  cs%coordinateInterfaces(nz+1) = g%Rlay(nz)+0.5*(g%Rlay(nz)-g%Rlay(nz-1))
613  do k = 2,nz
614  cs%coordinateInterfaces(k) = cs%coordinateInterfaces(k-1) + cs%coordinateResolution(k)
615  end do
616 
617  ! Build grid based on target interface densities
618  do i = g%isc-1,g%iec+1
619  do j = g%jsc-1,g%jec+1
620 
621  ! Copy T and S onto new variables so as to not alter the original values
622  ! of T and S (these are remapped at the end of the regridding iterations
623  ! once the final grid has been determined).
624  t_column = tv%T(i,j,:)
625  s_column = tv%S(i,j,:)
626 
627  ! Copy original grid
628  h0(1:nz) = h(i,j,1:nz)
629  x0(1) = 0.0
630  do k = 1,nz
631  x0(k+1) = x0(k) + h0(k)
632  end do
633 
634  ! Start iterations to build grid
635  m = 1
636  deviation = 1e10
637  do while ( ( m .le. nb_regridding_iterations ) .and. &
638  ( deviation .gt. deviation_tolerance ) )
639 
640  ! Count number of nonzero layers within current water column
641  count_nonzero_layers = 0
642  do k = 1,nz
643  if ( h0(k) .gt. threshold ) then
644  count_nonzero_layers = count_nonzero_layers + 1
645  end if
646  end do
647 
648  ! If there is at most one nonzero layer, stop here (no regridding)
649  if ( count_nonzero_layers .le. 1 ) then
650  h1(1:nz) = h0(1:nz)
651  exit ! stop iterations here
652  end if
653 
654  ! Build new grid containing only nonzero layers
655  map_index = 1
656  correction = 0.0
657  do k = 1,nz
658  if ( h0(k) .gt. threshold ) then
659  mapping(map_index) = k
660  htmp(map_index) = h0(k)
661  map_index = map_index + 1
662  else
663  correction = correction + h0(k)
664  end if
665  end do
666 
667  max_thickness = htmp(1)
668  k_found = 1
669  do k = 1,count_nonzero_layers
670  if ( htmp(k) .gt. max_thickness ) then
671  max_thickness = htmp(k)
672  k_found = k
673  end if
674  end do
675 
676  htmp(k_found) = htmp(k_found) + correction
677 
678  xtmp(1) = 0.0
679  do k = 1,count_nonzero_layers
680  xtmp(k+1) = xtmp(k) + htmp(k)
681  end do
682 
683 
684  ! Compute densities within current water column
685  call calculate_density( t_column, s_column, p_column, densities,&
686  1, nz, tv%eqn_of_state )
687 
688  do k = 1,count_nonzero_layers
689  densities(k) = densities(mapping(k))
690  end do
691 
692  ! One regridding iteration
693  call regridding_iteration( densities, cs%coordinateInterfaces, cs,&
694  count_nonzero_layers, htmp, xtmp, &
695  cs%ppoly_i, nz, h1, x1 )
696 
697  ! Remap T and S from previous grid to new grid
698  do k = 1,nz
699  h0(k) = x0(k+1) - x0(k)
700  h1(k) = x1(k+1) - x1(k)
701  end do
702  dx(:) = x1(:) - x0(:)
703  dx(1) = 0.
704  dx(nz+1) = 0.
705 
706  call remapping_core(remapcs, nz, h0, s_column, nz, dx, tmp_column)
707  s_column(:) = tmp_column(:)
708 
709  call remapping_core(remapcs, nz, h0, t_column, nz, dx, tmp_column)
710  t_column(:) = tmp_column(:)
711 
712  ! Compute the deviation between two successive grids
713  deviation = 0.0
714  do k = 2,nz
715  deviation = deviation + (x0(k)-x1(k))**2
716  end do
717  deviation = sqrt( deviation / (nz-1) )
718 
719 
720  m = m + 1
721 
722  ! Copy final grid onto start grid for next iteration
723  x0(:) = x1(:)
724  h0(:) = h1(:)
725 
726  end do ! end regridding iterations
727 
728  ! Local depth (G%bathyT is positive)
729  nominaldepth = g%bathyT(i,j)
730 
731  ! The rest of the model defines grids integrating up from the bottom
732  totalthickness = 0.0
733  zold(nz+1) = - nominaldepth
734  znew(nz+1) = - nominaldepth
735  do k = nz,1,-1
736  totalthickness = totalthickness + h(i,j,k)
737  znew(k) = znew(k+1) + h1(k)
738  ! Adjust interface position to accomodate inflating layers
739  ! without disturbing the interface above
740  ! if ( zNew(k) < (zNew(k+1) + CS%min_thickness) ) then
741  ! zNew(k) = zNew(k+1) + CS%min_thickness
742  ! endif
743  zold(k) = zold(k+1) + h(i,j,k)
744  enddo
745 
746  ! Define regridding in terms of a movement of interfaces
747  dzinterface(i,j,1) = 0.
748  do k = 2,nz
749  dzinterface(i,j,k) = znew(k) - zold(k)
750 #ifdef __DO_SAFTEY_CHECKS__
751  if (znew(k) > zold(1)) then
752  write(0,*) 'zOld=',zold
753  write(0,*) 'zNew=',znew
754  call mom_error( fatal, 'MOM_regridding, buildGridRho: '//&
755  'interior interface above surface!' )
756  endif
757  if (znew(k) > znew(k-1)) then
758  write(0,*) 'zOld=',zold
759  write(0,*) 'zNew=',znew
760  call mom_error( fatal, 'MOM_regridding, buildGridRho: '//&
761  'interior interfaces cross!' )
762  endif
763 #endif
764  enddo
765  dzinterface(i,j,nz+1) = 0.
766 
767 #ifdef __DO_SAFTEY_CHECKS__
768  dh=max(nominaldepth,totalthickness)
769  if (abs(znew(1)-zold(1))>(nz-1)*0.5*epsilon(dh)*dh) then
770  write(0,*) 'min_thickness=',cs%min_thickness
771  write(0,*) 'nominalDepth=',nominaldepth,'totalThickness=',totalthickness
772  write(0,*) 'zNew(1)-zOld(1) = ',znew(1)-zold(1),epsilon(dh),nz
773  do k=1,nz+1
774  write(0,*) k,zold(k),znew(k)
775  enddo
776  do k=1,nz
777  write(0,*) k,h(i,j,k),znew(k)-znew(k+1)
778  enddo
779  call mom_error( fatal, &
780  'MOM_regridding, buildGridRho: top surface has moved!!!' )
781  endif
782 #endif
783 
784  end do ! end loop on j
785  end do ! end loop on i
786 
787 end subroutine buildgridrho

790 !------------------------------------------------------------------------------
791 ! Build arbitrary grid
792 !------------------------------------------------------------------------------
793 subroutine build_grid_arbitrary( G, h, dzInterface, h_new, CS )
794 !------------------------------------------------------------------------------
795 ! This routine builds a grid based on arbitrary rules
796 !------------------------------------------------------------------------------
797 
798  ! Arguments
799  type(ocean_grid_type), intent(in) :: g
800  real, dimension(NIMEM_,NJMEM_, NKMEM_), intent(in) :: h
801  real, dimension(NIMEM_,NJMEM_, NK_INTERFACE_), intent(inout) :: dzinterface
802  real, intent(inout) :: h_new
803  type(regridding_cs), intent(in) :: cs
804 
805  ! Local variables
806  integer :: i, j, k
807  integer :: nz
808  real :: z_inter(szk_(g)+1)
809  real :: total_height
810  real :: delta_h
811  real :: max_depth
812  real :: min_thickness
813  real :: eta ! local elevation
814  real :: local_depth
815  real :: x1, y1, x2, y2
816  real :: x, t
817 
818  nz = g%ke
819 
820  max_depth = g%max_depth
821  min_thickness = cs%min_thickness
822 
823  do j = g%jsc-1,g%jec+1
824  do i = g%isc-1,g%iec+1
825 
826  ! Local depth
827  local_depth = g%bathyT(i,j)
828 
829  ! Determine water column height
830  total_height = 0.0
831  do k = 1,nz
832  total_height = total_height + h(i,j,k)
833  end do
834 
835  eta = total_height - local_depth
836 
837  ! Compute new thicknesses based on stretched water column
838  delta_h = (max_depth + eta) / nz
839 
840  ! Define interfaces
841  z_inter(1) = eta
842  do k = 1,nz
843  z_inter(k+1) = z_inter(k) - delta_h
844  end do
845 
846  ! Refine grid in the middle
847  do k = 1,nz+1
848  x1 = 0.35; y1 = 0.45; x2 = 0.65; y2 = 0.55
849 
850  x = - ( z_inter(k) - eta ) / max_depth
851 
852  if ( x .le. x1 ) then
853  t = y1*x/x1
854  else if ( (x .gt. x1 ) .and. ( x .lt. x2 )) then
855  t = y1 + (y2-y1) * (x-x1) / (x2-x1)
856  else
857  t = y2 + (1.0-y2) * (x-x2) / (1.0-x2)
858  end if
859 
860  z_inter(k) = -t * max_depth + eta
861 
862  end do
863 
864  ! Modify interface heights to account for topography
865  z_inter(nz+1) = - local_depth
866 
867  ! Modify interface heights to avoid layers of zero thicknesses
868  do k = nz,1,-1
869  if ( z_inter(k) .LT. (z_inter(k+1) + min_thickness) ) then
870  z_inter(k) = z_inter(k+1) + min_thickness
871  end if
872  end do
873 
874  ! Chnage in interface position
875  x = 0. ! Left boundary at x=0
876  dzinterface(i,j,1) = 0.
877  do k = 2,nz
878  x = x + h(i,j,k)
879  dzinterface(i,j,k) = z_inter(k) - x
880  end do
881  dzinterface(i,j,nz+1) = 0.
882 
883  end do
884  end do
885 
886 stop 'OOOOOOPS' ! For some reason the gnu compiler will not let me delete this
887  ! routine????
888 
889 end subroutine build_grid_arbitrary

892 !------------------------------------------------------------------------------
893 ! Regridding iterations
894 !------------------------------------------------------------------------------
895 subroutine regridding_iteration( densities, target_values, CS, &
896  n0, h0, x0, ppoly0, n1, h1, x1 )
897 ! ------------------------------------------------------------------------------
898 ! This routine performs one single iteration of the regridding based on target
899 ! interface densities. Given the set of target values and cell densities, this
900 ! routine builds an interpolated profile and determines the location of the new
901 ! grid. It may happen that, given a high-order interpolator, the number of
902 ! available layers is insufficient (e.g., there are two available layers for
903 ! a third-order PPM ih4 scheme). In these cases, we resort to the simplest
904 ! continuous linear scheme (P1M h2).
905 ! ------------------------------------------------------------------------------
906 
907  ! Arguments
908  real, dimension(:), intent(in) :: densities ! Actual cell densities
909  real, dimension(:), intent(in) :: target_values ! Target interface densities
910  integer, intent(in) :: n0 ! Number of cells on source grid
911  real, dimension(:), intent(in) :: h0 ! cell widths on source grid
912  real, dimension(:), intent(in) :: x0 ! interface positions on source grid
913  type(ppoly_t), intent(inout) :: ppoly0 ! Piecewise polynomial for density interpolation
914  integer, intent(in) :: n1 ! Number of cells on target grid
915  real, dimension(:), intent(out) :: h1 ! cell widths on target grid
916  real, dimension(:), intent(out) :: x1 ! interface positions on target grid
917  type(regridding_cs), intent(inout) :: cs ! Parameters used for regridding
918 
919  ! Local variables
920  integer :: degree, k
921 
922  ! Reset piecewise polynomials
923  ppoly0%E = 0.0
924  ppoly0%S = 0.0
925  ppoly0%coefficients = 0.0
926 
927  ! 1. Compute the interpolated profile of the density field and build grid
928  select case ( cs%interpolation_scheme )
929 
930  case ( interpolation_p1m_h2 )
931  degree = degree_1
932  call edge_values_explicit_h2( n0, h0, densities, ppoly0%E )
933  call p1m_interpolation( n0, h0, densities, ppoly0 )
934  if ( cs%boundary_extrapolation) then
935  call p1m_boundary_extrapolation( n0, h0, densities, ppoly0 )
936  end if
937 
938  case ( interpolation_p1m_h4 )
939  degree = degree_1
940  if ( n0 .ge. 4 ) then
941  call edge_values_explicit_h4( n0, h0, densities, ppoly0%E )
942  else
943  call edge_values_explicit_h2( n0, h0, densities, ppoly0%E )
944  end if
945  call p1m_interpolation( n0, h0, densities, ppoly0 )
946  if ( cs%boundary_extrapolation) then
947  call p1m_boundary_extrapolation( n0, h0, densities, ppoly0 )
948  end if
949 
950  case ( interpolation_p1m_ih4 )
951  degree = degree_1
952  if ( n0 .ge. 4 ) then
953  call edge_values_implicit_h4( n0, h0, densities, cs%edgeValueWrk, ppoly0%E )
954  else
955  call edge_values_explicit_h2( n0, h0, densities, ppoly0%E )
956  end if
957  call p1m_interpolation( n0, h0, densities, ppoly0 )
958  if ( cs%boundary_extrapolation) then
959  call p1m_boundary_extrapolation( n0, h0, densities, ppoly0 )
960  end if
961 
962  case ( interpolation_plm )
963  degree = degree_1
964  call plm_reconstruction( n0, h0, densities, ppoly0 )
965  if ( cs%boundary_extrapolation) then
966  call plm_boundary_extrapolation( n0, h0, densities, ppoly0 )
967  end if
968 
969  case ( interpolation_ppm_h4 )
970  if ( n0 .ge. 4 ) then
971  degree = degree_2
972  call edge_values_explicit_h4( n0, h0, densities, ppoly0%E )
973  call ppm_reconstruction( n0, h0, densities, ppoly0 )
974  if ( cs%boundary_extrapolation) then
975  call ppm_boundary_extrapolation( n0, h0, densities, ppoly0 )
976  end if
977  else
978  degree = degree_1
979  call edge_values_explicit_h2( n0, h0, densities, ppoly0%E )
980  call p1m_interpolation( n0, h0, densities, ppoly0 )
981  if ( cs%boundary_extrapolation) then
982  call p1m_boundary_extrapolation( n0, h0, densities, ppoly0 )
983  end if
984  end if
985 
986  case ( interpolation_ppm_ih4 )
987 
988  if ( n0 .ge. 4 ) then
989  degree = degree_2
990  call edge_values_implicit_h4( n0, h0, densities, cs%edgeValueWrk, ppoly0%E )
991  call ppm_reconstruction( n0, h0, densities, ppoly0 )
992  if ( cs%boundary_extrapolation) then
993  call ppm_boundary_extrapolation( n0, h0, densities, ppoly0 )
994  end if
995  else
996  degree = degree_1
997  call edge_values_explicit_h2( n0, h0, densities, ppoly0%E )
998  call p1m_interpolation( n0, h0, densities, ppoly0 )
999  if ( cs%boundary_extrapolation) then
1000  call p1m_boundary_extrapolation( n0, h0, densities, ppoly0 )
1001  end if
1002  end if
1003 
1004  case ( interpolation_p3m_ih4ih3 )
1005 
1006  if ( n0 .ge. 4 ) then
1007  degree = degree_3
1008  call edge_values_implicit_h4( n0, h0, densities, cs%edgeValueWrk, ppoly0%E )
1009  call edge_slopes_implicit_h3( n0, h0, densities, cs%edgeSlopeWrk, ppoly0%S )
1010  call p3m_interpolation( n0, h0, densities, ppoly0 )
1011  if ( cs%boundary_extrapolation) then
1012  call p3m_boundary_extrapolation( n0, h0, densities, ppoly0 )
1013  end if
1014  else
1015  degree = degree_1
1016  call edge_values_explicit_h2( n0, h0, densities, ppoly0%E )
1017  call p1m_interpolation( n0, h0, densities, ppoly0 )
1018  if ( cs%boundary_extrapolation) then
1019  call p1m_boundary_extrapolation( n0, h0, densities, ppoly0 )
1020  end if
1021  end if
1022 
1023  case ( interpolation_p3m_ih6ih5 )
1024  if ( n0 .ge. 6 ) then
1025  degree = degree_3
1026  call edge_values_implicit_h6( n0, h0, densities, cs%edgeValueWrk, ppoly0%E )
1027  call edge_slopes_implicit_h5( n0, h0, densities, cs%edgeSlopeWrk, ppoly0%S )
1028  call p3m_interpolation( n0, h0, densities, ppoly0 )
1029  if ( cs%boundary_extrapolation) then
1030  call p3m_boundary_extrapolation( n0, h0, densities, ppoly0 )
1031  end if
1032  else
1033  degree = degree_1
1034  call edge_values_explicit_h2( n0, h0, densities, ppoly0%E )
1035  call p1m_interpolation( n0, h0, densities, ppoly0 )
1036  if ( cs%boundary_extrapolation) then
1037  call p1m_boundary_extrapolation( n0, h0, densities, ppoly0 )
1038  end if
1039  end if
1040 
1041  case ( interpolation_pqm_ih4ih3 )
1042 
1043  if ( n0 .ge. 4 ) then
1044  degree = degree_4
1045  call edge_values_implicit_h4( n0, h0, densities, cs%edgeValueWrk, ppoly0%E )
1046  call edge_slopes_implicit_h3( n0, h0, densities, cs%edgeSlopeWrk, ppoly0%S )
1047  call pqm_reconstruction( n0, h0, densities, ppoly0 )
1048  if ( cs%boundary_extrapolation) then
1049  call pqm_boundary_extrapolation_v1( n0, h0, densities, ppoly0 )
1050  end if
1051  else
1052  degree = degree_1
1053  call edge_values_explicit_h2( n0, h0, densities, ppoly0%E )
1054  call p1m_interpolation( n0, h0, densities, ppoly0 )
1055  if ( cs%boundary_extrapolation) then
1056  call p1m_boundary_extrapolation( n0, h0, densities, ppoly0 )
1057  end if
1058  end if
1059 
1060  case ( interpolation_pqm_ih6ih5 )
1061  if ( n0 .ge. 6 ) then
1062  degree = degree_4
1063  call edge_values_implicit_h6( n0, h0, densities, cs%edgeValueWrk, ppoly0%E )
1064  call edge_slopes_implicit_h5( n0, h0, densities, cs%edgeSlopeWrk, ppoly0%S )
1065  call pqm_reconstruction( n0, h0, densities, ppoly0 )
1066  if ( cs%boundary_extrapolation) then
1067  call pqm_boundary_extrapolation_v1( n0, h0, densities, ppoly0 )
1068  end if
1069  else
1070  degree = degree_1
1071  call edge_values_explicit_h2( n0, h0, densities, ppoly0%E )
1072  call p1m_interpolation( n0, h0, densities, ppoly0 )
1073  if ( cs%boundary_extrapolation) then
1074  call p1m_boundary_extrapolation( n0, h0, densities, ppoly0 )
1075  end if
1076  end if
1077 
1078  end select
1079 
1080  ! Based on global density profile, interpolate new grid and
1081  ! inflate vanished layers
1082  call interpolate_grid( n0, h0, x0, ppoly0, target_values, degree, n1, h1, x1 )
1083  call inflate_vanished_layers( cs%min_thickness, n1, h1 )
1084  x1(1) = 0.0
1085  do k = 1,n1
1086  x1(k+1) = x1(k) + h1(k)
1087  end do
1088 
1089 end subroutine regridding_iteration

1093 ! Given target values (e.g., density), build new grid based on polynomial
1094 !------------------------------------------------------------------------------
1095 subroutine interpolate_grid( n0, h0, x0, ppoly0, target_values, degree, n1, h1, x1 )
1096 ! ------------------------------------------------------------------------------
1097 ! Given the grid 'grid0' and the piecewise polynomial interpolant
1098 ! 'ppoly0' (possibly discontinuous), the coordinates of the new grid 'grid1'
1099 ! are determined by finding the corresponding target interface densities.
1100 ! ------------------------------------------------------------------------------
1101 
1102  ! Arguments
1103  integer, intent(in) :: n0
1104  real, dimension(:), intent(in) :: h0
1105  real, dimension(:), intent(in) :: x0
1106  type(ppoly_t), intent(in) :: ppoly0
1107  real, dimension(:), intent(in) :: target_values
1108  integer, intent(in) :: degree
1109  integer, intent(in) :: n1
1110  real, dimension(:), intent(inout) :: h1
1111  real, dimension(:), intent(inout) :: x1
1112 
1113  ! Local variables
1114  integer :: k ! loop index
1115  real :: t ! current interface target density
1116 
1117  ! Make sure boundary coordinates of new grid coincide with boundary
1118  ! coordinates of previous grid
1119  x1(1) = x0(1)
1120  x1(n1+1) = x0(n0+1)
1121 
1122  ! Find coordinates for interior target values
1123  do k = 2,n1
1124  t = target_values(k)
1125  x1(k) = get_polynomial_coordinate( n0, h0, ppoly0, t, degree )
1126  h1(k-1) = x1(k) - x1(k-1)
1127  end do
1128  h1(n1) = x1(n1+1) - x1(n1)
1129 
1130 end subroutine interpolate_grid

1133 !------------------------------------------------------------------------------
