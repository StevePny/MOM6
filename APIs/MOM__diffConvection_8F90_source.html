109  ! (out) Modified vertical diffusivity (m2/s)
110 ! Local variables
111  integer :: i, j, k
112  real, dimension( G%ke+1 ) :: n2_1d ! Brunt-Vaisala frequency squared, at interfaces (1/s2)
113  real, dimension( G%ke+1 ) :: kd_1d ! Vertical diffusivity at interfaces (m2/s)
114  real :: gorho, pref, rhok, rhokm1
115 
116  gorho = g%g_Earth / g%Rho0
117 
118  n2_1d( 1 ) = 0.
119  n2_1d( g%ke+1 ) = 0.
120  kd_1d( 1 ) = 0.
121  kd_1d( g%ke+1 ) = 0.
122  do j = g%jsc, g%jec
123  do i = g%isc, g%iec
124 
125  ! This k-loop calculates external quantities independent of any iterations
126  ! Start at bottom of top level
127  pref = 0. ! Ignore atmospheric pressure
128  do k = 2, g%ke
129  ! Pressure at interface K is incremented by mass of level above
130  pref = pref + g%g_Earth * g%Rho0 * h(i,j,k-1) * g%H_to_m ! Boussinesq approximation!!!! ?????
131  ! Compute Brunt-Vaisala frequency (static stability) on interfaces
132  call calculate_density(temp(i,j,k), salt(i,j,k), pref, rhok, eos)
133  call calculate_density(temp(i,j,k-1), salt(i,j,k-1), pref, rhokm1, eos)
134  n2_1d(k) = gorho * (rhok - rhokm1) / (0.5*(h(i,j,k-1) + h(i,j,k))+g%H_subroundoff) ! Can be negative
135  kd_1d(k) = 0.
136  if (n2_1d(k) < 0.) kd_1d(k) = cs%Kd_convection
137  enddo ! k
138 
139  if (.not. cs%passiveMode) kd_int(i,j,:) = kd_int(i,j,:) + kd_1d(:)
140 
141  if (cs%id_N2 > 0) cs%N2(i,j,:) = n2_1d(:)
142  if (cs%id_Kd_conv > 0) cs%Kd_conv(i,j,:) = kd_1d(:)
