24 !* By Robert Hallberg, April 1994 - June 2002 *
25 !* *
26 !* This subroutine initializes the fields for the simulations. *
27 !* The one argument passed to initialize, Time, is set to the *
28 !* current time of the simulation. The fields which are initialized *
29 !* here are: *
30 !* u - Zonal velocity in m s-1. *
31 !* v - Meridional velocity in m s-1. *
32 !* h - Layer thickness in m. (Must be positive.) *
33 !* G%bathyT - Basin depth in m. (Must be positive.) *
34 !* G%CoriolisBu - The Coriolis parameter, in s-1. *
35 !* G%g_prime - The reduced gravity at each interface, in m s-2. *
36 !* G%Rlay - Layer potential density (coordinate variable) in kg m-3.*
37 !* If ENABLE_THERMODYNAMICS is defined: *
38 !* tv%T - Temperature in C. *
39 !* tv%S - Salinity in psu. *
40 !* If SPONGE is defined: *
41 !* A series of subroutine calls are made to set up the damping *
42 !* rates and reference profiles for all variables that are damped *
43 !* in the sponge. *
44 !* Any user provided tracer code is also first linked through this *
45 !* subroutine. *
46 !* *
47 !* Forcing-related fields (taux, tauy, buoy, ustar, etc.) are set *
48 !* in MOM_surface_forcing.F90. *
49 !* *
50 !* These variables are all set in the set of subroutines (in this *
51 !* file): initialize_topography, initialize_thickness, *
52 !* initialize_velocity, initialize_temp_sal, initialize_sponges, and *
53 !* set_coordinate. *
54 !* *
55 !* Macros written all in capital letters are defined in MOM_memory.h. *
56 !* *
57 !* A small fragment of the grid is shown below: *
58 !* *
59 !* j+1 x ^ x ^ x At x: q, CoriolisBu *
60 !* j+1 > o > o > At ^: v, tauy *
61 !* j x ^ x ^ x At >: u, taux *
62 !* j > o > o > At o: h, bathyT, buoy, tr, T, S, Rml, ustar *
63 !* j-1 x ^ x ^ x *
64 !* i-1 i i+1 At x & ^: *
65 !* i i+1 At > & o: *
66 !* *
67 !* The boundaries always run through q grid points (x). *
68 !* *

798 subroutine set_coord_from_ts_range(Rlay, g_prime, G, param_file, &
799  eqn_of_state, p_ref)
800  real, dimension(:), intent(out) :: rlay, g_prime
801  type(ocean_grid_type), intent(in) :: g
802  type(param_file_type), intent(in) :: param_file
803  type(eos_type), pointer :: eqn_of_state
804  real, intent(in) :: p_ref
805 ! Arguments: Rlay - the layers' target coordinate values (potential density).
806 ! (out) g_prime - the reduced gravity across the interfaces, in m s-2.
807 ! (in) G - The ocean's grid structure.
808 ! (in) param_file - A structure indicating the open file to parse for
809 ! model parameter values.
810 ! (in) eqn_of_state - integer that selects equation of state
811 ! (in) P_Ref - The coordinate-density reference pressure in Pa.
812 
813 ! This subroutine sets the layer densities (Rlay) and the interface !
814 ! reduced gravities (g). !
815  real, dimension(SZK_(G)) :: t0, s0, pref
816  real :: s_ref, s_light, s_dense ! Salnity range parameters in PSU.
817  real :: t_ref, t_light, t_dense ! Temperature range parameters in dec C.
818  real :: res_rat ! The ratio of density space resolution in the denser part
819  ! of the range to that in the lighter part of the range.
820  ! Setting this greater than 1 increases the resolution for
821  ! the denser water.
822  real :: g_fs ! Reduced gravity across the free surface, in m s-2.
823  real :: a1, frac_dense, k_frac
824  integer :: k, nz, k_light
825  character(len=40) :: mod = "set_coord_from_TS_range" ! This subroutine's name.
826  character(len=200) :: filename, coord_file, inputdir ! Strings for file/path
827  nz = g%ke
828 
829  call calltree_enter(trim(mod)//"(), MOM_initialization.F90")
830 
831  call get_param(param_file, mod, "T_REF", t_ref, &
832  "The default initial temperatures.", units="degC", default=10.0)
833  call get_param(param_file, mod, "TS_RANGE_T_LIGHT", t_light, &
834  "The initial temperature of the lightest layer when \n"//&
835  "COORD_CONFIG is set to ts_range.", units="degC", default=t_ref)
836  call get_param(param_file, mod, "TS_RANGE_T_DENSE", t_dense, &
837  "The initial temperature of the densest layer when \n"//&
838  "COORD_CONFIG is set to ts_range.", units="degC", default=t_ref)
839 
840  call get_param(param_file, mod, "S_REF", s_ref, &
841  "The default initial salinities.", units="PSU", default=35.0)
842  call get_param(param_file, mod, "TS_RANGE_S_LIGHT", s_light, &
843  "The initial lightest salinities when COORD_CONFIG \n"//&
844  "is set to ts_range.", default = s_ref, units="PSU")
845  call get_param(param_file, mod, "TS_RANGE_S_DENSE", s_dense, &
846  "The initial densest salinities when COORD_CONFIG \n"//&
847  "is set to ts_range.", default = s_ref, units="PSU")
848 
849  call get_param(param_file, mod, "TS_RANGE_RESOLN_RATIO", res_rat, &
850  "The ratio of density space resolution in the densest \n"//&
851  "part of the range to that in the lightest part of the \n"//&
852  "range when COORD_CONFIG is set to ts_range. Values \n"//&
853  "greater than 1 increase the resolution of the denser water.",&
854  default=1.0, units="nondim")
855 
856  call get_param(param_file, mod, "GFS", g_fs, &
857  "The reduced gravity at the free surface.", units="m s-2", &
858  default=g%g_Earth)
859 
860  k_light = g%nk_rho_varies + 1
861 
862  ! Set T0(k) to range from T_LIGHT to T_DENSE, and simliarly for S0(k).
863  t0(k_light) = t_light ; s0(k_light) = s_light
864  a1 = 2.0 * res_rat / (1.0 + res_rat)
865  do k=k_light+1,nz
866  k_frac = real(k-k_light)/real(nz-k_light)
867  frac_dense = a1 * k_frac + (1.0 - a1) * k_frac**2
868  t0(k) = frac_dense * (t_dense - t_light) + t_light
869  s0(k) = frac_dense * (s_dense - s_light) + s_light
870  enddo
871 
872  g_prime(1) = g_fs
873  do k=1,nz ; pref(k) = p_ref ; enddo
874  call calculate_density(t0, s0, pref, rlay, k_light,nz-k_light+1,eqn_of_state)
875  ! Extrapolate target densities for the variable density mixed and buffer layers.
876  do k=k_light-1,1,-1
877  rlay(k) = 2.0*rlay(k+1) - rlay(k+2)
878  enddo
879  do k=2,nz; g_prime(k) = (g%g_Earth/g%Rho0) * (rlay(k) - rlay(k-1)); enddo
880 
881  call calltree_leave(trim(mod)//'()')
882 end subroutine set_coord_from_ts_range
