95 subroutine calculate_density_array_wright(T, S, pressure, rho, start, npts)
96  real, intent(in), dimension(:) :: t, s, pressure
97  real, intent(out), dimension(:) :: rho
98  integer, intent(in) :: start, npts
99 ! * Arguments: T - potential temperature relative to the surface in C. *
100 ! * (in) S - salinity in PSU. *
101 ! * (in) pressure - pressure in Pa. *
102 ! * (out) rho - in situ density in kg m-3. *
103 ! * (in) start - the starting point in the arrays. *
104 ! * (in) npts - the number of values to calculate. *
105 
106 ! *====================================================================*
107 ! * This subroutine computes the in situ density of sea water (rho in *
108 ! * units of kg/m^3) from salinity (S in psu), potential temperature *
109 ! * (T in deg C), and pressure in Pa. It uses the expression from *
110 ! * Wright, 1997, J. Atmos. Ocean. Tech., 14, 735-740. *
111 ! * Coded by R. Hallberg, 7/00 *
112 ! *====================================================================*
113  real :: al0, p0, lambda
114  integer :: j
115 
116  do j=start,start+npts-1
117  al0 = a0 + a1*t(j) +a2*s(j)
118  p0 = b0 + b4*s(j) + t(j) * (b1 + t(j)*(b2 + b3*t(j)) + b5*s(j))
119  lambda = c0 +c4*s(j) + t(j) * (c1 + t(j)*(c2 + c3*t(j)) + c5*s(j))
120 
121  rho(j) = (pressure(j) + p0) / (lambda + al0*(pressure(j) + p0))
122  enddo
123 end subroutine calculate_density_array_wright

125 subroutine calculate_density_derivs_wright(T, S, pressure, drho_dT, drho_dS, start, npts)
126  real, intent(in), dimension(:) :: t, s, pressure
127  real, intent(out), dimension(:) :: drho_dt, drho_ds
128  integer, intent(in) :: start, npts
129 ! * Arguments: T - potential temperature relative to the surface in C. *
130 ! * (in) S - salinity in PSU. *
131 ! * (in) pressure - pressure in Pa. *
132 ! * (out) drho_dT - the partial derivative of density with *
133 ! * potential tempetature, in kg m-3 K-1. *
134 ! * (out) drho_dS - the partial derivative of density with *
135 ! * salinity, in kg m-3 psu-1. *
136 ! * (in) start - the starting point in the arrays. *
137 ! * (in) npts - the number of values to calculate. *
138  real :: al0, p0, lambda, i_denom2
139  integer :: j
140 
141  do j=start,start+npts-1
142  al0 = a0 + a1*t(j) + a2*s(j)
143  p0 = b0 + b4*s(j) + t(j) * (b1 + t(j)*(b2 + b3*t(j)) + b5*s(j))
144  lambda = c0 +c4*s(j) + t(j) * (c1 + t(j)*(c2 + c3*t(j)) + c5*s(j))
145 
146  i_denom2 = 1.0 / (lambda + al0*(pressure(j) + p0))
147  i_denom2 = i_denom2 *i_denom2
148  drho_dt(j) = i_denom2 * &
149  (lambda* (b1 + t(j)*(2.0*b2 + 3.0*b3*t(j)) + b5*s(j)) - &
150  (pressure(j)+p0) * ( (pressure(j)+p0)*a1 + &
151  (c1 + t(j)*(c2*2.0 + c3*3.0*t(j)) + c5*s(j)) ))
152  drho_ds(j) = i_denom2 * (lambda* (b4 + b5*t(j)) - &
153  (pressure(j)+p0) * ( (pressure(j)+p0)*a2 + (c4 + c5*t(j)) ))
154  enddo
155 
156 end subroutine calculate_density_derivs_wright

158 subroutine calculate_compress_wright(T, S, pressure, rho, drho_dp, start, npts)
159  real, intent(in), dimension(:) :: t, s, pressure
160  real, intent(out), dimension(:) :: rho, drho_dp
161  integer, intent(in) :: start, npts
162 ! * Arguments: T - potential temperature relative to the surface in C. *
163 ! * (in) S - salinity in PSU. *
164 ! * (in) pressure - pressure in Pa. *
165 ! * (out) rho - in situ density in kg m-3. *
166 ! * (out) drho_dp - the partial derivative of density with *
167 ! * pressure (also the inverse of the square of *
168 ! * sound speed) in s2 m-2. *
169 ! * (in) start - the starting point in the arrays. *
170 ! * (in) npts - the number of values to calculate. *
171 ! *====================================================================*
172 ! * This subroutine computes the in situ density of sea water (rho in *
173 ! * units of kg/m^3) and the compressibility (drho/dp = C_sound^-2) *
174 ! * (drho_dp in units of s2 m-2) from salinity (sal in psu), potential*
175 ! * temperature (T in deg C), and pressure in Pa. It uses the expres-*
176 ! * sions from Wright, 1997, J. Atmos. Ocean. Tech., 14, 735-740. *
177 ! * Coded by R. Hallberg, 1/01 *
178 ! *====================================================================*
179  real :: al0, p0, lambda, i_denom
180  integer :: j
181 
182  do j=start,start+npts-1
183  al0 = a0 + a1*t(j) +a2*s(j)
184  p0 = b0 + b4*s(j) + t(j) * (b1 + t(j)*(b2 + b3*t(j)) + b5*s(j))
185  lambda = c0 +c4*s(j) + t(j) * (c1 + t(j)*(c2 + c3*t(j)) + c5*s(j))
186 
187  i_denom = 1.0 / (lambda + al0*(pressure(j) + p0))
188  rho(j) = (pressure(j) + p0) * i_denom
189  drho_dp(j) = lambda * i_denom * i_denom
190  enddo
191 end subroutine calculate_compress_wright

227 subroutine int_density_dz_wright(T, S, z_t, z_b, rho_ref, rho_0, G_e, G, dpa, intz_dpa, &
228  intx_dpa, inty_dpa)
229  real, dimension(NIMEM_,NJMEM_), intent(in) :: t, s, z_t, z_b
230  real, intent(in) :: rho_ref, rho_0, g_e
231  type(ocean_grid_type), intent(in) :: g
232  real, dimension(NIMEM_,NJMEM_), intent(out) :: dpa
233  real, dimension(NIMEM_,NJMEM_), optional, intent(out) :: intz_dpa
234  real, dimension(NIMEMB_,NJMEM_), optional, intent(out) :: intx_dpa
235  real, dimension(NIMEM_,NJMEMB_), optional, intent(out) :: inty_dpa
236 ! This subroutine calculates analytical and nearly-analytical integrals of
237 ! pressure anomalies across layers, which are required for calculating the
238 ! finite-volume form pressure accelerations in a Boussinesq model.
239 !
240 ! Arguments: T - potential temperature relative to the surface in C.
241 ! (in) S - salinity in PSU.
242 ! (in) z_t - height at the top of the layer in m.
243 ! (in) z_b - height at the top of the layer in m.
244 ! (in) rho_ref - A mean density, in kg m-3, that is subtracted out to reduce
245 ! the magnitude of each of the integrals.
246 ! (The pressure is calucated as p~=-z*rho_0*G_e.)
247 ! (in) rho_0 - A density, in kg m-3, that is used to calculate the pressure
248 ! (as p~=-z*rho_0*G_e) used in the equation of state.
249 ! (in) G_e - The Earth's gravitational acceleration, in m s-2.
250 ! (in) G - The ocean's grid structure.
251 ! (out) dpa - The change in the pressure anomaly across the layer,
252 ! in Pa.
253 ! (out,opt) intz_dpa - The integral through the thickness of the layer of the
254 ! pressure anomaly relative to the anomaly at the top of
255 ! the layer, in Pa m.
256 ! (out,opt) intx_dpa - The integral in x of the difference between the
257 ! pressure anomaly at the top and bottom of the layer
258 ! divided by the x grid spacing, in Pa.
259 ! (out,opt) inty_dpa - The integral in y of the difference between the
260 ! pressure anomaly at the top and bottom of the layer
261 ! divided by the y grid spacing, in Pa.
262 
263  real, dimension(SZI_(G),SZJ_(G)) :: al0_2d, p0_2d, lambda_2d
264  real :: al0, p0, lambda
265  real :: eps, eps2, rho_anom, rem
266  real :: w_left, w_right, intz(5)
267  real, parameter :: c1_3 = 1.0/3.0, c1_7 = 1.0/7.0 ! Rational constants.
268  real, parameter :: c1_9 = 1.0/9.0, c1_90 = 1.0/90.0 ! Rational constants.
269  real :: gxrho, i_rho
270  real :: dz, p_ave, i_al0, i_lzz
271  integer :: isq, ieq, jsq, jeq, i, j, m
272 
273  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
274 
275  gxrho = g_e * rho_0
276  i_rho = 1.0 / rho_0
277 
278  do j=jsq,jeq+1 ; do i=isq,ieq+1
279  al0_2d(i,j) = a0 + a1*t(i,j) + a2*s(i,j)
280  p0_2d(i,j) = b0 + b4*s(i,j) + t(i,j) * (b1 + t(i,j)*(b2 + b3*t(i,j)) + b5*s(i,j))
281  lambda_2d(i,j) = c0 +c4*s(i,j) + t(i,j) * (c1 + t(i,j)*(c2 + c3*t(i,j)) + c5*s(i,j))
282 
283  al0 = al0_2d(i,j) ; p0 = p0_2d(i,j) ; lambda = lambda_2d(i,j)
284 
285  dz = z_t(i,j) - z_b(i,j)
286  p_ave = -0.5*gxrho*(z_t(i,j)+z_b(i,j))
287 
288  i_al0 = 1.0 / al0
289  i_lzz = 1.0 / (p0 + (lambda * i_al0) + p_ave)
290  eps = 0.5*gxrho*dz*i_lzz ; eps2 = eps*eps
291 
292 ! rho(j) = (pressure(j) + p0) / (lambda + al0*(pressure(j) + p0))
293 
294  rho_anom = (p0 + p_ave)*(i_lzz*i_al0) - rho_ref
295  rem = i_rho * (lambda * i_al0**2) * eps2 * &
296  (c1_3 + eps2*(0.2 + eps2*(c1_7 + c1_9*eps2)))
297  dpa(i,j) = g_e*rho_anom*dz - 2.0*eps*rem
298  if (present(intz_dpa)) &
299  intz_dpa(i,j) = 0.5*g_e*rho_anom*dz**2 - dz*(1.0+eps)*rem
300  enddo ; enddo
301 
302  if (present(intx_dpa)) then ; do j=g%jsc,g%jec ; do i=isq,ieq
303  intz(1) = dpa(i,j) ; intz(5) = dpa(i+1,j)
304  do m=2,4
305  w_left = 0.25*real(m-1) ; w_right = 1.0-w_left
306  al0 = w_left*al0_2d(i,j) + w_right*al0_2d(i+1,j)
307  p0 = w_left*p0_2d(i,j) + w_right*p0_2d(i+1,j)
308  lambda = w_left*lambda_2d(i,j) + w_right*lambda_2d(i+1,j)
309 
310  dz = w_left*(z_t(i,j) - z_b(i,j)) + w_right*(z_t(i+1,j) - z_b(i+1,j))
311  p_ave = -0.5*gxrho*(w_left*(z_t(i,j)+z_b(i,j)) + &
312  w_right*(z_t(i+1,j)+z_b(i+1,j)))
313 
314  i_al0 = 1.0 / al0
315  i_lzz = 1.0 / (p0 + (lambda * i_al0) + p_ave)
316  eps = 0.5*gxrho*dz*i_lzz ; eps2 = eps*eps
317 
318  intz(m) = g_e*dz*((p0 + p_ave)*(i_lzz*i_al0) - rho_ref) - 2.0*eps * &
319  i_rho * (lambda * i_al0**2) * eps2 * &
320  (c1_3 + eps2*(0.2 + eps2*(c1_7 + c1_9*eps2)))
321  enddo
322  ! Use Bode's rule to integrate the values.
323  intx_dpa(i,j) = c1_90*(7.0*(intz(1)+intz(5)) + 32.0*(intz(2)+intz(4)) + &
324  12.0*intz(3))
325  enddo ; enddo ; endif
326 
327  if (present(inty_dpa)) then ; do j=jsq,jeq ; do i=g%isc,g%iec
328  intz(1) = dpa(i,j) ; intz(5) = dpa(i,j+1)
329  do m=2,4
330  w_left = 0.25*real(m-1) ; w_right = 1.0-w_left
331  al0 = w_left*al0_2d(i,j) + w_right*al0_2d(i,j+1)
332  p0 = w_left*p0_2d(i,j) + w_right*p0_2d(i,j+1)
333  lambda = w_left*lambda_2d(i,j) + w_right*lambda_2d(i,j+1)
334 
335  dz = w_left*(z_t(i,j) - z_b(i,j)) + w_right*(z_t(i,j+1) - z_b(i,j+1))
336  p_ave = -0.5*gxrho*(w_left*(z_t(i,j)+z_b(i,j)) + &
337  w_right*(z_t(i,j+1)+z_b(i,j+1)))
338 
339  i_al0 = 1.0 / al0
340  i_lzz = 1.0 / (p0 + (lambda * i_al0) + p_ave)
341  eps = 0.5*gxrho*dz*i_lzz ; eps2 = eps*eps
342 
343  intz(m) = g_e*dz*((p0 + p_ave)*(i_lzz*i_al0) - rho_ref) - 2.0*eps * &
344  i_rho * (lambda * i_al0**2) * eps2 * &
345  (c1_3 + eps2*(0.2 + eps2*(c1_7 + c1_9*eps2)))
346  enddo
347  ! Use Bode's rule to integrate the values.
348  inty_dpa(i,j) = c1_90*(7.0*(intz(1)+intz(5)) + 32.0*(intz(2)+intz(4)) + &
349  12.0*intz(3))
350  enddo ; enddo ; endif
351 end subroutine int_density_dz_wright

353 subroutine int_spec_vol_dp_wright(T, S, p_t, p_b, alpha_ref, G, dza, &
354  intp_dza, intx_dza, inty_dza, halo_size)
355  real, dimension(NIMEM_,NJMEM_), intent(in) :: t, s, p_t, p_b
356  real, intent(in) :: alpha_ref
357  type(ocean_grid_type), intent(in) :: g
358  real, dimension(NIMEM_,NJMEM_), intent(out) :: dza
359  real, dimension(NIMEM_,NJMEM_), optional, intent(out) :: intp_dza
360  real, dimension(NIMEMB_,NJMEM_), optional, intent(out) :: intx_dza
361  real, dimension(NIMEM_,NJMEMB_), optional, intent(out) :: inty_dza
362  integer, optional, intent(in) :: halo_size
363 ! This subroutine calculates analytical and nearly-analytical integrals in
364 ! pressure across layers of geopotential anomalies, which are required for
365 ! calculating the finite-volume form pressure accelerations in a non-Boussinesq
366 ! model. There are essentially no free assumptions, apart from the use of
367 ! Bode's rule to do the horizontal integrals, and from a truncation in the
368 ! series for log(1-eps/1+eps) that assumes that |eps| < 0.34.
369 !
370 ! Arguments: T - potential temperature relative to the surface in C.
371 ! (in) S - salinity in PSU.
372 ! (in) p_t - pressure at the top of the layer in Pa.
373 ! (in) p_b - pressure at the top of the layer in Pa.
374 ! (in) alpha_ref - A mean specific volume that is subtracted out to reduce
375 ! the magnitude of each of the integrals, m3 kg-1.
376 ! The calculation is mathematically identical with
377 ! different values of alpha_ref, but this reduces the
378 ! effects of roundoff.
379 ! (in) G - The ocean's grid structure.
380 ! (out) dza - The change in the geopotential anomaly across the layer,
381 ! in m2 s-2.
382 ! (out,opt) intp_dza - The integral in pressure through the layer of the
383 ! geopotential anomaly relative to the anomaly at the
384 ! bottom of the layer, in Pa m2 s-2.
385 ! (out,opt) intx_dza - The integral in x of the difference between the
386 ! geopotential anomaly at the top and bottom of the layer
387 ! divided by the x grid spacing, in m2 s-2.
388 ! (out,opt) inty_dza - The integral in y of the difference between the
389 ! geopotential anomaly at the top and bottom of the layer
390 ! divided by the y grid spacing, in m2 s-2.
391 ! (in,opt) halo_size - The width of halo points on which to calculate dza.
392 
393  real, dimension(SZI_(G),SZJ_(G)) :: al0_2d, p0_2d, lambda_2d
394  real :: al0, p0, lambda
395  real :: alpha_anom, dp, p_ave
396  real :: rem, eps, eps2
