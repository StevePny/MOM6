23 !* *
24 !* By Robert Hallberg, April 1994 - June 2002. *
25 !* *
26 !* This program contains the subroutine that calculates the *
27 !* effects of horizontal viscosity, including parameterizations of *
28 !* the value of the viscosity itself. horizontal_viscosity calc- *
29 !* ulates the acceleration due to some combination of a biharmonic *
30 !* viscosity and a Laplacian viscosity. Either or both may use a *
31 !* coefficient that depends on the shear and strain of the flow. *
32 !* All metric terms are retained. The Laplacian is calculated as *
33 !* the divergence of a stress tensor, using the form suggested by *
34 !* Smagorinsky (1993). The biharmonic is calculated by twice *
35 !* applying the divergence of the stress tensor that is used to *
36 !* calculate the Laplacian, but without the dependence on thickness *
37 !* in the first pass. This form permits a variable viscosity, and *
38 !* indicates no acceleration for either resting fluid or solid body *
39 !* rotation. *
40 !* *
41 !* set_up_hor_visc calculates and stores the values of a number of *
42 !* metric functions that are used in horizontal_viscosity. It is *
43 !* called by horizontal_viscosity the first time that the latter is *
44 !* called. *
45 !* *
46 !* The form of the Laplacian viscosity is: *
47 !* *
48 !* diffu = 1/h * {d/dx[KH*h*sh_xx] + d/dy[KH*h*sh_xy]} *
49 !* diffv = 1/h * {d/dx[KH*h*sh_xy] - d/dy[KH*h*sh_xx]} *
50 !* *
51 !* sh_xx = du/dx - dv/dy sh_xy = du/dy + dv/dx *
52 !* *
53 !* with appropriate metric terms thrown in. KH may either be a *
54 !* constant or may vary with the shear, as proposed by Smagorinsky. *
55 !* The form of this term is discussed extensively in Griffies and *
56 !* Hallberg (MWR, 2000), and the implementation here follows that *
57 !* discussion closely. *
58 !* *
59 !* Only free slip boundary conditions have been coded, although *
60 !* no slip boundary conditions could be used with the Laplacian *
61 !* viscosity. For a western boundary, for example, the boundary *
62 !* conditions with the biharmonic operator would be written as: *
63 !* dv/dx = 0, d^3v/dx^3 = 0, u = 0, d^2u/dx^2 = 0 , *
64 !* while for a Laplacian operator, they are simply: *
65 !* dv/dx = 0, u = 0 . *
66 !* These boundary conditions are largely dictated by the use of *
67 !* a an Arakawa C-grid and by the varying layer thickness. *
68 !* *
69 !* Macros written all in capital letters are defined in MOM_memory.h. *
70 !* *
71 !* A small fragment of the C-grid is shown below: *
72 !* *
73 !* j+1 x ^ x ^ x At x: q, CoriolisBu, hq, str_xy, sh_xy *
74 !* j+1 > o > o > At ^: v, diffv, v0 *
75 !* j x ^ x ^ x At >: u, diffu, u0 *
76 !* j > o > o > At o: h, str_xx, sh_xx *
77 !* j-1 x ^ x ^ x *
78 !* i-1 i i+1 At x & ^: *
79 !* i i+1 At > & o: *
80 !* *
81 !* The boundaries always run through q grid points (x). *
82 !* *

193  real, dimension(NIMEMB_,NJMEM_,NKMEM_), intent(in) :: u
194  real, dimension(NIMEM_,NJMEMB_,NKMEM_), intent(in) :: v
195  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h
196  real, dimension(NIMEMB_,NJMEM_,NKMEM_), intent(out) :: diffu

204 ! Arguments: u - Zonal velocity, in m s-1.
205 ! (in) v - Meridional velocity, in m s-1.
206 ! (in) h - Layer thickness, in m or kg m-2. The units of h are
207 ! referred to below as H.
208 ! (out) diffu - Zonal acceleration due to convergence of the
209 ! along-isopycnal stress tensor, in m s-2.
210 ! (out) diffv - Meridional acceleration due to convergence of
211 ! the along-isopycnal stress tensor, in m s-2.
212 ! (inout) MEKE - A pointer to a structure containing fields related to
213 ! the Mesoscale Eddy Kinetic Energy.
214 ! (in) VarMix - A pointer to a structure with fields that specify the
215 ! spatially variable viscosities.
216 ! (in) G - The ocean's grid structure.
217 ! (in) CS - The control structure returned by a previous call to
218 ! hor_visc_init.
219 ! (in) OBC - A pointer to an open boundary condition type.
220 
221 ! By R. Hallberg, August 1998 - November 1998.
222 ! This subroutine determines the acceleration due to the
223 ! horizontal viscosity. A combination of biharmonic and Laplacian
224 ! forms can be used. The coefficient may either be a constant or
225 ! a shear-dependent form. The biharmonic is determined by twice
226 ! taking the divergence of an appropriately defined stress tensor.
227 ! The Laplacian is determined by doing so once.
228 ! To work, the following fields must be set outside of the usual
229 ! is to ie range before this subroutine is called:
230 ! v[is-2,is-1,ie+1,ie+2], u[is-2,is-1,ie+1,ie+2], and h[is-1,ie+1],
231 ! with a similarly sized halo in the y-direction.
232 
233  real, dimension(SZIB_(G),SZJ_(G)) :: u0
234  real, dimension(SZI_(G),SZJB_(G)) :: v0
235  ! u0 and v0 are the Laplacian of the velocities, in m-1 s-1.
236  real, dimension(SZI_(G),SZJ_(G)) :: &
237  sh_xx, & ! sh_xx is the horizontal tension (du/dx - dv/dy) including
238  ! all metric terms, in s-1.
239  str_xx ! str_xx is the diagonal term in the stress tensor, in H m2 s-2.
240  real, dimension(SZIB_(G),SZJB_(G)) :: &
241  sh_xy, & ! sh_xy is the horizontal shearing strain (du/dy + dv/dx)
242  ! including all metric terms, in s-1.
243  str_xy ! str_xy is the cross term in the stress tensor, in H m2 s-2.
244  real, dimension(SZIB_(G),SZJB_(G),SZK_(G)) :: &
245  ah_q, & ! The biharmonic viscosity at corner points, in m4 s-1.
246  kh_q, & ! The Laplacian viscosity at corner points, in m2 s-1.
247  frictwork ! Work released by lateral friction terms in W m-2.
248  real, dimension(SZI_(G),SZJ_(G),SZK_(G)) :: &
249  ah_h, & ! The biharmonic viscosity at thickness points, in m4 s-1.
250  kh_h ! The Laplacian viscosity at thickness points, in m2 s-1.
251  real :: ah ! Biharmonic viscosity in m4 s-1.
252  real :: kh ! Laplacian viscosity in m2 s-1.
253  real :: ahsm ! Smagorinsky biharmonic viscosity in m4 s-1.
254  real :: khsm ! Smagorinsky Laplacian viscosity in m2 s-1.
255  real :: shear_mag ! The magnitude of the shear, in s-1.
256  real :: huq, hvq ! Temporary variables in units of H (i.e. m2 or kg2 m-4).
257  real :: hq ! The harmonic mean of the harmonic means of the u- & v-
258  ! point thicknesses, in H. This guarantees that hq/hu < 4.
259  real :: h_neglect ! A thickness that is so small it is usually lost
260  ! in roundoff and can be neglected, in H.
261  real :: h_neglect3 ! h_neglect^3, in H3.
262  real :: hrat_min ! The minimum of the thicknesses at the 4 neighboring
263  ! velocity points divided by the thickness at the stress
264  ! point (h or q point), nondimensional.
265  real :: visc_bound_rem ! The fraction of the overall viscous bounds that
266  ! remain to be applied. Nondim.
267  real :: kh_scale ! A factor between 0 and 1 by which the horizontal
268  ! Laplacian viscosity is rescaled.
269  logical :: rescale_kh, find_frictwork, apply_obc = .false.
270  integer :: is, ie, js, je, isq, ieq, jsq, jeq, nz
271  integer :: i, j, k
272  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
273  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
274  h_neglect = g%H_subroundoff
275  h_neglect3 = h_neglect**3
276 
277  if (present(obc)) then ; if (associated(obc)) then
278  apply_obc = obc%apply_OBC_u_flather_east .or. obc%apply_OBC_u_flather_west .or. &
279  obc%apply_OBC_v_flather_north .or. obc%apply_OBC_v_flather_south
280  endif ; endif
281 
282  if (.not.associated(cs)) call mom_error(fatal, &
283  "MOM_hor_visc: Module must be initialized before it is used.")
284 
285  find_frictwork = (cs%id_FrictWork > 0)
286  if (associated(meke)) then
287  if (associated(meke%mom_src)) find_frictwork = .true.
288  endif
289 
290  rescale_kh = .false.
291  if (associated(varmix)) then
292  rescale_kh = varmix%Resoln_scaled_Kh
293  if (rescale_kh .and. &
294  (.not.associated(varmix%Res_fn_h) .or. .not.associated(varmix%Res_fn_q))) &
295  call mom_error(fatal, "MOM_hor_visc: VarMix%Res_fn_h and " //&
296  "VarMix%Res_fn_q both need to be associated with Resoln_scaled_Kh.")
297  endif

299 !GOMP(parallel do default(shared) private(i, j, k, u0, v0, sh_xx, str_xx, &)
300 !GOMP( sh_xy, str_xy, Ah, Kh, AhSm, KhSm, &)
301 !GOMP( Shear_mag, huq, hvq, hq, Kh_scale))
302  do k=1,nz
303 
304 ! This code uses boundary conditions that are consistent with
305 ! free slip and no normal flow boundary conditions. The boundary
306 ! conditions for the western boundary, for example, are:
307 ! dv/dx = 0, d^3v/dx^3 = 0, u = 0, d^2u/dx^2 = 0 .
308 ! The overall scheme is second order accurate.
309 ! All of the metric terms are retained, and the repeated use of
310 ! the symmetric stress tensor insures that no stress is applied with
311 ! no flow or solid-body rotation, even with non-constant values of
312 ! of the biharmonic viscosity.
313 
314 ! The following are the forms of the horizontal tension and hori-
315 ! shearing strain advocated by Smagorinsky (1993) and discussed in
316 ! Griffies and Hallberg (MWR, 2000).
317  do j=jsq-1,jeq+2 ; do i=isq-1,ieq+2
318  sh_xx(i,j) = (cs%DY_dxT(i,j)*(g%IdyCu(i,j) * u(i,j,k) - &
319  g%IdyCu(i-1,j) * u(i-1,j,k)) - &
320  cs%DX_dyT(i,j)*(g%IdxCv(i,j) * v(i,j,k) - &
321  g%IdxCv(i,j-1)*v(i,j-1,k)))
322 
323  enddo ; enddo
324 
325  if (cs%no_slip) then
326  do j=js-2,jeq+1 ; do i=is-2,ieq+1
327  sh_xy(i,j) = (2.0-g%mask2dBu(i,j)) * &
328  (cs%DX_dyBu(i,j)*(u(i,j+1,k)*g%IdxCu(i,j+1) - u(i,j,k)*g%IdxCu(i,j)) + &
329  cs%DY_dxBu(i,j)*(v(i+1,j,k)*g%IdyCv(i+1,j) - v(i,j,k)*g%IdyCv(i,j)))
330  enddo ; enddo
331  else
332  do j=js-2,jeq+1 ; do i=is-2,ieq+1
333  sh_xy(i,j) = g%mask2dBu(i,j) * &
334  (cs%DX_dyBu(i,j)*(u(i,j+1,k)*g%IdxCu(i,j+1) - u(i,j,k)*g%IdxCu(i,j)) + &
335  cs%DY_dxBu(i,j)*(v(i+1,j,k)*g%IdyCv(i+1,j) - v(i,j,k)*g%IdyCv(i,j)))
336  enddo ; enddo
337  endif
338 
339 ! Evaluate u0 = x.Div(Grad u) and v0 = y.Div( Grad u)
340  if (cs%biharmonic) then
341  do j=js-1,jeq+1 ; do i=isq-1,ieq+1
342  u0(i,j) = cs%IDXDY2u(i,j)*(cs%DY2h(i+1,j)*sh_xx(i+1,j) - cs%DY2h(i,j)*sh_xx(i,j)) + &
343  cs%IDX2dyCu(i,j)*(cs%DX2q(i,j)*sh_xy(i,j) - cs%DX2q(i,j-1)*sh_xy(i,j-1))
344  enddo ; enddo
345  do j=jsq-1,jeq+1 ; do i=is-1,ieq+1
346  v0(i,j) = cs%IDXDY2v(i,j)*(cs%DY2q(i,j)*sh_xy(i,j) - cs%DY2q(i-1,j)*sh_xy(i-1,j)) - &
347  cs%IDX2dyCv(i,j)*(cs%DX2h(i,j+1)*sh_xx(i,j+1) - cs%DX2h(i,j)*sh_xx(i,j))
348  enddo ; enddo
349  endif
350 
351  do j=jsq,jeq+1 ; do i=isq,ieq+1
352  if ((cs%Smagorinsky_Kh) .or. (cs%Smagorinsky_Ah)) &
353  shear_mag = sqrt(sh_xx(i,j)*sh_xx(i,j) + &
354  0.25*((sh_xy(i-1,j-1)*sh_xy(i-1,j-1) + sh_xy(i,j)*sh_xy(i,j)) + &
355  (sh_xy(i-1,j)*sh_xy(i-1,j) + sh_xy(i,j-1)*sh_xy(i,j-1))))
356  if (cs%better_bound_Ah .or. cs%better_bound_Kh) then
357  hrat_min = min(1.0, &
358  0.5*min((h(i+1,j,k) + h(i,j,k)), (h(i,j,k) + h(i-1,j,k)), &
359  (h(i,j+1,k) + h(i,j,k)), (h(i,j-1,k) + h(i,j,k))) / &
360  (h(i,j,k) + h_neglect) )
361  visc_bound_rem = 1.0
362  endif
363 
364  if (cs%Laplacian) then
365  ! Determine the Laplacian viscosity at h points, using the
366  ! largest value from several parameterizations.
367  kh_scale = 1.0 ; if (rescale_kh) kh_scale = varmix%Res_fn_h(i,j)
368  if (cs%Smagorinsky_Kh) then
369  khsm = cs%LAPLAC_CONST_xx(i,j) * shear_mag
370  kh = kh_scale * max(cs%Kh_bg_xx(i,j), khsm)
371  if (cs%bound_Kh .and. .not.cs%better_bound_Kh) &
372  kh = min(kh, cs%Kh_Max_xx(i,j))
373  else
374  kh = kh_scale * cs%Kh_bg_xx(i,j)
375  endif
376  if (cs%better_bound_Kh) then
377  if (kh >= hrat_min*cs%Kh_Max_xx(i,j)) then
378  visc_bound_rem = 0.0
379  kh = hrat_min*cs%Kh_Max_xx(i,j)
380  else
381  visc_bound_rem = 1.0 - kh / (hrat_min*cs%Kh_Max_xx(i,j))
382  endif
383  endif
384 
385  if (cs%id_Kh_h>0) kh_h(i,j,k) = kh
386 
387  str_xx(i,j) = -kh * sh_xx(i,j)
388  else ! not Laplacian
389  str_xx(i,j) = 0.0
390  endif ! Laplacian
391 
392  if (cs%biharmonic) then
393 ! Determine the biharmonic viscosity at h points, using the
394 ! largest value from several parameterizations.
395  if (cs%Smagorinsky_Ah) then
396  if (cs%bound_Coriolis) then
397  ahsm = shear_mag * (cs%BIHARM_CONST_xx(i,j) + &
398  cs%Biharm_Const2_xx(i,j)*shear_mag)
399  else
400  ahsm = cs%BIHARM_CONST_xx(i,j) * shear_mag
401  endif
402  ah = max(cs%Ah_bg_xx(i,j), ahsm)
403  if (cs%bound_Ah .and. .not.cs%better_bound_Ah) &
404  ah = min(ah, cs%Ah_Max_xx(i,j))
405  else
406  ah = cs%Ah_bg_xx(i,j)
407  endif ! Smagorinsky_Ah
408  if (cs%better_bound_Ah) then
409  ah = min(ah, visc_bound_rem*hrat_min*cs%Ah_Max_xx(i,j))
410  endif
411 
412  if (cs%id_Ah_h>0) ah_h(i,j,k) = ah
413 
414  str_xx(i,j) = str_xx(i,j) + ah * &
415  (cs%DY_dxT(i,j)*(g%IdyCu(i,j)*u0(i,j) - g%IdyCu(i-1,j)*u0(i-1,j)) - &
416  cs%DX_dyT(i,j) *(g%IdxCv(i,j)*v0(i,j) - g%IdxCv(i,j-1)*v0(i,j-1)))
417 
418  endif ! biharmonic
419 
420  str_xx(i,j) = str_xx(i,j) * (h(i,j,k) * cs%reduction_xx(i,j))
421  enddo ; enddo
422 
423  do j=js-1,jeq ; do i=is-1,ieq
424  if ((cs%Smagorinsky_Kh) .or. (cs%Smagorinsky_Ah)) &
425  shear_mag = sqrt(sh_xy(i,j)*sh_xy(i,j) + &
426  0.25*((sh_xx(i,j)*sh_xx(i,j) + sh_xx(i+1,j+1)*sh_xx(i+1,j+1)) + &
427  (sh_xx(i,j+1)*sh_xx(i,j+1) + sh_xx(i+1,j)*sh_xx(i+1,j))))
428 
429  huq = (h(i,j,k) + h(i+1,j,k)) * (h(i,j+1,k) + h(i+1,j+1,k))
430  hvq = (h(i,j,k) + h(i,j+1,k)) * (h(i+1,j,k) + h(i+1,j+1,k))
431  hq = 2.0 * huq * hvq / (h_neglect3 + (huq + hvq) * &
432  ((h(i,j,k) + h(i+1,j+1,k)) + (h(i,j+1,k) + h(i+1,j,k))))
433 
434  if (cs%better_bound_Ah .or. cs%better_bound_Kh) then
435  hrat_min = min(1.0, &
436  0.5*min((h(i,j,k) + h(i+1,j,k)), (h(i,j+1,k) + h(i+1,j+1,k)), &
437  (h(i,j,k) + h(i,j+1,k)), (h(i+1,j,k) + h(i+1,j+1,k))) / &
438  (hq + h_neglect) )
439  visc_bound_rem = 1.0
440  endif
441 
442  if (cs%Laplacian) then
443  ! Determine the Laplacian viscosity at q points, using the
444  ! largest value from several parameterizations.
445  kh_scale = 1.0 ; if (rescale_kh) kh_scale = varmix%Res_fn_q(i,j)
446  if (cs%Smagorinsky_Kh) then
447  khsm = cs%LAPLAC_CONST_xy(i,j) * shear_mag
448  kh = kh_scale * max(cs%Kh_bg_xy(i,j), khsm)
449  if (cs%bound_Kh .and. .not.cs%better_bound_Kh) &
450  kh = min(kh, cs%Kh_Max_xy(i,j))
451  else
452  kh = kh_scale * cs%Kh_bg_xy(i,j)
453  endif
454  if (cs%better_bound_Kh) then
455  if (kh >= hrat_min*cs%Kh_Max_xy(i,j)) then
456  visc_bound_rem = 0.0
457  kh = hrat_min*cs%Kh_Max_xy(i,j)
458  else
459  visc_bound_rem = 1.0 - kh / (hrat_min*cs%Kh_Max_xy(i,j))
460  endif
461  endif
462 
463  if (cs%id_Kh_q>0) kh_q(i,j,k) = kh
464 
465  str_xy(i,j) = -kh * sh_xy(i,j)
466  else ! not Laplacian
467  str_xy(i,j) = 0.0
468  endif ! Laplacian
469 
470  if (cs%biharmonic) then
471  ! Determine the biharmonic viscosity at q points, using the
472  ! largest value from several parameterizations.
473  if (cs%Smagorinsky_Ah) then
474  if (cs%bound_Coriolis) then
475  ahsm = shear_mag * (cs%BIHARM_CONST_xy(i,j) + &
476  cs%Biharm_Const2_xy(i,j)*shear_mag)
477  else
478  ahsm = cs%BIHARM_CONST_xy(i,j) * shear_mag
479  endif
480  ah = max(cs%Ah_bg_xy(i,j), ahsm)
481  if (cs%bound_Ah .and. .not.cs%better_bound_Ah) &
482  ah = min(ah, cs%Ah_Max_xy(i,j))
483  else
484  ah = cs%Ah_bg_xy(i,j)
485  endif ! Smagorinsky_Ah
486  if (cs%better_bound_Ah) then
487  ah = min(ah, visc_bound_rem*hrat_min*cs%Ah_Max_xy(i,j))
488  endif
489 
490  if (cs%id_Ah_q>0) ah_q(i,j,k) = ah
491 
492  str_xy(i,j) = str_xy(i,j) + ah * &
493  (cs%DX_dyBu(i,j)*(u0(i,j+1)*g%IdxCu(i,j+1) - u0(i,j)*g%IdxCu(i,j)) + &
494  cs%DY_dxBu(i,j)*(v0(i+1,j)*g%IdyCv(i+1,j) - v0(i,j)*g%IdyCv(i,j)))
495 
496  endif ! biharmonic
497 
498  str_xy(i,j) = str_xy(i,j) * (hq * g%mask2dBu(i,j) * cs%reduction_xy(i,j))
499  enddo ; enddo
500 
501  do j=js,je ; do i=isq,ieq
502 ! Evaluate 1/h x.Div(h Grad u) or the biharmonic equivalent.
503  diffu(i,j,k) = ((g%IdyCu(i,j)*(cs%DY2h(i,j) *str_xx(i,j) - &
504  cs%DY2h(i+1,j)*str_xx(i+1,j)) + &
505  g%IdxCu(i,j)*(cs%DX2q(i,j-1)*str_xy(i,j-1) - &
506  cs%DX2q(i,j) *str_xy(i,j))) * &
507  g%IareaCu(i,j)) / (0.5*(h(i+1,j,k) + h(i,j,k)) + h_neglect)
508 
509  if (apply_obc) then ; if (obc%OBC_mask_u(i,j)) then
510  if ((obc%OBC_kind_u(i,j) == obc_flather_e) .or. &
511  (obc%OBC_kind_u(i,j) == obc_flather_w)) diffu(i,j,k) = 0.0
512  endif ; endif
513  enddo ; enddo
514 
515 ! Evaluate 1/h y.Div(h Grad u) or the biharmonic equivalent.
516  do j=jsq,jeq ; do i=is,ie
517  diffv(i,j,k) = ((g%IdyCv(i,j)*(cs%DY2q(i-1,j)*str_xy(i-1,j) - &
518  cs%DY2q(i,j) *str_xy(i,j)) - &
519  g%IdxCv(i,j)*(cs%DX2h(i,j) *str_xx(i,j) - &
520  cs%DX2h(i,j+1)*str_xx(i,j+1))) * &
521  g%IareaCv(i,j)) / (0.5*(h(i,j+1,k) + h(i,j,k)) + h_neglect)
522  if (apply_obc) then ; if (obc%OBC_mask_v(i,j)) then
523  if ((obc%OBC_kind_v(i,j) == obc_flather_n) .or. &
524  (obc%OBC_kind_v(i,j) == obc_flather_s)) diffv(i,j,k) = 0.0
525  endif ; endif
526  enddo ; enddo
527 
528  if (find_frictwork) then ; do j=js,je ; do i=is,ie
529  ! Diagnose str_xx*d_x u - str_yy*d_y v + str_xy*(d_y u + d_x v)
530  frictwork(i,j,k) = g%H_to_kg_m2 * ( &
531  (str_xx(i,j)*(u(i,j,k)-u(i-1,j,k))*g%IdxT(i,j) &
532  -str_xx(i,j)*(v(i,j,k)-v(i,j-1,k))*g%IdyT(i,j)) &
533  +0.25*((str_xy(i,j)*( &
534  (u(i,j+1,k)-u(i,j,k))*g%IdyBu(i,j) &
535  +(v(i+1,j,k)-v(i,j,k))*g%IdxBu(i,j) ) &
536  +str_xy(i-1,j-1)*( &
537  (u(i-1,j,k)-u(i-1,j-1,k))*g%IdyBu(i-1,j-1) &
538  +(v(i,j-1,k)-v(i-1,j-1,k))*g%IdxBu(i-1,j-1) )) &
539  +(str_xy(i-1,j)*( &
540  (u(i-1,j+1,k)-u(i-1,j,k))*g%IdyBu(i-1,j) &
541  +(v(i,j,k)-v(i-1,j,k))*g%IdxBu(i-1,j) ) &
542  +str_xy(i,j-1)*( &
543  (u(i,j,k)-u(i,j-1,k))*g%IdyBu(i,j-1) &
544  +(v(i+1,j-1,k)-v(i,j-1,k))*g%IdxBu(i,j-1) )) ) )
545  enddo ; enddo ; endif
546 
547  enddo ! end of k loop
548 
549  if (associated(meke)) then ; if (associated(meke%mom_src)) then
550  do j=js,je ; do i=is,ie ; meke%mom_src(i,j) = frictwork(i,j,1) ; enddo ; enddo
551  do k=2,nz ; do j=js,je ; do i=is,ie
552  meke%mom_src(i,j) = meke%mom_src(i,j) + frictwork(i,j,k)
553  enddo ; enddo ; enddo
554  endif ; endif
555 
556 ! Offer fields for averaging.
557  if (cs%id_diffu>0) call post_data(cs%id_diffu, diffu, cs%diag)
558  if (cs%id_diffv>0) call post_data(cs%id_diffv, diffv, cs%diag)
559  if (cs%id_FrictWork>0) call post_data(cs%id_FrictWork, frictwork, cs%diag)
560  if (cs%id_Ah_h>0) call post_data(cs%id_Ah_h, ah_h, cs%diag)
561  if (cs%id_Ah_q>0) call post_data(cs%id_Ah_q, ah_q, cs%diag)
562  if (cs%id_Kh_h>0) call post_data(cs%id_Kh_h, kh_h, cs%diag)
563  if (cs%id_Kh_q>0) call post_data(cs%id_Kh_q, kh_q, cs%diag)
564 
565 end subroutine horizontal_viscosity
